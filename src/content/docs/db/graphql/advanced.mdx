# GraphQL Advanced

## Apollo Federation

Apollo Federation lets you split a large GraphQL API into multiple independent **subgraphs** (microservices), each owning a portion of the schema. A **router** (Apollo Router or Apollo Gateway) composes them into a unified supergraph.

### Subgraph Setup

```bash
npm install @apollo/subgraph @apollo/server graphql
```

#### Users Subgraph

```javascript
// users-subgraph/index.js
import { ApolloServer } from '@apollo/server';
import { buildSubgraphSchema } from '@apollo/subgraph';
import { startStandaloneServer } from '@apollo/server/standalone';
import { gql } from 'graphql-tag';

const typeDefs = gql`
  extend schema
    @link(
      url: "https://specs.apollo.dev/federation/v2.0"
      import: ["@key", "@shareable"]
    )

  type User @key(fields: "id") {
    id: ID!
    name: String!
    email: String!
    createdAt: String!
  }

  type Query {
    user(id: ID!): User
    users: [User!]!
    me: User
  }
`;

const resolvers = {
  User: {
    // Entity resolver — called by router when other subgraphs need a User
    __resolveReference: async ({ id }, { db }) => db.users.findById(id),
  },
  Query: {
    user: (_, { id }, { db }) => db.users.findById(id),
    users: (_, __, { db }) => db.users.findAll(),
  },
};

const server = new ApolloServer({
  schema: buildSubgraphSchema({ typeDefs, resolvers }),
});
const { url } = await startStandaloneServer(server, { listen: { port: 4001 } });
console.log(`Users subgraph ready at ${url}`);
```

#### Orders Subgraph

```javascript
// orders-subgraph/index.js
const typeDefs = gql`
  extend schema @link(url: "https://specs.apollo.dev/federation/v2.0",
    import: ["@key", "@external", "@requires", "@provides"])

  # Reference the User type defined in the users subgraph
  type User @key(fields: "id") {
    id: ID!
    # @external means this field is defined in another subgraph
    name: String! @external
    # Access fields from the users subgraph to compute this field
    orders: [Order!]!
  }

  type Order @key(fields: "id") {
    id: ID!
    total: Float!
    status: OrderStatus!
    createdAt: String!
    # @requires: needs the `name` field from users subgraph (via @external)
    customerName: String! @requires(fields: "name")
    customerId: ID!
    customer: User!
  }

  enum OrderStatus { PENDING PROCESSING SHIPPED DELIVERED CANCELLED }

  type Query {
    order(id: ID!): Order
    orders(userId: ID): [Order!]!
  }
`;
```

### Federation Directives

| Directive                     | Purpose                                                            |
| ----------------------------- | ------------------------------------------------------------------ |
| `@key(fields: "id")`          | Marks a type as an entity (can be referenced across subgraphs)     |
| `@external`                   | Field is defined in another subgraph                               |
| `@requires(fields: "...")`    | Field needs values from @external fields (computes cross-subgraph) |
| `@provides(fields: "...")`    | Optimizer hint that this resolver can provide fields cheaply       |
| `@shareable`                  | Type/field can be defined in multiple subgraphs                    |
| `@override(from: "subgraph")` | Migrate a field from one subgraph to another                       |
| `@inaccessible`               | Hide field from the supergraph schema                              |

## Code-First with TypeGraphQL

TypeGraphQL lets you define your schema using TypeScript classes and decorators.

```bash
npm install type-graphql graphql reflect-metadata class-validator
```

```typescript
// schema/types/User.ts
import { ObjectType, Field, ID, Int } from 'type-graphql';

@ObjectType({ description: 'Represents a registered user' })
export class User {
  @Field(() => ID)
  id: string;

  @Field()
  name: string;

  @Field()
  email: string;

  @Field(() => Int)
  age: number;

  @Field({ nullable: true })
  bio?: string;

  @Field(() => [Post])
  posts: Post[];

  @Field()
  createdAt: Date;
}

// schema/inputs/CreateUserInput.ts
import { InputType, Field } from 'type-graphql';
import { IsEmail, MinLength, MaxLength, IsEnum } from 'class-validator';

@InputType()
export class CreateUserInput {
  @Field()
  @MinLength(2)
  @MaxLength(100)
  name: string;

  @Field()
  @IsEmail()
  email: string;

  @Field()
  @MinLength(8)
  password: string;

  @Field(() => UserRole, { defaultValue: UserRole.USER })
  @IsEnum(UserRole)
  role: UserRole;
}

// schema/resolvers/UserResolver.ts
import {
  Resolver,
  Query,
  Mutation,
  Arg,
  Ctx,
  ID,
  Authorized,
} from 'type-graphql';

@Resolver(() => User)
export class UserResolver {
  @Query(() => User, { nullable: true })
  async user(
    @Arg('id', () => ID) id: string,
    @Ctx() ctx: Context,
  ): Promise<User | null> {
    return ctx.db.users.findById(id);
  }

  @Query(() => [User])
  @Authorized(['ADMIN']) // role-based authorization
  async users(@Ctx() ctx: Context): Promise<User[]> {
    return ctx.db.users.findAll();
  }

  @Mutation(() => CreateUserPayload)
  async createUser(
    @Arg('input') input: CreateUserInput,
    @Ctx() ctx: Context,
  ): Promise<CreateUserPayload> {
    const user = await ctx.db.users.create(input);
    return { user, errors: [] };
  }

  // Field resolver for computed/relational fields
  @FieldResolver(() => [Post])
  async posts(@Root() user: User, @Ctx() ctx: Context): Promise<Post[]> {
    return ctx.loaders.postsByUser.load(user.id);
  }
}

// Building the schema
import { buildSchema } from 'type-graphql';
import 'reflect-metadata';

const schema = await buildSchema({
  resolvers: [UserResolver, PostResolver, OrderResolver],
  authChecker: ({ context: { currentUser } }, roles) => {
    if (!currentUser) return false;
    if (roles.length === 0) return true;
    return roles.some((role) => currentUser.roles.includes(role));
  },
  validate: true, // use class-validator automatically
});
```

## Strawberry (Python Code-First)

```python
# schema.py
import strawberry
from typing import Optional, List
from datetime import datetime

@strawberry.type
class User:
    id: strawberry.ID
    name: str
    email: str
    created_at: datetime

    @strawberry.field
    async def posts(self, info: strawberry.types.Info, first: int = 10) -> List["Post"]:
        return await info.context["loaders"]["posts_by_user"].load(self.id)

@strawberry.input
class CreateUserInput:
    name: str
    email: str
    password: str

@strawberry.type
class CreateUserPayload:
    user: Optional[User]
    errors: List[str]

@strawberry.type
class Query:
    @strawberry.field
    async def user(self, info: strawberry.types.Info, id: strawberry.ID) -> Optional[User]:
        db = info.context["db"]
        return await db.users.find_by_id(id)

    @strawberry.field
    async def users(self, info: strawberry.types.Info) -> List[User]:
        db = info.context["db"]
        return await db.users.find_all()

@strawberry.type
class Mutation:
    @strawberry.mutation
    async def create_user(
        self,
        info: strawberry.types.Info,
        input: CreateUserInput
    ) -> CreateUserPayload:
        db = info.context["db"]
        user = await db.users.create(input)
        return CreateUserPayload(user=user, errors=[])

schema = strawberry.Schema(query=Query, mutation=Mutation)

# FastAPI integration
import strawberry.fastapi
from fastapi import FastAPI

app = FastAPI()

async def get_context(request: Request) -> dict:
    return {
        "db": db,
        "current_user": await get_current_user(request),
        "loaders": create_loaders(db),
    }

graphql_app = strawberry.fastapi.GraphQLRouter(
    schema,
    context_getter=get_context,
)
app.include_router(graphql_app, prefix="/graphql")
```

## Authentication Patterns

```javascript
// Pattern 1: JWT in Authorization header → context
const server = new ApolloServer({ typeDefs, resolvers });

await startStandaloneServer(server, {
  context: async ({ req }) => {
    const token = req.headers.authorization?.replace('Bearer ', '');
    let currentUser = null;

    if (token) {
      try {
        const payload = jwt.verify(token, process.env.JWT_SECRET);
        currentUser = await db.users.findById(payload.sub);
      } catch {
        /* invalid or expired token */
      }
    }

    return { db, currentUser, loaders: createLoaders(db) };
  },
});

// Pattern 2: @auth custom directive
const { mapSchema, getDirective, MapperKind } = require('@graphql-tools/utils');
const { defaultFieldResolver } = require('graphql');

function authDirectiveTransformer(schema) {
  return mapSchema(schema, {
    [MapperKind.OBJECT_FIELD]: (fieldConfig) => {
      const authDirective = getDirective(schema, fieldConfig, 'auth')?.[0];
      if (!authDirective) return fieldConfig;

      const { requires } = authDirective;
      const { resolve = defaultFieldResolver } = fieldConfig;

      fieldConfig.resolve = async function (source, args, context, info) {
        if (!context.currentUser) {
          throw new GraphQLError('Unauthenticated', {
            extensions: { code: 'UNAUTHENTICATED' },
          });
        }
        if (requires && !context.currentUser.roles.includes(requires)) {
          throw new GraphQLError('Forbidden', {
            extensions: { code: 'FORBIDDEN' },
          });
        }
        return resolve(source, args, context, info);
      };

      return fieldConfig;
    },
  });
}
```

## File Uploads

```bash
npm install graphql-upload
```

```graphql
# Schema
scalar Upload

type Mutation {
  uploadFile(file: Upload!): UploadResult!
  uploadProfilePhoto(userId: ID!, photo: Upload!): User!
}
```

```javascript
import { GraphQLUpload, processRequest } from 'graphql-upload';

const resolvers = {
  Upload: GraphQLUpload,

  Mutation: {
    uploadFile: async (_, { file }) => {
      const { createReadStream, filename, mimetype, encoding } = await file;

      // Validate
      const allowedTypes = ['image/jpeg', 'image/png', 'application/pdf'];
      if (!allowedTypes.includes(mimetype)) {
        throw new GraphQLError('Unsupported file type', {
          extensions: { code: 'UNSUPPORTED_MEDIA_TYPE' },
        });
      }

      const MAX_SIZE = 10 * 1024 * 1024; // 10 MB
      let size = 0;
      const stream = createReadStream();
      stream.on('data', (chunk) => {
        size += chunk.length;
        if (size > MAX_SIZE) stream.destroy(new Error('File too large'));
      });

      // Stream to storage (S3, local filesystem, etc.)
      const key = `uploads/${Date.now()}-${filename}`;
      await s3
        .upload({
          Bucket: 'my-bucket',
          Key: key,
          Body: stream,
          ContentType: mimetype,
        })
        .promise();

      return { url: `https://cdn.example.com/${key}`, filename, mimetype };
    },
  },
};
```

## Subscriptions at Scale with Redis PubSub

The default in-memory PubSub doesn't work across multiple server instances. Use Redis for horizontal scaling:

```bash
npm install graphql-redis-subscriptions ioredis
```

```javascript
import { RedisPubSub } from 'graphql-redis-subscriptions';
import Redis from 'ioredis';

const options = {
  host: process.env.REDIS_HOST,
  port: parseInt(process.env.REDIS_PORT ?? '6379'),
  retryStrategy: (times) => Math.min(times * 50, 2000),
};

const pubsub = new RedisPubSub({
  publisher: new Redis(options),
  subscriber: new Redis(options),
});

// Usage is identical to in-memory PubSub
await pubsub.publish('ORDER_UPDATED', { orderUpdated: order });
pubsub.asyncIterator(['ORDER_UPDATED']);
```

## GraphQL Codegen

GraphQL Code Generator generates TypeScript types and React hooks from your schema and operations.

```bash
npm install -D @graphql-codegen/cli @graphql-codegen/typescript @graphql-codegen/typescript-operations @graphql-codegen/typescript-react-apollo
npx graphql-codegen init
```

```yaml
# codegen.yml
schema: http://localhost:4000/graphql
documents: 'src/**/*.graphql'
generates:
  src/generated/graphql.ts:
    plugins:
      - typescript
      - typescript-operations
      - typescript-react-apollo
    config:
      withHooks: true
      withComponent: false
      withRefetchFn: true
      scalars:
        DateTime: string
        Upload: File
```

```bash
# Generate types
npx graphql-codegen
# Watch mode
npx graphql-codegen --watch
```

Generated code usage:

```typescript
// Instead of writing hooks manually, use generated typed hooks
import { useGetUsersQuery, useCreateUserMutation } from './generated/graphql';

function UserList() {
  // Fully typed! data.users is User[]
  const { data, loading } = useGetUsersQuery({
    variables: { role: 'ADMIN' },
  });

    refetchQueries: [{ query: GetUsersDocument }],
  });
}
```

## Testing GraphQL

### Server-Side Testing (Jest + Supertest)

```javascript
// __tests__/users.test.js
import { ApolloServer } from '@apollo/server';
import request from 'supertest';
import express from 'express';
import { expressMiddleware } from '@apollo/server/express4';
import { typeDefs, resolvers } from '../schema';

describe('User API', () => {
  let app;
  let server;
  let mockDb;

  beforeAll(async () => {
    mockDb = {
      users: {
        findById: jest.fn(),
        findAll: jest.fn(),
        create: jest.fn(),
      },
    };

    server = new ApolloServer({ typeDefs, resolvers });
    await server.start();

    app = express();
    app.use(
      '/graphql',
      express.json(),
      expressMiddleware(server, {
        context: async () => ({ db: mockDb }),
      }),
    );
  });

  afterAll(async () => await server.stop());

  test('query user by id', async () => {
    mockDb.users.findById.mockResolvedValue({
      id: '1',
      name: 'Alice',
      email: 'alice@test.com',
    });

    const res = await request(app).post('/graphql').send({
      query: `query { user(id: "1") { id name email } }`,
    });

    expect(res.status).toBe(200);
    expect(res.body.errors).toBeUndefined();
    expect(res.body.data.user).toEqual({
      id: '1',
      name: 'Alice',
      email: 'alice@test.com',
    });
  });

  test('returns null for non-existent user', async () => {
    mockDb.users.findById.mockResolvedValue(null);

    const res = await request(app)
      .post('/graphql')
      .send({ query: `query { user(id: "999") { id } }` });

    expect(res.body.data.user).toBeNull();
  });
});
```

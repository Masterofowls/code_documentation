# GraphQL Queries & Mutations

This section covers writing GraphQL operations from the client's perspective — queries, mutations, subscriptions, and the patterns that make them powerful.

## Query Syntax

### Basic Query

```graphql
# Simple query — get a user by ID
query GetUser {
  user(id: "42") {
    id
    name
    email
  }
}
```

### Operation Names

Always name your operations in production code. It makes debugging, logging, and tracing much easier.

```graphql
# Named operation — always use in production
query GetUserProfile {
  user(id: "42") {
    id
    name
    email
    bio
    createdAt
  }
}

# Multiple operations in a document — must specify which to execute
query GetAllUsers {
  users { id name }
}

query GetSingleUser($id: ID!) {
  user(id: $id) { id name email }
}
```

### Variables

Variables keep queries reusable and prevent injection by separating the query from the data:

```graphql
# Declare variables with $ prefix and type
query GetUser($id: ID!, $withPosts: Boolean = false) {
  user(id: $id) {
    id
    name
    email
    posts @include(if: $withPosts) {
      id
      title
    }
  }
}

# Variables are passed as a separate JSON object
# { "id": "42", "withPosts": true }
```

### Aliases

Use aliases to rename fields in the response, or to query the same field multiple times:

```graphql
query GetTwoUsers {
  alice: user(id: "1") {
    id
    name
    email
  }
  bob: user(id: "2") {
    id
    name
    email
  }
}

# Response:
# { "alice": { "id": "1", "name": "Alice" }, "bob": { "id": "2", "name": "Bob" } }

# Alias to rename a field
query GetPosts {
  allPosts: posts(first: 10) {
    postId: id
    postTitle: title
    createdDate: createdAt
  }
}
```

### @skip and @include Directives

```graphql
query GetUserConditional(
  $id: ID!,
  $showEmail: Boolean!,
  $skipBio: Boolean!
) {
  user(id: $id) {
    id
    name
    email @include(if: $showEmail)   # Include only if showEmail is true
    bio   @skip(if: $skipBio)         # Skip if skipBio is true
  }
}
# Variables: { "id": "1", "showEmail": true, "skipBio": false }
```

## Fragments

### Named Fragments

```graphql
# Define a fragment once, reuse across multiple queries
fragment UserBasic on User {
  id
  name
  email
  createdAt
}

fragment PostSummary on Post {
  id
  title
  slug
  publishedAt
  author {
    ...UserBasic
  }
}

query GetFeed {
  posts(first: 20) {
    edges {
      node {
        ...PostSummary
        excerpt
        tags
      }
    }
    pageInfo {
      hasNextPage
      endCursor
    }
  }
}

query GetPost($slug: String!) {
  post(slug: $slug) {
    ...PostSummary
    body
    comments(first: 10) {
      id
      text
      author {
        ...UserBasic
      }
    }
  }
}
```

### Inline Fragments

```graphql
# For union types and interfaces — specify type-specific fields
query Search($query: String!) {
  search(query: $query) {
    __typename  # Always include to distinguish types client-side
    ... on User {
      id
      name
      email
    }
    ... on Post {
      id
      title
      slug
    }
    ... on Comment {
      id
      text
      post {
        title
      }
    }
  }
}
```

### Fragment Colocation

Co-locate fragments with the components that use them for better maintainability:

```typescript
// UserCard.tsx
import { gql, useFragment } from '@apollo/client';

export const USER_CARD_FRAGMENT = gql`
  fragment UserCardData on User {
    id
    name
    email
    avatar
  }
`;

function UserCard({ user }: { user: UserCardData }) {
  return <div>{user.name}</div>;
}

// FeedPage.tsx
const FEED_QUERY = gql`
  ${USER_CARD_FRAGMENT}
  query GetFeed {
    posts(first: 10) {
      edges {
        node {
          id
          title
          author {
            ...UserCardData  # Compose the component's fragment here
          }
        }
      }
    }
  }
`;
```

## Mutations

### Basic Mutation

```graphql
mutation CreatePost($input: CreatePostInput!) {
  createPost(input: $input) {
    id
    title
    slug
    status
    createdAt
    author {
      id
      name
    }
  }
}

# Variables:
# {
#   "input": {
#     "title": "Getting Started with GraphQL",
#     "body": "GraphQL is...",
#     "status": "DRAFT",
#     "tags": ["graphql", "api"]
#   }
# }
```

### Multiple Root Fields in a Mutation

```graphql
# Mutations execute sequentially (unlike query fields which are parallel)
mutation UpdateAndPublish($id: ID!, $input: UpdatePostInput!) {
  updatePost(id: $id, input: $input) {
    id
    title
    status
  }
  publishPost(id: $id) {
    id
    status
    publishedAt
  }
}
```

### Error Patterns in Mutations

<Tabs>
  <TabItem label="Errors Field (Union Pattern)">

The union return type pattern gives typed errors as part of the schema:

```graphql
union CreateUserResult = User | ValidationError | DuplicateEmailError

type ValidationError {
  message: String!
  fields: [FieldError!]!
}

type FieldError {
  field: String!
  message: String!
}

type DuplicateEmailError {
  message: String!
  email: String!
}

type Mutation {
  createUser(input: CreateUserInput!): CreateUserResult!
}
```

```graphql
mutation CreateUser($input: CreateUserInput!) {
  createUser(input: $input) {
    ... on User {
      id
      name
      email
    }
    ... on ValidationError {
      message
      fields {
        field
        message
      }
    }
    ... on DuplicateEmailError {
      message
      email
    }
  }
}
```

  </TabItem>
  <TabItem label="HTTP 200 + errors Array">

GraphQL always returns HTTP 200 (even for errors). Errors are in the `errors` array:

```graphql
mutation CreateUser($input: CreateUserInput!) {
  createUser(input: $input) {
    id
    name
    email
  }
}
```

```json
// Error response (still HTTP 200)
{
  "data": { "createUser": null },
  "errors": [
    {
      "message": "Email already in use",
      "locations": [{ "line": 2, "column": 3 }],
      "path": ["createUser"],
      "extensions": {
        "code": "DUPLICATE_EMAIL",
        "field": "email"
      }
    }
  ]
}
```

## Pagination Patterns

### Offset-Based Pagination

```graphql
query GetPosts($page: Int!, $perPage: Int!) {
  posts(page: $page, perPage: $perPage) {
    data {
      id
      title
    }
    total
    page
    perPage
    totalPages
  }
}
```

### Cursor-Based (Connection) Pagination

```graphql
query GetPosts($first: Int!, $after: String) {
  posts(first: $first, after: $after, orderBy: CREATED_AT, order: DESC) {
    edges {
      cursor
      node {
        id
        title
        slug
        publishedAt
      }
    }
    pageInfo {
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    totalCount
  }
}

# Load next page using endCursor from previous response:
# { "first": 10, "after": "Y3Vyc29yOnYyOpHOA..." }
```

## Apollo Client Patterns

### useQuery

```typescript
import { gql, useQuery } from '@apollo/client';

const GET_POSTS = gql`
  query GetPosts($first: Int!, $after: String) {
    posts(first: $first, after: $after) {
      edges { node { id title slug } }
      pageInfo { hasNextPage endCursor }
    }
  }
`;

function PostList() {
  const { data, loading, error, fetchMore } = useQuery(GET_POSTS, {
    variables: { first: 10 },
    notifyOnNetworkStatusChange: true,
  });

  if (loading) return <Spinner />;
  if (error) return <ErrorMessage error={error} />;

  const loadMore = () => {
    fetchMore({
      variables: { after: data.posts.pageInfo.endCursor },
      updateQuery: (prev, { fetchMoreResult }) => ({
        posts: {
          ...fetchMoreResult.posts,
          edges: [...prev.posts.edges, ...fetchMoreResult.posts.edges],
        },
      }),
    });
  };

  return (
    <>
      {data.posts.edges.map(({ node }) => <PostCard key={node.id} post={node} />)}
      {data.posts.pageInfo.hasNextPage && <button onClick={loadMore}>Load More</button>}
    </>
  );
}
```

### useMutation with Optimistic Updates

```typescript
const CREATE_POST = gql`
  mutation CreatePost($input: CreatePostInput!) {
    createPost(input: $input) {
      id
      title
      slug
      status
      createdAt
    }
  }
`;

function CreatePostForm() {
  const [createPost, { loading, error }] = useMutation(CREATE_POST, {
    // Optimistic update: assume success and update UI immediately
    optimisticResponse: ({ input }) => ({
      createPost: {
        __typename: 'Post',
        id: 'temp-id',
        title: input.title,
        slug: slugify(input.title),
        status: 'DRAFT',
        createdAt: new Date().toISOString(),
      },
    }),

    // Update the cache after mutation
    update(cache, { data }) {
      cache.modify({
        fields: {
          posts(existingPosts = { edges: [] }) {
            const newPostRef = cache.writeFragment({
              data: data.createPost,
              fragment: gql`fragment NewPost on Post { id title slug status createdAt }`,
            });
            return {
              ...existingPosts,
              edges: [{ node: newPostRef, cursor: '' }, ...existingPosts.edges],
            };
          },
        },
      });
    },

    // Alternatively, just refetch the affected query
    refetchQueries: ['GetPosts'],
  });
}
```

### Cache Manipulation

```typescript
const client = useApolloClient();

// Write to cache directly (without a network request)
client.cache.writeQuery({
  query: GET_USER,
  variables: { id: '42' },
  data: { user: updatedUser },
});

// Modify cached fields
client.cache.modify({
  id: client.cache.identify({ __typename: 'User', id: '42' }),
  fields: {
    name: () => 'New Name',
    posts: (existingPosts, { readField }) => existingPosts,
  },
});

// Evict a specific object from cache (forces refetch on next access)
client.cache.evict({
  id: client.cache.identify({ __typename: 'User', id: '42' }),
});
client.cache.gc(); // Garbage collect orphaned references
```

## Subscriptions

```typescript
import { gql, useSubscription } from '@apollo/client';

const POST_PUBLISHED = gql`
  subscription OnPostPublished {
    postPublished {
      id
      title
      slug
      author {
        name
      }
    }
  }
`;

function LiveFeed() {
  const { data, loading } = useSubscription(POST_PUBLISHED, {
    onData: ({ data, client }) => {
      // Update the cache when a new post arrives
      client.cache.modify({
        fields: {
          posts(existing = { edges: [] }) {
            const newRef = client.cache.writeFragment({
              data: data.data.postPublished,
              fragment: gql`fragment NewPost on Post { id title slug }`,
            });
            return {
              ...existing,
              edges: [{ node: newRef }, ...existing.edges],
            };
          },
        },
      });
    },
  });

  return <div>Listening for new posts...</div>;
}
```

## HTTP Caching GraphQL

GraphQL's default use of `POST` prevents HTTP caching. Solutions:

<Tabs>
  <TabItem label="GET for Queries">

```typescript
// Apollo Client: use GET for queries
const link = new HttpLink({
  uri: '/graphql',
  useGETForQueries: true,  // Queries use GET, mutations use POST
  fetchOptions: { cache: 'default' },
});
```

CDN then caches GET responses by URL (query + variables).

  </TabItem>
  <TabItem label="Persisted Queries + CDN">

```typescript
// 1. Hash the query
import { sha256 } from 'crypto-hash';
import { createPersistedQueryLink } from '@apollo/client/link/persisted-queries';

const link = createPersistedQueryLink({ sha256 })
  .concat(new HttpLink({ uri: '/graphql', useGETForQueries: true }));

// 2. First request: GET /graphql?extensions={"persistedQuery":{"version":1,"sha256Hash":"abc..."}}
// 3. Server responds with PersistedQueryNotFound if unknown
// 4. Client retries with full query
// 5. Subsequent requests: hash-only GET → CDN can cache
```

  </TabItem>
  <TabItem label="Cache-Control Headers">

```typescript
// Server: set Cache-Control per resolver using @cacheControl directive
// Apollo Server plugin reads these and sets HTTP headers

import responseCachePlugin from '@apollo/server-plugin-response-cache';

const server = new ApolloServer({
  typeDefs: `
    type Query {
      posts: [Post!]! @cacheControl(maxAge: 60)
      me: User @cacheControl(maxAge: 0, scope: PRIVATE)
    }
  `,
  plugins: [responseCachePlugin()],
});
```

  </TabItem>
</Tabs>

:::tip[Query Batching]
Apollo Client automatically batches multiple queries fired in the same event loop tick into a single HTTP request when using `BatchHttpLink`:

```typescript
import { BatchHttpLink } from '@apollo/client/link/batch-http';
const link = new BatchHttpLink({ uri: '/graphql', batchMax: 5, batchInterval: 20 });
```

This reduces HTTP overhead at the cost of increased latency per individual query.
:::

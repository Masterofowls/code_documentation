# MySQL Advanced Features

## Views

Views are stored SQL queries that act like virtual tables. They simplify complex queries,
provide an abstraction layer, and can restrict column/row access.

```sql
-- Create a simple view
CREATE VIEW active_users AS
  SELECT id, name, email, created_at
  FROM users
  WHERE status = 1;

-- Query the view like a table
SELECT * FROM active_users WHERE name LIKE 'A%';

-- Create or replace (no error if exists)
CREATE OR REPLACE VIEW active_users AS
  SELECT id, name, email, created_at
  FROM users
  WHERE status = 1;

-- View with WITH CHECK OPTION
-- Prevents INSERT/UPDATE through view that would make row invisible to view
CREATE VIEW premium_users AS
  SELECT * FROM users WHERE subscription_tier = 'premium'
  WITH CHECK OPTION;
-- Now: INSERT INTO premium_users (name, subscription_tier) VALUES ('Bob', 'free')
-- ERROR 1369: CHECK OPTION failed — 'free' not visible through this view

-- Updatable views (restrictions apply)
-- A view is updatable only if it:
-- 1. Does NOT have: DISTINCT, GROUP BY, HAVING, UNION, subquery in SELECT list
-- 2. References only ONE base table
-- 3. Does NOT use aggregate functions
UPDATE active_users SET name = 'Updated' WHERE id = 1;   -- Works on simple view

-- View with JOIN (not updatable)
CREATE VIEW order_summary AS
  SELECT u.name, COUNT(o.id) AS order_count, SUM(o.total) AS total_spent
  FROM users u
  LEFT JOIN orders o ON u.id = o.user_id
  GROUP BY u.id, u.name;

-- List views
SHOW FULL TABLES WHERE Table_type = 'VIEW';
SELECT TABLE_NAME, VIEW_DEFINITION FROM information_schema.VIEWS WHERE TABLE_SCHEMA = DATABASE();
```

## Stored Procedures

Stored procedures are precompiled SQL code blocks stored in the database.
They reduce network round-trips and can encapsulate business logic.

```sql
-- Change delimiter so semicolons inside procedure don't end it
DELIMITER //

CREATE PROCEDURE GetUserOrders(
  IN  p_user_id   INT,
  IN  p_limit     INT,
  OUT p_total     DECIMAL(12,2)
)
BEGIN
  -- Declare variables
  DECLARE v_count INT DEFAULT 0;

  -- Handle errors
  DECLARE EXIT HANDLER FOR SQLEXCEPTION
  BEGIN
    ROLLBACK;
    SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'An error occurred in GetUserOrders';
  END;

  -- Validate input
  IF p_limit <= 0 THEN
    SET p_limit = 10;
  END IF;

  -- Main query
  SELECT * FROM orders
  WHERE user_id = p_user_id
  ORDER BY created_at DESC
  LIMIT p_limit;

  -- Set OUT parameter
  SELECT SUM(total) INTO p_total
  FROM orders
  WHERE user_id = p_user_id;

END //

DELIMITER ;

-- Call the procedure
CALL GetUserOrders(1, 5, @total);
SELECT @total;

-- Procedure with CURSOR (iterate over a result set)
DELIMITER //

CREATE PROCEDURE ProcessPendingOrders()
BEGIN
  DECLARE v_done     BOOLEAN DEFAULT FALSE;
  DECLARE v_order_id INT;
  DECLARE v_total    DECIMAL(10,2);

  DECLARE cur CURSOR FOR
    SELECT id, total FROM orders WHERE status = 'pending';

  DECLARE CONTINUE HANDLER FOR NOT FOUND SET v_done = TRUE;

  OPEN cur;

  order_loop: LOOP
    FETCH cur INTO v_order_id, v_total;
    IF v_done THEN
      LEAVE order_loop;
    END IF;

    -- Process each order
    UPDATE orders SET status = 'processing' WHERE id = v_order_id;

    IF v_total > 1000 THEN
      INSERT INTO high_value_alerts (order_id, total) VALUES (v_order_id, v_total);
    END IF;
  END LOOP;

  CLOSE cur;
END //

DELIMITER ;

-- Show all stored procedures
SHOW PROCEDURE STATUS WHERE Db = DATABASE();
SHOW CREATE PROCEDURE GetUserOrders\G

-- Drop a procedure
DROP PROCEDURE IF EXISTS GetUserOrders;
```

## Stored Functions

Stored functions return a single value and can be used in SQL expressions.

```sql
DELIMITER //

CREATE FUNCTION FormatCurrency(amount DECIMAL(12,2), currency CHAR(3))
  RETURNS VARCHAR(20)
  DETERMINISTIC              -- Same inputs always produce same output
  NO SQL                     -- Does not read or write database tables
BEGIN
  RETURN CONCAT(currency, ' ', FORMAT(amount, 2));
END //

CREATE FUNCTION GetUserOrderCount(p_user_id INT)
  RETURNS INT
  READS SQL DATA             -- Reads but does not modify data
  DETERMINISTIC
BEGIN
  DECLARE v_count INT;
  SELECT COUNT(*) INTO v_count FROM orders WHERE user_id = p_user_id;
  RETURN v_count;
END //

DELIMITER ;

-- Use function in query
SELECT name, FormatCurrency(balance, 'USD') AS formatted_balance FROM accounts;
SELECT name, GetUserOrderCount(id) AS order_count FROM users ORDER BY order_count DESC LIMIT 10;

-- Function characteristics:
-- DETERMINISTIC: cached, safer for replication
-- NOT DETERMINISTIC: default if not specified
-- NO SQL: no SQL statements
-- CONTAINS SQL: has SQL but no reads/writes (e.g., SET @var = 1)
-- READS SQL DATA: SELECT only
-- MODIFIES SQL DATA: INSERT/UPDATE/DELETE
```

## Triggers

Triggers execute automatically in response to INSERT, UPDATE, or DELETE events.

```sql
DELIMITER //

-- BEFORE INSERT: validate or modify data before it's saved
CREATE TRIGGER before_user_insert
BEFORE INSERT ON users
FOR EACH ROW
BEGIN
  -- Normalize email to lowercase
  SET NEW.email = LOWER(NEW.email);
  -- Set created_at if not provided
  IF NEW.created_at IS NULL THEN
    SET NEW.created_at = NOW();
  END IF;
END //

-- AFTER INSERT: log or cascade actions
CREATE TRIGGER after_user_insert
AFTER INSERT ON users
FOR EACH ROW
BEGIN
  INSERT INTO audit_log (table_name, action, record_id, changed_at)
  VALUES ('users', 'INSERT', NEW.id, NOW());
END //

-- BEFORE UPDATE: track changes
CREATE TRIGGER before_order_update
BEFORE UPDATE ON orders
FOR EACH ROW
BEGIN
  IF OLD.total != NEW.total THEN
    INSERT INTO order_history (order_id, old_total, new_total, changed_at, changed_by)
    VALUES (OLD.id, OLD.total, NEW.total, NOW(), CURRENT_USER());
  END IF;
END //

-- AFTER DELETE: cascade to related tables (alternative to FK ON DELETE)
CREATE TRIGGER after_user_delete
AFTER DELETE ON users
FOR EACH ROW
BEGIN
  INSERT INTO deleted_users_archive (id, name, email, deleted_at)
  VALUES (OLD.id, OLD.name, OLD.email, NOW());
END //

DELIMITER ;

-- Show triggers
SHOW TRIGGERS FROM mydb;
SHOW TRIGGERS LIKE 'users%';

-- Drop trigger
DROP TRIGGER IF EXISTS before_user_insert;

-- Trigger limitations in MySQL:
-- Cannot call stored procedures with OUT/INOUT parameters
-- Cannot use CALL for procedures that return result sets
-- Cannot use transactions (BEGIN/COMMIT/ROLLBACK) in trigger body
-- Cannot modify the triggering table itself (would cause infinite loop)
-- One trigger per event/timing per table (MySQL 5.7+: multiple allowed)
```

## Events (Scheduled Tasks)

```sql
-- Enable event scheduler (must be ON for events to run)
SET GLOBAL event_scheduler = ON;
SHOW VARIABLES LIKE 'event_scheduler';

-- One-time event
CREATE EVENT cleanup_old_sessions
ON SCHEDULE AT CURRENT_TIMESTAMP + INTERVAL 1 HOUR
DO
  DELETE FROM sessions WHERE last_active < DATE_SUB(NOW(), INTERVAL 24 HOUR);

-- Recurring event
CREATE EVENT daily_stats_rollup
ON SCHEDULE EVERY 1 DAY
STARTS '2024-01-01 02:00:00'     -- Start at 2 AM
ON COMPLETION PRESERVE           -- Keep after last run (PRESERVE vs NOT PRESERVE)
ENABLE
COMMENT 'Daily statistics aggregation'
DO
BEGIN
  INSERT INTO daily_stats (date, total_orders, total_revenue)
  SELECT DATE(created_at), COUNT(*), SUM(total)
  FROM orders
  WHERE DATE(created_at) = CURDATE() - INTERVAL 1 DAY
  ON DUPLICATE KEY UPDATE
    total_orders = VALUES(total_orders),
    total_revenue = VALUES(total_revenue);
END;

-- Show events
SHOW EVENTS FROM mydb;
SHOW CREATE EVENT daily_stats_rollup\G

-- Enable/disable an event
ALTER EVENT daily_stats_rollup DISABLE;
ALTER EVENT daily_stats_rollup ENABLE;

-- Drop an event
DROP EVENT IF EXISTS daily_stats_rollup;
```

## MySQL JSON Functions

```sql
CREATE TABLE products (
  id       INT AUTO_INCREMENT PRIMARY KEY,
  details  JSON NOT NULL
);

INSERT INTO products (details) VALUES
  ('{"name":"Laptop","price":999.99,"specs":{"ram":"16GB","cpu":"i7"},"tags":["electronics","sale"]}');

-- Extracting values
SELECT details->>'$.name' FROM products;                         -- Unquoted string
SELECT details->'$.price' FROM products;                         -- Quoted: "999.99"
SELECT JSON_EXTRACT(details, '$.specs.ram') FROM products;       -- "16GB"
SELECT JSON_UNQUOTE(JSON_EXTRACT(details, '$.name')) FROM products; -- Laptop (no quotes)

-- Modifying JSON
UPDATE products SET details = JSON_SET(details, '$.price', 899.99) WHERE id = 1;
UPDATE products SET details = JSON_INSERT(details, '$.discount', 0.10) WHERE id = 1;
UPDATE products SET details = JSON_REPLACE(details, '$.name', 'Gaming Laptop') WHERE id = 1;
UPDATE products SET details = JSON_REMOVE(details, '$.discount') WHERE id = 1;

-- Building JSON
SELECT JSON_OBJECT('name', 'Alice', 'age', 30);
SELECT JSON_ARRAY('tag1', 'tag2', 'tag3');
SELECT JSON_MERGE_PATCH('{"a":1,"b":2}', '{"b":3,"c":4}');  -- {"a":1,"b":3,"c":4}

-- Searching within JSON
SELECT * FROM products WHERE JSON_CONTAINS(details->'$.tags', '"sale"');
SELECT * FROM products WHERE JSON_SEARCH(details, 'one', 'electronics') IS NOT NULL;

-- Aggregating to JSON
SELECT JSON_ARRAYAGG(name) AS names FROM users;
SELECT JSON_OBJECTAGG(id, name) AS id_name_map FROM users;

-- JSON_TABLE — expand JSON array into rows
SELECT jt.*
FROM products,
     JSON_TABLE(details->'$.tags', '$[*]' COLUMNS (tag VARCHAR(50) PATH '$')) AS jt;

-- Check if valid JSON
SELECT JSON_VALID('{"key":"value"}');   -- 1
SELECT JSON_VALID('not json');           -- 0

-- JSON path expressions
-- $.key            — property access
-- $[0]             — array index
-- $[*]             — all array elements
-- $.key[*].nested  — nested in array elements
-- **               — recursive wildcard
```

## Table Partitioning

Partitioning divides a large table into smaller physical pieces while appearing as one table.

```sql
-- RANGE partitioning — divide by column value ranges
CREATE TABLE orders_partitioned (
  id         INT NOT NULL AUTO_INCREMENT,
  created_at DATETIME NOT NULL,
  total      DECIMAL(10,2),
  PRIMARY KEY (id, created_at)  -- Partition key must be part of PRIMARY KEY
) PARTITION BY RANGE (YEAR(created_at)) (
  PARTITION p2022 VALUES LESS THAN (2023),
  PARTITION p2023 VALUES LESS THAN (2024),
  PARTITION p2024 VALUES LESS THAN (2025),
  PARTITION p_future VALUES LESS THAN MAXVALUE
);

-- LIST partitioning — divide by explicit values
CREATE TABLE employees_partitioned (
  id         INT NOT NULL,
  department ENUM('HR','Engineering','Sales','Marketing'),
  name       VARCHAR(100),
  PRIMARY KEY (id, department)
) PARTITION BY LIST COLUMNS(department) (
  PARTITION p_people   VALUES IN ('HR', 'Marketing'),
  PARTITION p_tech     VALUES IN ('Engineering'),
  PARTITION p_revenue  VALUES IN ('Sales')
);

-- HASH partitioning — distribute evenly across N partitions
CREATE TABLE logs_hash (
  id         BIGINT NOT NULL AUTO_INCREMENT,
  message    TEXT,
  PRIMARY KEY (id)
) PARTITION BY HASH(id) PARTITIONS 8;

-- KEY partitioning — similar to HASH, uses MySQL's hashing function
CREATE TABLE sessions_key (
  session_id CHAR(32) NOT NULL,
  user_id    INT,
  data       TEXT,
  PRIMARY KEY (session_id)
) PARTITION BY KEY(session_id) PARTITIONS 4;

-- Partition pruning — MySQL skips irrelevant partitions
EXPLAIN SELECT * FROM orders_partitioned WHERE created_at BETWEEN '2024-01-01' AND '2024-12-31';
-- partitions column shows only 'p2024'

-- Managing partitions
ALTER TABLE orders_partitioned ADD PARTITION (PARTITION p2025 VALUES LESS THAN (2026));
ALTER TABLE orders_partitioned DROP PARTITION p2022;
ALTER TABLE orders_partitioned TRUNCATE PARTITION p2023;
ALTER TABLE orders_partitioned REORGANIZE PARTITION p_future INTO (
  PARTITION p2025 VALUES LESS THAN (2026),
  PARTITION p_future VALUES LESS THAN MAXVALUE
);

-- Show partition info
SELECT PARTITION_NAME, TABLE_ROWS, DATA_LENGTH
FROM information_schema.PARTITIONS
WHERE TABLE_SCHEMA = DATABASE() AND TABLE_NAME = 'orders_partitioned';
```

## Replication

MySQL replication copies data from a master (source) to one or more replica servers.

```sql
-- Binary log must be enabled on source
-- server_id must be unique per server

-- On SOURCE server:
SHOW MASTER STATUS;  -- Shows current binary log file and position

-- GTID-based replication (recommended for MySQL 5.6+)
-- GTID = Global Transaction Identifier: server_uuid:transaction_id
SHOW VARIABLES LIKE 'gtid_mode';
-- Set in my.cnf: gtid_mode=ON, enforce_gtid_consistency=ON

-- Replication formats:
-- STATEMENT: logs SQL statements — compact but may be non-deterministic
-- ROW: logs actual row changes — safe but verbose
-- MIXED: uses STATEMENT usually, ROW when needed

-- Create replication user on source
CREATE USER 'replicator'@'replica_host' IDENTIFIED BY 'replpass'
  REQUIRE SSL;  -- Enforce SSL for replication
GRANT REPLICATION SLAVE ON *.* TO 'replicator'@'replica_host';

-- On REPLICA server (GTID-based):
CHANGE MASTER TO
  MASTER_HOST = 'source_server',
  MASTER_USER = 'replicator',
  MASTER_PASSWORD = 'replpass',
  MASTER_AUTO_POSITION = 1;   -- GTID auto-position

START REPLICA;
SHOW REPLICA STATUS\G  -- Check Replica_IO_Running and Replica_SQL_Running

-- Monitor replication lag
SELECT TIMESTAMPDIFF(SECOND, 
  MIN(TIMESTAMP(CONCAT(MASTER_LOG_FILE, MASTER_LOG_POS))),
  NOW()) AS lag_seconds
FROM information_schema.PROCESSLIST;
-- Easier: check Seconds_Behind_Source in SHOW REPLICA STATUS
```

## InnoDB Internals

Understanding InnoDB internals helps diagnose performance issues.

<CardGrid>
  <Card title="Buffer Pool" icon="seti:db">
    The buffer pool caches data pages and index pages in memory. The most important
    InnoDB setting. Set to 70-80% of available RAM for a dedicated database server.
  </Card>
  <Card title="MVCC" icon="document">
    Multi-Version Concurrency Control allows readers to see a consistent snapshot
    of data without blocking writers. Uses undo logs to reconstruct old row versions.
  </Card>
  <Card title="Redo Log" icon="approve-check">
    Write-ahead log for crash recovery. All changes are written here before the
    buffer pool pages are flushed to disk. Controls durability.
  </Card>
  <Card title="Change Buffer" icon="magnifier">
    Caches changes to secondary index pages when those pages are not in the buffer
    pool. Merged when pages are loaded. Reduces random I/O for writes.
  </Card>
</CardGrid>

## Performance Schema

The Performance Schema collects granular performance data at runtime.

```sql
-- Check if Performance Schema is enabled (default: ON in MySQL 5.6+)
SHOW VARIABLES LIKE 'performance_schema';

-- Top queries by total execution time
SELECT DIGEST_TEXT, COUNT_STAR, SUM_TIMER_WAIT/1e12 AS total_sec,
       AVG_TIMER_WAIT/1e12 AS avg_sec, SUM_ROWS_EXAMINED
FROM performance_schema.events_statements_summary_by_digest
ORDER BY SUM_TIMER_WAIT DESC LIMIT 10;

-- Current connections summary
SELECT USER, HOST, DB, COMMAND, TIME, STATE
FROM performance_schema.threads WHERE PROCESSLIST_ID IS NOT NULL;

-- Table I/O statistics
SELECT OBJECT_SCHEMA, OBJECT_NAME,
       COUNT_READ, COUNT_WRITE, COUNT_FETCH,
       SUM_TIMER_READ/1e12 AS read_sec, SUM_TIMER_WRITE/1e12 AS write_sec
FROM performance_schema.table_io_waits_summary_by_table
ORDER BY SUM_TIMER_WAIT DESC LIMIT 10;

-- Mutex/lock waits
SELECT EVENT_NAME, COUNT_STAR, SUM_TIMER_WAIT/1e12 AS wait_sec
FROM performance_schema.events_waits_summary_global_by_event_name
WHERE COUNT_STAR > 0
ORDER BY SUM_TIMER_WAIT DESC LIMIT 10;
```

## sys Schema

The `sys` schema (available since MySQL 5.7) wraps Performance Schema into human-readable views.

```sql
-- Top 10 statements by total latency
SELECT * FROM sys.statements_with_full_table_scans LIMIT 10;

-- Find tables not using indexes
SELECT * FROM sys.schema_tables_with_full_table_scans;

-- Memory usage by user
SELECT * FROM sys.memory_by_user_by_current_bytes;

-- Redundant/duplicate indexes
SELECT * FROM sys.schema_redundant_indexes;

-- Unused indexes (never used since last restart)
SELECT * FROM sys.schema_unused_indexes;

-- Current InnoDB locks
SELECT * FROM sys.innodb_lock_waits;

-- Host activity summary
SELECT * FROM sys.host_summary ORDER BY total_connections DESC;

-- I/O by file
SELECT * FROM sys.io_global_by_file_by_bytes LIMIT 10;

-- Statement analysis
SELECT * FROM sys.statement_analysis ORDER BY full_scan DESC LIMIT 10;
```

:::tip[sys Schema vs Performance Schema]
Use `sys` schema for quick human-readable diagnostics. Use `performance_schema`
directly when you need raw data, custom aggregations, or historical data not
surfaced by `sys` views. Both are read-only diagnostic tools — they do not
affect query execution.
:::

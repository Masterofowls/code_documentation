# MySQL Advanced

## Views

A view is a named SQL query stored in the database. It behaves like a virtual table.

```sql
-- Create a view
CREATE VIEW v_active_orders AS
  SELECT o.id, o.created_at, u.name AS customer, o.total
  FROM orders o
  JOIN users u ON o.user_id = u.id
  WHERE o.status != 'cancelled';

-- Use it like a table
SELECT * FROM v_active_orders WHERE total > 100;

-- Replace existing view
CREATE OR REPLACE VIEW v_active_orders AS
  SELECT o.id, o.created_at, o.status, u.name AS customer, o.total
  FROM orders o
  JOIN users u ON o.user_id = u.id
  WHERE o.status NOT IN ('cancelled', 'refunded');

-- Drop a view
DROP VIEW IF EXISTS v_active_orders;

-- List views
SELECT TABLE_NAME FROM information_schema.VIEWS
WHERE TABLE_SCHEMA = 'myapp';
```

### WITH CHECK OPTION

Prevents INSERT/UPDATE through a view that would make the row "invisible" through the view's WHERE clause.

```sql
CREATE VIEW v_premium_users AS
  SELECT * FROM users WHERE account_type = 'premium'
WITH CHECK OPTION;

-- This succeeds (account_type = 'premium')
UPDATE v_premium_users SET name = 'Bob' WHERE id = 42;

-- This fails (would change account_type away from 'premium')
UPDATE v_premium_users SET account_type = 'free' WHERE id = 42;
-- ERROR: CHECK OPTION failed 'myapp.v_premium_users'
```

### Updatable Views

A view is updatable if it:

- References exactly one base table
- Contains no DISTINCT, GROUP BY, HAVING, UNION, or aggregate functions
- Uses no subqueries in SELECT
- Has no LIMIT clause (with UPDATE/DELETE)
- Has no non-updatable view in FROM

## Stored Procedures

```sql
-- Change delimiter so semicolons in the body don't end the statement
DELIMITER //

CREATE PROCEDURE transfer_funds(
  IN  from_account INT,
  IN  to_account   INT,
  IN  amount       DECIMAL(15,2),
  OUT result_msg   VARCHAR(255)
)
BEGIN
  DECLARE from_balance DECIMAL(15,2);
  DECLARE EXIT HANDLER FOR SQLEXCEPTION
  BEGIN
    ROLLBACK;
    SET result_msg = 'Transfer failed: database error';
  END;

  START TRANSACTION;

  SELECT balance INTO from_balance FROM accounts WHERE id = from_account FOR UPDATE;

  IF from_balance < amount THEN
    SET result_msg = 'Insufficient funds';
    ROLLBACK;
  ELSE
    UPDATE accounts SET balance = balance - amount WHERE id = from_account;
    UPDATE accounts SET balance = balance + amount WHERE id = to_account;
    COMMIT;
    SET result_msg = CONCAT('Transfer of ', amount, ' completed successfully');
  END IF;
END //

DELIMITER ;

-- Call the procedure
CALL transfer_funds(101, 202, 500.00, @msg);
SELECT @msg;
```

### Procedure Control Flow

```sql
DELIMITER //

CREATE PROCEDURE process_pending_orders()
BEGIN
  DECLARE done INT DEFAULT FALSE;
  DECLARE order_id INT;
  DECLARE order_total DECIMAL(10,2);

  -- Cursor declaration
  DECLARE order_cursor CURSOR FOR
    SELECT id, total FROM orders WHERE status = 'pending' ORDER BY created_at;

  -- Handler for when cursor runs out of rows
  DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;

  OPEN order_cursor;

  read_loop: LOOP
    FETCH order_cursor INTO order_id, order_total;

    IF done THEN
      LEAVE read_loop;
    END IF;

    -- Process each order
    IF order_total > 1000 THEN
      UPDATE orders SET status = 'flagged', flagged_at = NOW() WHERE id = order_id;
    ELSE
      UPDATE orders SET status = 'processing' WHERE id = order_id;
    END IF;

  END LOOP read_loop;

  CLOSE order_cursor;
END //

-- WHILE loop example
CREATE PROCEDURE generate_sequence(IN n INT)
BEGIN
  DECLARE i INT DEFAULT 1;
  WHILE i <= n DO
    INSERT INTO sequence_table (num) VALUES (i);
    SET i = i + 1;
  END WHILE;
END //

DELIMITER ;
```

## Stored Functions

Unlike procedures, functions return a single value and can be used in SQL expressions.

```sql
DELIMITER //

CREATE FUNCTION get_age(birth_date DATE)
  RETURNS INT
  DETERMINISTIC
  NO SQL
BEGIN
  RETURN TIMESTAMPDIFF(YEAR, birth_date, CURDATE());
END //

-- Function characteristics:
-- DETERMINISTIC: same inputs always produce same output (enables caching/optimization)
-- NOT DETERMINISTIC: default; outputs may vary (e.g., uses NOW())
-- NO SQL: doesn't read or write any tables
-- READS SQL DATA: reads but doesn't modify tables (SELECT only)
-- MODIFIES SQL DATA: may modify tables (allows INSERT/UPDATE/DELETE)
-- CONTAINS SQL: default; may contain SQL (but not read/write data)

CREATE FUNCTION calculate_discount(
  base_price   DECIMAL(10,2),
  account_type ENUM('standard', 'premium', 'vip')
)
  RETURNS DECIMAL(10,2)
  DETERMINISTIC
  NO SQL
BEGIN
  RETURN CASE account_type
    WHEN 'vip'     THEN base_price * 0.80
    WHEN 'premium' THEN base_price * 0.90
    ELSE                base_price
  END;
END //

DELIMITER ;

-- Use in a query
SELECT
  name,
  get_age(birth_date) AS age,
  calculate_discount(price, account_type) AS discounted_price
FROM users
JOIN products;
```

## Triggers

Triggers execute automatically when a DML event occurs on a table.

```sql
DELIMITER //

-- Audit log trigger: log every update to the users table
CREATE TRIGGER trg_users_after_update
  AFTER UPDATE ON users
  FOR EACH ROW
BEGIN
  INSERT INTO audit_log (
    table_name, record_id, action,
    old_values, new_values, changed_at, changed_by
  ) VALUES (
    'users',
    OLD.id,
    'UPDATE',
    JSON_OBJECT('name', OLD.name, 'email', OLD.email, 'status', OLD.status),
    JSON_OBJECT('name', NEW.name, 'email', NEW.email, 'status', NEW.status),
    NOW(),
    CURRENT_USER()
  );
END //

-- BEFORE INSERT: auto-populate derived fields
CREATE TRIGGER trg_products_before_insert
  BEFORE INSERT ON products
  FOR EACH ROW
BEGIN
  IF NEW.slug IS NULL OR NEW.slug = '' THEN
    SET NEW.slug = LOWER(REPLACE(NEW.name, ' ', '-'));
  END IF;
  SET NEW.created_at = NOW();
END //

-- BEFORE DELETE: prevent deletion of referenced records
CREATE TRIGGER trg_orders_before_delete
  BEFORE DELETE ON orders
  FOR EACH ROW
BEGIN
  IF OLD.status NOT IN ('cancelled', 'refunded') THEN
    SIGNAL SQLSTATE '45000'
      SET MESSAGE_TEXT = 'Cannot delete non-cancelled orders';
  END IF;
END //

DELIMITER ;
```

### Trigger Limitations

- Cannot use `CALL` to invoke a stored procedure that modifies the same table being triggered
- Cannot modify tables that are already being used by the statement that invoked the trigger
- A table can have at most one trigger per event/timing combination (BEFORE INSERT, AFTER INSERT, BEFORE UPDATE, etc.)
- Triggers don't fire for `TRUNCATE TABLE`
- Cannot use transactions within a trigger (transactions are inherited from the calling statement)

```sql
-- List triggers
SHOW TRIGGERS FROM myapp\G
SELECT TRIGGER_NAME, EVENT_MANIPULATION, EVENT_OBJECT_TABLE, ACTION_TIMING
FROM information_schema.TRIGGERS WHERE TRIGGER_SCHEMA = 'myapp';

-- Drop trigger
DROP TRIGGER IF EXISTS trg_users_after_update;
```

## Events (Scheduled Tasks)

MySQL Events are like cron jobs inside the database.

```sql
-- Enable the event scheduler
SET GLOBAL event_scheduler = ON;
SHOW VARIABLES LIKE 'event_scheduler';

-- One-time event
CREATE EVENT evt_send_welcome_emails
  ON SCHEDULE AT CURRENT_TIMESTAMP + INTERVAL 1 HOUR
  DO
    UPDATE users SET welcome_sent = 1 WHERE created_at > DATE_SUB(NOW(), INTERVAL 1 DAY);

-- Recurring event
CREATE EVENT evt_cleanup_old_sessions
  ON SCHEDULE EVERY 1 HOUR
  STARTS CURRENT_TIMESTAMP
  ON COMPLETION PRESERVE      -- keep event after it runs (default is DROP)
  ENABLE
  COMMENT 'Remove expired sessions every hour'
  DO
    DELETE FROM sessions WHERE expires_at < NOW() LIMIT 10000;

-- Complex recurring event with BEGIN/END
DELIMITER //
CREATE EVENT evt_daily_summary
  ON SCHEDULE EVERY 1 DAY
  STARTS '2024-01-01 02:00:00'
  ON COMPLETION PRESERVE
  DO
  BEGIN
    -- Build daily summary
    INSERT INTO daily_stats (date, order_count, revenue)
    SELECT CURDATE() - INTERVAL 1 DAY,
           COUNT(*),
           SUM(total)
    FROM orders
    WHERE DATE(created_at) = CURDATE() - INTERVAL 1 DAY;

    -- Purge old data
    DELETE FROM temp_calculations WHERE created_at < DATE_SUB(NOW(), INTERVAL 7 DAY);
  END //
DELIMITER ;

-- Manage events
SHOW EVENTS FROM myapp\G
ALTER EVENT evt_cleanup_old_sessions DISABLE;
ALTER EVENT evt_cleanup_old_sessions ENABLE;
DROP EVENT IF EXISTS evt_cleanup_old_sessions;
```

## JSON Functions

MySQL 5.7 introduced a native JSON type and rich set of functions. MySQL 8.0 further expanded them.

```sql
-- Creating JSON
SELECT JSON_OBJECT('name', 'Alice', 'age', 30, 'active', true);
-- '{"name": "Alice", "age": 30, "active": true}'

SELECT JSON_ARRAY(1, 2, 'three', null, true);
-- '[1, 2, "three", null, true]'

-- Extracting values
SELECT JSON_EXTRACT('{"name":"Alice","age":30}', '$.name');   -- "Alice"
SELECT '{"name":"Alice","age":30}'->'$.name';                  -- "Alice" (quoted)
SELECT '{"name":"Alice","age":30}'->>'$.name';                 -- Alice (unquoted)

-- Nested paths
SELECT data->>'$.address.city' FROM users;
SELECT data->>'$.tags[0]' FROM articles;       -- first array element

-- Modifying JSON
SELECT JSON_SET('{"a":1,"b":2}', '$.c', 3);                -- add/update key
SELECT JSON_INSERT('{"a":1}', '$.b', 2);                   -- add only (skip if exists)
SELECT JSON_REPLACE('{"a":1,"b":2}', '$.a', 99);           -- update only (skip if not exists)
SELECT JSON_REMOVE('{"a":1,"b":2,"c":3}', '$.b');          -- remove key

-- Searching
SELECT JSON_CONTAINS('{"tags":["mysql","index"]}', '"mysql"', '$.tags');  -- 1
SELECT JSON_CONTAINS_PATH('{"a":{"b":1}}', 'one', '$.a.b');  -- 1
SELECT JSON_SEARCH('{"tags":["mysql","index"]}', 'one', 'mysql');  -- "$.tags[0]"

-- Array operations
SELECT JSON_LENGTH('[1,2,3,4,5]');                          -- 5
SELECT JSON_LENGTH('{"a":1,"b":2}');                        -- 2 (key count)
SELECT JSON_DEPTH('{"a":{"b":{"c":1}}}');                   -- 3
SELECT JSON_TYPE('"hello"');                                 -- STRING
SELECT JSON_TYPE('42');                                      -- INTEGER
SELECT JSON_KEYS('{"name":"Alice","age":30}');               -- ["name", "age"]

-- Aggregation (MySQL 5.7.22+)
SELECT JSON_ARRAYAGG(name) FROM users WHERE active = 1;
-- ["Alice", "Bob", "Charlie"]

SELECT JSON_OBJECTAGG(id, name) FROM users WHERE active = 1;
-- {1: "Alice", 2: "Bob", 3: "Charlie"}

-- JSON_TABLE: shred JSON into relational rows (MySQL 8.0)
SELECT product_name, tag
FROM products,
JSON_TABLE(tags, '$[*]' COLUMNS (tag VARCHAR(50) PATH '$')) AS jt;

-- Validate JSON
SELECT JSON_VALID('{"key": "value"}');   -- 1
SELECT JSON_VALID('not json');            -- 0
```

## Partitioning

Partitioning divides a large table into smaller physical pieces while appearing as a single table to SQL. MySQL supports four partition types.

```sql
-- RANGE partitioning (numeric ranges)
CREATE TABLE orders_partitioned (
  id BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  created_year SMALLINT NOT NULL,
  total DECIMAL(10,2),
  PRIMARY KEY (id, created_year)
) PARTITION BY RANGE (created_year) (
  PARTITION p2021 VALUES LESS THAN (2022),
  PARTITION p2022 VALUES LESS THAN (2023),
  PARTITION p2023 VALUES LESS THAN (2024),
  PARTITION p2024 VALUES LESS THAN (2025),
  PARTITION p_future VALUES LESS THAN MAXVALUE
);

-- RANGE COLUMNS (for DATE ranges)
CREATE TABLE logs (
  id BIGINT AUTO_INCREMENT,
  log_date DATE NOT NULL,
  message TEXT,
  PRIMARY KEY (id, log_date)
) PARTITION BY RANGE COLUMNS (log_date) (
  PARTITION p_2023_q1 VALUES LESS THAN ('2023-04-01'),
  PARTITION p_2023_q2 VALUES LESS THAN ('2023-07-01'),
  PARTITION p_2023_q3 VALUES LESS THAN ('2023-10-01'),
  PARTITION p_2023_q4 VALUES LESS THAN ('2024-01-01'),
  PARTITION p_future   VALUES LESS THAN (MAXVALUE)
);

-- LIST partitioning (explicit value lists)
CREATE TABLE employees_by_region (
  id INT,
  region VARCHAR(20),
  name VARCHAR(100)
) PARTITION BY LIST COLUMNS (region) (
  PARTITION p_north VALUES IN ('NY', 'MA', 'CT'),
  PARTITION p_south VALUES IN ('FL', 'GA', 'SC'),
  PARTITION p_west  VALUES IN ('CA', 'OR', 'WA')
);

-- HASH partitioning (distributes by hash of column)
CREATE TABLE user_data (
  user_id INT NOT NULL,
  data TEXT
) PARTITION BY HASH (user_id) PARTITIONS 8;

-- KEY partitioning (MySQL's own hash function, handles more types)
CREATE TABLE sessions (
  session_id CHAR(32) NOT NULL,
  data TEXT
) PARTITION BY KEY (session_id) PARTITIONS 16;
```

### Partition Management

```sql
-- Add a partition
ALTER TABLE orders_partitioned ADD PARTITION (
  PARTITION p2025 VALUES LESS THAN (2026)
);

-- Drop a partition (deletes data in that partition!)
ALTER TABLE logs DROP PARTITION p_2023_q1;

-- Truncate a partition (fast delete of all data in partition)
ALTER TABLE logs TRUNCATE PARTITION p_2023_q1;

-- Reorganize partitions (split or merge without data loss)
ALTER TABLE orders_partitioned REORGANIZE PARTITION p_future INTO (
  PARTITION p2025 VALUES LESS THAN (2026),
  PARTITION p_future VALUES LESS THAN MAXVALUE
);

-- Check partition pruning in EXPLAIN
EXPLAIN SELECT * FROM logs WHERE log_date BETWEEN '2023-01-01' AND '2023-03-31'\G
-- Look for "partitions: p_2023_q1" — only one partition scanned

-- List partitions and their row counts
SELECT PARTITION_NAME, TABLE_ROWS, AVG_ROW_LENGTH, DATA_LENGTH
FROM information_schema.PARTITIONS
WHERE TABLE_SCHEMA = 'myapp' AND TABLE_NAME = 'logs';
```

## Replication

MySQL replication copies data from a source (primary/master) to one or more replicas (secondaries/slaves).

```sql
-- Source server: enable binary logging (my.cnf)
-- log_bin = /var/log/mysql/mysql-bin
-- server_id = 1
-- binlog_format = ROW       (recommended)
-- gtid_mode = ON            (recommended for GTID-based replication)
-- enforce_gtid_consistency = ON

-- Create replication user on source
CREATE USER 'repl'@'10.0.0.%' IDENTIFIED WITH mysql_native_password BY 'ReplPass!1';
GRANT REPLICATION SLAVE ON *.* TO 'repl'@'10.0.0.%';

-- Get current binary log position (for traditional replication)
SHOW MASTER STATUS\G

-- On replica server (my.cnf)
-- server_id = 2
-- relay-log = /var/log/mysql/relay-bin

-- Configure replication on replica
CHANGE REPLICATION SOURCE TO
  SOURCE_HOST = '10.0.0.1',
  SOURCE_USER = 'repl',
  SOURCE_PASSWORD = 'ReplPass!1',
  SOURCE_AUTO_POSITION = 1;   -- GTID-based (preferred)

-- Or for traditional position-based:
CHANGE REPLICATION SOURCE TO
  SOURCE_HOST = '10.0.0.1',
  SOURCE_USER = 'repl',
  SOURCE_PASSWORD = 'ReplPass!1',
  SOURCE_LOG_FILE = 'mysql-bin.000042',
  SOURCE_LOG_POS = 154;

-- Start/stop/check replication
START REPLICA;
STOP REPLICA;
SHOW REPLICA STATUS\G

-- Key fields in SHOW REPLICA STATUS:
-- Replica_IO_Running: Yes
-- Replica_SQL_Running: Yes
-- Seconds_Behind_Source: 0  (replication lag)
-- Last_Error: (empty = good)
```

### Binary Log Formats

| Format      | Description                           | Pros                | Cons                                                       |
| ----------- | ------------------------------------- | ------------------- | ---------------------------------------------------------- |
| `STATEMENT` | Replicates SQL statements             | Small log size      | Non-deterministic functions cause issues                   |
| `ROW`       | Replicates actual row changes         | Fully deterministic | Larger log files (e.g., UPDATE of 1M rows = 1M row images) |
| `MIXED`     | STATEMENT by default, ROW when needed | Balance             | Less predictable                                           |

## InnoDB Internals

### MVCC (Multi-Version Concurrency Control)

InnoDB uses MVCC to provide consistent reads without blocking writers:

- Each row has hidden columns: `DB_TRX_ID` (last modifying transaction), `DB_ROLL_PTR` (undo log pointer), `DB_ROW_ID`
- When a transaction reads a row, it sees the version consistent with its transaction start time
- Older versions are stored in the **undo log** (tablespace in `ibdata1` or separate undo tablespaces)
- Readers never block writers; writers never block readers (except for SELECT ... FOR UPDATE)

### Key InnoDB Components

| Component               | Purpose                                                                                                           |
| ----------------------- | ----------------------------------------------------------------------------------------------------------------- |
| **Buffer Pool**         | In-memory cache of data pages and index pages. Size = `innodb_buffer_pool_size`. Most important tuning parameter. |
| **Redo Log**            | Write-ahead log for crash recovery. Transactions write to redo log before committing.                             |
| **Undo Log**            | Stores old row versions for MVCC and transaction rollback.                                                        |
| **Change Buffer**       | Caches changes to secondary index pages not in the buffer pool (deferred writes).                                 |
| **Adaptive Hash Index** | Automatically built in-memory for frequently accessed index pages.                                                |
| **Doublewrite Buffer**  | Protects against partial page writes during crash.                                                                |

```sql
-- Key InnoDB status
SHOW ENGINE INNODB STATUS\G

-- Buffer pool hit rate (should be > 99%)
SHOW STATUS LIKE 'Innodb_buffer_pool_read%';
-- hit rate = 1 - (Innodb_buffer_pool_reads / Innodb_buffer_pool_read_requests)

-- InnoDB metrics
SELECT NAME, COUNT, STATUS
FROM information_schema.INNODB_METRICS
WHERE STATUS = 'enabled'
ORDER BY COUNT DESC
LIMIT 20;
```

## Performance Schema

The Performance Schema (`performance_schema`) database collects server performance data with minimal overhead.

```sql
-- Check if enabled
SHOW VARIABLES LIKE 'performance_schema';

-- Top slow queries
SELECT
  DIGEST_TEXT,
  COUNT_STAR AS exec_count,
  SUM_TIMER_WAIT / 1e12 AS total_seconds,
  AVG_TIMER_WAIT / 1e12 AS avg_seconds,
  SUM_ROWS_EXAMINED,
  SUM_ROWS_SENT
FROM performance_schema.events_statements_summary_by_digest
WHERE SCHEMA_NAME = 'myapp'
ORDER BY SUM_TIMER_WAIT DESC
LIMIT 10;

-- Table I/O hotspots
SELECT OBJECT_SCHEMA, OBJECT_NAME,
  COUNT_READ, COUNT_WRITE,
  SUM_TIMER_READ / 1e12 AS read_seconds,
  SUM_TIMER_WRITE / 1e12 AS write_seconds
FROM performance_schema.table_io_waits_summary_by_table
WHERE OBJECT_SCHEMA = 'myapp'
ORDER BY SUM_TIMER_READ + SUM_TIMER_WRITE DESC;

-- Lock waits
SELECT * FROM performance_schema.events_waits_history_long
WHERE EVENT_NAME LIKE '%lock%'
ORDER BY TIMER_WAIT DESC
LIMIT 10;

-- Enable/disable instrumentation
UPDATE performance_schema.setup_instruments
SET ENABLED = 'YES', TIMED = 'YES'
WHERE NAME LIKE '%statement%';
```

## sys Schema

The `sys` schema (MySQL 5.7.7+) provides human-readable views on top of Performance Schema.

```sql
-- Statements causing the most I/O
SELECT * FROM sys.statements_with_errors_or_warnings LIMIT 10;
SELECT * FROM sys.statements_with_full_table_scans LIMIT 10;
SELECT * FROM sys.statements_with_sorting LIMIT 10;
SELECT * FROM sys.statements_with_temp_tables LIMIT 10;

-- Table I/O statistics
SELECT * FROM sys.schema_table_statistics
WHERE table_schema = 'myapp'
ORDER BY total_latency DESC;

-- Unused indexes
SELECT * FROM sys.schema_unused_indexes
WHERE object_schema = 'myapp';

-- Redundant indexes
SELECT * FROM sys.schema_redundant_indexes;

-- Largest tables by size
SELECT
  table_schema,
  table_name,
  sys.format_bytes(data_length + index_length) AS total_size,
  sys.format_bytes(data_length) AS data_size,
  sys.format_bytes(index_length) AS index_size,
  table_rows
FROM information_schema.tables
WHERE table_schema = 'myapp'
ORDER BY data_length + index_length DESC;

-- Current waits
SELECT * FROM sys.session WHERE command = 'Query' ORDER BY time DESC;

-- Memory usage per user
SELECT * FROM sys.memory_by_user_by_current_bytes LIMIT 10;
```

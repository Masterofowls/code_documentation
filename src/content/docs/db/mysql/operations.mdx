# MySQL Operations

This section covers MySQL data types, MySQL-specific syntax extensions, built-in functions, transactions, and session management tools.

## Data Types

Choosing the right data type is one of the most impactful decisions for storage efficiency and query performance.

### Integer Types

| Type | Bytes | Signed Range | Unsigned Range |
|---|---|---|---|
| `TINYINT` | 1 | -128 to 127 | 0 to 255 |
| `SMALLINT` | 2 | -32,768 to 32,767 | 0 to 65,535 |
| `MEDIUMINT` | 3 | -8,388,608 to 8,388,607 | 0 to 16,777,215 |
| `INT` / `INTEGER` | 4 | -2,147,483,648 to 2,147,483,647 | 0 to 4,294,967,295 |
| `BIGINT` | 8 | -9.2 × 10¹⁸ to 9.2 × 10¹⁸ | 0 to 1.8 × 10¹⁹ |

```sql
CREATE TABLE example_ints (
  flag        TINYINT(1) UNSIGNED,   -- Boolean-like (0 or 1)
  age         TINYINT UNSIGNED,      -- 0-255 for ages
  quantity    SMALLINT UNSIGNED,
  population  MEDIUMINT UNSIGNED,
  user_id     INT UNSIGNED NOT NULL AUTO_INCREMENT,
  order_count BIGINT UNSIGNED DEFAULT 0,
  PRIMARY KEY (user_id)
);
```

:::tip[INT(11) Display Width is Irrelevant]
In MySQL, `INT(11)` does not limit the stored value — the number in parentheses is only a display width hint used with `ZEROFILL`. It has no effect on storage or value range. Just use `INT` without a width specifier.
:::

### Floating-Point & Exact Numeric

```sql
CREATE TABLE prices (
  -- DECIMAL/NUMERIC: exact, use for money
  price         DECIMAL(10, 2),    -- Up to 9,999,999.99
  tax_rate      DECIMAL(5, 4),     -- e.g., 0.0875 (8.75%)
  
  -- FLOAT: 4-byte approximate (7 significant digits)
  lat           FLOAT,
  
  -- DOUBLE: 8-byte approximate (15-16 significant digits)
  precise_val   DOUBLE,
  
  -- BIT: fixed-length bit string
  permissions   BIT(8)             -- 8 bits, store as b'10110011'
);
```

:::caution[Never Use FLOAT or DOUBLE for Money]
`FLOAT` and `DOUBLE` are approximate numeric types. `0.1 + 0.2` may not equal `0.3` exactly. Always use `DECIMAL` for financial calculations.
:::

### String Types

```sql
CREATE TABLE string_examples (
  -- CHAR: fixed-length, padded with spaces (fast for fixed-width data)
  country_code  CHAR(2),           -- Always exactly 2 chars
  uuid          CHAR(36),          -- UUID stored as string
  
  -- VARCHAR: variable-length (1-2 byte length prefix + data)
  email         VARCHAR(255),      -- Up to 255 chars efficiently
  username      VARCHAR(50) NOT NULL,
  
  -- TEXT family (stored off-page for large values in InnoDB)
  summary       TINYTEXT,          -- Up to 255 bytes
  description   TEXT,              -- Up to 65,535 bytes
  content       MEDIUMTEXT,        -- Up to 16,777,215 bytes (~16MB)
  document      LONGTEXT,          -- Up to 4,294,967,295 bytes (~4GB)
  
  -- BLOB family (binary, no charset)
  thumbnail     TINYBLOB,
  image         BLOB,
  file_data     MEDIUMBLOB,
  large_file    LONGBLOB,
  
  -- BINARY and VARBINARY (like CHAR/VARCHAR but binary)
  hash          BINARY(32),        -- SHA-256 hash stored as bytes
  token         VARBINARY(256)
);
```

### Date and Time Types

```sql
CREATE TABLE time_examples (
  -- DATE: 'YYYY-MM-DD', range: 1000-01-01 to 9999-12-31
  birth_date      DATE,
  
  -- TIME: 'HH:MM:SS', range: -838:59:59 to 838:59:59
  duration        TIME,
  
  -- DATETIME: 'YYYY-MM-DD HH:MM:SS', range: 1000-01-01 to 9999-12-31
  -- NOT timezone-aware; stored as-is
  scheduled_at    DATETIME,
  
  -- TIMESTAMP: stored as UTC, auto-converts on display
  -- Range: 1970-01-01 to 2038-01-19 (Year 2038 problem)
  created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  
  -- YEAR: 1901-2155 (1-byte storage)
  model_year      YEAR
);
```

:::tip[DATETIME vs TIMESTAMP]
Use `TIMESTAMP` when you need automatic timezone conversion and auto-update on write — perfect for `created_at`/`updated_at` audit columns. Use `DATETIME` when you need to store a specific calendar date and time regardless of timezone (appointments, schedules). Be aware of the TIMESTAMP Year 2038 limitation on 32-bit systems.
:::

### Enumerated and Set Types

```sql
CREATE TABLE orders (
  id     INT AUTO_INCREMENT PRIMARY KEY,
  status ENUM('pending', 'processing', 'shipped', 'delivered', 'cancelled')
           NOT NULL DEFAULT 'pending',
  -- SET: can hold zero or more values
  tags   SET('sale', 'featured', 'new', 'clearance')
);

-- ENUM is stored as an integer internally (1-byte for ≤255 values)
-- SET is stored as a bitfield
INSERT INTO orders (status, tags) VALUES ('pending', 'sale,featured');
SELECT * FROM orders WHERE FIND_IN_SET('sale', tags) > 0;
```

### JSON Type (MySQL 5.7.8+)

```sql
CREATE TABLE products (
  id         INT AUTO_INCREMENT PRIMARY KEY,
  name       VARCHAR(255) NOT NULL,
  attributes JSON,         -- Validated JSON, stored in binary format
  metadata   JSON
);

INSERT INTO products (name, attributes) VALUES (
  'Laptop',
  '{"brand":"Dell","ram":16,"tags":["electronics","laptop"],"dimensions":{"w":35,"h":23}}'
);

-- Extract fields
SELECT
  name,
  attributes->>'$.brand' AS brand,          -- ->> extracts as text
  attributes->'$.ram' AS ram_json,           -- -> extracts as JSON value
  JSON_EXTRACT(attributes, '$.dimensions.w') AS width
FROM products;

-- Filter on JSON field
SELECT * FROM products WHERE attributes->>'$.brand' = 'Dell';
SELECT * FROM products WHERE attributes->'$.ram' >= 8;
```

## AUTO_INCREMENT

```sql
CREATE TABLE users (
  id       INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  email    VARCHAR(255) UNIQUE NOT NULL,
  name     VARCHAR(100)
);

-- Insert and get the generated ID
INSERT INTO users (email, name) VALUES ('alice@example.com', 'Alice');
SELECT LAST_INSERT_ID();  -- Returns the ID of the most recently inserted row

-- LAST_INSERT_ID() is connection-specific — safe in concurrent environments
-- For multi-row inserts, it returns the FIRST generated ID

-- View current AUTO_INCREMENT value
SHOW TABLE STATUS LIKE 'users'\G

-- Reset AUTO_INCREMENT (can only be set higher than max existing value)
ALTER TABLE users AUTO_INCREMENT = 1000;

-- BIGINT for tables that may have billions of rows
CREATE TABLE events (
  id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  -- ...
);
```

## MySQL-Specific DML Syntax

### REPLACE INTO

`REPLACE INTO` deletes the existing row and inserts a new one if a unique key conflict occurs. This means AUTO_INCREMENT IDs change and foreign key references break.

```sql
-- If a row with the same PRIMARY KEY or UNIQUE KEY exists, delete it and insert fresh
REPLACE INTO users (id, email, name) VALUES (1, 'alice@example.com', 'Alice Updated');

-- Effectively equivalent to DELETE + INSERT
-- Use with caution: triggers fire for both DELETE and INSERT
```

### INSERT IGNORE

`INSERT IGNORE` silently discards rows that would violate a unique constraint, rather than erroring.

```sql
-- Insert without error if duplicate key exists
INSERT IGNORE INTO users (email, name) VALUES ('alice@example.com', 'Alice Duplicate');

-- Also ignores other "ignorable" errors (division by zero, etc.)
-- Returns 0 affected rows if the row was ignored
SELECT ROW_COUNT();  -- 0 if ignored, 1 if inserted
```

### INSERT ... ON DUPLICATE KEY UPDATE

The most useful upsert pattern in MySQL: insert if new, update specific columns if duplicate.

```sql
-- Upsert: insert or update on conflict
INSERT INTO page_views (page_id, view_count, last_viewed)
VALUES (42, 1, NOW())
ON DUPLICATE KEY UPDATE
  view_count  = view_count + 1,
  last_viewed = NOW();

-- Multiple value upsert
INSERT INTO inventory (product_id, quantity)
VALUES (1, 100), (2, 200), (3, 300)
ON DUPLICATE KEY UPDATE
  quantity = VALUES(quantity);  -- VALUES() refers to the proposed insert value

-- MySQL 8.0.20+ prefers alias syntax
INSERT INTO inventory (product_id, quantity)
VALUES (1, 100)
AS new_vals
ON DUPLICATE KEY UPDATE
  quantity = new_vals.quantity;

-- ROW_COUNT() returns 1 for insert, 2 for update-on-duplicate, 0 if unchanged
```

### LOAD DATA INFILE

Extremely fast bulk import from a CSV or delimited text file:

```sql
LOAD DATA INFILE '/var/lib/mysql-files/users.csv'
INTO TABLE users
CHARACTER SET utf8mb4
FIELDS TERMINATED BY ','
OPTIONALLY ENCLOSED BY '"'
LINES TERMINATED BY '\n'
IGNORE 1 LINES              -- Skip header row
(email, name, @created_str)
SET created_at = STR_TO_DATE(@created_str, '%Y-%m-%d');

-- LOAD DATA LOCAL INFILE reads from the client machine
-- Requires --local-infile=1 on server and client
```

## GROUP BY and ONLY_FULL_GROUP_BY

MySQL 5.7+ enables `ONLY_FULL_GROUP_BY` by default (SQL standard compliance):

```sql
-- ERROR in ONLY_FULL_GROUP_BY mode:
-- 'users.name' is not in GROUP BY or aggregate function
SELECT department_id, name, COUNT(*) FROM employees GROUP BY department_id;

-- CORRECT: select only grouped or aggregated columns
SELECT department_id, COUNT(*), MAX(salary) FROM employees GROUP BY department_id;

-- Use ANY_VALUE() to explicitly select non-aggregated column (MySQL 5.7+)
SELECT department_id, ANY_VALUE(name), COUNT(*) FROM employees GROUP BY department_id;

-- Check and modify SQL mode
SELECT @@sql_mode;
-- To disable (not recommended for new code):
SET SESSION sql_mode = REPLACE(@@sql_mode, 'ONLY_FULL_GROUP_BY', '');
```

## LIMIT with UPDATE and DELETE

MySQL allows `LIMIT` in `UPDATE` and `DELETE` statements (not standard SQL):

```sql
-- Delete only the first 100 old records
DELETE FROM logs WHERE created_at < NOW() - INTERVAL 30 DAY LIMIT 100;

-- Update in batches (important for large tables to avoid long locks)
UPDATE products SET discounted = 1 WHERE category = 'sale' LIMIT 500;
```

## String Functions

```sql
-- CONCAT: join strings (NULL-safe alternative: CONCAT_WS)
SELECT CONCAT('Hello', ' ', 'World');                        -- 'Hello World'
SELECT CONCAT_WS('-', '2024', '01', '15');                   -- '2024-01-15'
SELECT CONCAT('Hello', NULL, 'World');                       -- NULL (any NULL = NULL)
SELECT CONCAT_WS(',', 'a', NULL, 'b');                       -- 'a,b' (NULLs skipped)

-- GROUP_CONCAT: aggregate strings within a group
SELECT department,
  GROUP_CONCAT(name ORDER BY name SEPARATOR ', ') AS members,
  GROUP_CONCAT(DISTINCT job_title ORDER BY job_title) AS roles
FROM employees
GROUP BY department;

-- Set max length (default 1024, often too short)
SET SESSION group_concat_max_len = 65536;

-- SUBSTRING / SUBSTR / MID
SELECT SUBSTRING('Hello World', 1, 5);    -- 'Hello' (1-indexed)
SELECT SUBSTRING('Hello World', 7);       -- 'World'
SELECT SUBSTRING('Hello World', -5);      -- 'World' (from end)

-- LOCATE / INSTR / POSITION: find substring position
SELECT LOCATE('World', 'Hello World');    -- 7
SELECT INSTR('Hello World', 'World');     -- 7

-- REPLACE
SELECT REPLACE('foo bar foo', 'foo', 'baz');  -- 'baz bar baz'

-- TRIM, LTRIM, RTRIM
SELECT TRIM('  hello  ');                 -- 'hello'
SELECT TRIM(LEADING 'x' FROM 'xxhello'); -- 'hello'

-- LOWER / UPPER / LPAD / RPAD
SELECT LOWER('MYSQL'), UPPER('mysql');
SELECT LPAD('42', 5, '0');                -- '00042'
SELECT RPAD('hello', 8, '.');            -- 'hello...'

-- LENGTH (bytes) vs CHAR_LENGTH (characters)
SELECT LENGTH('héllo');       -- 6 (é is 2 bytes in UTF-8)
SELECT CHAR_LENGTH('héllo'); -- 5 (5 characters)

-- FORMAT: format numbers with commas
SELECT FORMAT(1234567.891, 2);  -- '1,234,567.89'

-- CONVERT / CAST
SELECT CAST('42' AS UNSIGNED);
SELECT CONVERT('2024-01-15', DATE);

-- LEFT / RIGHT
SELECT LEFT('Hello World', 5);   -- 'Hello'
SELECT RIGHT('Hello World', 5);  -- 'World'

-- REPEAT / REVERSE / SPACE
SELECT REPEAT('ab', 3);    -- 'ababab'
SELECT REVERSE('hello');   -- 'olleh'
SELECT SPACE(5);           -- '     '
```

## Date and Time Functions

```sql
-- Current date/time
SELECT NOW();              -- '2024-01-15 14:30:00' (datetime)
SELECT CURDATE();          -- '2024-01-15' (date only)
SELECT CURTIME();          -- '14:30:00' (time only)
SELECT UTC_TIMESTAMP();    -- UTC datetime
SELECT UNIX_TIMESTAMP();   -- Unix epoch seconds

-- Format dates
SELECT DATE_FORMAT(NOW(), '%Y-%m-%d %H:%i:%s');    -- '2024-01-15 14:30:00'
SELECT DATE_FORMAT(NOW(), '%W, %M %D %Y');         -- 'Monday, January 15th 2024'
SELECT DATE_FORMAT(NOW(), '%d/%m/%Y');             -- '15/01/2024'

-- Parse date strings
SELECT STR_TO_DATE('15/01/2024', '%d/%m/%Y');      -- '2024-01-15'
SELECT STR_TO_DATE('Jan 15 2024 2:30PM', '%b %d %Y %h:%i%p');

-- Date arithmetic
SELECT DATE_ADD(NOW(), INTERVAL 7 DAY);
SELECT DATE_ADD(NOW(), INTERVAL 3 MONTH);
SELECT DATE_ADD(NOW(), INTERVAL -1 YEAR);
SELECT DATE_SUB('2024-01-15', INTERVAL 30 DAY);

-- DATEDIFF: difference in days
SELECT DATEDIFF('2024-12-31', '2024-01-01');  -- 365

-- TIMEDIFF: difference as time
SELECT TIMEDIFF('14:30:00', '12:00:00');  -- '02:30:00'

-- TIMESTAMPDIFF: difference in specific unit
SELECT TIMESTAMPDIFF(YEAR, '1990-05-15', CURDATE())  AS age;
SELECT TIMESTAMPDIFF(MINUTE, '2024-01-15 10:00:00', NOW()) AS minutes_ago;

-- Extract parts
SELECT YEAR(NOW()), MONTH(NOW()), DAY(NOW());
SELECT HOUR(NOW()), MINUTE(NOW()), SECOND(NOW());
SELECT DAYOFWEEK(NOW());   -- 1=Sunday, 7=Saturday
SELECT DAYOFYEAR(NOW());
SELECT WEEK(NOW(), 1);     -- Week number (1=Monday as first day)
SELECT QUARTER(NOW());     -- 1-4

-- Convert between Unix timestamps and datetimes
SELECT UNIX_TIMESTAMP('2024-01-15 00:00:00');  -- 1705276800
SELECT FROM_UNIXTIME(1705276800);              -- '2024-01-15 00:00:00'
SELECT FROM_UNIXTIME(1705276800, '%Y-%m-%d'); -- '2024-01-15'

-- LAST_DAY of month
SELECT LAST_DAY('2024-02-01');  -- '2024-02-29' (2024 is leap year)
```

## Regular Expressions

```sql
-- REGEXP / RLIKE (case-insensitive by default)
SELECT * FROM users WHERE email REGEXP '^[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,}$';
SELECT * FROM products WHERE name REGEXP 'pro|premium|enterprise';
SELECT * FROM orders WHERE reference_number REGEXP '^ORD-[0-9]{8}$';

-- REGEXP_LIKE (MySQL 8.0+, explicit)
SELECT * FROM users WHERE REGEXP_LIKE(phone, '^\\+?[0-9\\s\\-]{10,15}$');

-- REGEXP_REPLACE (MySQL 8.0+)
SELECT REGEXP_REPLACE('foo123bar456', '[0-9]+', '#');  -- 'foo#bar#'

-- REGEXP_SUBSTR (MySQL 8.0+)
SELECT REGEXP_SUBSTR('2024-01-15', '[0-9]{4}');  -- '2024'

-- REGEXP_INSTR (MySQL 8.0+): position of match
SELECT REGEXP_INSTR('MySQL 8.0', '[0-9]+');  -- 7
```

## Conditional Expressions

```sql
-- IF(condition, true_val, false_val)
SELECT name,
  IF(salary > 50000, 'Senior', 'Junior') AS level,
  IF(deleted_at IS NULL, 'Active', 'Deleted') AS status
FROM employees;

-- IFNULL(value, fallback): return fallback if value is NULL
SELECT IFNULL(nickname, name) AS display_name FROM users;
SELECT IFNULL(discount, 0) + price AS final_price FROM products;

-- NULLIF(a, b): return NULL if a = b, otherwise a (useful to avoid division by zero)
SELECT total_sales / NULLIF(total_visits, 0) AS conversion_rate FROM stats;

-- COALESCE: return first non-NULL value
SELECT COALESCE(nickname, first_name, 'Anonymous') AS display_name FROM users;

-- CASE expressions
SELECT
  order_id,
  CASE status
    WHEN 'pending'    THEN 'Awaiting Payment'
    WHEN 'processing' THEN 'Being Fulfilled'
    WHEN 'shipped'    THEN 'On Its Way'
    WHEN 'delivered'  THEN 'Completed'
    ELSE 'Unknown'
  END AS status_label
FROM orders;

-- Searched CASE
SELECT
  product_id,
  price,
  CASE
    WHEN price < 10    THEN 'Budget'
    WHEN price < 50    THEN 'Mid-Range'
    WHEN price < 200   THEN 'Premium'
    ELSE 'Luxury'
  END AS price_tier
FROM products;
```

## Transactions and AUTOCOMMIT

```sql
-- MySQL defaults to AUTOCOMMIT=1 (each statement is its own transaction)
SELECT @@autocommit;  -- 1 by default

-- Disable autocommit for a session
SET autocommit = 0;

-- Explicit transaction management
START TRANSACTION;
-- or: BEGIN;

  INSERT INTO orders (user_id, total) VALUES (1, 150.00);
  SET @order_id = LAST_INSERT_ID();
  
  INSERT INTO order_items (order_id, product_id, quantity, price)
  VALUES (@order_id, 42, 2, 75.00);
  
  UPDATE inventory SET stock = stock - 2 WHERE product_id = 42;
  
  -- Check if we have enough stock
  -- If not, we can ROLLBACK

COMMIT;  -- Commit if everything is fine

-- ROLLBACK: undo everything since START TRANSACTION
-- START TRANSACTION;
-- ...
-- ROLLBACK;

-- SAVEPOINT: partial rollback
START TRANSACTION;
  INSERT INTO a_table VALUES (1);
  SAVEPOINT sp1;
  INSERT INTO b_table VALUES (1);
  ROLLBACK TO SAVEPOINT sp1;  -- Undoes b_table insert, keeps a_table
  RELEASE SAVEPOINT sp1;
COMMIT;

-- Transaction isolation levels
SET SESSION TRANSACTION ISOLATION LEVEL READ COMMITTED;
-- READ UNCOMMITTED: dirty reads possible
-- READ COMMITTED: no dirty reads (default in PostgreSQL)
-- REPEATABLE READ: no dirty or non-repeatable reads (MySQL InnoDB default)
-- SERIALIZABLE: fully isolated, slowest
```

## SHOW PROCESSLIST and KILL

```sql
-- Show all running queries and connections
SHOW PROCESSLIST;
-- Shows: Id, User, Host, db, Command, Time, State, Info

-- Show full query text (not truncated)
SHOW FULL PROCESSLIST;

-- Information Schema equivalent (more filterable)
SELECT
  id,
  user,
  host,
  db,
  command,
  time,
  state,
  LEFT(info, 100) AS query_snippet
FROM information_schema.PROCESSLIST
WHERE command != 'Sleep'
  AND time > 5
ORDER BY time DESC;

-- Performance Schema (more detailed)
SELECT
  processlist_id,
  processlist_user,
  processlist_host,
  processlist_db,
  processlist_command,
  processlist_time,
  processlist_state,
  LEFT(processlist_info, 200) AS query
FROM performance_schema.threads
WHERE processlist_command != 'Sleep'
ORDER BY processlist_time DESC;

-- Kill a specific connection or query
KILL 42;            -- Kill connection 42 (both query and connection)
KILL QUERY 42;      -- Kill only the running query (keep connection)
KILL CONNECTION 42; -- Same as KILL 42

-- Kill all queries running longer than 60 seconds (scripted approach)
SELECT CONCAT('KILL QUERY ', id, ';')
FROM information_schema.PROCESSLIST
WHERE command = 'Query'
  AND time > 60
  AND user != 'replication';
```

:::tip[Monitoring Long-Running Queries]
Keep an eye on queries with `State = 'Copying to tmp table'`, `'Sorting result'`, or `'Waiting for table metadata lock'` — these often indicate missing indexes, lock contention, or expensive operations that need optimization.
:::

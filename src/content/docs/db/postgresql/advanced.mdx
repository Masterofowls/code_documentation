## Table Partitioning

Partitioning splits a large table into smaller physical pieces (partitions) while maintaining a single logical table interface.

### RANGE Partitioning

```sql
-- Create partitioned table
CREATE TABLE orders (
  id          BIGSERIAL,
  customer_id BIGINT NOT NULL,
  total       NUMERIC(12,2) NOT NULL DEFAULT 0,
  status      TEXT NOT NULL DEFAULT 'pending',
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
) PARTITION BY RANGE (created_at);

-- Create partitions
CREATE TABLE orders_2023 PARTITION OF orders
  FOR VALUES FROM ('2023-01-01') TO ('2024-01-01');

CREATE TABLE orders_2024_q1 PARTITION OF orders
  FOR VALUES FROM ('2024-01-01') TO ('2024-04-01');

CREATE TABLE orders_2024_q2 PARTITION OF orders
  FOR VALUES FROM ('2024-04-01') TO ('2024-07-01');

CREATE TABLE orders_2024_q3 PARTITION OF orders
  FOR VALUES FROM ('2024-07-01') TO ('2024-10-01');

CREATE TABLE orders_2024_q4 PARTITION OF orders
  FOR VALUES FROM ('2024-10-01') TO ('2025-01-01');

-- Default partition for values not matching any other partition
CREATE TABLE orders_default PARTITION OF orders DEFAULT;

-- Each partition can have its own indexes
CREATE INDEX ON orders_2024_q1 (customer_id);
CREATE INDEX ON orders_2024_q2 (customer_id);

-- Queries automatically prune to relevant partitions
EXPLAIN SELECT * FROM orders WHERE created_at >= '2024-04-01' AND created_at < '2024-07-01';
-- Only scans orders_2024_q2

-- Detach a partition (becomes a standalone table)
ALTER TABLE orders DETACH PARTITION orders_2023;

-- Attach an existing table as a partition
ALTER TABLE orders ATTACH PARTITION orders_2023
  FOR VALUES FROM ('2023-01-01') TO ('2024-01-01');
```

### LIST Partitioning

```sql
CREATE TABLE products (
  id       BIGSERIAL,
  name     TEXT NOT NULL,
  region   TEXT NOT NULL,
  status   TEXT NOT NULL
) PARTITION BY LIST (region);

CREATE TABLE products_us     PARTITION OF products FOR VALUES IN ('US', 'CA');
CREATE TABLE products_europe PARTITION OF products FOR VALUES IN ('DE', 'FR', 'GB', 'ES', 'IT');
CREATE TABLE products_apac   PARTITION OF products FOR VALUES IN ('JP', 'AU', 'SG', 'IN');
CREATE TABLE products_other  PARTITION OF products DEFAULT;
```

### HASH Partitioning

```sql
-- Distribute rows evenly across N partitions
CREATE TABLE events (
  id         BIGSERIAL,
  user_id    BIGINT NOT NULL,
  event_type TEXT NOT NULL,
  payload    JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
) PARTITION BY HASH (user_id);

CREATE TABLE events_0 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 0);
CREATE TABLE events_1 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 1);
CREATE TABLE events_2 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 2);
CREATE TABLE events_3 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 3);
```

## LISTEN / NOTIFY

PostgreSQL has a built-in publish/subscribe mechanism for notifying connected clients of changes.

```sql
-- Publisher (any session can notify)
NOTIFY channel_name;
NOTIFY user_events, 'payload_string';  -- Payload is a text string (max 8KB)

-- Dynamic notification from a trigger
CREATE OR REPLACE FUNCTION notify_order_change()
RETURNS TRIGGER AS $$
BEGIN
  PERFORM pg_notify(
    'order_updates',
    json_build_object(
      'operation', TG_OP,
      'id',        NEW.id,
      'status',    NEW.status,
      'customer',  NEW.customer_id
    )::text
  );
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_order_notify
  AFTER INSERT OR UPDATE ON orders
  FOR EACH ROW
  EXECUTE FUNCTION notify_order_change();

-- Subscriber (listening connection)
LISTEN order_updates;

-- Application-level (Node.js with pg)
-- client.query('LISTEN order_updates');
-- client.on('notification', (msg) => console.log(msg.payload));
```

```python
# Python example with psycopg2
import psycopg2, select, json

conn = psycopg2.connect("dbname=myapp")
conn.set_isolation_level(0)  # Autocommit required for LISTEN
cursor = conn.cursor()
cursor.execute("LISTEN order_updates;")

while True:
    if select.select([conn], [], [], 5) == ([], [], []):
        print("Timeout, no notification")
    else:
        conn.poll()
        while conn.notifies:
            notify = conn.notifies.pop(0)
            payload = json.loads(notify.payload)
            print(f"Order {payload['id']} changed to {payload['status']}")
```

## Advisory Locks

Application-level locks that are not tied to specific rows or tables. Useful for preventing concurrent execution of application logic.

```sql
-- Session-level advisory lock (held until explicit release or session end)
SELECT pg_advisory_lock(12345);           -- Blocks until acquired (integer ID)
SELECT pg_advisory_lock(42, 99);          -- Two-integer form (for namespacing)

-- Non-blocking try version
SELECT pg_try_advisory_lock(12345);       -- Returns true if acquired, false if not

-- Release session lock
SELECT pg_advisory_unlock(12345);
SELECT pg_advisory_unlock_all();          -- Release all session advisory locks

-- Transaction-level advisory lock (released automatically at transaction end)
SELECT pg_advisory_xact_lock(99999);      -- Blocks
SELECT pg_try_advisory_xact_lock(99999); -- Non-blocking

-- Check held locks
SELECT pid, classid, objid, objsubid, mode, granted
FROM pg_locks
WHERE locktype = 'advisory';
```

```sql
-- Use case: ensure only one instance runs a cron job at a time
DO $$
DECLARE
  lock_acquired BOOLEAN;
BEGIN
  SELECT pg_try_advisory_lock(1234567) INTO lock_acquired;
  IF NOT lock_acquired THEN
    RAISE NOTICE 'Another instance is running, skipping.';
    RETURN;
  END IF;

  -- Do the work
  PERFORM process_pending_notifications();

  PERFORM pg_advisory_unlock(1234567);
END $$;
```

## Sequences

```sql
-- Create a sequence
CREATE SEQUENCE order_number_seq
  START WITH 100000
  INCREMENT BY 1
  MINVALUE 100000
  MAXVALUE 999999
  CYCLE            -- Restart at MINVALUE after MAXVALUE
  CACHE 10;        -- Pre-allocate 10 values for performance

-- Use sequence
SELECT nextval('order_number_seq');    -- Get and increment
SELECT currval('order_number_seq');    -- Current value (must call nextval first)
SELECT lastval();                       -- Last value returned by any sequence this session

-- Reset sequence
SELECT setval('order_number_seq', 100000, false);  -- Next call returns 100000
SELECT setval('order_number_seq', 100000, true);   -- Next call returns 100001

-- Inspect sequence
SELECT * FROM order_number_seq;

-- Modify sequence
ALTER SEQUENCE order_number_seq RESTART WITH 200000;
ALTER SEQUENCE order_number_seq INCREMENT BY 5;
```

## Writable CTEs (Data-Modifying CTEs)

CTEs can contain DML statements, making complex multi-step operations readable and atomic:

```sql
-- Move rows between tables atomically
WITH deleted_orders AS (
  DELETE FROM orders
  WHERE created_at < NOW() - INTERVAL '7 years'
    AND status = 'completed'
  RETURNING *
)
INSERT INTO orders_archive
SELECT * FROM deleted_orders;

-- Update and return for use in insert
WITH updated AS (
  UPDATE inventory
  SET reserved = reserved + 1
  WHERE product_id = 42 AND available >= 1
  RETURNING product_id, available - 1 AS remaining
)
INSERT INTO pending_reservations (product_id, user_id, expires_at)
SELECT u.product_id, 99, NOW() + INTERVAL '30 minutes'
FROM updated u;

-- Cascade an operation
WITH completed_order AS (
  UPDATE orders SET status = 'completed', completed_at = NOW()
  WHERE id = 123
  RETURNING id, customer_id, total
),
loyalty_points AS (
  UPDATE customers
  SET points = points + (co.total * 10)::integer
  FROM completed_order co
  WHERE customers.id = co.customer_id
  RETURNING customers.id
)
SELECT 'Order completed and points awarded' AS result
FROM completed_order;
```

## UPSERT with ON CONFLICT

```sql
-- On conflict, update specific fields
INSERT INTO product_inventory (product_id, warehouse_id, quantity)
VALUES (42, 1, 100)
ON CONFLICT (product_id, warehouse_id)
DO UPDATE SET
  quantity   = EXCLUDED.quantity,
  updated_at = NOW()
WHERE inventory.quantity != EXCLUDED.quantity;  -- Only update if changed

-- On conflict, do nothing (idempotent insert)
INSERT INTO user_role_assignments (user_id, role_id)
VALUES (1, 5)
ON CONFLICT (user_id, role_id) DO NOTHING;

-- Upsert with full row replacement
INSERT INTO user_settings (user_id, theme, language, timezone)
VALUES (42, 'dark', 'en', 'UTC')
ON CONFLICT (user_id)
DO UPDATE SET
  theme    = EXCLUDED.theme,
  language = EXCLUDED.language,
  timezone = EXCLUDED.timezone,
  updated_at = NOW();
```

## MERGE Statement (PostgreSQL 15+)

```sql
MERGE INTO target_table AS t
USING source_table AS s ON t.id = s.id
WHEN MATCHED AND s.is_deleted = true THEN
  DELETE
WHEN MATCHED THEN
  UPDATE SET
    name       = s.name,
    email      = s.email,
    updated_at = NOW()
WHEN NOT MATCHED THEN
  INSERT (id, name, email, created_at)
  VALUES (s.id, s.name, s.email, NOW());
```

## PL/pgSQL Functions and Procedures

### Function vs Procedure

| Feature | Function | Procedure |
|---|---|---|
| Returns value | Yes | No (or via OUT params) |
| Call with | SELECT | CALL |
| Transaction control | No | Yes (COMMIT/ROLLBACK) |
| Used in queries | Yes | No |

### PL/pgSQL Reference

```sql
CREATE OR REPLACE FUNCTION process_order(
  p_order_id  BIGINT,
  p_user_id   BIGINT
)
RETURNS TABLE (success BOOLEAN, message TEXT, updated_at TIMESTAMPTZ) AS $$
DECLARE
  v_order     orders%ROWTYPE;
  v_balance   NUMERIC;
  v_result    BOOLEAN := false;
  v_message   TEXT;
BEGIN
  -- SELECT INTO
  SELECT * INTO v_order FROM orders WHERE id = p_order_id FOR UPDATE;

  IF NOT FOUND THEN
    RAISE EXCEPTION 'Order % not found', p_order_id
      USING ERRCODE = 'no_data_found';
  END IF;

  IF v_order.status != 'pending' THEN
    v_message := format('Order %s is already %s', p_order_id, v_order.status);
    RETURN QUERY SELECT false, v_message, NOW();
    RETURN;
  END IF;

  -- Get user balance
  SELECT balance INTO v_balance FROM accounts WHERE user_id = p_user_id;

  IF v_balance < v_order.total THEN
    RAISE NOTICE 'Insufficient balance: % < %', v_balance, v_order.total;
    RETURN QUERY SELECT false, 'Insufficient balance'::text, NOW();
    RETURN;
  END IF;

  -- Process
  UPDATE accounts SET balance = balance - v_order.total WHERE user_id = p_user_id;
  UPDATE orders SET status = 'confirmed', confirmed_at = NOW() WHERE id = p_order_id;

  -- Log
  INSERT INTO order_events (order_id, event_type, user_id, created_at)
  VALUES (p_order_id, 'confirmed', p_user_id, NOW());

  RETURN QUERY SELECT true, 'Order confirmed'::text, NOW();

EXCEPTION
  WHEN OTHERS THEN
    RAISE WARNING 'Error processing order %: %', p_order_id, SQLERRM;
    RETURN QUERY SELECT false, SQLERRM, NOW();
END;
$$ LANGUAGE plpgsql;

-- Call the function
SELECT * FROM process_order(42, 99);
```

## pg_cron Extension

Run periodic jobs directly inside PostgreSQL:

```sql
-- Enable the extension
CREATE EXTENSION IF NOT EXISTS pg_cron;

-- Schedule a job
SELECT cron.schedule(
  'daily-cleanup',          -- Job name
  '0 2 * * *',             -- Cron expression: 2am daily
  'DELETE FROM sessions WHERE expires_at < NOW()'
);

-- Other scheduling examples
SELECT cron.schedule('0 */4 * * *', $$REFRESH MATERIALIZED VIEW CONCURRENTLY stats$$);
SELECT cron.schedule('*/30 * * * *', 'SELECT process_pending_emails()');

-- List scheduled jobs
SELECT jobid, jobname, schedule, command, active FROM cron.job;

-- View job run history
SELECT jobid, status, return_message, start_time, end_time
FROM cron.job_run_details
ORDER BY start_time DESC
LIMIT 20;

-- Disable a job
SELECT cron.unschedule('daily-cleanup');

-- Update schedule
UPDATE cron.job SET schedule = '0 3 * * *' WHERE jobname = 'daily-cleanup';
```

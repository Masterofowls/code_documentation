PostgreSQL has one of the richest type systems of any relational database, including JSON, arrays, ranges, geometric types, and user-defined types.

## Identity and Serial Types

### SERIAL and BIGSERIAL (Legacy)

```sql
-- SERIAL is shorthand for INTEGER + SEQUENCE + DEFAULT
CREATE TABLE items (
  id SERIAL PRIMARY KEY     -- Equivalent to the block below
);

-- What SERIAL actually creates:
CREATE SEQUENCE items_id_seq;
CREATE TABLE items (
  id INTEGER NOT NULL DEFAULT nextval('items_id_seq') PRIMARY KEY
);
ALTER SEQUENCE items_id_seq OWNED BY items.id;
```

### IDENTITY (SQL Standard — Recommended)

```sql
-- GENERATED ALWAYS: cannot manually insert a value (use OVERRIDING SYSTEM VALUE)
CREATE TABLE users (
  id BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY
);

-- GENERATED BY DEFAULT: can insert a value or let the sequence handle it
CREATE TABLE users (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
);

-- Customize the underlying sequence
CREATE TABLE orders (
  id BIGINT GENERATED ALWAYS AS IDENTITY
    (START WITH 1000 INCREMENT BY 1 MINVALUE 1000)
  PRIMARY KEY
);
```

:::tip[Prefer IDENTITY over SERIAL]
`GENERATED ... AS IDENTITY` is the SQL standard (since SQL:2003). SERIAL is PostgreSQL-specific and deprecated in new code. Use `BIGINT GENERATED ALWAYS AS IDENTITY` for new primary keys.
:::

## UUID

```sql
-- PostgreSQL 13+: gen_random_uuid() built-in (no extension needed)
CREATE TABLE sessions (
  id         UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id    BIGINT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- PostgreSQL 12 and earlier: requires pgcrypto
CREATE EXTENSION IF NOT EXISTS pgcrypto;
CREATE TABLE sessions (
  id UUID DEFAULT gen_random_bytes(16)::uuid PRIMARY KEY
);

-- Ordered UUID for better index performance (UUID v7 concept)
-- Use a library or custom function that generates time-ordered UUIDs
```

## Numeric Types

```sql
-- Exact numeric (no rounding — use for money, measurements)
price    NUMERIC(10, 2)   -- Up to 10 digits, 2 decimal places
rate     DECIMAL(5, 4)    -- DECIMAL and NUMERIC are identical
unlimited NUMERIC         -- Unlimited precision

-- Integer types
small_id  SMALLINT        -- 2 bytes, -32768 to 32767
regular   INTEGER         -- 4 bytes, ±2.1 billion
large_id  BIGINT          -- 8 bytes, ±9.2 × 10^18

-- Floating point (approximate — do NOT use for currency)
ratio     REAL            -- 4 bytes, ~6 decimal digits precision
score     DOUBLE PRECISION -- 8 bytes, ~15 decimal digits precision

-- Money type (stores as cents internally, locale-aware display)
price     MONEY
-- Note: MONEY type has locale issues; prefer NUMERIC(10,2) for money
```

## Text Types

```sql
-- Fixed length (padded with spaces)
code      CHAR(2)         -- Always exactly 2 chars, e.g., 'US', 'CA'

-- Variable length with limit
username  VARCHAR(50)     -- Up to 50 chars, error if exceeded

-- Unlimited (PostgreSQL-specific, no length limit)
body      TEXT            -- No practical length limit

-- Case-insensitive text (requires citext extension)
CREATE EXTENSION IF NOT EXISTS citext;
email     CITEXT          -- Comparisons are case-insensitive automatically
```

## Boolean

```sql
is_active BOOLEAN NOT NULL DEFAULT true

-- Accepted literals (all case-insensitive):
-- TRUE:  true, 't', 'yes', 'y', 'on', '1'
-- FALSE: false, 'f', 'no', 'n', 'off', '0'

SELECT * FROM users WHERE is_active IS TRUE;
SELECT * FROM users WHERE NOT is_active;
```

## Date and Time Types

```sql
birthday    DATE           -- Date only: '2024-01-15'
start_time  TIME           -- Time only: '14:30:00', '14:30:00.123456'
start_tz    TIMETZ         -- Time with timezone (avoid — use TIMESTAMPTZ)
event_at    TIMESTAMP      -- Date + time, no timezone: '2024-01-15 14:30:00'
created_at  TIMESTAMPTZ    -- Date + time + UTC offset (recommended)
duration    INTERVAL       -- '2 hours', '3 days 4 hours', '1 year 2 months'
```

### Timestamp Operations

```sql
-- Current timestamps
SELECT NOW();               -- Current timestamp with timezone
SELECT CURRENT_TIMESTAMP;   -- SQL standard equivalent
SELECT CURRENT_DATE;        -- Date only
SELECT CLOCK_TIMESTAMP();   -- Current time (changes within transaction)

-- Timezone conversion
SELECT NOW() AT TIME ZONE 'America/New_York';
SELECT NOW() AT TIME ZONE 'UTC';
SELECT '2024-01-15 14:30:00'::TIMESTAMPTZ AT TIME ZONE 'Asia/Tokyo';

-- Date arithmetic
SELECT NOW() + INTERVAL '7 days';
SELECT NOW() - INTERVAL '1 year 3 months';
SELECT '2024-12-31'::DATE - '2024-01-01'::DATE;  -- Returns integer (364)

-- Date truncation
SELECT DATE_TRUNC('month', NOW());   -- First day of current month
SELECT DATE_TRUNC('week', NOW());    -- Monday of current week
SELECT DATE_TRUNC('year', created_at) FROM orders;

-- Extract parts
SELECT EXTRACT(YEAR FROM NOW());
SELECT EXTRACT(DOW FROM NOW());     -- Day of week (0=Sunday)
SELECT EXTRACT(EPOCH FROM NOW());   -- Unix timestamp
SELECT DATE_PART('hour', NOW());    -- Same as EXTRACT
```

## Arrays

```sql
-- Array column declaration
CREATE TABLE articles (
  id    SERIAL PRIMARY KEY,
  title TEXT NOT NULL,
  tags  TEXT[]    DEFAULT '{}',
  nums  INTEGER[] DEFAULT '{}'
);

-- Insert with arrays
INSERT INTO articles (title, tags)
VALUES ('PostgreSQL Arrays', ARRAY['postgresql', 'sql', 'arrays']);

INSERT INTO articles (title, tags)
VALUES ('SQL Basics', '{"sql", "basics", "tutorial"}');  -- Literal syntax

-- Array operations
SELECT tags[1] FROM articles;          -- 1-based indexing (not 0!)
SELECT tags[1:2] FROM articles;        -- Slice: first two elements
SELECT array_length(tags, 1) FROM articles;  -- Length of 1st dimension

-- Update arrays
UPDATE articles SET tags = array_append(tags, 'new-tag') WHERE id = 1;
UPDATE articles SET tags = array_remove(tags, 'old-tag') WHERE id = 1;
UPDATE articles SET tags = tags || ARRAY['extra'] WHERE id = 1;

-- Array operators
SELECT * FROM articles WHERE tags @> ARRAY['postgresql'];   -- Contains
SELECT * FROM articles WHERE tags && ARRAY['sql', 'nosql']; -- Overlap (any match)
SELECT * FROM articles WHERE 'sql' = ANY(tags);             -- Element exists

-- Unnest (expand array to rows)
SELECT id, unnest(tags) AS tag FROM articles;

-- Aggregate into array
SELECT ARRAY_AGG(id ORDER BY created_at) FROM articles WHERE 'sql' = ANY(tags);
```

## JSONB

JSONB stores JSON in a binary format — parsed on insert, efficient to query, and indexable.

```sql
CREATE TABLE events (
  id      BIGSERIAL PRIMARY KEY,
  type    TEXT NOT NULL,
  payload JSONB NOT NULL DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Insert
INSERT INTO events (type, payload)
VALUES ('user.created', '{"user_id": 42, "email": "alice@example.com", "plan": "pro"}');
```

### JSONB Operators

```sql
-- Navigate JSON (returns JSON)
SELECT payload->'plan'        FROM events;   -- Returns JSON: "pro"
SELECT payload->'address'->'city' FROM events;   -- Nested

-- Navigate JSON (returns TEXT — unwrapped)
SELECT payload->>'plan'       FROM events;   -- Returns text: pro
SELECT payload->>'email'      FROM events;

-- Path operators
SELECT payload #> '{address,city}' FROM events;   -- Returns JSON at path
SELECT payload #>> '{address,city}' FROM events;  -- Returns text at path

-- Containment
SELECT * FROM events WHERE payload @> '{"plan": "pro"}';
SELECT * FROM events WHERE '{"plan": "pro"}' <@ payload;

-- Key existence
SELECT * FROM events WHERE payload ? 'email';           -- Has key
SELECT * FROM events WHERE payload ?| ARRAY['email', 'phone']; -- Any key
SELECT * FROM events WHERE payload ?& ARRAY['email', 'plan'];  -- All keys

-- Modify JSON
SELECT jsonb_set(payload, '{plan}', '"enterprise"')           FROM events;
SELECT jsonb_insert(payload, '{tags, 0}', '"vip"')            FROM events;
SELECT payload - 'password'                                    FROM events; -- Remove key
SELECT jsonb_strip_nulls(payload)                              FROM events;
```

### JSONB Functions

```sql
-- Build JSON
SELECT jsonb_build_object('name', 'Alice', 'age', 30);
SELECT jsonb_build_array(1, 2, 3, 'four');
SELECT json_object('{a, b}', '{1, 2}');

-- Expand
SELECT jsonb_each('{"a": 1, "b": 2}'::jsonb);              -- Rows: (key, value)
SELECT jsonb_each_text('{"a": "x", "b": "y"}'::jsonb);     -- Text values
SELECT jsonb_array_elements('[1, 2, 3]'::jsonb);            -- One row per element
SELECT jsonb_array_elements_text('["a","b"]'::jsonb);       -- Text elements

-- Aggregate into JSON
SELECT jsonb_agg(jsonb_build_object('id', id, 'name', name))
FROM products;

-- Merge JSONB objects
SELECT '{"a": 1}'::jsonb || '{"b": 2}'::jsonb;   -- {"a": 1, "b": 2}
SELECT jsonb_object_agg(key, value) FROM config_table;
```

### Indexing JSONB

```sql
-- GIN index for containment and key operators
CREATE INDEX ON events USING GIN (payload);

-- GIN with jsonb_path_ops (smaller index, only @> operator)
CREATE INDEX ON events USING GIN (payload jsonb_path_ops);

-- B-tree index on a specific JSON field (for equality/range on one field)
CREATE INDEX ON events ((payload->>'user_id'));
CREATE INDEX ON events ((payload->>'created_at')::timestamptz);
```

## Range Types

Range types represent a range of values with configurable inclusivity.

```sql
-- Built-in range types
int4range    -- Integer range
int8range    -- BigInt range
numrange     -- Numeric range
tsrange      -- Timestamp without timezone range
tstzrange    -- Timestamp with timezone range (most useful)
daterange    -- Date range

-- Usage
CREATE TABLE subscriptions (
  id       SERIAL PRIMARY KEY,
  user_id  INTEGER,
  plan     TEXT,
  active   tstzrange NOT NULL DEFAULT tstzrange(NOW(), 'infinity')
);

-- Insert a range
INSERT INTO subscriptions (user_id, plan, active)
VALUES (42, 'pro', '[2024-01-01, 2025-01-01)');
-- [ = inclusive, ) = exclusive

-- Range operators
SELECT * FROM subscriptions WHERE active @> NOW();          -- Contains timestamp
SELECT * FROM subscriptions WHERE active && tstzrange(NOW(), 'infinity'); -- Overlap
SELECT lower(active), upper(active) FROM subscriptions;      -- Bounds
SELECT isempty(active), lower_inc(active), upper_inc(active) FROM subscriptions;
```

## Enum Types

```sql
-- Create a custom enum type
CREATE TYPE order_status AS ENUM (
  'pending', 'processing', 'shipped', 'delivered', 'cancelled', 'refunded'
);

CREATE TABLE orders (
  id     SERIAL PRIMARY KEY,
  status order_status NOT NULL DEFAULT 'pending'
);

-- Use like any type
UPDATE orders SET status = 'shipped' WHERE id = 1;
SELECT * FROM orders WHERE status = 'pending';

-- Add a value (at end, or at a specific position)
ALTER TYPE order_status ADD VALUE 'on_hold' AFTER 'processing';

-- List enum values
SELECT enum_range(NULL::order_status);
SELECT unnest(enum_range(NULL::order_status));

-- Drop enum type (must drop dependent columns first)
DROP TYPE order_status;
```

## Full-Text Search Types

```sql
-- tsvector: pre-processed searchable document
-- tsquery: search query expression
SELECT to_tsvector('english', 'The quick brown fox jumps over the lazy dog');
-- Returns: 'brown':3 'dog':9 'fox':4 'jump':5 'lazi':8 'quick':2

SELECT to_tsquery('english', 'quick & fox');
SELECT plainto_tsquery('english', 'quick fox');  -- Implicit AND
SELECT phraseto_tsquery('english', 'quick fox'); -- Phrase match
SELECT websearch_to_tsquery('english', 'quick fox -lazy');

-- Match operator
SELECT 'quick brown fox'::tsvector @@ 'fox'::tsquery;  -- true

-- Store pre-computed tsvector
ALTER TABLE articles ADD COLUMN search_vector tsvector;
UPDATE articles SET search_vector = to_tsvector('english', title || ' ' || body);
CREATE INDEX ON articles USING GIN (search_vector);

-- Generated column for automatic updates (PostgreSQL 12+)
ALTER TABLE articles ADD COLUMN search_vector tsvector
  GENERATED ALWAYS AS (
    to_tsvector('english', coalesce(title, '') || ' ' || coalesce(body, ''))
  ) STORED;

-- Search with ranking
SELECT title, ts_rank(search_vector, query) AS rank
FROM articles, to_tsquery('english', 'postgresql & index') AS query
WHERE search_vector @@ query
ORDER BY rank DESC
LIMIT 10;

-- Highlighted snippets
SELECT ts_headline('english', body, to_tsquery('postgresql'))
FROM articles WHERE search_vector @@ to_tsquery('postgresql');
```

:::tip[Use `websearch_to_tsquery` for user input]
`websearch_to_tsquery` handles user-provided search strings gracefully — it supports `"quoted phrases"`, `-excluded words`, and boolean operators, similar to Google search syntax. It's much more user-friendly than `to_tsquery` which requires strict syntax.
:::

## Installation

<Tabs>
<TabItem label="Ubuntu/Debian">
```bash
# Add official PostgreSQL apt repository
sudo apt-get install -y postgresql-common
sudo /usr/share/postgresql-common/pgdg/apt.postgresql.org.sh

# Install PostgreSQL 16
sudo apt-get install -y postgresql-16

# Start and enable the service
sudo systemctl start postgresql
sudo systemctl enable postgresql

# Check status
sudo systemctl status postgresql
```
</TabItem>
<TabItem label="RHEL/Rocky/CentOS">
```bash
# Install the repository RPM
sudo dnf install -y https://download.postgresql.org/pub/repos/yum/reporpms/EL-9-x86_64/pgdg-redhat-repo-latest.noarch.rpm

# Disable built-in PostgreSQL module
sudo dnf -qy module disable postgresql

# Install PostgreSQL 16
sudo dnf install -y postgresql16-server

# Initialize and start
sudo /usr/pgsql-16/bin/postgresql-16-setup initdb
sudo systemctl enable postgresql-16
sudo systemctl start postgresql-16
```
</TabItem>
<TabItem label="macOS (Homebrew)">
```bash
# Install via Homebrew
brew install postgresql@16

# Add to PATH
echo 'export PATH="/opt/homebrew/opt/postgresql@16/bin:$PATH"' >> ~/.zshrc
source ~/.zshrc

# Start as a service
brew services start postgresql@16

# Or start manually
pg_ctl -D /opt/homebrew/var/postgresql@16 start
```
</TabItem>
<TabItem label="Docker">

# Quick start
docker run -d \

## Service Management

```bash
# systemd (Linux)
sudo systemctl start   postgresql
sudo systemctl stop    postgresql
sudo systemctl restart postgresql
sudo systemctl reload  postgresql   # Reload config without restart
sudo systemctl status  postgresql

# pg_ctl (cross-platform)
pg_ctl start  -D /var/lib/postgresql/16/main
pg_ctl stop   -D /var/lib/postgresql/16/main -m fast
pg_ctl reload -D /var/lib/postgresql/16/main
pg_ctl status -D /var/lib/postgresql/16/main
```

## Connecting with psql

```bash
# Connect as the postgres superuser (Unix socket, no password needed locally)
sudo -u postgres psql

# Specify host, user, database, port
psql -h localhost -p 5432 -U myuser -d mydb

# Using a connection URI
psql "postgres://myuser:mypassword@localhost:5432/mydb"
psql "postgresql://myuser:mypassword@localhost:5432/mydb?sslmode=require"

# Connect to a remote server with SSL
psql "host=db.example.com port=5432 dbname=myapp user=myuser sslmode=verify-full"
```

### The `.pgpass` File

Store credentials to avoid repeated password prompts:

```bash
# ~/.pgpass (Unix) or %APPDATA%\postgresql\pgpass.conf (Windows)
# Format: hostname:port:database:username:password
localhost:5432:*:myuser:mypassword
db.example.com:5432:production:appuser:secretpassword

# Set permissions (required on Unix)
chmod 600 ~/.pgpass
```

## Creating Users and Roles

```sql
-- Create a basic user with password
CREATE USER appuser WITH PASSWORD 'secure_password_here';

-- Create a user who can create databases
CREATE USER devuser WITH PASSWORD 'devpass' CREATEDB;

-- Create a superuser (for administration only — avoid for application users)
CREATE USER admin_user WITH PASSWORD 'adminpass' SUPERUSER;

-- Create a role with specific capabilities
CREATE ROLE readonly_role;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO readonly_role;

-- Assign role to user
GRANT readonly_role TO appuser;

-- Create user using ROLE syntax (more flexible)
CREATE ROLE appuser
  WITH LOGIN
  PASSWORD 'secure_password'
  NOSUPERUSER
  NOCREATEDB
  NOCREATEROLE
  CONNECTION LIMIT 20;   -- Max concurrent connections for this user

-- Change password
ALTER USER appuser WITH PASSWORD 'new_password';

-- List all users
\du

-- In SQL:
SELECT usename, usesuper, usecreatedb, usecreaterole FROM pg_user;
```

## GRANT and REVOKE

```sql
-- Grant access to a specific database
GRANT CONNECT ON DATABASE myapp TO appuser;

-- Grant schema usage
GRANT USAGE ON SCHEMA public TO appuser;

-- Grant on all current tables in schema
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO appuser;

-- Grant on all future tables (default privileges)
ALTER DEFAULT PRIVILEGES IN SCHEMA public
  GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO appuser;

-- Grant on sequences (needed for SERIAL / BIGSERIAL columns)
GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO appuser;
ALTER DEFAULT PRIVILEGES IN SCHEMA public
  GRANT USAGE, SELECT ON SEQUENCES TO appuser;

-- Read-only access
GRANT CONNECT ON DATABASE myapp TO readonly_user;
GRANT USAGE ON SCHEMA public TO readonly_user;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO readonly_user;
ALTER DEFAULT PRIVILEGES IN SCHEMA public
  GRANT SELECT ON TABLES TO readonly_user;

-- Revoke
REVOKE DELETE ON ALL TABLES IN SCHEMA public FROM appuser;
REVOKE CONNECT ON DATABASE myapp FROM olduser;
```

## Creating Databases

```sql
-- Basic
CREATE DATABASE myapp;

-- With specific owner and encoding
CREATE DATABASE myapp
  OWNER     = appuser
  ENCODING  = 'UTF8'
  LC_COLLATE = 'en_US.UTF-8'
  LC_CTYPE   = 'en_US.UTF-8'
  TEMPLATE  = template0;   -- Use template0 for custom encoding

-- Clone an existing database
CREATE DATABASE myapp_backup TEMPLATE myapp;

-- Drop
DROP DATABASE IF EXISTS myapp_backup;

-- List databases (SQL)
SELECT datname, datowner::regrole, pg_encoding_to_char(encoding)
FROM pg_database
WHERE datistemplate = false;
```

## Essential psql Commands

```sql
\?          -- Help for psql commands
\h SELECT   -- SQL syntax help for specific command

-- Connection
\c dbname [user] [host] [port]   -- Connect to another database
\conninfo                         -- Show current connection info

-- Listing
\l          -- List databases
\l+         -- List databases with sizes
\dt         -- List tables in current schema
\dt *.*     -- List all tables in all schemas
\dt public.* -- List tables in public schema
\d tablename  -- Describe table structure
\d+ tablename -- Describe with extra detail (storage, comments)
\dn         -- List schemas
\du         -- List users/roles
\dv         -- List views
\dm         -- List materialized views
\df         -- List functions
\di         -- List indexes

-- Editing and execution
\e          -- Open query in $EDITOR
\i file.sql -- Execute SQL file
\o file.txt -- Send output to file
\o          -- Stop sending to file (back to stdout)

-- Display options
\timing     -- Toggle query timing
\x          -- Toggle expanded display (column per line for wide tables)
\a          -- Toggle aligned output
\H          -- Toggle HTML output
\t          -- Toggle tuple-only mode (no headers/footers)

-- Output modes
\pset format aligned   -- Default table format
\pset format csv       -- CSV output
\pset format json      -- JSON output
\pset format unaligned -- Pipe-separated values

-- Variables
\set VARIABLE value    -- Set psql variable
\echo :VARIABLE        -- Print variable value
\set ON_ERROR_STOP 1   -- Stop on first error (useful in scripts)

-- Transaction shortcuts
\begin      -- BEGIN
\commit     -- COMMIT
\rollback   -- ROLLBACK
```

## pg_dump and pg_restore

### Backup

```bash
# Plain SQL dump (human-readable, easy to restore)
pg_dump -U postgres -d myapp > myapp_backup.sql

# Custom format (compressed, supports parallel restore)
pg_dump -U postgres -d myapp -Fc -f myapp_backup.dump

# Directory format (parallel, best for large databases)
pg_dump -U postgres -d myapp -Fd -j 4 -f myapp_backup_dir/

# Dump with schema only (no data)
pg_dump -U postgres -d myapp --schema-only -f schema.sql

# Dump with data only (no schema)
pg_dump -U postgres -d myapp --data-only -f data.sql

# Dump specific tables
pg_dump -U postgres -d myapp -t orders -t order_items -f orders_backup.dump -Fc

# Dump all databases
pg_dumpall -U postgres > all_databases.sql
```

### Restore

```bash
# Restore plain SQL
psql -U postgres -d myapp < myapp_backup.sql

# Restore custom format
pg_restore -U postgres -d myapp myapp_backup.dump

# Restore with parallel jobs (directory format)
pg_restore -U postgres -d myapp -j 4 myapp_backup_dir/

# Restore with verbose output
pg_restore -U postgres -d myapp -v myapp_backup.dump

# List contents of a dump file
pg_restore -l myapp_backup.dump
```

## pg_hba.conf — Authentication

The `pg_hba.conf` file controls client authentication:

```
# TYPE  DATABASE  USER     ADDRESS       METHOD
# Local Unix socket connections
local   all       postgres               peer
local   all       all                    md5

# IPv4 local connections
host    all       all      127.0.0.1/32  scram-sha-256

# IPv6 local connections
host    all       all      ::1/128       scram-sha-256

# Remote connections for specific database/user
host    myapp     appuser  10.0.0.0/8   scram-sha-256

# Allow all from internal network (use with caution)
host    all       all      192.168.1.0/24 scram-sha-256
```

:::caution[Reload After Changes]
After editing `pg_hba.conf` or `postgresql.conf`, reload the config:
```bash
sudo systemctl reload postgresql
# Or in psql:
SELECT pg_reload_conf();
```
:::

## psql Variables and Scripting

```bash
# Set variables via command line
psql -U postgres -d myapp -v ON_ERROR_STOP=1 -v schema=public

# Use in SQL
SELECT * FROM :schema.users;

# Conditional logic in psql scripts
\set db_name 'myapp'
\set schema 'public'

SELECT current_database() = :'db_name' AS correct_database \gset

\if :correct_database
  \echo Connected to correct database
\else
  \echo Wrong database! Stopping.
  \q
\endif
```

This guide covers everything you need to define and manage database schemas in SQL — from creating your first table to modifying and dropping structures safely.

## Creating a Database

```sql
-- PostgreSQL
CREATE DATABASE myapp
  OWNER = myuser
  ENCODING = 'UTF8'
  LC_COLLATE = 'en_US.UTF-8'
  LC_CTYPE = 'en_US.UTF-8';

-- Connect to the database (psql)
\c myapp

-- MySQL
CREATE DATABASE myapp
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE myapp;

-- SQLite (database created automatically when you open the file)
-- sqlite3 myapp.db
```

## Creating Tables

### Column Data Types

```sql
CREATE TABLE users (
  -- Integer types
  id              BIGINT,           -- 8-byte integer (-9.2e18 to 9.2e18)
  age             INTEGER,          -- 4-byte integer (-2.1B to 2.1B)
  score           SMALLINT,         -- 2-byte integer (-32768 to 32767)

  -- Exact numeric
  price           NUMERIC(10, 2),   -- Up to 10 digits, 2 decimal places
  balance         DECIMAL(15, 4),   -- Same as NUMERIC
  percentage      NUMERIC(5, 2),    -- e.g. 99.99

  -- Floating point (approximate — don't use for money)
  latitude        DOUBLE PRECISION, -- 8-byte float
  longitude       REAL,             -- 4-byte float

  -- Character types
  code            CHAR(3),          -- Fixed-length, padded with spaces
  username        VARCHAR(50),      -- Variable-length, max 50 chars
  bio             TEXT,             -- Unlimited variable-length text

  -- Boolean
  is_active       BOOLEAN,          -- true / false / NULL

  -- Date and time
  birth_date      DATE,             -- '2024-01-15'
  start_time      TIME,             -- '09:30:00'
  created_at      TIMESTAMP,        -- '2024-01-15 09:30:00' (no timezone)
  updated_at      TIMESTAMPTZ,      -- '2024-01-15 09:30:00+00' (with timezone)

  -- UUID
  uuid            UUID,             -- PostgreSQL: gen_random_uuid()

  -- JSON (PostgreSQL)
  metadata        JSONB,            -- Binary JSON with indexing support
  raw_data        JSON,             -- Text JSON (no indexing)

  -- Binary
  avatar          BYTEA             -- Binary data
);
```

### Constraints

```sql
CREATE TABLE products (
  id          BIGSERIAL PRIMARY KEY,              -- Auto-incrementing PK

  sku         VARCHAR(50) NOT NULL,               -- Cannot be NULL
  name        TEXT        NOT NULL,

  price       NUMERIC(10,2) NOT NULL
              CHECK (price >= 0),                  -- Value constraint

  stock       INTEGER NOT NULL DEFAULT 0,          -- Default value
              CHECK (stock >= 0),

  slug        VARCHAR(100) UNIQUE NOT NULL,        -- Must be unique

  category_id BIGINT REFERENCES categories(id)     -- Foreign key (default: NO ACTION)
              ON DELETE SET NULL
              ON UPDATE CASCADE,

  status      TEXT NOT NULL DEFAULT 'active'
              CHECK (status IN ('active', 'inactive', 'discontinued')),

  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### Foreign Key Actions

When a referenced row is deleted or updated, the foreign key constraint controls what happens:

| Action | ON DELETE behavior | ON UPDATE behavior |
|---|---|---|
| `CASCADE` | Delete the referencing row | Update the FK column to match |
| `SET NULL` | Set FK column to NULL | Set FK column to NULL |
| `SET DEFAULT` | Set FK column to its default | Set FK column to its default |
| `RESTRICT` | Error: cannot delete parent | Error: cannot update parent |
| `NO ACTION` | Same as RESTRICT (deferred check) | Same as RESTRICT |

```sql
CREATE TABLE order_items (
  id         BIGSERIAL PRIMARY KEY,
  order_id   BIGINT NOT NULL REFERENCES orders(id)
             ON DELETE CASCADE,    -- Delete items when order is deleted
  product_id BIGINT NOT NULL REFERENCES products(id)
             ON DELETE RESTRICT,   -- Cannot delete product with open orders
  quantity   INTEGER NOT NULL CHECK (quantity > 0),
  unit_price NUMERIC(10,2) NOT NULL
);
```

### Named Constraints

```sql
CREATE TABLE accounts (
  id      BIGSERIAL,
  email   TEXT      NOT NULL,
  balance NUMERIC(15,4) NOT NULL DEFAULT 0,

  -- Named constraints can be referenced in error messages and ALTER TABLE
  CONSTRAINT accounts_pkey PRIMARY KEY (id),
  CONSTRAINT accounts_email_unique UNIQUE (email),
  CONSTRAINT accounts_balance_positive CHECK (balance >= 0),
  CONSTRAINT accounts_email_format CHECK (email ~* '^[^@]+@[^@]+\.[^@]+$')
);
```

## ALTER TABLE

Modify an existing table's structure without recreating it.

### Add and Remove Columns

```sql
-- Add a column (defaults to NULL for existing rows unless DEFAULT is specified)
ALTER TABLE users
  ADD COLUMN phone_number VARCHAR(20);

-- Add column with a default value
ALTER TABLE users
  ADD COLUMN subscription_tier TEXT NOT NULL DEFAULT 'free';

-- Add column and set its value for existing rows
ALTER TABLE users
  ADD COLUMN full_name TEXT;

UPDATE users SET full_name = first_name || ' ' || last_name;

ALTER TABLE users
  ALTER COLUMN full_name SET NOT NULL;

-- Drop a column
ALTER TABLE users
  DROP COLUMN phone_number;

-- Drop with CASCADE (also drops dependent views, constraints)
ALTER TABLE users
  DROP COLUMN phone_number CASCADE;
```

### Rename Operations

```sql
-- Rename a column
ALTER TABLE users
  RENAME COLUMN username TO handle;

-- Rename the table itself
ALTER TABLE users
  RENAME TO accounts;

-- Rename a constraint
ALTER TABLE accounts
  RENAME CONSTRAINT accounts_email_unique TO accounts_email_key;
```

### Modify Column Types

```sql
-- Change data type (only if values are compatible or a cast is specified)
ALTER TABLE products
  ALTER COLUMN price TYPE NUMERIC(12, 4);

-- Change type with explicit cast
ALTER TABLE logs
  ALTER COLUMN event_id TYPE BIGINT
  USING event_id::BIGINT;

-- Set or remove default
ALTER TABLE users
  ALTER COLUMN subscription_tier SET DEFAULT 'free';

ALTER TABLE users
  ALTER COLUMN subscription_tier DROP DEFAULT;

-- Set NOT NULL
ALTER TABLE users
  ALTER COLUMN email SET NOT NULL;

ALTER TABLE users
  ALTER COLUMN phone_number DROP NOT NULL;
```

### Constraints

```sql
-- Add a constraint
ALTER TABLE orders
  ADD CONSTRAINT orders_total_positive CHECK (total >= 0);

ALTER TABLE users
  ADD CONSTRAINT users_email_unique UNIQUE (email);

ALTER TABLE order_items
  ADD CONSTRAINT order_items_fk_order
  FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE;

-- Drop a constraint
ALTER TABLE orders
  DROP CONSTRAINT orders_total_positive;

-- Drop primary key
ALTER TABLE old_table
  DROP CONSTRAINT old_table_pkey;
```

## DROP TABLE

```sql
-- Drop if it exists (no error if missing)
DROP TABLE IF EXISTS temp_imports;

-- Drop and remove all dependent objects (views, foreign keys referencing this table)
DROP TABLE customers CASCADE;

-- Drop multiple tables
DROP TABLE IF EXISTS order_items, orders, products CASCADE;

-- Truncate (remove all rows but keep the table structure)
TRUNCATE TABLE audit_logs;

-- Truncate and reset sequences
TRUNCATE TABLE users RESTART IDENTITY;

-- Truncate with cascade (truncate referenced tables too)
TRUNCATE TABLE orders CASCADE;
```

:::caution[DROP TABLE is Irreversible]
`DROP TABLE` permanently deletes the table and all its data. Always check your current database before dropping: `SELECT current_database();` in PostgreSQL, `SELECT database();` in MySQL.
:::

## Schema Organization

Schemas are namespaces within a database that group related objects.

```sql
-- Create a schema
CREATE SCHEMA billing;
CREATE SCHEMA analytics;

-- Create a table in a specific schema
CREATE TABLE billing.invoices (
  id         BIGSERIAL PRIMARY KEY,
  customer_id BIGINT NOT NULL,
  amount     NUMERIC(12,2) NOT NULL,
  issued_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE analytics.page_views (
  id         BIGSERIAL PRIMARY KEY,
  page_url   TEXT NOT NULL,
  viewed_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Set search path (PostgreSQL) — schema resolution order
SET search_path TO billing, analytics, public;

-- Now you can omit the schema prefix for tables in billing or analytics
SELECT * FROM invoices;  -- Finds billing.invoices first
```

## `information_schema`

The `information_schema` is a standard SQL interface to query the database's own metadata:

```sql
-- List all tables in current database
SELECT table_schema, table_name, table_type
FROM information_schema.tables
WHERE table_schema NOT IN ('information_schema', 'pg_catalog')
ORDER BY table_schema, table_name;

-- List all columns for a specific table
SELECT
  column_name,
  data_type,
  character_maximum_length,
  is_nullable,
  column_default
FROM information_schema.columns
WHERE table_name = 'users'
  AND table_schema = 'public'
ORDER BY ordinal_position;

-- List all constraints
SELECT
  tc.constraint_name,
  tc.constraint_type,
  tc.table_name,
  kcu.column_name,
  ccu.table_name  AS foreign_table_name,
  ccu.column_name AS foreign_column_name
FROM information_schema.table_constraints AS tc
JOIN information_schema.key_column_usage  AS kcu
  ON tc.constraint_name = kcu.constraint_name
LEFT JOIN information_schema.constraint_column_usage AS ccu
  ON tc.constraint_name = ccu.constraint_name
WHERE tc.table_schema = 'public';

-- List all indexes (PostgreSQL)
SELECT
  indexname,
  tablename,
  indexdef
FROM pg_indexes
WHERE schemaname = 'public'
ORDER BY tablename, indexname;
```

## COMMENT ON

Add documentation to database objects:

```sql
COMMENT ON TABLE users IS 'Application users — includes both customers and admin accounts.';
COMMENT ON COLUMN users.status IS 'One of: active, suspended, deleted. Never hard-delete users.';
COMMENT ON CONSTRAINT users_email_unique ON users IS 'Email addresses must be globally unique across all user types.';
COMMENT ON INDEX idx_users_email IS 'Used for login and uniqueness checks.';
```

View comments in PostgreSQL:

```sql
SELECT obj_description('users'::regclass);   -- Table comment
\d+ users                                     -- psql: shows comments inline
```

## Complete Schema Example

```sql
-- Create schema
CREATE SCHEMA IF NOT EXISTS store;

-- Enum type for product status
CREATE TYPE store.product_status AS ENUM ('draft', 'active', 'discontinued');

-- Categories with self-referencing hierarchy
CREATE TABLE store.categories (
  id          BIGSERIAL PRIMARY KEY,
  name        TEXT    NOT NULL,
  slug        TEXT    NOT NULL UNIQUE,
  parent_id   BIGINT  REFERENCES store.categories(id) ON DELETE SET NULL,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Products
CREATE TABLE store.products (
  id           BIGSERIAL PRIMARY KEY,
  sku          VARCHAR(100) UNIQUE NOT NULL,
  name         TEXT         NOT NULL,
  description  TEXT,
  price        NUMERIC(12,2) NOT NULL CHECK (price >= 0),
  compare_at   NUMERIC(12,2) CHECK (compare_at IS NULL OR compare_at >= price),
  weight_kg    NUMERIC(8,3),
  category_id  BIGINT REFERENCES store.categories(id) ON DELETE SET NULL,
  status       store.product_status NOT NULL DEFAULT 'draft',
  metadata     JSONB NOT NULL DEFAULT '{}',
  created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

COMMENT ON TABLE store.products IS 'Product catalog. Never hard-delete — use status = discontinued.';
COMMENT ON COLUMN store.products.compare_at IS 'Original price for strike-through display. Must be >= price.';

-- Index for common queries
CREATE INDEX idx_products_category ON store.products(category_id)
  WHERE status = 'active';

CREATE INDEX idx_products_status_created ON store.products(status, created_at DESC);
```

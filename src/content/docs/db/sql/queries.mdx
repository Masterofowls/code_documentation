## SELECT Basics

```sql
-- Select all columns (avoid in production — explicit columns are clearer and faster)
SELECT * FROM users;

-- Select specific columns
SELECT id, email, created_at FROM users;

-- Column aliases with AS
SELECT
  id,
  email,
  CONCAT(first_name, ' ', last_name) AS full_name,
  EXTRACT(YEAR FROM created_at)      AS join_year
FROM users;

-- DISTINCT — remove duplicate rows
SELECT DISTINCT country FROM users;
SELECT DISTINCT ON (user_id) user_id, created_at, status
FROM orders
ORDER BY user_id, created_at DESC;   -- PostgreSQL: latest order per user

-- Limit and offset (pagination)
SELECT * FROM products
ORDER BY created_at DESC
LIMIT 20 OFFSET 40;      -- page 3 (0-indexed, 20 per page)

-- SQL standard syntax (more portable)
SELECT * FROM products
ORDER BY created_at DESC
FETCH FIRST 20 ROWS ONLY
OFFSET 40 ROWS;
```

## WHERE Clause

### Comparison Operators

```sql
WHERE price = 9.99
WHERE status != 'cancelled'       -- or <>
WHERE price < 100
WHERE price >= 50 AND price <= 200

-- Range (inclusive)
WHERE price BETWEEN 50 AND 200    -- equivalent to price >= 50 AND price <= 200
WHERE created_at BETWEEN '2024-01-01' AND '2024-12-31'

-- List membership
WHERE status IN ('active', 'pending', 'processing')
WHERE country NOT IN ('US', 'CA', 'MX')

-- Null checks (never use = NULL or != NULL)
WHERE deleted_at IS NULL
WHERE deleted_at IS NOT NULL

-- Boolean
WHERE is_active = true
WHERE is_active               -- equivalent in PostgreSQL
WHERE NOT is_active
```

### String Pattern Matching

```sql
-- LIKE — case-sensitive in most databases
WHERE email LIKE '%@gmail.com'      -- ends with @gmail.com
WHERE name LIKE 'J%'                -- starts with J
WHERE code LIKE '___'               -- exactly 3 characters
WHERE name LIKE '%smith%'           -- contains "smith"

-- ILIKE — case-insensitive (PostgreSQL)
WHERE name ILIKE '%smith%'

-- Wildcards: % = any number of chars, _ = exactly one char

-- Regular expressions (PostgreSQL)
WHERE email ~ '^[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,}$'
WHERE name ~* 'john'       -- ~* is case-insensitive regex
WHERE code !~ '[0-9]'      -- does NOT match
```

### Complex WHERE Conditions

```sql
SELECT *
FROM orders
WHERE
  (status = 'pending' OR status = 'processing')
  AND total > 100
  AND created_at >= NOW() - INTERVAL '30 days'
  AND user_id IS NOT NULL
  AND notes NOT ILIKE '%cancelled%';
```

## ORDER BY

```sql
-- Basic sort
SELECT * FROM users ORDER BY created_at;           -- ASC by default
SELECT * FROM users ORDER BY created_at DESC;

-- Multiple columns
SELECT * FROM users ORDER BY last_name ASC, first_name ASC;

-- NULL placement
SELECT * FROM products ORDER BY price ASC NULLS LAST;
SELECT * FROM products ORDER BY price DESC NULLS FIRST;

-- Sort by expression
SELECT * FROM users ORDER BY LOWER(last_name), LOWER(first_name);

-- Sort by column alias or position (avoid position in production)
SELECT id, price * quantity AS total FROM order_items
ORDER BY total DESC;       -- using alias
```

## Subqueries

### Non-Correlated Subqueries

The inner query runs once and returns a value or set:

```sql
-- Scalar subquery in WHERE
SELECT * FROM products
WHERE price > (SELECT AVG(price) FROM products);

-- IN subquery
SELECT * FROM users
WHERE id IN (
  SELECT DISTINCT user_id
  FROM orders
  WHERE status = 'cancelled'
  AND created_at >= NOW() - INTERVAL '7 days'
);

-- NOT IN (careful with NULLs — if subquery returns any NULL, NOT IN returns nothing)
SELECT * FROM products
WHERE id NOT IN (
  SELECT product_id FROM order_items WHERE order_items.product_id IS NOT NULL
);
```

### Correlated Subqueries

The inner query references the outer query and runs once per outer row:

```sql
-- Find users who have placed an order in the last 30 days
SELECT u.*
FROM users u
WHERE EXISTS (
  SELECT 1 FROM orders o
  WHERE o.user_id = u.id
  AND o.created_at >= NOW() - INTERVAL '30 days'
);

-- Find users who have NEVER ordered (anti-join via NOT EXISTS)
SELECT u.*
FROM users u
WHERE NOT EXISTS (
  SELECT 1 FROM orders o WHERE o.user_id = u.id
);

-- Scalar correlated subquery in SELECT
SELECT
  p.*,
  (SELECT COUNT(*) FROM order_items oi WHERE oi.product_id = p.id) AS times_ordered
FROM products p;
```

:::tip[EXISTS vs IN for Subqueries]
Prefer `EXISTS` over `IN` with subqueries. `EXISTS` short-circuits on the first match and handles NULLs correctly. `IN` with a nullable subquery can produce surprising results.
:::

## CTEs (Common Table Expressions)

CTEs make complex queries readable by naming intermediate result sets.

### Basic CTE

```sql
WITH
  active_users AS (
    SELECT id, email, name
    FROM users
    WHERE is_active = true
      AND created_at >= NOW() - INTERVAL '1 year'
  ),
  recent_orders AS (
    SELECT user_id, COUNT(*) AS order_count, SUM(total) AS total_spent
    FROM orders
    WHERE created_at >= NOW() - INTERVAL '90 days'
    GROUP BY user_id
  )
SELECT
  u.name,
  u.email,
  COALESCE(ro.order_count, 0)   AS recent_orders,
  COALESCE(ro.total_spent, 0)   AS recent_spend
FROM active_users u
LEFT JOIN recent_orders ro ON ro.user_id = u.id
ORDER BY recent_spend DESC;
```

### Recursive CTE

Recursive CTEs can traverse hierarchical data:

```sql
-- Employee hierarchy tree
WITH RECURSIVE org_tree AS (
  -- Anchor: start with root employees (no manager)
  SELECT id, name, manager_id, 0 AS depth, name::TEXT AS path
  FROM employees
  WHERE manager_id IS NULL

  UNION ALL

  -- Recursive: join each employee to their manager
  SELECT
    e.id,
    e.name,
    e.manager_id,
    ot.depth + 1,
    ot.path || ' > ' || e.name
  FROM employees e
  JOIN org_tree ot ON e.manager_id = ot.id
)
SELECT depth, path, name
FROM org_tree
ORDER BY path;

-- Category tree (unlimited depth)
WITH RECURSIVE category_tree AS (
  SELECT id, name, parent_id, ARRAY[name] AS breadcrumb
  FROM categories WHERE parent_id IS NULL

  UNION ALL

  SELECT c.id, c.name, c.parent_id, ct.breadcrumb || c.name
  FROM categories c
  JOIN category_tree ct ON c.parent_id = ct.id
)
SELECT id, array_to_string(breadcrumb, ' / ') AS full_path
FROM category_tree ORDER BY breadcrumb;
```

## CASE Expression

### Simple CASE

```sql
SELECT
  name,
  status,
  CASE status
    WHEN 'active'   THEN 'green'
    WHEN 'pending'  THEN 'yellow'
    WHEN 'inactive' THEN 'red'
    ELSE 'gray'
  END AS status_color
FROM users;
```

### Searched CASE

```sql
SELECT
  id,
  total,
  CASE
    WHEN total >= 1000 THEN 'enterprise'
    WHEN total >= 500  THEN 'premium'
    WHEN total >= 100  THEN 'standard'
    ELSE 'basic'
  END AS tier
FROM orders;
```

### CASE in Aggregation (Conditional Aggregation)

```sql
SELECT
  DATE_TRUNC('month', created_at)           AS month,
  COUNT(*)                                   AS total_orders,
  COUNT(*) FILTER (WHERE status = 'delivered') AS delivered,
  COUNT(*) FILTER (WHERE status = 'cancelled') AS cancelled,
  SUM(CASE WHEN status = 'delivered' THEN total ELSE 0 END) AS delivered_revenue
FROM orders
GROUP BY DATE_TRUNC('month', created_at)
ORDER BY month;
```

## INSERT

```sql
-- Single row
INSERT INTO users (email, name, role)
VALUES ('alice@example.com', 'Alice Smith', 'user');

-- Multiple rows (much faster than individual inserts)
INSERT INTO products (name, price, category_id)
VALUES
  ('Widget A', 9.99, 1),
  ('Widget B', 14.99, 1),
  ('Gadget C', 24.99, 2);

-- INSERT ... SELECT (copy/transform data)
INSERT INTO order_archive (user_id, total, created_at)
SELECT user_id, total, created_at
FROM orders
WHERE created_at < NOW() - INTERVAL '1 year';

-- INSERT with RETURNING (get back generated values)
INSERT INTO users (email, name)
VALUES ('bob@example.com', 'Bob Jones')
RETURNING id, created_at;
```

## UPDATE

```sql
-- Basic update with WHERE
UPDATE products
SET price = price * 1.10,
    updated_at = NOW()
WHERE category_id = 3;

-- Update with a join (FROM clause in PostgreSQL)
UPDATE orders o
SET status = 'shipped',
    shipped_at = NOW()
FROM shipments s
WHERE s.order_id = o.id
  AND s.status = 'dispatched'
  AND o.status = 'processing';

-- Update from a subquery
UPDATE products
SET price = subq.new_price
FROM (
  SELECT id, AVG(price) OVER (PARTITION BY category_id) AS new_price
  FROM products
) subq
WHERE products.id = subq.id
AND ABS(products.price - subq.new_price) > 10;

-- Update with RETURNING
UPDATE users
SET last_login = NOW()
WHERE id = 42
RETURNING id, email, last_login;
```

## DELETE

```sql
-- Delete specific rows
DELETE FROM sessions WHERE expires_at < NOW();

-- Delete with RETURNING
DELETE FROM users
WHERE is_active = false
  AND last_login < NOW() - INTERVAL '2 years'
RETURNING id, email;

-- Delete with a join (PostgreSQL USING clause)
DELETE FROM order_items oi
USING orders o
WHERE oi.order_id = o.id
  AND o.status = 'cancelled'
  AND o.created_at < NOW() - INTERVAL '30 days';
```

## UPSERT (INSERT ... ON CONFLICT)

```sql
-- Do nothing on conflict
INSERT INTO user_preferences (user_id, key, value)
VALUES (42, 'theme', 'dark')
ON CONFLICT (user_id, key) DO NOTHING;

-- Update on conflict (upsert)
INSERT INTO user_preferences (user_id, key, value, updated_at)
VALUES (42, 'theme', 'dark', NOW())
ON CONFLICT (user_id, key) DO UPDATE SET
  value      = EXCLUDED.value,
  updated_at = EXCLUDED.updated_at;

-- Conditional upsert — only update if the new value is different
INSERT INTO page_views (page_id, views, last_viewed)
VALUES (101, 1, NOW())
ON CONFLICT (page_id) DO UPDATE SET
  views       = page_views.views + 1,
  last_viewed = NOW()
WHERE page_views.last_viewed < NOW() - INTERVAL '1 minute';

-- Upsert with RETURNING
INSERT INTO products (sku, name, price)
VALUES ('WIDGET-001', 'Widget', 9.99)
ON CONFLICT (sku) DO UPDATE SET
  price = EXCLUDED.price,
  name  = EXCLUDED.name
RETURNING id, sku, (xmax = 0) AS was_inserted;
-- xmax = 0 means it was an INSERT; xmax > 0 means it was an UPDATE
```

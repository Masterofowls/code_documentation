This page covers the full range of SQL query capabilities — from basic SELECT statements to recursive CTEs, CASE expressions, and all DML operations.

## SELECT Basics

```sql
-- Select all columns
SELECT * FROM products;

-- Select specific columns
SELECT id, name, price FROM products;

-- Column aliases
SELECT
  id,
  name      AS product_name,
  price     AS unit_price,
  price * 1.1 AS price_with_tax   -- Computed column alias
FROM products;

-- DISTINCT — deduplicate results
SELECT DISTINCT category_id FROM products;
SELECT DISTINCT ON (category_id) * FROM products ORDER BY category_id, price DESC;

-- Limit and offset for pagination
SELECT * FROM products
ORDER BY created_at DESC
LIMIT 20 OFFSET 40;              -- Page 3 with 20 items per page

-- FETCH FIRST (SQL standard, equivalent to LIMIT)
SELECT * FROM products
ORDER BY created_at DESC
FETCH FIRST 20 ROWS ONLY
OFFSET 40 ROWS;
```

## WHERE Clause

### Comparison Operators

```sql
SELECT * FROM products WHERE price = 29.99;       -- Equals
SELECT * FROM products WHERE price != 29.99;      -- Not equals
SELECT * FROM products WHERE price <> 29.99;      -- Not equals (alternate)
SELECT * FROM products WHERE price > 50;
SELECT * FROM products WHERE price < 50;
SELECT * FROM products WHERE price >= 50;
SELECT * FROM products WHERE price <= 50;
```

### Logical Operators

```sql
SELECT * FROM products
WHERE category_id = 5
  AND price >= 10
  AND price <= 100
  AND status = 'active';

SELECT * FROM products
WHERE category_id = 5
   OR category_id = 7
   OR category_id = 12;

-- NOT
SELECT * FROM products WHERE NOT status = 'discontinued';
SELECT * FROM products WHERE status != 'discontinued';   -- Equivalent
```

### IN and NOT IN

```sql
-- IN — equivalent to OR
SELECT * FROM products
WHERE category_id IN (5, 7, 12);

SELECT * FROM users
WHERE status NOT IN ('suspended', 'deleted');

-- IN with subquery
SELECT * FROM products
WHERE category_id IN (
  SELECT id FROM categories WHERE parent_id = 1
);
```

### BETWEEN

```sql
SELECT * FROM products
WHERE price BETWEEN 10 AND 100;   -- Inclusive on both ends

SELECT * FROM orders
WHERE created_at BETWEEN '2024-01-01' AND '2024-12-31 23:59:59';
```

### LIKE and Pattern Matching

```sql
-- % matches any sequence of characters
SELECT * FROM users WHERE email LIKE '%@gmail.com';
SELECT * FROM products WHERE name LIKE 'Widget%';        -- Starts with
SELECT * FROM products WHERE name LIKE '%Widget%';       -- Contains

-- _ matches exactly one character
SELECT * FROM products WHERE sku LIKE 'SKU-___-2024';

-- ILIKE — case-insensitive (PostgreSQL)
SELECT * FROM products WHERE name ILIKE '%widget%';

-- SIMILAR TO — SQL standard regex
SELECT * FROM users WHERE phone SIMILAR TO '[0-9]{10}';

-- ~ and ~* — POSIX regex (PostgreSQL)
SELECT * FROM users WHERE email ~ '^[a-z]+@';           -- Case-sensitive
SELECT * FROM users WHERE email ~* '^[a-z]+@';          -- Case-insensitive
```

### NULL Handling

```sql
-- NULL comparisons always use IS NULL / IS NOT NULL
SELECT * FROM products WHERE description IS NULL;
SELECT * FROM products WHERE description IS NOT NULL;

-- NULL != NULL — these return 0 rows:
SELECT * FROM products WHERE description = NULL;   -- WRONG!
SELECT * FROM products WHERE description != NULL;  -- WRONG!

-- COALESCE — return first non-NULL value
SELECT name, COALESCE(description, 'No description provided') AS description
FROM products;

-- NULLIF — return NULL if two values are equal
SELECT NULLIF(quantity, 0)  -- Returns NULL instead of 0 (avoids division by zero)
FROM inventory;
```

## ORDER BY

```sql
-- Single column ascending (default)
SELECT * FROM products ORDER BY price;
SELECT * FROM products ORDER BY price ASC;

-- Descending
SELECT * FROM products ORDER BY price DESC;

-- Multiple columns
SELECT * FROM products ORDER BY category_id ASC, price DESC;

-- NULL placement
SELECT * FROM products ORDER BY discount NULLS FIRST;
SELECT * FROM products ORDER BY discount NULLS LAST;

-- Order by expression
SELECT * FROM products ORDER BY price * (1 - discount) DESC;

-- Order by alias
SELECT price * quantity AS total_value FROM order_items
ORDER BY total_value DESC;

-- Order by column position (fragile — avoid in production)
SELECT id, name, price FROM products ORDER BY 3 DESC;
```

## Subqueries

### Non-Correlated Subquery

Executes once; the result is used by the outer query.

```sql
-- Products priced above average
SELECT name, price
FROM products
WHERE price > (SELECT AVG(price) FROM products);

-- Customers who have placed orders
SELECT * FROM customers
WHERE id IN (
  SELECT DISTINCT customer_id FROM orders
);

-- Scalar subquery in SELECT
SELECT
  name,
  price,
  (SELECT AVG(price) FROM products) AS avg_price,
  price - (SELECT AVG(price) FROM products) AS diff_from_avg
FROM products;
```

### Correlated Subquery

References columns from the outer query; executes once per outer row.

```sql
-- Customers whose most recent order was this year
SELECT c.name
FROM customers c
WHERE (
  SELECT MAX(created_at)
  FROM orders o
  WHERE o.customer_id = c.id    -- Correlation
) >= '2024-01-01';

-- Products that are the most expensive in their category
SELECT *
FROM products p1
WHERE price = (
  SELECT MAX(price)
  FROM products p2
  WHERE p2.category_id = p1.category_id  -- Correlation
);
```

### EXISTS and NOT EXISTS

More efficient than `IN` with subqueries for large datasets:

```sql
-- Customers with at least one order
SELECT * FROM customers c
WHERE EXISTS (
  SELECT 1 FROM orders o WHERE o.customer_id = c.id
);

-- Customers with no orders
SELECT * FROM customers c
WHERE NOT EXISTS (
  SELECT 1 FROM orders o WHERE o.customer_id = c.id
);
```

## Common Table Expressions (CTEs)

CTEs make complex queries readable by naming intermediate result sets.

### Basic CTE

```sql
WITH expensive_products AS (
  SELECT id, name, price, category_id
  FROM products
  WHERE price > 100
    AND status = 'active'
),
high_value_categories AS (
  SELECT category_id, COUNT(*) AS product_count, AVG(price) AS avg_price
  FROM expensive_products
  GROUP BY category_id
  HAVING COUNT(*) >= 3
)
SELECT
  c.name            AS category_name,
  hvc.product_count,
  hvc.avg_price
FROM high_value_categories hvc
JOIN categories c ON c.id = hvc.category_id
ORDER BY hvc.avg_price DESC;
```

### Recursive CTE

Recursive CTEs process hierarchical or graph data.

```sql
-- Traverse a category hierarchy (parent → children)
WITH RECURSIVE category_tree AS (
  -- Base case: top-level categories (no parent)
  SELECT
    id,
    name,
    parent_id,
    0           AS depth,
    name        AS path
  FROM categories
  WHERE parent_id IS NULL

  UNION ALL

  -- Recursive case: children of already-found categories
  SELECT
    c.id,
    c.name,
    c.parent_id,
    ct.depth + 1,
    ct.path || ' > ' || c.name
  FROM categories c
  INNER JOIN category_tree ct ON c.parent_id = ct.id
)
SELECT id, name, depth, path
FROM category_tree
ORDER BY path;

-- Employee reporting chain
WITH RECURSIVE reporting_chain AS (
  SELECT id, name, manager_id, 0 AS level, name AS chain
  FROM employees
  WHERE id = 42   -- Start from specific employee

  UNION ALL

  SELECT
    e.id, e.name, e.manager_id,
    rc.level + 1,
    e.name || ' → ' || rc.chain
  FROM employees e
  JOIN reporting_chain rc ON e.id = rc.manager_id
)
SELECT * FROM reporting_chain ORDER BY level;
```

:::tip[Recursive CTE Cycle Detection]
Always add a depth limit or cycle detection to recursive CTEs to prevent infinite loops on circular data:
```sql
WHERE ct.depth < 10    -- Depth limit
```
:::

## CASE Expression

### Simple CASE

```sql
SELECT
  name,
  price,
  CASE status
    WHEN 'active'       THEN 'Available'
    WHEN 'discontinued' THEN 'No longer sold'
    WHEN 'draft'        THEN 'Coming soon'
    ELSE 'Unknown'
  END AS status_label
FROM products;
```

### Searched CASE

```sql
SELECT
  name,
  price,
  CASE
    WHEN price < 25   THEN 'budget'
    WHEN price < 100  THEN 'mid-range'
    WHEN price < 500  THEN 'premium'
    ELSE 'luxury'
  END AS tier
FROM products
ORDER BY price;

-- CASE in ORDER BY
SELECT name, status
FROM products
ORDER BY
  CASE status
    WHEN 'active'       THEN 1
    WHEN 'draft'        THEN 2
    WHEN 'discontinued' THEN 3
    ELSE 4
  END;

-- CASE in aggregate
SELECT
  COUNT(*) AS total_orders,
  COUNT(CASE WHEN status = 'completed' THEN 1 END) AS completed,
  COUNT(CASE WHEN status = 'cancelled' THEN 1 END) AS cancelled,
  SUM(CASE WHEN status = 'completed' THEN total ELSE 0 END) AS completed_revenue
FROM orders;
```

## INSERT

```sql
-- Single row
INSERT INTO products (sku, name, price, category_id)
VALUES ('WIDGET-001', 'Blue Widget', 29.99, 1);

-- Multi-row insert
INSERT INTO products (sku, name, price, category_id)
VALUES
  ('WIDGET-002', 'Red Widget',   24.99, 1),
  ('WIDGET-003', 'Green Widget', 19.99, 1),
  ('GADGET-001', 'Power Gadget', 99.99, 2);

-- INSERT ... SELECT (copy from another table)
INSERT INTO archived_orders (id, customer_id, total, created_at)
SELECT id, customer_id, total, created_at
FROM orders
WHERE created_at < '2023-01-01';

-- RETURNING — get the inserted row(s) back
INSERT INTO products (sku, name, price)
VALUES ('NEW-001', 'New Product', 49.99)
RETURNING id, created_at;
```

## UPDATE

```sql
-- Basic update
UPDATE products
SET price = 39.99
WHERE id = 1;

-- Update multiple columns
UPDATE users
SET
  last_login  = NOW(),
  login_count = login_count + 1,
  status      = 'active'
WHERE id = $1;

-- Update with FROM (PostgreSQL) — join another table
UPDATE products p
SET price = p.price * 0.90
FROM categories c
WHERE p.category_id = c.id
  AND c.name = 'Clearance';

-- RETURNING
UPDATE orders
SET status = 'shipped', shipped_at = NOW()
WHERE id = 42
RETURNING id, status, shipped_at;
```

## DELETE

```sql
-- Delete specific rows
DELETE FROM products WHERE status = 'discontinued' AND stock = 0;

-- Delete with subquery
DELETE FROM order_items
WHERE order_id IN (
  SELECT id FROM orders WHERE created_at < '2020-01-01'
);

-- Delete using JOIN (PostgreSQL: using)
DELETE FROM products p
USING categories c
WHERE p.category_id = c.id
  AND c.name = 'Deprecated';

-- RETURNING
DELETE FROM sessions
WHERE expires_at < NOW()
RETURNING id, user_id;
```

## UPSERT (INSERT ON CONFLICT)

```sql
-- Do nothing on conflict
INSERT INTO product_views (product_id, view_date, count)
VALUES (1, CURRENT_DATE, 1)
ON CONFLICT (product_id, view_date) DO NOTHING;

-- Update on conflict (the typical upsert)
INSERT INTO user_settings (user_id, setting_key, setting_value)
VALUES (42, 'theme', 'dark')
ON CONFLICT (user_id, setting_key)
DO UPDATE SET
  setting_value = EXCLUDED.setting_value,  -- EXCLUDED = the rejected row
  updated_at    = NOW();

-- Update only if the new value changes something meaningful
INSERT INTO products (sku, name, price)
VALUES ('SKU-001', 'Widget', 29.99)
ON CONFLICT (sku)
DO UPDATE SET
  name  = EXCLUDED.name,
  price = EXCLUDED.price
WHERE products.price != EXCLUDED.price   -- Only update if price changed
   OR products.name  != EXCLUDED.name;
```

The **App Router** is the modern routing system in Next.js, built on React Server Components. It uses a **file-system-based router** where the folder hierarchy maps directly to URL segments, and special file names define the behavior of each route segment.

## File Conventions

Each route segment is a folder. The following files have special meaning within a segment:

| File | Purpose |
|---|---|
| `page.tsx` | The UI for this route (makes it publicly accessible) |
| `layout.tsx` | Persistent layout wrapping the page and its children |
| `loading.tsx` | Loading skeleton shown instantly while the segment loads |
| `error.tsx` | Error boundary — shown when an error is thrown in the segment |
| `not-found.tsx` | UI for `notFound()` calls or unmatched routes |
| `template.tsx` | Like layout, but re-mounts on navigation (no state persistence) |
| `route.ts` | API Route Handler (no UI — only JSON/Response) |
| `default.tsx` | Fallback for parallel routes |

```
src/app/
├── layout.tsx          ← Root layout (required — wraps entire app)
├── page.tsx            ← / (home)
├── loading.tsx         ← Root loading state
├── error.tsx           ← Root error boundary
├── not-found.tsx       ← Root 404
├── blog/
│   ├── layout.tsx      ← Blog layout (wraps all /blog/... routes)
│   ├── page.tsx        ← /blog
│   └── [slug]/
│       ├── page.tsx    ← /blog/my-post-title
│       └── loading.tsx ← Per-post loading state
└── api/
    └── posts/
        └── route.ts    ← /api/posts
```

## Root Layout

The root `layout.tsx` is **required** and must define the `<html>` and `<body>` tags. It wraps every route in the app and is never unmounted during navigation.

```tsx
// src/app/layout.tsx
import type { Metadata } from 'next';
import { Inter } from 'next/font/google';
import { Providers } from '@/components/Providers';
import './globals.css';

const inter = Inter({
  subsets: ['latin'],
  variable: '--font-inter',
  display: 'swap',
});

export const metadata: Metadata = {
  metadataBase: new URL('https://myapp.com'),
  title: {
    default: 'My App',
    template: '%s | My App',   // page titles render as: "Blog Post | My App"
  },
  description: 'The best app ever built.',
  openGraph: {
    type: 'website',
    url: 'https://myapp.com',
    siteName: 'My App',
    images: [{ url: '/og-image.jpg', width: 1200, height: 630 }],
  },
  twitter: {
    card: 'summary_large_image',
    creator: '@yourhandle',
  },
  robots: {
    index: true,
    follow: true,
  },
};

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en" className={inter.variable} suppressHydrationWarning>
      <body className="min-h-screen bg-background text-foreground antialiased">
        <Providers>
          {children}
        </Providers>
      </body>
    </html>
  );
}
```

:::tip[suppressHydrationWarning on html]
Add `suppressHydrationWarning` to the `<html>` tag to avoid hydration mismatch warnings caused by browser extensions or theme scripts that modify the DOM before React hydrates.
:::

## Nested Layouts

Layouts are automatically nested based on the folder hierarchy. A child layout wraps its children **without unmounting** on navigation within its subtree.

```tsx
// src/app/dashboard/layout.tsx
import { redirect } from 'next/navigation';
import { getSession } from '@/lib/auth';
import { Sidebar } from '@/components/dashboard/Sidebar';
import { DashboardHeader } from '@/components/dashboard/DashboardHeader';

export default async function DashboardLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  const session = await getSession();

  if (!session) {
    redirect('/login?from=/dashboard');
  }

  return (
    <div className="flex h-screen overflow-hidden">
      <Sidebar user={session.user} />
      <div className="flex flex-col flex-1 overflow-hidden">
        <DashboardHeader user={session.user} />
        <main className="flex-1 overflow-y-auto p-6">
          {children}
        </main>
      </div>
    </div>
  );
}
```

## Loading and Error States

### loading.tsx

`loading.tsx` is automatically wrapped in a `<Suspense>` boundary. It shows instantly while the page (or layout) is streaming:

```tsx
// src/app/dashboard/loading.tsx
export default function DashboardLoading() {
  return (
    <div className="space-y-4 animate-pulse">
      <div className="h-8 bg-gray-200 rounded w-1/3" />
      <div className="grid grid-cols-3 gap-4">
        {Array.from({ length: 6 }).map((_, i) => (
          <div key={i} className="h-32 bg-gray-200 rounded-xl" />
        ))}
      </div>
    </div>
  );
}
```

### error.tsx

`error.tsx` must be a **Client Component** (it uses the `error` prop and `reset` function):

```tsx
// src/app/dashboard/error.tsx
'use client';

import { useEffect } from 'react';

interface ErrorProps {
  error: Error & { digest?: string };
  reset: () => void;
}

export default function DashboardError({ error, reset }: ErrorProps) {
  useEffect(() => {
    // Log to error tracking service:
    console.error('Dashboard error:', error);
  }, [error]);

  return (
    <div className="flex flex-col items-center justify-center h-64 gap-4">
      <h2 className="text-xl font-semibold text-gray-900">Something went wrong</h2>
      <p className="text-gray-500">{error.message}</p>
      <button
        onClick={reset}
        className="px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700"
      >
        Try again
      </button>
    </div>
  );
}
```

### not-found.tsx

```tsx
// src/app/not-found.tsx
import Link from 'next/link';

export default function NotFound() {
  return (
    <div className="flex flex-col items-center justify-center min-h-screen gap-4">
      <h1 className="text-6xl font-bold text-gray-900">404</h1>
      <p className="text-xl text-gray-600">The page you are looking for does not exist.</p>
      <Link href="/" className="text-blue-600 hover:underline">
        Return home
      </Link>
    </div>
  );
}
```

## template.tsx

Unlike `layout.tsx`, `template.tsx` creates a **new instance** on each navigation. Use it when you need re-mounting behavior (e.g., page transition animations, resetting useEffect on navigation):

```tsx
// src/app/template.tsx
'use client';

import { motion } from 'framer-motion';

export default function Template({ children }: { children: React.ReactNode }) {
  return (
    <motion.div
      initial={{ opacity: 0, y: 8 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.2 }}
    >
      {children}
    </motion.div>
  );
}
```

## Route Groups

Parentheses in a folder name `(groupName)` create a **route group** — it organizes files without affecting the URL. Use route groups to:

- Share a layout among some routes but not others
- Split the app into meaningful sections
- Create multiple root layouts

```
src/app/
├── (marketing)/           ← No URL segment — routes are at /
│   ├── layout.tsx         ← Marketing-specific layout (no sidebar)
│   ├── page.tsx           ← /
│   ├── about/page.tsx     ← /about
│   └── pricing/page.tsx   ← /pricing
├── (auth)/                ← No URL segment
│   ├── layout.tsx         ← Centered auth layout
│   ├── login/page.tsx     ← /login
│   └── register/page.tsx  ← /register
└── (app)/                 ← No URL segment
    ├── layout.tsx          ← App layout (with sidebar, auth-gated)
    ├── dashboard/page.tsx  ← /dashboard
    └── settings/page.tsx   ← /settings
```

## Private Folders

Folders prefixed with `_` are excluded from routing — useful for colocating non-route files:

```
src/app/
├── _components/           ← Not a route — safe to colocate helpers
│   ├── PageHeader.tsx
│   └── Breadcrumbs.tsx
├── _utils/
│   └── formatDate.ts
├── dashboard/
│   └── page.tsx
```

## Parallel Routes

Parallel routes (`@slotName`) let you simultaneously render multiple pages in the same layout — useful for modals, split views, or conditional slots:

```
src/app/
├── layout.tsx             ← Receives @team and @analytics as props
├── page.tsx               ← Default content
├── @team/
│   ├── page.tsx           ← Team slot content
│   └── default.tsx        ← Fallback when no match
└── @analytics/
    ├── page.tsx           ← Analytics slot content
    └── default.tsx
```

```tsx
// src/app/layout.tsx
export default function Layout({
  children,
  team,
  analytics,
}: {
  children: React.ReactNode;
  team: React.ReactNode;
  analytics: React.ReactNode;
}) {
  return (
    <div className="grid grid-cols-2 gap-4">
      <div>{children}</div>
      <div>{team}</div>
      <div className="col-span-2">{analytics}</div>
    </div>
  );
}
```

## Intercepting Routes

Intercepting routes `(.)`, `(..)`, `(..)(..)`, `(...)` let you load a route **within a different layout** — the classic use case is a modal photo gallery where the URL updates but the background stays:

```
src/app/
├── feed/
│   └── page.tsx           ← /feed — full feed page
├── photo/
│   └── [id]/
│       └── page.tsx       ← /photo/123 — full photo page
└── @modal/
    ├── default.tsx         ← null (no modal by default)
    └── (.)photo/           ← Intercept /photo/* when navigating from /feed
        └── [id]/
            └── page.tsx    ← Renders as modal overlay on top of /feed
```

## Metadata API

### Static Metadata

Export a `metadata` object from any `page.tsx` or `layout.tsx`:

```typescript
// src/app/blog/page.tsx
import type { Metadata } from 'next';

export const metadata: Metadata = {
  title: 'Blog',
  description: 'Read the latest articles from our team.',
  keywords: ['blog', 'articles', 'news'],
  alternates: {
    canonical: 'https://myapp.com/blog',
  },
  openGraph: {
    title: 'Blog – My App',
    description: 'Read the latest articles.',
    images: ['/og/blog.jpg'],
  },
};
```

### Dynamic Metadata

Use `generateMetadata` for dynamic content:

```typescript
// src/app/blog/[slug]/page.tsx
import type { Metadata, ResolvingMetadata } from 'next';

interface Props {
  params: Promise<{ slug: string }>;
}

export async function generateMetadata(
  { params }: Props,
  parent: ResolvingMetadata,
): Promise<Metadata> {
  const { slug } = await params;
  const post = await fetchPost(slug);

  // Merge with parent metadata:
  const previousImages = (await parent).openGraph?.images ?? [];

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      title: post.title,
      description: post.excerpt,
      images: [post.coverImage, ...previousImages],
      type: 'article',
      publishedTime: post.publishedAt,
      authors: [post.author.name],
    },
    twitter: {
      card: 'summary_large_image',
      title: post.title,
      description: post.excerpt,
      images: [post.coverImage],
    },
  };
}

export default async function BlogPost({ params }: Props) {
  const { slug } = await params;
  const post = await fetchPost(slug);
  return <article>{/* ... */}</article>;
}
```

## generateStaticParams

Pre-render dynamic routes at build time:

```typescript
// src/app/blog/[slug]/page.tsx
export async function generateStaticParams() {
  const posts = await fetchAllPosts();

  return posts.map((post) => ({
    slug: post.slug,
  }));
}

// Control what happens for slugs not in generateStaticParams:
export const dynamicParams = true; // true = generate on demand (default)
// export const dynamicParams = false; // false = 404 for unknown slugs
```

:::caution[generateStaticParams Runs at Build Time]
The `generateStaticParams` function always runs at build time, even in development. Use it only for content that exists before deployment (or is fetched from a stable API). It uses the same `fetch` caching as the rest of the app.
:::

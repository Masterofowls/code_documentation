# Docker Compose

Docker Compose is a tool for defining and running multi-container Docker applications. With a single YAML file, you configure your application's services, networks, and volumes — then bring everything up with one command.

:::tip[Compose v2 vs v1]
Docker Compose v2 is integrated directly into the Docker CLI as `docker compose` (a plugin). The legacy standalone `docker-compose` (v1, written in Python) is deprecated. Always use `docker compose` (no hyphen) for new projects.
:::

---

## `compose.yml` Structure

```yaml
# compose.yml (also accepted: docker-compose.yml, docker-compose.yaml)
name: my-project                    # Optional: override project name

services:
  web:                              # Service name
    image: nginx:alpine
    ports:
      - "80:80"

networks:
  default:                          # Networks section
    driver: bridge

volumes:
  db-data:                          # Named volumes section

configs:                            # Docker configs (Swarm / Compose v2.3+)
  nginx_config:
    file: ./nginx.conf

secrets:                            # Docker secrets
  db_password:
    file: ./secrets/db_password.txt
```

## Service Configuration

### Core Options

```yaml
services:
  api:
    # Image to use (mutually exclusive with 'build' — or use both)
    image: node:20-alpine

    # Build configuration
    build:
      context: .                    # Build context path
      dockerfile: Dockerfile.prod   # Dockerfile path (default: Dockerfile)
      args:
        NODE_ENV: production
        APP_VERSION: "1.0.0"
      target: runner                # Multi-stage build target
      cache_from:
        - my-registry/api:latest
      labels:
        com.example.version: "1.0"

    # Container name (overrides generated name)
    container_name: my-api

    # Command to run (overrides CMD in Dockerfile)
    command: ["node", "dist/server.js"]

    # Entrypoint (overrides ENTRYPOINT in Dockerfile)
    entrypoint: ["/sbin/tini", "--"]

    # Working directory
    working_dir: /app

    # User
    user: "1001:1001"
```

### Ports

```yaml
services:
  web:
    ports:
      - "80:80"                     # host:container
      - "443:443"
      - "127.0.0.1:8080:80"        # Bind to specific host interface
      - "3000"                      # Random host port maps to 3000
      - target: 80                  # Long-form syntax
        published: 8080
        protocol: tcp
        mode: host
```

### Environment Variables

```yaml
services:
  api:
    # Inline values
    environment:
      NODE_ENV: production
      PORT: "3000"
      DATABASE_URL: postgres://postgres:password@db:5432/mydb

    # From host environment (variable must exist on host)
    environment:
      - NODE_ENV                    # Takes value from host
      - DATABASE_URL

    # From a .env file
    env_file:
      - .env
      - .env.production             # Later files override earlier ones

    # Combination
    environment:
      - NODE_ENV=production
    env_file:
      - .env.secrets
```

### Volumes in Services

```yaml
services:
  api:
    volumes:
      # Named volume
      - api-data:/app/data

      # Bind mount (long form)
      - type: bind
        source: ./src
        target: /app/src
        read_only: false

      # Bind mount (short form)
      - ./nginx.conf:/etc/nginx/nginx.conf:ro

      # tmpfs (in-memory)
      - type: tmpfs
        target: /tmp

      # Named volume with options
      - type: volume
        source: db-data
        target: /var/lib/postgresql/data
        volume:
          nocopy: true              # Don't copy container data to volume on creation
```

### Health Checks

```yaml
services:
  api:
    healthcheck:
      test: ["CMD", "wget", "-qO-", "http://localhost:3000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 30s             # Don't count failures during initial startup
      # Disable inherited healthcheck:
      # test: ["NONE"]
```

### Dependencies and Restart Policies

```yaml
services:
  api:
    depends_on:
      db:
        condition: service_healthy  # Wait for healthy status
      redis:
        condition: service_started  # Wait for container to start (default)
      migrations:
        condition: service_completed_successfully  # Wait for exit 0

    restart: unless-stopped         # no | always | on-failure | unless-stopped
    # restart: on-failure:3         # Retry up to 3 times

    # Stop grace period (default: 10s)
    stop_grace_period: 30s

  db:
    image: postgres:16-alpine
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 10s
      timeout: 5s
      retries: 5

  migrations:
    image: my-app
    command: ["npm", "run", "db:migrate"]
    restart: "no"
    depends_on:
      db:
        condition: service_healthy
```

### Profiles

```yaml
services:
  web:
    image: nginx
    # No profile — always starts

  debug-tools:
    image: nicolaka/netshoot
    profiles:
      - debug                       # Only starts with --profile debug

  docs:
    image: my-docs
    profiles:
      - docs
```

```bash
# Start all services including debug profile
docker compose --profile debug up

# Start multiple profiles
docker compose --profile debug --profile docs up
```

## Networks

### Default Network

By default, Compose creates a single bridge network named `<project>_default`. All services join it and can reach each other by service name.

```yaml
services:
  web:
    image: nginx
    # Can reach 'api' at http://api:3000

  api:
    image: my-api
    # Can reach 'db' at postgres://db:5432
```

### Custom Networks

```yaml
services:
  nginx:
    networks:
      - frontend
      - backend

  api:
    networks:
      - backend
      - db-net

  db:
    networks:
      - db-net

networks:
  frontend:
    driver: bridge
    ipam:
      config:
        - subnet: 172.20.0.0/16

  backend:
    driver: bridge

  db-net:
    driver: bridge
    internal: true                  # No external access

  # Use a pre-existing external network
  existing-net:
    external: true
    name: my-existing-network
```

### Network Aliases

```yaml
services:
  api:
    networks:
      backend:
        aliases:
          - app-server
          - api.internal
```

## Named Volumes

```yaml
services:
  db:
    volumes:
      - postgres-data:/var/lib/postgresql/data

  redis:
    volumes:
      - redis-data:/data

volumes:
  postgres-data:                    # Managed by Docker
    driver: local

  redis-data:
    driver: local

  # Use an external (pre-existing) volume
  external-vol:
    external: true
    name: my-existing-volume

  # NFS volume
  nfs-data:
    driver: local
    driver_opts:
      type: nfs
      o: addr=192.168.1.100,rw
      device: ":/exports/data"
```

## Override Files

Compose automatically merges `compose.override.yml` with `compose.yml`. Use this for dev/production splits.

```yaml
# compose.yml — base configuration (committed to git)
services:
  api:
    image: my-api:${TAG:-latest}
    environment:
      NODE_ENV: production

  db:
    image: postgres:16-alpine
    volumes:
      - db-data:/var/lib/postgresql/data

volumes:
  db-data:
```

```yaml
# compose.override.yml — development overrides (committed or gitignored)
services:
  api:
    build: .                        # Build from source in dev
    volumes:
      - ./src:/app/src              # Live reload
    environment:
      NODE_ENV: development
    ports:
      - "9229:9229"                 # Node debugger

  db:
    ports:
      - "5432:5432"                 # Expose DB locally in dev
```

```yaml
# compose.prod.yml — production overrides
services:
  api:
    deploy:
      replicas: 3
      resources:
        limits:
          cpus: "0.5"
          memory: 512M
    restart: always
    logging:
      driver: "json-file"
      options:
        max-size: "10m"
        max-file: "3"
```

```bash
# Development (auto-merges compose.override.yml)
docker compose up

# Production (explicit files, no override)
docker compose -f compose.yml -f compose.prod.yml up -d

# Override project name
docker compose -p myproject up
```

## Docker Compose CLI

```bash
# Start all services (detached)
docker compose up -d

# Build images before starting
docker compose up -d --build

# Pull latest images before starting
docker compose up -d --pull always

# Stop and remove containers, networks (keep volumes)
docker compose down

# Also remove volumes (DATA LOSS)
docker compose down -v

# Also remove images
docker compose down --rmi all

# List services and their status
docker compose ps

# View logs
docker compose logs
docker compose logs -f api         # Follow logs for one service
docker compose logs --tail 50 api

# Execute command in a running container
docker compose exec api sh
docker compose exec db psql -U postgres

# Run a one-off command (creates a new container)
docker compose run --rm api npm run db:seed
docker compose run --rm --no-deps api npm test

# Build images
docker compose build
docker compose build api           # Build specific service
docker compose build --no-cache

# Pull images
docker compose pull

# Restart services
docker compose restart api

# Scale a service
docker compose up -d --scale api=3

# Show running processes in containers
docker compose top

# Validate and view config
docker compose config
docker compose config --services
docker compose config --volumes
```

:::caution[docker compose run vs exec]
`docker compose run` creates a **new** container. `docker compose exec` runs a command in an **existing** running container. For one-off tasks like migrations, use `run --rm` to avoid leaving behind stopped containers.
:::

## Full Stack Example

A production-ready stack: Node.js API + PostgreSQL + Redis + Nginx reverse proxy.

```yaml
# compose.yml
name: myapp

services:
  nginx:
    image: nginx:1.25-alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/certs:/etc/nginx/certs:ro
    depends_on:
      api:
        condition: service_healthy
    restart: unless-stopped
    networks:
      - frontend

  api:
    build:
      context: .
      target: production
    environment:
      NODE_ENV: production
      PORT: 3000
      DATABASE_URL: postgres://${POSTGRES_USER}:${POSTGRES_PASSWORD}@db:5432/${POSTGRES_DB}
      REDIS_URL: redis://redis:6379
    env_file:
      - .env
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "wget", "-qO-", "http://localhost:3000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 20s
    restart: unless-stopped
    networks:
      - frontend
      - backend
    deploy:
      resources:
        limits:
          memory: 512M
          cpus: "0.5"

  db:
    image: postgres:16-alpine
    environment:
      POSTGRES_USER: ${POSTGRES_USER}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
      POSTGRES_DB: ${POSTGRES_DB}
    volumes:
      - postgres-data:/var/lib/postgresql/data
      - ./db/init:/docker-entrypoint-initdb.d:ro
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${POSTGRES_USER} -d ${POSTGRES_DB}"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - backend

  redis:
    image: redis:7-alpine
    command: redis-server --appendonly yes --maxmemory 256mb --maxmemory-policy allkeys-lru
    volumes:
      - redis-data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 3
    restart: unless-stopped
    networks:
      - backend

  migrations:
    build:
      context: .
      target: production
    command: ["npm", "run", "db:migrate"]
    environment:
      DATABASE_URL: postgres://${POSTGRES_USER}:${POSTGRES_PASSWORD}@db:5432/${POSTGRES_DB}
    env_file:
      - .env
    depends_on:
      db:
        condition: service_healthy
    restart: "no"
    networks:
      - backend

networks:
  frontend:
    driver: bridge
  backend:
    driver: bridge
    internal: true                  # DB and Redis not reachable from outside

volumes:
  postgres-data:
  redis-data:
```

```bash
# .env file
POSTGRES_USER=myapp
POSTGRES_PASSWORD=supersecretpassword
POSTGRES_DB=myapp_production
TAG=1.2.3
```

```bash
# Start the full stack
docker compose up -d

# Check everything is healthy
docker compose ps

# View all logs
docker compose logs -f

# Run database migrations
docker compose run --rm migrations

# Scale the API
docker compose up -d --scale api=3
```

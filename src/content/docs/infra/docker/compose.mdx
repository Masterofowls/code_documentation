# Docker Compose

Docker Compose lets you define and run multi-container applications from a single YAML file. Instead of juggling multiple `docker run` commands, you declare all services, networks, and volumes in `compose.yml` and manage them with a single command.

:::tip[Compose v2]
`docker compose` (with a space) is the modern v2 plugin, written in Go and bundled with Docker Desktop and Docker Engine. The older `docker-compose` (with a hyphen, v1) was a separate Python tool and is now deprecated. Always use `docker compose`.
:::

## compose.yml Structure

```yaml
# compose.yml (formerly docker-compose.yml)
name: myapp # project name (default: directory name)

services:
  web:
    # ... service definition

networks:
  frontend:
    # ... network definition

volumes:
  db-data:
    # ... volume definition

configs: # Docker Swarm / Compose v2.3+
  nginx-config:
    file: ./nginx.conf

secrets: # Docker Swarm / Compose v2.3+
  db-password:
    file: ./secrets/db-password.txt
```

## Service Definition

```yaml
services:
  app:
    # ── Image ────────────────────────────────────────────────────
    image: node:20-alpine # use existing image
    build: # or build from Dockerfile
      context: . # build context directory
      dockerfile: Dockerfile.prod # optional custom Dockerfile name
      target: runtime # multi-stage build target
      args:
        BUILD_VERSION: '1.2.3' # build ARGs
      cache_from:
        - myregistry/myapp:cache
      labels:
        com.example.version: '1.2.3'

    # ── Container settings ───────────────────────────────────────
    container_name: myapp # custom container name
    hostname: app # container hostname

    # ── Ports ────────────────────────────────────────────────────
    ports:
      - '3000:3000' # host:container
      - '127.0.0.1:9229:9229' # bind to localhost only
      - '8080-8090:8080-8090' # range

    # ── Environment ──────────────────────────────────────────────
    environment:
      NODE_ENV: production
      PORT: '3000'
      DB_HOST: db # service name resolves via DNS
    env_file:
      - .env # load from file
      - .env.production # merged (later overrides earlier)

    # ── Volumes ──────────────────────────────────────────────────
    volumes:
      - app-data:/app/data # named volume
      - ./src:/app/src:ro # bind mount, read-only
      - /etc/localtime:/etc/localtime:ro # host path

    # ── Dependencies ─────────────────────────────────────────────
    depends_on:
      db:
        condition: service_healthy # wait for health check to pass
      redis:
        condition: service_started # just wait for container to start

    # ── Restart policy ───────────────────────────────────────────
    restart: unless-stopped
    # Options: "no", "always", "on-failure", "unless-stopped"
    # on-failure with limit:
    # restart: "on-failure:3"

    # ── Health check ─────────────────────────────────────────────
    healthcheck:
      test: ['CMD', 'wget', '-qO-', 'http://localhost:3000/health']
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

    # ── Command/Entrypoint ───────────────────────────────────────
    command: ['node', 'dist/server.js']
    entrypoint: ['/sbin/tini', '--']

    # ── Resources ────────────────────────────────────────────────
    deploy:
      resources:
        limits:
          cpus: '1.0'
          memory: 512M
        reservations:
          cpus: '0.25'
          memory: 128M

    # ── Networks ─────────────────────────────────────────────────
    networks:
      - frontend
      - backend

    # ── Profiles ─────────────────────────────────────────────────
    profiles:
      - debug # only starts when --profile debug is used

    # ── User ─────────────────────────────────────────────────────
    user: '1001:1001'

    # ── Logging ──────────────────────────────────────────────────
    logging:
      driver: json-file
      options:
        max-size: '10m'
        max-file: '3'

    # ── Working directory ─────────────────────────────────────────
    working_dir: /app

    # ── Stdin / TTY ──────────────────────────────────────────────
    stdin_open: true # -i flag
    tty: true # -t flag
```

## Networks

```yaml
services:
  frontend:
    networks:
      - public-net
      - internal-net
    # Network alias (other containers can reach this service by alias)
    networks:
      internal-net:
        aliases:
          - frontend-service

  backend:
    networks:
      - internal-net

networks:
  public-net:
    driver: bridge

  internal-net:
    driver: bridge
    internal: true          # no external access — isolated network

  existing-net:
    external: true          # use a network created outside Compose
    name: my-existing-network

  overlay-net:
    driver: overlay         # for Docker Swarm multi-host networking
    attachable: true        # allow standalone containers to attach

  custom-bridge:
    driver: bridge
    driver_opts:
      com.docker.network.bridge.name: my_bridge
    ipam:
      config:
        - subnet: "172.28.0.0/16"
          gateway: "172.28.0.1"
```

:::tip[Service Name DNS]
In a user-defined network, containers resolve each other by **service name**. `app` can reach `db` at the hostname `db` — no need to hardcode IPs.
:::

## Named Volumes vs Bind Mounts

```yaml
services:
  db:
    image: postgres:16-alpine
    volumes:
      # Named volume — managed by Docker, survives container removal
      - postgres-data:/var/lib/postgresql/data

      # Bind mount — maps host directory into container
      - ./postgres/init:/docker-entrypoint-initdb.d:ro

      # tmpfs — in-memory, not persisted
      - type: tmpfs
        target: /tmp

volumes:
  postgres-data: # declare named volume
    driver: local
    labels:
      com.example.description: 'PostgreSQL data'

  # Use existing external volume (not managed by Compose)
  external-vol:
    external: true
    name: my-existing-volume
```

## Override Files

Compose merges multiple files. Use this pattern for dev/prod differences:

```
compose.yml              # base configuration (shared)
compose.override.yml     # auto-loaded in dev (not committed)
compose.prod.yml         # production overrides
compose.test.yml         # test overrides
```

**`compose.yml`** — base:

```yaml
services:
  app:
    build: .
    environment:
      NODE_ENV: production
    restart: unless-stopped

  db:
    image: postgres:16-alpine
    volumes:
      - db-data:/var/lib/postgresql/data

volumes:
  db-data:
```

**`compose.override.yml`** — dev additions (auto-loaded):

```yaml
services:
  app:
    build:
      target: development # use dev build stage
    environment:
      NODE_ENV: development
      DEBUG: '*'
    volumes:
      - .:/app # hot-reload: mount source
      - /app/node_modules # preserve container's node_modules
    ports:
      - '3000:3000'
      - '9229:9229' # Node debugger
    command: ['npm', 'run', 'dev']
    restart: 'no'

  db:
    ports:
      - '5432:5432' # expose DB port for dev tools
```

**`compose.prod.yml`** — production:

```yaml
services:
  app:
    image: myregistry/myapp:${TAG:-latest}
    deploy:
      replicas: 3
      resources:
        limits:
          memory: 512M

  db:
    # No ports exposed in production
    environment:
      POSTGRES_PASSWORD_FILE: /run/secrets/db-password
    secrets:
      - db-password

secrets:
  db-password:
    external: true
```

```bash
# Development (uses compose.yml + compose.override.yml automatically)
docker compose up

# Production
docker compose -f compose.yml -f compose.prod.yml up -d

# Testing
docker compose -f compose.yml -f compose.test.yml run --rm tests
```

## Docker Compose CLI Reference

```bash
# Start all services (build if needed)
docker compose up

# Start in background (detached)
docker compose up -d

# Build images before starting
docker compose up --build

# Start specific services
docker compose up app db

# Stop and remove containers, networks
docker compose down

# Also remove volumes (caution: data loss)
docker compose down --volumes

# Also remove images
docker compose down --rmi all

# List running services
docker compose ps

# Show logs
docker compose logs

# Follow logs for all services
docker compose logs -f

# Follow logs for specific service
docker compose logs -f app

# Execute command in running service
docker compose exec app sh
docker compose exec db psql -U postgres

# Run one-off command (creates new container, removed after)
docker compose run --rm app node scripts/migrate.js

# Build images
docker compose build
docker compose build --no-cache app

# Pull latest images
docker compose pull

# Scale a service
docker compose up -d --scale app=3

# Restart services
docker compose restart app

# Show service configuration (merged)
docker compose config

# Show running processes in services
docker compose top
```

## Full-Stack Example

A complete application: Node.js API + PostgreSQL + Redis + Nginx reverse proxy.

```yaml
# compose.yml
name: fullstack-app

services:
  # ── Nginx Reverse Proxy ────────────────────────────────────────
  nginx:
    image: nginx:1.25-alpine
    ports:
      - '80:80'
      - '443:443'
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/conf.d:/etc/nginx/conf.d:ro
      - certbot-certs:/etc/letsencrypt:ro
      - nginx-logs:/var/log/nginx
    depends_on:
      app:
        condition: service_healthy
    networks:
      - frontend
    restart: unless-stopped
    healthcheck:
      test: ['CMD', 'nginx', '-t']
      interval: 30s
      timeout: 5s
      retries: 3

  # ── Application ────────────────────────────────────────────────
  app:
    build:
      context: .
      target: runtime
    environment:
      NODE_ENV: production
      PORT: '3000'
      DB_HOST: db
      DB_PORT: '5432'
      DB_NAME: ${POSTGRES_DB:-appdb}
      DB_USER: ${POSTGRES_USER:-appuser}
      DB_PASSWORD: ${POSTGRES_PASSWORD}
      REDIS_HOST: redis
      REDIS_PORT: '6379'
    env_file:
      - .env
    volumes:
      - app-uploads:/app/uploads
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    networks:
      - frontend
      - backend
    restart: unless-stopped
    healthcheck:
      test: ['CMD', 'wget', '-qO-', 'http://localhost:3000/health']
      interval: 30s
      timeout: 10s
      start_period: 30s
      retries: 3
    logging:
      driver: json-file
      options:
        max-size: '10m'
        max-file: '5'
    deploy:
      resources:
        limits:
          cpus: '1.0'
          memory: 512M

  # ── PostgreSQL ────────────────────────────────────────────────
  db:
    image: postgres:16-alpine
    environment:
      POSTGRES_DB: ${POSTGRES_DB:-appdb}
      POSTGRES_USER: ${POSTGRES_USER:-appuser}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
      PGDATA: /var/lib/postgresql/data/pgdata
    volumes:
      - postgres-data:/var/lib/postgresql/data
      - ./postgres/init:/docker-entrypoint-initdb.d:ro
    networks:
      - backend
    restart: unless-stopped
    healthcheck:
      test:
        [
          'CMD-SHELL',
          'pg_isready -U ${POSTGRES_USER:-appuser} -d ${POSTGRES_DB:-appdb}',
        ]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 30s
    logging:
      driver: json-file
      options:
        max-size: '10m'
        max-file: '3'

  # ── Redis ─────────────────────────────────────────────────────
  redis:
    image: redis:7-alpine
    command: >
      redis-server
      --appendonly yes
      --maxmemory 256mb
      --maxmemory-policy allkeys-lru
    volumes:
      - redis-data:/data
    networks:
      - backend
    restart: unless-stopped
    healthcheck:
      test: ['CMD', 'redis-cli', 'ping']
      interval: 10s
      timeout: 5s
      retries: 5
    logging:
      driver: json-file
      options:
        max-size: '5m'
        max-file: '2'

networks:
  frontend:
    driver: bridge
  backend:
    driver: bridge
    internal: true # no external network access

volumes:
  postgres-data:
  redis-data:
  app-uploads:
  certbot-certs:
  nginx-logs:
```

## Profiles — Optional Services

```yaml
services:
  app:
    image: myapp:latest
    # No profile — always starts

  db:
    image: postgres:16-alpine
    # No profile — always starts

  pgadmin:
    image: dpage/pgadmin4
    profiles:
      - tools # only starts with: docker compose --profile tools up

  mailhog:
    image: mailhog/mailhog
    profiles:
      - tools
      - debug

  debug-proxy:
    image: mitmproxy/mitmproxy
    profiles:
      - debug
```

```bash
# Start base services only
docker compose up -d

# Start with tools profile
docker compose --profile tools up -d

# Start with multiple profiles
docker compose --profile tools --profile debug up -d
```

:::caution[compose.override.yml in Git]
Do not commit `compose.override.yml` if it contains developer-specific settings. Add it to `.gitignore`. Instead, provide a `compose.override.yml.example` that developers copy and customise locally.
:::

## Environment Variable Interpolation

```yaml
# compose.yml — uses shell env vars and .env file
services:
  app:
    image: myapp:${APP_VERSION:-latest} # default to "latest" if not set
    environment:
      DB_URL: postgresql://${DB_USER}:${DB_PASSWORD}@db/${DB_NAME}
```

```bash
# .env file (loaded automatically)
APP_VERSION=2.1.0
DB_USER=myuser
DB_PASSWORD=secret
DB_NAME=mydb
```

Compose loads `.env` from the directory where you run `docker compose`. You can override:

```bash
docker compose --env-file .env.staging up -d
```

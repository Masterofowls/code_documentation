## Job Structure

A job is a collection of steps that run on the same runner. Here is the full job configuration reference:

```yaml
jobs:
  build:
    name: Build Application           # display name
    runs-on: ubuntu-latest            # runner
    needs: [lint, test]               # job dependencies
    if: github.ref == 'refs/heads/main'
    environment: staging              # GitHub Environment (triggers approvals)
    timeout-minutes: 30               # job-level timeout
    continue-on-error: false          # fail the workflow if this job fails
    permissions:
      contents: read
      packages: write
    env:
      BUILD_ENV: production
    outputs:                          # values other jobs can read
      image-tag: ${{ steps.tag.outputs.tag }}
      artifact-name: build-${{ github.run_id }}
    strategy:
      matrix:
        os: [ubuntu-latest, windows-latest]
    steps:
      - name: Checkout
        uses: actions/checkout@v4
```

## Job Outputs

Jobs can pass values to downstream jobs via outputs. The pattern:
1. Write to `$GITHUB_OUTPUT` in a step
2. Declare the output in the job's `outputs:` block
3. Reference it in another job with `needs.JOBID.outputs.KEY`

```yaml
jobs:
  build:
    runs-on: ubuntu-latest
    outputs:
      version: ${{ steps.get-version.outputs.version }}
      image-digest: ${{ steps.build-image.outputs.digest }}
    steps:
      - uses: actions/checkout@v4

      - name: Get version
        id: get-version
        run: |
          VERSION=$(node -p "require('./package.json').version")
          echo "version=$VERSION" >> $GITHUB_OUTPUT

      - name: Build Docker image
        id: build-image
        run: |
          docker build -t myapp:$VERSION .
          DIGEST=$(docker inspect myapp:$VERSION --format='{{.Id}}')
          echo "digest=$DIGEST" >> $GITHUB_OUTPUT

  deploy:
    runs-on: ubuntu-latest
    needs: build
    steps:
      - run: |
          echo "Version: ${{ needs.build.outputs.version }}"
          echo "Digest: ${{ needs.build.outputs.image-digest }}"
```

:::caution[Deprecated: set-output]
The old `echo "::set-output name=key::value"` syntax is deprecated. Always use `echo "key=value" >> $GITHUB_OUTPUT`.
:::

## Service Containers

Service containers spin up Docker containers alongside your job — perfect for databases, caches, or external services your tests depend on.

### PostgreSQL Service

```yaml
jobs:
  integration-test:
    runs-on: ubuntu-latest

    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_USER: testuser
          POSTGRES_PASSWORD: testpass
          POSTGRES_DB: testdb
        ports:
          - 5432:5432
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    env:
      DATABASE_URL: postgres://testuser:testpass@localhost:5432/testdb

    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
      - run: npm ci
      - run: npm run db:migrate
      - run: npm run test:integration
```

### Redis Service

```yaml
services:
  redis:
    image: redis:7-alpine
    ports:
      - 6379:6379
    options: >-
      --health-cmd "redis-cli ping"
      --health-interval 10s
      --health-timeout 5s
      --health-retries 5
```

### Multiple Services

```yaml
services:
  postgres:
    image: postgres:16
    env:
      POSTGRES_PASSWORD: secret
    options: --health-cmd pg_isready --health-interval 10s --health-retries 5
    ports:
      - 5432:5432
  redis:
    image: redis:7
    options: --health-cmd "redis-cli ping" --health-interval 10s --health-retries 5
    ports:
      - 6379:6379
  localstack:
    image: localstack/localstack:3
    env:
      SERVICES: s3,sqs
    ports:
      - 4566:4566
```

:::tip[Service containers in container jobs]
When running a job inside a container (`container:` block), services connect via hostname (the service key), not `localhost`. So `postgres:5432` instead of `localhost:5432`.
:::

## Container Jobs

Run the entire job inside a Docker container:

```yaml
jobs:
  test:
    runs-on: ubuntu-latest
    container:
      image: node:20-alpine
      credentials:
        username: ${{ secrets.DOCKER_USER }}
        password: ${{ secrets.DOCKER_PAT }}
      env:
        NODE_ENV: test
      options: --user 1001
      volumes:
        - /host/path:/container/path

    steps:
      - uses: actions/checkout@v4
      - run: node --version     # uses container's Node.js
      - run: npm ci && npm test
```

## `actions/cache@v4`

Cache dependencies to speed up workflows dramatically.

```yaml
- name: Cache npm dependencies
  uses: actions/cache@v4
  id: npm-cache
  with:
    path: ~/.npm
    key: npm-${{ runner.os }}-${{ hashFiles('**/package-lock.json') }}
    restore-keys: |
      npm-${{ runner.os }}-

- name: Install dependencies
  if: steps.npm-cache.outputs.cache-hit != 'true'
  run: npm ci
```

### Cache Key Strategy

```yaml
# Exact key — matches only if lockfile unchanged
key: ${{ runner.os }}-node20-${{ hashFiles('package-lock.json') }}

# Restore keys — fallback to nearest match if exact miss
restore-keys: |
  ${{ runner.os }}-node20-
  ${{ runner.os }}-
```

### Common Cache Paths

```yaml
# npm
path: ~/.npm
key: npm-${{ runner.os }}-${{ hashFiles('**/package-lock.json') }}

# yarn
path: |
  ~/.yarn/cache
  .yarn/cache
key: yarn-${{ runner.os }}-${{ hashFiles('**/yarn.lock') }}

# pnpm
path: ~/.pnpm-store
key: pnpm-${{ runner.os }}-${{ hashFiles('**/pnpm-lock.yaml') }}

# pip
path: ~/.cache/pip
key: pip-${{ runner.os }}-${{ hashFiles('**/requirements*.txt') }}

# cargo
path: |
  ~/.cargo/registry
  ~/.cargo/git
  target/
key: cargo-${{ runner.os }}-${{ hashFiles('**/Cargo.lock') }}

# Go modules
path: |
  ~/go/pkg/mod
  ~/.cache/go-build
key: go-${{ runner.os }}-${{ hashFiles('**/go.sum') }}
```

## `actions/upload-artifact@v4` and `actions/download-artifact@v4`

Share files between jobs or preserve build outputs:

```yaml
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: npm ci && npm run build

      - name: Upload build artifacts
        uses: actions/upload-artifact@v4
        with:
          name: dist-${{ github.sha }}
          path: |
            dist/
            !dist/**/*.map    # exclude source maps
          retention-days: 7   # how long to keep (max 90)
          if-no-files-found: error  # error | warn | ignore

  deploy:
    runs-on: ubuntu-latest
    needs: build
    steps:
      - name: Download build artifacts
        uses: actions/download-artifact@v4
        with:
          name: dist-${{ github.sha }}
          path: ./dist

      - run: ls -la dist/

  # Download all artifacts from the workflow
  report:
    runs-on: ubuntu-latest
    needs: [build, test]
    steps:
      - uses: actions/download-artifact@v4
        with:
          pattern: '*-report'    # glob pattern
          path: reports/
          merge-multiple: true   # merge into single directory
```

## `actions/github-script`

Run JavaScript with full access to the Octokit GitHub API client:

```yaml
- name: Create PR comment
  uses: actions/github-script@v7
  with:
    script: |
      const { data: comments } = await github.rest.issues.listComments({
        owner: context.repo.owner,
        repo: context.repo.repo,
        issue_number: context.issue.number,
      });

      const botComment = comments.find(c =>
        c.user.login === 'github-actions[bot]' &&
        c.body.includes('<!-- CI-COMMENT -->')
      );

      const body = `<!-- CI-COMMENT -->
      ## ✅ CI Results
      - Tests: passed
      - Coverage: 87%
      - Build: succeeded
      `;

      if (botComment) {
        await github.rest.issues.updateComment({
          owner: context.repo.owner,
          repo: context.repo.repo,
          comment_id: botComment.id,
          body,
        });
      } else {
        await github.rest.issues.createComment({
          owner: context.repo.owner,
          repo: context.repo.repo,
          issue_number: context.issue.number,
          body,
        });
      }
```

## Job Summary (`$GITHUB_STEP_SUMMARY`)

Write rich Markdown content to the workflow run's summary page:

```yaml
- name: Generate test summary
  if: always()
  run: |
    cat >> $GITHUB_STEP_SUMMARY << 'EOF'
    ## 🧪 Test Results

    | Suite | Tests | Passed | Failed | Skipped |
    |-------|-------|--------|--------|---------|
    | Unit  | 142   | 140    | 1      | 1       |
    | E2E   | 38    | 38     | 0      | 0       |

    **Total: 180 tests, 178 passed, 1 failed, 1 skipped**

    ### Failed Tests
    - `UserService > should handle expired tokens` — assertion error on line 47
    EOF

- name: Upload coverage badge data
  run: |
    COVERAGE=$(cat coverage/coverage-summary.json | jq '.total.lines.pct')
    echo "### 📊 Coverage: ${COVERAGE}%" >> $GITHUB_STEP_SUMMARY
```

## Masking Sensitive Values

Prevent values from appearing in logs:

```yaml
- name: Fetch and mask API key
  run: |
    # Mask a dynamically fetched secret
    API_KEY=$(curl -s https://vault.example.com/api-key)
    echo "::add-mask::$API_KEY"
    echo "KEY=$API_KEY" >> $GITHUB_ENV

- name: Use masked value
  run: echo "Key is: $KEY"
  # Output: Key is: ***
```

## Complete Job Reference Example

```yaml
jobs:
  full-example:
    name: Full Job Example
    runs-on: ubuntu-latest
    needs: [lint]
    if: success()
    timeout-minutes: 20
    environment:
      name: staging
      url: ${{ steps.deploy.outputs.url }}
    permissions:
      contents: read
      id-token: write
    env:
      APP_ENV: staging
    outputs:
      deploy-url: ${{ steps.deploy.outputs.url }}

    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_PASSWORD: test
        options: --health-cmd pg_isready --health-interval 10s --health-retries 5
        ports:
          - 5432:5432

    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Cache build
        uses: actions/cache@v4
        with:
          path: .next/cache
          key: nextjs-${{ runner.os }}-${{ hashFiles('**/package-lock.json') }}-${{ hashFiles('**/*.ts', '**/*.tsx') }}
          restore-keys: nextjs-${{ runner.os }}-${{ hashFiles('**/package-lock.json') }}-

      - name: Install
        run: npm ci

      - name: Build
        run: npm run build
        env:
          DATABASE_URL: postgres://postgres:test@localhost:5432/mydb

      - name: Deploy
        id: deploy
        run: |
          URL=$(./scripts/deploy.sh)
          echo "url=$URL" >> $GITHUB_OUTPUT

      - name: Upload artifacts
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: build-output
          path: .next/
          retention-days: 3

      - name: Summary
        if: always()
        run: |
          echo "## Deploy Summary" >> $GITHUB_STEP_SUMMARY
          echo "- URL: ${{ steps.deploy.outputs.url }}" >> $GITHUB_STEP_SUMMARY
          echo "- SHA: ${{ github.sha }}" >> $GITHUB_STEP_SUMMARY
```

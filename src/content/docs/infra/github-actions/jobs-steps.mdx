## Job Structure Reference

A job is the fundamental unit of work in GitHub Actions. Each job runs on its own fresh runner instance.

```yaml
jobs:
  my-job:
    name: Human-readable name # shown in GitHub UI
    runs-on: ubuntu-latest # runner label
    needs: [previous-job] # dependencies
    if: github.ref == 'refs/heads/main' # conditional execution
    timeout-minutes: 30 # max runtime
    continue-on-error: false # whether to fail the workflow on error
    environment: # deployment environment
      name: production
      url: https://app.example.com
    concurrency:
      group: deploy-prod
      cancel-in-progress: false
    outputs: # values to share with downstream jobs
      artifact-id: ${{ steps.upload.outputs.artifact-id }}
      version: ${{ steps.version.outputs.value }}
    env: # job-level environment variables
      NODE_ENV: production
    strategy: # matrix configuration
      matrix:
        os: [ubuntu-latest, macos-latest]
    steps: []
```

## Job Outputs

Job outputs allow passing data between jobs. Set them via `$GITHUB_OUTPUT` in a step, then declare them in the job's `outputs` block.

```yaml
jobs:
  build:
    runs-on: ubuntu-latest
    outputs:
      version: ${{ steps.get-version.outputs.version }}
      image-digest: ${{ steps.build.outputs.digest }}
    steps:
      - uses: actions/checkout@v4

      - name: Get version from package.json
        id: get-version
        run: |
          VERSION=$(node -p "require('./package.json').version")
          echo "version=$VERSION" >> $GITHUB_OUTPUT

      - name: Build Docker image
        id: build
        run: |
          docker build -t myapp:latest .
          DIGEST=$(docker inspect --format='{{index .RepoDigests 0}}' myapp:latest)
          echo "digest=$DIGEST" >> $GITHUB_OUTPUT

  deploy:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - name: Use version from build job
        run: |
          echo "Deploying version: ${{ needs.build.outputs.version }}"
          echo "Image digest: ${{ needs.build.outputs.image-digest }}"
```

:::caution[Deprecated set-output]
The old `echo "::set-output name=key::value"` syntax is **deprecated and disabled**. Always use `echo "key=value" >> $GITHUB_OUTPUT`. The new `$GITHUB_OUTPUT` file approach is the only supported method.
:::

## Service Containers

Service containers spin up Docker images (databases, caches, etc.) alongside your job. They're available on the job's virtual network.

### PostgreSQL Service

```yaml
jobs:
  integration-test:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_USER: testuser
          POSTGRES_PASSWORD: testpass
          POSTGRES_DB: testdb
        ports:
          - 5432:5432 # map container port to host
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    steps:
      - uses: actions/checkout@v4

      - name: Run migrations
        run: npm run migrate
        env:
          DATABASE_URL: postgres://testuser:testpass@localhost:5432/testdb

      - name: Run integration tests
        run: npm run test:integration
        env:
          DATABASE_URL: postgres://testuser:testpass@localhost:5432/testdb
```

### Redis Service

```yaml
services:
  redis:
```

```yaml
services:
  postgres:

      MINIO_ROOT_PASSWORD: minioadmin

      - 9000:9000
```

## Container Jobs

Run the entire job inside a Docker container instead of directly on the runner VM:

```yaml
jobs:
  build:
    runs-on: ubuntu-latest
    container:
      image: node:20-alpine
      env:
        CI: true
      options: --cpus 2 --memory 4g
      volumes:
        - /tmp:/tmp

    steps:
      - uses: actions/checkout@v4
      - run: node --version # uses the container's node, not the runner's
      - run: npm ci && npm test
```

## Caching with `actions/cache@v4`

### Node.js (npm)

```yaml
- name: Cache npm packages
  uses: actions/cache@v4
  id: npm-cache
  with:
    path: ~/.npm
    key: ${{ runner.os }}-npm-${{ hashFiles('**/package-lock.json') }}
    restore-keys: |
      ${{ runner.os }}-npm-

- name: Install dependencies
  # Skip install if exact cache was restored
  if: steps.npm-cache.outputs.cache-hit != 'true'
  run: npm ci
```

### Multiple Cache Paths

```yaml
- uses: actions/cache@v4
  with:
    path: |
      ~/.npm
      node_modules
      .next/cache
    key: ${{ runner.os }}-nextjs-${{ hashFiles('package-lock.json') }}-${{ hashFiles('**/*.ts', '**/*.tsx') }}
    restore-keys: |
      ${{ runner.os }}-nextjs-${{ hashFiles('package-lock.json') }}-
      ${{ runner.os }}-nextjs-
```

### Python (pip)

```yaml
- uses: actions/cache@v4
  with:
    path: ~/.cache/pip
    key: ${{ runner.os }}-pip-${{ hashFiles('requirements*.txt') }}
    restore-keys: ${{ runner.os }}-pip-
```

### Cargo (Rust)

```yaml
- uses: actions/cache@v4
  with:
    path: |
      ~/.cargo/bin/
      ~/.cargo/registry/index/
      ~/.cargo/registry/cache/
      ~/.cargo/git/db/
      target/
    key: ${{ runner.os }}-cargo-${{ hashFiles('Cargo.lock') }}
```

:::tip[Cache Invalidation Strategy]
Build your cache key from the content hash of your lockfile (`hashFiles`). When dependencies change, the key changes and the cache is rebuilt. The `restore-keys` fallback lets you still restore a partial cache to speed up installs.
:::

## Artifacts

### Uploading Artifacts

```yaml
- name: Upload test results
  uses: actions/upload-artifact@v4
  if: always() # upload even if tests fail
  with:
    name: test-results-${{ matrix.os }}
    path: |
      coverage/
      test-results/
      playwright-report/
    retention-days: 30 # default: 90 days
    compression-level: 6 # 0=none, 9=max (default: 6)
    overwrite: false # fail if artifact name already exists

- name: Upload build artifact
  uses: actions/upload-artifact@v4
  with:
    name: dist-${{ github.sha }}
    path: dist/
    if-no-files-found: error # error, warn, or ignore
```

### Downloading Artifacts

```yaml
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - run: npm run build
      - uses: actions/upload-artifact@v4
        with:
          name: dist
          path: dist/

  deploy:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - name: Download build artifact
        uses: actions/download-artifact@v4
        with:
          name: dist
          path: ./dist

      - name: Download all artifacts
        uses: actions/download-artifact@v4
        # Omit 'name' to download all artifacts
        with:
          path: ./artifacts
          # Each artifact goes into artifacts/<artifact-name>/

      - run: ls -la ./dist
```

## `actions/github-script`

Run JavaScript with access to the Octokit API client:

```yaml
- uses: actions/github-script@v7
  with:
    script: |
      // Create a PR comment
      const { data: comment } = await github.rest.issues.createComment({
        issue_number: context.issue.number,
        owner: context.repo.owner,
        repo: context.repo.repo,
        body: `## 🚀 Deploy Preview\n\nPreview URL: https://preview-${context.sha.substring(0,7)}.example.com`
      });
      console.log(`Created comment ${comment.id}`);

      // Get PR files
      const { data: files } = await github.rest.pulls.listFiles({
        owner: context.repo.owner,
        repo: context.repo.repo,
        pull_number: context.issue.number
      });
      console.log(`PR changes ${files.length} files`);
```

Output from github-script:

```yaml
- id: script-output
  uses: actions/github-script@v7
  with:
    result-encoding: string
    script: return 'hello from script'

- run: echo "Output was: ${{ steps.script-output.outputs.result }}"
```

## Job Summary (`$GITHUB_STEP_SUMMARY`)

Write rich Markdown to the job summary page (visible in the Actions run UI):

```yaml
- name: Generate test summary
  if: always()
  run: |
    cat >> $GITHUB_STEP_SUMMARY << 'EOF'
    ## 🧪 Test Results

    | Suite | Tests | Passed | Failed | Skipped |
    |-------|-------|--------|--------|---------|
    | Unit  | 142   | 140    | 2      | 0       |
    | E2E   | 38    | 38     | 0      | 0       |

    ### Failed Tests
    - `AuthService > should reject invalid tokens`
    - `UserController > should return 404 for unknown user`
    EOF

- name: Summary with dynamic data
  run: |
    PASSED=$(cat test-results.json | jq '.numPassedTests')
    FAILED=$(cat test-results.json | jq '.numFailedTests')
    echo "## Test Results: $PASSED passed, $FAILED failed" >> $GITHUB_STEP_SUMMARY
    if [ "$FAILED" -gt "0" ]; then
      echo "❌ Some tests failed" >> $GITHUB_STEP_SUMMARY
    else
      echo "✅ All tests passed" >> $GITHUB_STEP_SUMMARY
    fi
```

## Secret Masking

Mask sensitive values so they never appear in logs:

```yaml
- name: Mask a computed secret
  run: |
    GENERATED_TOKEN=$(./scripts/generate-token.sh)
    echo "::add-mask::$GENERATED_TOKEN"
    echo "TOKEN=$GENERATED_TOKEN" >> $GITHUB_ENV

- name: Use masked token
  run: |
    # $TOKEN appears as *** in logs
    curl -H "Authorization: Bearer $TOKEN" https://api.example.com/data
```

## Persisting Environment Variables Across Steps

```yaml
- name: Set variables for downstream steps
  run: |
    echo "DEPLOY_VERSION=1.2.3" >> $GITHUB_ENV
    echo "DEPLOY_TIMESTAMP=$(date -u +%Y%m%d%H%M%S)" >> $GITHUB_ENV

- name: Add binary to PATH
  run: |
    echo "$HOME/.local/bin" >> $GITHUB_PATH
    # ~/.local/bin is now available to all subsequent steps

- name: Use the variables
  run: |
    echo "Version: $DEPLOY_VERSION"
    echo "Timestamp: $DEPLOY_TIMESTAMP"
    my-binary --version    # available because we added to PATH
```

## Complete Job with All Features

```yaml
jobs:
  full-example:
    runs-on: ubuntu-latest
    timeout-minutes: 30
    outputs:
      artifact-id: ${{ steps.upload.outputs.artifact-id }}
      version: ${{ steps.version.outputs.value }}

    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_PASSWORD: secret
        ports:
          - 5432:5432
        options: --health-cmd pg_isready --health-interval 10s --health-retries 5

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm

      - uses: actions/cache@v4
        with:
          path: .next/cache
          key: nextjs-${{ hashFiles('package-lock.json') }}

      - run: npm ci

      - id: version
        run: echo "value=$(node -p "require('./package.json').version")" >> $GITHUB_OUTPUT

      - name: Run tests against real DB
        run: npm run test:integration
        env:
          DATABASE_URL: postgres://postgres:secret@localhost:5432/postgres

      - name: Build
        run: npm run build

      - id: upload
        uses: actions/upload-artifact@v4
        with:
          name: build-${{ github.sha }}
          path: .next/

      - name: Write job summary
        if: always()
        run: |
          echo "## Build Summary" >> $GITHUB_STEP_SUMMARY
          echo "- Version: ${{ steps.version.outputs.value }}" >> $GITHUB_STEP_SUMMARY
          echo "- Commit: ${{ github.sha }}" >> $GITHUB_STEP_SUMMARY
          echo "- Artifact: ${{ steps.upload.outputs.artifact-id }}" >> $GITHUB_STEP_SUMMARY
```

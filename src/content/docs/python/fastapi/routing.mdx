Routing in FastAPI is built around **path operation decorators** that map HTTP methods and URL paths to Python functions. This guide covers everything from basic GET endpoints to advanced streaming responses.

## HTTP Path Operations

FastAPI provides a decorator for every standard HTTP method:

```python
from fastapi import FastAPI

app = FastAPI()


@app.get("/items")         # Retrieve a collection
@app.post("/items")        # Create a new resource
@app.put("/items/{id}")    # Full update (replace)
@app.patch("/items/{id}")  # Partial update
@app.delete("/items/{id}") # Remove a resource
@app.head("/items")        # Like GET, but no response body
@app.options("/items")     # Describe communication options
```

:::tip[Semantic HTTP Methods]
Use the correct verb for the action: `GET` for reads (safe, idempotent), `POST` for creates (neither), `PUT` for full replacements (idempotent), `PATCH` for partial updates, and `DELETE` for removals (idempotent). The correct method improves caching behavior and API usability.
:::

## Path Parameters

Declare dynamic segments in the path with curly braces. The type annotation defines validation:

```python
from fastapi import FastAPI
from enum import Enum

app = FastAPI()


class ModelName(str, Enum):
    alexnet = "alexnet"
    resnet = "resnet"
    lenet = "lenet"


@app.get("/users/{user_id}")
async def get_user(user_id: int):
    """user_id is coerced to int — non-integers return 422."""
    return {"user_id": user_id}


@app.get("/files/{file_path:path}")
async def get_file(file_path: str):
    """The :path converter captures slashes, e.g. /files/docs/report.pdf"""
    return {"file_path": file_path}


@app.get("/models/{model_name}")
async def get_model(model_name: ModelName):
    """Enum parameter — only 'alexnet', 'resnet', 'lenet' are valid."""
    return {"model": model_name, "message": f"Loaded {model_name.value}"}
```

Test path parameter validation:

```bash
curl http://localhost:8000/users/42
```

```json
{ "user_id": 42 }
```

```bash
curl http://localhost:8000/models/resnet
```

```json
{ "model": "resnet", "message": "Loaded resnet" }
```

```bash
curl http://localhost:8000/models/unknown
```

```json
{
  "detail": [
    {
      "type": "enum",
      "loc": ["path", "model_name"],
      "msg": "Input should be 'alexnet', 'resnet' or 'lenet'",
      "input": "unknown"
    }
  ]
}
```

:::caution[Route Declaration Order]
FastAPI matches routes in declaration order. Declare fixed paths **before** parameterized ones to avoid conflicts:

```python
@app.get("/users/me")        # Must come FIRST
@app.get("/users/{user_id}") # Otherwise "me" is parsed as user_id
```

:::

## Query Parameters

Any function parameter not in the path is treated as a query parameter:

```python
from fastapi import FastAPI, Query

app = FastAPI()

ITEMS_DB = [{"name": "Widget"}, {"name": "Gadget"}, {"name": "Doohickey"}]


@app.get("/items")
async def list_items(
    skip: int = 0,
    limit: int = 10,
    search: str | None = None,
    sort_by: str = "name",
    descending: bool = False,
):
    """
    List items with pagination, optional search, and sorting.
    - skip: number of items to skip (default 0)
    - limit: max items to return (default 10)
    - search: optional filter by name substring
    - sort_by: field to sort by (default 'name')
    - descending: reverse sort order (default false)
    """
    results = ITEMS_DB
    if search:
        results = [i for i in results if search.lower() in i["name"].lower()]
    return {
        "total": len(results),
        "skip": skip,
        "limit": limit,
        "items": results[skip : skip + limit],
    }
```

```bash
curl "http://localhost:8000/items?skip=0&limit=2&search=get"
```

```json
{ "total": 1, "skip": 0, "limit": 2, "items": [{ "name": "Gadget" }] }
```

### Query Parameter Validation with `Query()`

```python
from fastapi import FastAPI, Query

app = FastAPI()


@app.get("/search")
async def search_items(
    q: str = Query(
        min_length=3,
        max_length=50,
        pattern=r"^[a-zA-Z0-9 ]+$",
        description="Search query — alphanumeric only",
        examples=["blue widget"],
    ),
    tags: list[str] = Query(default=[], description="Filter by one or more tags"),
    page: int = Query(default=1, ge=1, description="Page number (1-indexed)"),
    page_size: int = Query(default=20, ge=1, le=100, description="Items per page"),
):
    return {"q": q, "tags": tags, "page": page, "page_size": page_size}
```

```bash
curl "http://localhost:8000/search?q=blue+widget&tags=sale&tags=new&page=2"
```

```json
{
  "q": "blue widget",
  "tags": ["sale", "new"],
  "page": 2,
  "page_size": 20
}
```

## APIRouter — Organizing Routes into Modules

Split routes across multiple files using `APIRouter`:

```python
# app/routers/items.py
from fastapi import APIRouter, HTTPException, status
from app.schemas.item import ItemCreate, ItemRead, ItemUpdate

router = APIRouter(
    prefix="/items",
    tags=["items"],
    responses={
        404: {"description": "Item not found"},
        401: {"description": "Not authenticated"},
    },
)

FAKE_DB: dict[int, dict] = {}
_counter = 0


@router.get("/", response_model=list[ItemRead], summary="List all items")
async def list_items(skip: int = 0, limit: int = 20):
    return list(FAKE_DB.values())[skip : skip + limit]


@router.post("/", response_model=ItemRead, status_code=status.HTTP_201_CREATED)
async def create_item(item: ItemCreate):
    global _counter
    _counter += 1
    record = {"id": _counter, **item.model_dump()}
    FAKE_DB[_counter] = record
    return record


@router.get("/{item_id}", response_model=ItemRead)
async def get_item(item_id: int):
    if item_id not in FAKE_DB:
        raise HTTPException(status_code=404, detail=f"Item {item_id} not found")
    return FAKE_DB[item_id]


@router.patch("/{item_id}", response_model=ItemRead)
async def update_item(item_id: int, item: ItemUpdate):
    if item_id not in FAKE_DB:
        raise HTTPException(status_code=404, detail=f"Item {item_id} not found")
    FAKE_DB[item_id].update(item.model_dump(exclude_unset=True))
    return FAKE_DB[item_id]


@router.delete("/{item_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete_item(item_id: int):
    if item_id not in FAKE_DB:
        raise HTTPException(status_code=404, detail=f"Item {item_id} not found")
    del FAKE_DB[item_id]
```

Register the router in `main.py`:

```python
# main.py
from fastapi import FastAPI
from app.routers import items, users, auth

app = FastAPI()

app.include_router(auth.router, prefix="/auth", tags=["auth"])
app.include_router(users.router, prefix="/users", tags=["users"])
app.include_router(items.router)  # prefix already set in router
```

## HTTP Status Codes

Use `status_code` on the decorator and FastAPI's `status` constants for clarity:

```python
from fastapi import FastAPI, status

app = FastAPI()


@app.post("/items/", status_code=status.HTTP_201_CREATED)
async def create_item():
    return {"created": True}


@app.delete("/items/{item_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete_item(item_id: int):
    # 204 responses have no body — return None
    return None


@app.post("/process/", status_code=status.HTTP_202_ACCEPTED)
async def start_processing():
    return {"task_id": "abc123", "status": "queued"}
```

Common status codes in FastAPI:

| Code | Constant                         | Use Case                               |
| ---- | -------------------------------- | -------------------------------------- |
| 200  | `HTTP_200_OK`                    | Successful GET, PUT, PATCH             |
| 201  | `HTTP_201_CREATED`               | Successful POST (resource created)     |
| 202  | `HTTP_202_ACCEPTED`              | Async task accepted (not yet complete) |
| 204  | `HTTP_204_NO_CONTENT`            | Successful DELETE                      |
| 400  | `HTTP_400_BAD_REQUEST`           | Client sent malformed data             |
| 401  | `HTTP_401_UNAUTHORIZED`          | Authentication required                |
| 403  | `HTTP_403_FORBIDDEN`             | Authenticated but not authorized       |
| 404  | `HTTP_404_NOT_FOUND`             | Resource does not exist                |
| 409  | `HTTP_409_CONFLICT`              | Conflict with current state            |
| 422  | `HTTP_422_UNPROCESSABLE_ENTITY`  | Validation failure (auto by FastAPI)   |
| 429  | `HTTP_429_TOO_MANY_REQUESTS`     | Rate limit exceeded                    |
| 500  | `HTTP_500_INTERNAL_SERVER_ERROR` | Server error                           |

## Response Models

Use `response_model` to control the shape of the JSON response and exclude sensitive fields:

```python
from fastapi import FastAPI
from pydantic import BaseModel, EmailStr

app = FastAPI()


class UserCreate(BaseModel):
    username: str
    email: EmailStr
    password: str          # Sensitive — must not be returned


class UserRead(BaseModel):
    id: int
    username: str
    email: EmailStr
    is_active: bool = True
    # No 'password' field — it is stripped from the response


@app.post("/users/", response_model=UserRead, status_code=201)
async def create_user(user: UserCreate):
    # In reality: hash password, save to DB
    saved = {"id": 1, "username": user.username, "email": user.email, "is_active": True}
    return saved  # 'password' is silently excluded by response_model
```

```bash
curl -X POST http://localhost:8000/users/ \
  -H "Content-Type: application/json" \
  -d '{"username": "alice", "email": "alice@example.com", "password": "secret123"}'
```

```json
{
  "id": 1,
  "username": "alice",
  "email": "alice@example.com",
  "is_active": true
}
```

### `response_model_exclude_unset`

Return only fields that were explicitly set, omitting defaults:

```python
@app.patch("/users/{user_id}", response_model=UserRead, response_model_exclude_unset=True)
async def patch_user(user_id: int, user: UserUpdate):
    # Only returns fields that changed — useful for PATCH responses
    ...
```

## Background Tasks

Run work after the response is sent without making the client wait:

```python
from fastapi import FastAPI, BackgroundTasks
import logging

app = FastAPI()
logger = logging.getLogger(__name__)


def send_welcome_email(email: str, username: str) -> None:
    """Runs in the background after the response is returned."""
    logger.info("Sending welcome email to %s (%s)", email, username)
    # Call your email service here (smtplib, sendgrid, etc.)


def log_analytics_event(event: str, user_id: int) -> None:
    logger.info("Analytics: %s for user %d", event, user_id)


@app.post("/users/", status_code=201)
async def create_user(background_tasks: BackgroundTasks, username: str, email: str):
    user = {"id": 99, "username": username, "email": email}

    # These run after the response is sent to the client
    background_tasks.add_task(send_welcome_email, email=email, username=username)
    background_tasks.add_task(log_analytics_event, "user_registered", user["id"])

    return user  # Client receives this immediately
```

```bash
curl -X POST "http://localhost:8000/users/?username=bob&email=bob@example.com"
```

```json
{ "id": 99, "username": "bob", "email": "bob@example.com" }
```

:::caution[BackgroundTasks Limitations]
`BackgroundTasks` runs in the same process as the web server — on the same event loop. It is suitable for lightweight, fast tasks (emails, audit logs). For **CPU-intensive** or **long-running** tasks (video encoding, ML inference, bulk data processing), use a dedicated task queue like **Celery** with a Redis or RabbitMQ broker.
:::

## Streaming Responses

Stream large responses without buffering the entire payload in memory:

```python
import asyncio
import json
from fastapi import FastAPI
from fastapi.responses import StreamingResponse

app = FastAPI()


async def generate_large_dataset():
    """Generator that yields data chunks — nothing is buffered in memory."""
    yield b'{"data": ['
    for i in range(1, 10_001):
        item = json.dumps({"id": i, "value": i * 2})
        yield item.encode()
        if i < 10_000:
            yield b","
        await asyncio.sleep(0)  # Yield control back to the event loop
    yield b"]}"


@app.get("/stream/data")
async def stream_data():
    return StreamingResponse(
        generate_large_dataset(),
        media_type="application/json",
        headers={"X-Total-Count": "10000"},
    )


async def generate_events():
    """Server-Sent Events stream."""
    for i in range(10):
        yield f"data: {json.dumps({'event': 'update', 'count': i})}\n\n"
        await asyncio.sleep(1)
    yield "data: {\"event\": \"done\"}\n\n"


@app.get("/events")
async def sse_stream():
    return StreamingResponse(generate_events(), media_type="text/event-stream")
```

## File Responses

Serve files directly from disk:

```python
from pathlib import Path
from fastapi import FastAPI, HTTPException
from fastapi.responses import FileResponse

app = FastAPI()
UPLOAD_DIR = Path("./uploads")


@app.get("/download/{filename}")
async def download_file(filename: str):
    file_path = UPLOAD_DIR / filename

    # Sanitize: prevent path traversal attacks
    if not file_path.resolve().is_relative_to(UPLOAD_DIR.resolve()):
        raise HTTPException(status_code=400, detail="Invalid filename")

    if not file_path.exists():
        raise HTTPException(status_code=404, detail="File not found")

    return FileResponse(
        path=file_path,
        filename=filename,                    # Sets Content-Disposition header
        media_type="application/octet-stream",
    )
```

## Router Dependencies

Apply dependencies to every route in a router:

```python
from fastapi import APIRouter, Depends
from app.dependencies.auth import get_current_active_user

# All routes in this router require authentication
router = APIRouter(
    prefix="/admin",
    tags=["admin"],
    dependencies=[Depends(get_current_active_user)],
)

@router.get("/stats")
async def get_stats():
    return {"active_users": 1234, "total_requests": 99999}
```

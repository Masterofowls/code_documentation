Layouts are the backbone of any PyQt6 UI. They automatically position and resize child widgets when the window is resized — no manual coordinate math required.

---

## Why Use Layouts?

Without layouts you'd have to call `widget.setGeometry(x, y, w, h)` on every widget and recalculate everything when the window resizes. Layouts handle this for you:

- Widgets stay proportionally sized as the window grows.
- Text remains readable regardless of the system DPI.
- Your UI adapts to translated strings that may be longer or shorter.
- Internationalisation, HiDPI, and accessibility just work.

:::tip[Always use layouts]
Never hard-code `move()` or `setGeometry()` for normal widgets. Reserve absolute positioning only for custom painting or game-style overlays.
:::

## QVBoxLayout — Vertical Stack

Arranges widgets in a single column from top to bottom.

```python
from PyQt6.QtWidgets import QWidget, QVBoxLayout, QPushButton, QLabel

container = QWidget()
layout = QVBoxLayout(container)

layout.addWidget(QLabel("Header"))
layout.addWidget(QPushButton("Action A"))
layout.addWidget(QPushButton("Action B"))
layout.addStretch()              # flexible space — pushes remaining widgets up
layout.addWidget(QPushButton("Footer Button"))
```

### Spacing & Margins

```python
layout.setSpacing(8)             # pixels between widgets (default: platform-dependent)
layout.setContentsMargins(16, 16, 16, 16)   # left, top, right, bottom padding

# Remove all padding for edge-to-edge layouts
layout.setContentsMargins(0, 0, 0, 0)
layout.setSpacing(0)
```

### Stretch Factors

```python
layout.addWidget(header_widget, stretch=0)   # fixed size
layout.addWidget(content_widget, stretch=1)  # takes remaining space
layout.addWidget(footer_widget, stretch=0)   # fixed size
```

## QHBoxLayout — Horizontal Row

Same API as `QVBoxLayout`, but arranges widgets left to right.

```python
from PyQt6.QtWidgets import QHBoxLayout

row = QHBoxLayout()
row.addWidget(QLabel("Label:"))
row.addWidget(QLineEdit())
row.addWidget(QPushButton("OK"))

# Typical button bar at the bottom of a dialog
button_bar = QHBoxLayout()
button_bar.addStretch()           # push buttons to the right
button_bar.addWidget(QPushButton("Cancel"))
button_bar.addWidget(QPushButton("Apply"))
button_bar.addWidget(QPushButton("OK"))
```

## QGridLayout — 2-D Grid

Places widgets in rows and columns. A widget can span multiple cells.

```python
from PyQt6.QtWidgets import QGridLayout

grid = QGridLayout()

# addWidget(widget, row, col, rowSpan=1, colSpan=1)
grid.addWidget(QLabel("First Name:"),  0, 0)
grid.addWidget(QLineEdit(),            0, 1)
grid.addWidget(QLabel("Last Name:"),   1, 0)
grid.addWidget(QLineEdit(),            1, 1)
grid.addWidget(QLabel("Bio:"),         2, 0)
grid.addWidget(QTextEdit(),            2, 1, 3, 1)   # spans 3 rows

# Column stretch — second column gets all extra width
grid.setColumnStretch(0, 0)
grid.setColumnStretch(1, 1)

# Row stretch
grid.setRowStretch(2, 1)   # bio row expands vertically

# Cell spacing
grid.setHorizontalSpacing(12)
grid.setVerticalSpacing(8)
```

### Calculator-style Grid

```python
buttons = [
    ("7", 0, 0), ("8", 0, 1), ("9", 0, 2), ("÷", 0, 3),
    ("4", 1, 0), ("5", 1, 1), ("6", 1, 2), ("×", 1, 3),
    ("1", 2, 0), ("2", 2, 1), ("3", 2, 2), ("−", 2, 3),
    ("0", 3, 0), (".", 3, 1), ("=", 3, 2), ("+", 3, 3),
]

grid = QGridLayout()
for label, row, col in buttons:
    btn = QPushButton(label)
    btn.setFixedSize(52, 52)
    grid.addWidget(btn, row, col)
```

## QFormLayout — Label–Field Pairs

Designed for data-entry forms. It lines up labels with their corresponding input fields.

```python
from PyQt6.QtWidgets import QFormLayout, QLineEdit, QSpinBox, QCheckBox

form = QFormLayout()
form.setRowWrapPolicy(QFormLayout.RowWrapPolicy.WrapLongRows)
form.setLabelAlignment(Qt.AlignmentFlag.AlignRight)

name_edit  = QLineEdit()
age_spin   = QSpinBox()
email_edit = QLineEdit()
active_cb  = QCheckBox("Active")

form.addRow("Full Name:",    name_edit)
form.addRow("Age:",          age_spin)
form.addRow("Email:",        email_edit)
form.addRow("",              active_cb)   # no label for checkboxes

# Access rows programmatically
label_item, field_item = form.itemAt(0, QFormLayout.ItemRole.LabelRole), \
                         form.itemAt(0, QFormLayout.ItemRole.FieldRole)
```

## QStackedLayout — Page Switcher

Holds multiple "pages" of widgets but shows only one at a time. Perfect for multi-step wizards, settings panels, and tabbed-like views.

```python
from PyQt6.QtWidgets import QStackedLayout, QWidget, QPushButton, QLabel

stack = QStackedLayout()

page1 = QWidget()
p1_layout = QVBoxLayout(page1)
p1_layout.addWidget(QLabel("Page 1: Welcome"))

page2 = QWidget()
p2_layout = QVBoxLayout(page2)
p2_layout.addWidget(QLabel("Page 2: Settings"))

page3 = QWidget()
p3_layout = QVBoxLayout(page3)
p3_layout.addWidget(QLabel("Page 3: Finish"))

stack.addWidget(page1)   # index 0
stack.addWidget(page2)   # index 1
stack.addWidget(page3)   # index 2

stack.setCurrentIndex(0)   # show page 1

# Navigate
next_btn = QPushButton("Next →")
next_btn.clicked.connect(lambda: stack.setCurrentIndex(stack.currentIndex() + 1))

# Signal fires when page changes
stack.currentChanged.connect(lambda idx: print("Now on page:", idx))
```

## Nesting Layouts

Layouts can be nested arbitrarily. The pattern is always:

1. Create a parent layout.
2. Create a child layout.
3. Call `parent.addLayout(child)`.

```python
from PyQt6.QtWidgets import (
    QWidget, QVBoxLayout, QHBoxLayout, QGroupBox,
    QLabel, QLineEdit, QPushButton, QTextEdit,
)

container = QWidget()
root = QVBoxLayout(container)

# ── Header row ────────────────────────────────────────────────────
header_row = QHBoxLayout()
header_row.addWidget(QLabel("Document Editor"))
header_row.addStretch()
header_row.addWidget(QPushButton("New"))
header_row.addWidget(QPushButton("Open"))
header_row.addWidget(QPushButton("Save"))
root.addLayout(header_row)

# ── Form group ────────────────────────────────────────────────────
form_group = QGroupBox("Document Info")
form_layout = QFormLayout(form_group)
form_layout.addRow("Title:", QLineEdit())
form_layout.addRow("Author:", QLineEdit())
root.addWidget(form_group)

# ── Editor ────────────────────────────────────────────────────────
root.addWidget(QTextEdit(), stretch=1)

# ── Button bar ────────────────────────────────────────────────────
btn_row = QHBoxLayout()
btn_row.addStretch()
btn_row.addWidget(QPushButton("Cancel"))
btn_row.addWidget(QPushButton("Submit"))
root.addLayout(btn_row)
```

## QSplitter — Resizable Panes

`QSplitter` divides its children with a draggable handle so the user can resize each pane at runtime.

```python
from PyQt6.QtWidgets import QSplitter, QTextEdit, QListWidget
from PyQt6.QtCore import Qt

splitter = QSplitter(Qt.Orientation.Horizontal)

left_pane  = QListWidget()
right_pane = QTextEdit()

splitter.addWidget(left_pane)
splitter.addWidget(right_pane)

# Initial proportions (list of pixel sizes, relative to each other)
splitter.setSizes([200, 600])

# Prevent a pane from collapsing to zero width
splitter.setCollapsible(0, False)
splitter.setCollapsible(1, False)

# Vertical split
vsplitter = QSplitter(Qt.Orientation.Vertical)
vsplitter.addWidget(QTextEdit())
vsplitter.addWidget(QTextEdit())

# Nested splitters (3-pane IDE-like layout)
main_splitter = QSplitter(Qt.Orientation.Horizontal)
main_splitter.addWidget(QListWidget())       # file tree

editor_splitter = QSplitter(Qt.Orientation.Vertical)
editor_splitter.addWidget(QTextEdit())       # code editor
editor_splitter.addWidget(QTextEdit())       # terminal / output
main_splitter.addWidget(editor_splitter)

# Save/restore state
state = splitter.saveState()
splitter.restoreState(state)
```

## QScrollArea — Scrollable Containers

When content may exceed the visible area, wrap it in a `QScrollArea`.

```python
from PyQt6.QtWidgets import QScrollArea, QWidget, QVBoxLayout, QLabel
from PyQt6.QtCore import Qt

scroll = QScrollArea()
scroll.setWidgetResizable(True)          # content widget resizes with scroll area
scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)

# The widget that holds all the content
content = QWidget()
content_layout = QVBoxLayout(content)
for i in range(50):
    content_layout.addWidget(QLabel(f"Row {i + 1}: some content here"))
content_layout.addStretch()

scroll.setWidget(content)

# Programmatically scroll
scroll.verticalScrollBar().setValue(0)               # top
scroll.ensureWidgetVisible(some_widget)              # scroll to a specific child
```

:::caution[setWidgetResizable]
Always call `scroll.setWidgetResizable(True)` unless you explicitly want the content widget to maintain a fixed size. Without it, the content widget won't expand to fill the scroll area.
:::

## QSizePolicy — How Widgets Resize

`QSizePolicy` tells the layout manager how a widget wants to be sized.

```python
from PyQt6.QtWidgets import QSizePolicy

# Common policies:
# Fixed          — exactly sizeHint(), never grows or shrinks
# Minimum        — can grow, not shrink below sizeHint()
# Maximum        — can shrink, not grow above sizeHint()
# Preferred      — sizeHint() is preferred, but can grow/shrink
# Expanding      — take as much space as possible
# MinimumExpanding — at least sizeHint(), but prefers more
# Ignored        — sizeHint() is ignored, takes all available space

widget.setSizePolicy(
    QSizePolicy.Policy.Expanding,   # horizontal
    QSizePolicy.Policy.Fixed,       # vertical
)

# Horizontal stretch factor (relative to other widgets in same layout)
widget.setSizePolicy(
    QSizePolicy.Policy.Expanding,
    QSizePolicy.Policy.Preferred,
)
sp = widget.sizePolicy()
sp.setHorizontalStretch(2)         # this widget gets 2× more width than stretch=1 widgets
widget.setSizePolicy(sp)
```

## Layout Tips & Patterns

### Centering a Widget

```python
outer = QHBoxLayout()
outer.addStretch()
outer.addWidget(centered_widget)
outer.addStretch()
```

### Equal-Width Buttons

```python
button_row = QHBoxLayout()
for label in ["Cancel", "Apply", "OK"]:
    btn = QPushButton(label)
    btn.setSizePolicy(QSizePolicy.Policy.Expanding, QSizePolicy.Policy.Fixed)
    button_row.addWidget(btn)
```

### Preventing a Layout from Stretching

```python
layout.setSizeConstraint(QLayout.SizeConstraint.SetFixedSize)
```

### Removing a Widget from a Layout

```python
layout.removeWidget(widget)
widget.setParent(None)   # also removes from scene and hides it
```

### Dynamically Adding/Removing Rows

```python
# Add
new_label = QLabel("Dynamic row")
layout.insertWidget(0, new_label)   # insert at position 0 (top)

# Remove
layout.removeWidget(existing_widget)
existing_widget.deleteLater()        # schedule Qt to clean up memory
```

## Full Example: Settings Panel with Multiple Layouts

```python
import sys
from PyQt6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout, QHBoxLayout,
    QFormLayout, QGroupBox, QLabel, QLineEdit, QSpinBox, QCheckBox,
    QPushButton, QComboBox, QSplitter, QListWidget, QStackedLayout,
)
from PyQt6.QtCore import Qt


class SettingsWindow(QMainWindow):
    def __init__(self) -> None:
        super().__init__()
        self.setWindowTitle("Settings")
        self.resize(700, 500)

        splitter = QSplitter(Qt.Orientation.Horizontal)

        # ── Left: category list ───────────────────────────────────
        self.category_list = QListWidget()
        for cat in ["General", "Appearance", "Network", "Advanced"]:
            self.category_list.addItem(cat)
        self.category_list.setFixedWidth(160)
        splitter.addWidget(self.category_list)

        # ── Right: stacked pages ──────────────────────────────────
        self.pages_container = QWidget()
        self.stack = QStackedLayout(self.pages_container)
        self.stack.addWidget(self._make_general_page())
        self.stack.addWidget(self._make_appearance_page())
        self.stack.addWidget(QLabel("Network settings…"))
        self.stack.addWidget(QLabel("Advanced settings…"))
        splitter.addWidget(self.pages_container)
        splitter.setSizes([160, 540])

        # ── Bottom: OK / Cancel ───────────────────────────────────
        wrapper = QWidget()
        vbox = QVBoxLayout(wrapper)
        vbox.addWidget(splitter, stretch=1)

        btn_row = QHBoxLayout()
        btn_row.addStretch()
        btn_row.addWidget(QPushButton("Cancel"))
        ok_btn = QPushButton("OK")
        ok_btn.setDefault(True)
        btn_row.addWidget(ok_btn)
        vbox.addLayout(btn_row)

        self.setCentralWidget(wrapper)
        self.category_list.currentRowChanged.connect(self.stack.setCurrentIndex)
        self.category_list.setCurrentRow(0)

    def _make_general_page(self) -> QWidget:
        page = QWidget()
        form = QFormLayout(page)
        form.addRow("Username:", QLineEdit())
        form.addRow("Language:", QComboBox())
        form.addRow("Max recent files:", QSpinBox())
        form.addRow("", QCheckBox("Check for updates on startup"))
        return page

    def _make_appearance_page(self) -> QWidget:
        page = QWidget()
        form = QFormLayout(page)
        form.addRow("Theme:", QComboBox())
        form.addRow("Font size:", QSpinBox())
        form.addRow("", QCheckBox("Use system font"))
        return page


if __name__ == "__main__":
    app = QApplication(sys.argv)
    win = SettingsWindow()
    win.show()
    sys.exit(app.exec())
```

## QLabel

The simplest display widget. Supports plain text, rich HTML, and images.

```python
from PyQt6.QtWidgets import QLabel
from PyQt6.QtCore import Qt
from PyQt6.QtGui import QPixmap

# Plain text
label = QLabel("Hello, World!")

# Rich HTML
label = QLabel("<b>Bold</b> and <i>italic</i> text with <span style='color:red'>colour</span>")

# Image
label = QLabel()
label.setPixmap(QPixmap("logo.png"))
label.setScaledContents(True)      # stretch to label size

# Alignment
label.setAlignment(Qt.AlignmentFlag.AlignCenter)

# Word wrap for long text
label.setWordWrap(True)

# Selection (click and drag)
label.setTextInteractionFlags(Qt.TextInteractionFlag.TextSelectableByMouse)

# Clickable link (HTML href)
label.setText('<a href="https://qt.io">Visit Qt.io</a>')
label.setOpenExternalLinks(True)
```

## QPushButton

```python
from PyQt6.QtWidgets import QPushButton
from PyQt6.QtGui import QIcon

btn = QPushButton("Click Me")
btn.setFixedHeight(36)
btn.setIcon(QIcon("icons/save.png"))
btn.setToolTip("Save the file (Ctrl+S)")

# Flat style (no raised border)
btn.setFlat(True)

# Checkable toggle button
btn.setCheckable(True)
btn.toggled.connect(lambda checked: print("Checked:", checked))

# Signals
btn.clicked.connect(my_slot)        # user click
btn.pressed.connect(on_press)       # mouse button down
btn.released.connect(on_release)    # mouse button up

# Disable / enable
btn.setEnabled(False)

# Default button (triggered by Enter key)
btn.setDefault(True)
```

## QLineEdit

Single-line text input.

```python
from PyQt6.QtWidgets import QLineEdit
from PyQt6.QtGui import QIntValidator, QRegularExpressionValidator
from PyQt6.QtCore import QRegularExpression

edit = QLineEdit()
edit.setPlaceholderText("Enter value…")
edit.setMaxLength(50)
edit.setClearButtonEnabled(True)    # shows ✕ button to clear

# Echo modes
edit.setEchoMode(QLineEdit.EchoMode.Password)       # • • • •
edit.setEchoMode(QLineEdit.EchoMode.PasswordEchoOnEdit)

# Validators — reject invalid input as the user types
edit.setValidator(QIntValidator(0, 9999))

# Regex validator (only lowercase letters + digits)
rx = QRegularExpression("[a-z0-9]+")
edit.setValidator(QRegularExpressionValidator(rx))

# Read current value
text = edit.text()

# Signals
edit.textChanged.connect(lambda t: print("Changed:", t))
edit.returnPressed.connect(submit_form)
edit.editingFinished.connect(validate_field)

# Programmatic changes
edit.setText("default value")
edit.clear()
edit.selectAll()
```

## QTextEdit

Multi-line rich-text editor.

```python
from PyQt6.QtWidgets import QTextEdit

editor = QTextEdit()
editor.setPlaceholderText("Start typing your document…")
editor.setReadOnly(False)

# Plain text
editor.setPlainText("Line 1\nLine 2")
content = editor.toPlainText()

# HTML
editor.setHtml("<h2>Title</h2><p>Paragraph with <b>bold</b>.</p>")
html_content = editor.toHtml()

# Append without replacing
editor.append("New line appended")

# Scroll to bottom (useful for log views)
editor.verticalScrollBar().setValue(editor.verticalScrollBar().maximum())

# Signals
editor.textChanged.connect(on_text_changed)

# Limit height for a "log" widget
editor.setFixedHeight(150)
```

:::tip[QPlainTextEdit for logs]
`QPlainTextEdit` is a lighter variant optimised for monospaced plain text (logs, terminals). It handles very large documents more efficiently than `QTextEdit`.
:::

## QCheckBox

```python
from PyQt6.QtWidgets import QCheckBox
from PyQt6.QtCore import Qt

cb = QCheckBox("Enable notifications")

# Tristate (unchecked / partially checked / checked)
cb.setTristate(True)
cb.setCheckState(Qt.CheckState.PartiallyChecked)

# Read state
if cb.isChecked():
    print("Checked")

state = cb.checkState()   # Qt.CheckState enum

# Signals
cb.stateChanged.connect(lambda state: print("State:", state))
cb.toggled.connect(lambda checked: print("Toggled:", checked))

# Pre-check
cb.setChecked(True)
```

## QRadioButton

Radio buttons are mutually exclusive within the same parent widget or `QButtonGroup`.

```python
from PyQt6.QtWidgets import QRadioButton, QButtonGroup, QVBoxLayout, QWidget

container = QWidget()
layout = QVBoxLayout(container)

rb1 = QRadioButton("Option A")
rb2 = QRadioButton("Option B")
rb3 = QRadioButton("Option C")
rb1.setChecked(True)   # default selection

# Group them so only one can be selected at a time
group = QButtonGroup(container)
group.addButton(rb1, id=1)
group.addButton(rb2, id=2)
group.addButton(rb3, id=3)

# Signal fires when selection changes
group.buttonClicked.connect(lambda btn: print("Selected:", btn.text()))

# Which button is selected?
checked_id = group.checkedId()    # returns the id passed to addButton

layout.addWidget(rb1)
layout.addWidget(rb2)
layout.addWidget(rb3)
```

## QComboBox

Drop-down selector.

```python
from PyQt6.QtWidgets import QComboBox
from PyQt6.QtGui import QIcon

combo = QComboBox()

# Add items
combo.addItem("Apple")
combo.addItem("Banana")
combo.addItems(["Cherry", "Date", "Elderberry"])

# Add with icon and user data
combo.addItem(QIcon("icons/flag_us.png"), "United States", "US")
combo.addItem(QIcon("icons/flag_uk.png"), "United Kingdom", "GB")

# Editable combo (free text + dropdown list)
combo.setEditable(True)
combo.setInsertPolicy(QComboBox.InsertPolicy.InsertAlphabetically)

# Read current selection
text = combo.currentText()
index = combo.currentIndex()
data = combo.currentData()          # the userData passed to addItem

# Programmatic selection
combo.setCurrentIndex(2)
combo.setCurrentText("Banana")

# Signals
combo.currentIndexChanged.connect(lambda i: print("Index:", i))
combo.currentTextChanged.connect(lambda t: print("Text:", t))
```

## QProgressBar

```python
from PyQt6.QtWidgets import QProgressBar

bar = QProgressBar()
bar.setRange(0, 100)
bar.setValue(0)
bar.setTextVisible(True)
bar.setFormat("%p% complete")    # %p = percentage, %v = value, %m = max

# Indeterminate "busy" mode
bar.setRange(0, 0)   # no numeric range = spinning animation

# Update from a loop (combined with QApplication.processEvents for same-thread use)
for i in range(101):
    bar.setValue(i)
    QApplication.processEvents()
```

:::caution[processEvents in loops]
Calling `QApplication.processEvents()` in a tight loop keeps the UI responsive but can cause re-entrant issues. For production use, run the work in a `QThread` and update the bar via a signal.
:::

## QListWidget

```python
from PyQt6.QtWidgets import QListWidget, QListWidgetItem
from PyQt6.QtCore import Qt

lw = QListWidget()

# Add items
lw.addItem("First item")
lw.addItems(["Second", "Third", "Fourth"])

# Custom item with data
item = QListWidgetItem("Custom")
item.setData(Qt.ItemDataRole.UserRole, {"id": 42, "type": "document"})
item.setCheckState(Qt.CheckState.Unchecked)
lw.addItem(item)

# Read selection
selected = lw.currentItem()
if selected:
    print(selected.text())

all_selected = [lw.item(i).text() for i in range(lw.count()) if lw.item(i).isSelected()]

# Signals
lw.itemClicked.connect(lambda item: print("Clicked:", item.text()))
lw.itemDoubleClicked.connect(on_double_click)
lw.currentRowChanged.connect(lambda row: print("Row:", row))

# Remove item
lw.takeItem(lw.currentRow())

# Reordering via drag-and-drop
lw.setDragDropMode(QListWidget.DragDropMode.InternalMove)
```

## QTableWidget

```python
from PyQt6.QtWidgets import QTableWidget, QTableWidgetItem
from PyQt6.QtCore import Qt

table = QTableWidget(5, 3)   # 5 rows, 3 columns
table.setHorizontalHeaderLabels(["Name", "Age", "City"])
table.setAlternatingRowColors(True)
table.setSelectionBehavior(QTableWidget.SelectionBehavior.SelectRows)

# Populate
data = [
    ("Alice", "30", "London"),
    ("Bob",   "25", "Paris"),
    ("Carol", "35", "Berlin"),
]
for row, (name, age, city) in enumerate(data):
    table.setItem(row, 0, QTableWidgetItem(name))
    table.setItem(row, 1, QTableWidgetItem(age))
    table.setItem(row, 2, QTableWidgetItem(city))

# Make column fill available width
from PyQt6.QtWidgets import QHeaderView
table.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeMode.Stretch)

# Read current cell
row = table.currentRow()
col = table.currentColumn()
cell = table.item(row, col)
if cell:
    print(cell.text())

# Signals
table.cellChanged.connect(lambda r, c: print(f"Cell ({r},{c}) changed"))
table.cellDoubleClicked.connect(lambda r, c: print(f"Double-clicked ({r},{c})"))
```

## QTreeWidget

```python
from PyQt6.QtWidgets import QTreeWidget, QTreeWidgetItem

tree = QTreeWidget()
tree.setHeaderLabels(["Name", "Size", "Modified"])
tree.setColumnCount(3)

# Root-level item
root = QTreeWidgetItem(tree, ["Documents", "", ""])
root.setExpanded(True)

# Child items
child1 = QTreeWidgetItem(root, ["report.pdf", "1.2 MB", "2024-01-10"])
child2 = QTreeWidgetItem(root, ["notes.txt", "4 KB",   "2024-01-08"])

# Nested (grandchildren)
sub = QTreeWidgetItem(child1, ["page1.png", "340 KB", "2024-01-09"])

# Iterate all items
def iterate_tree(item: QTreeWidgetItem) -> None:
    print(item.text(0))
    for i in range(item.childCount()):
        iterate_tree(item.child(i))

for i in range(tree.topLevelItemCount()):
    iterate_tree(tree.topLevelItem(i))

# Signals
tree.itemClicked.connect(lambda item, col: print(item.text(0)))
tree.itemExpanded.connect(lambda item: print("Expanded:", item.text(0)))
```

## QGroupBox

Groups related widgets with an optional title and border.

```python
from PyQt6.QtWidgets import QGroupBox, QVBoxLayout, QRadioButton

group = QGroupBox("Alignment")
layout = QVBoxLayout(group)
layout.addWidget(QRadioButton("Left"))
layout.addWidget(QRadioButton("Center"))
layout.addWidget(QRadioButton("Right"))

# Collapsible group box
group.setCheckable(True)
group.setChecked(True)   # False = all children disabled
group.toggled.connect(lambda on: print("Group active:", on))
```

## Common Widget Properties

Every `QWidget` subclass inherits these useful properties:

```python
widget.setEnabled(False)          # grey out and ignore input
widget.setVisible(False)          # hide (still occupies layout space if using hide())
```

## Event Filters

Intercept events on any widget without subclassing it.

```python
from PyQt6.QtCore import QObject, QEvent
from PyQt6.QtWidgets import QLineEdit


class EnterFilter(QObject):
    """Treats Enter key in a QLineEdit as Tab (focus next widget)."""

    def eventFilter(self, watched: QObject, event: QEvent) -> bool:
        if (
            isinstance(watched, QLineEdit)
            and event.type() == QEvent.Type.KeyPress
        ):
            from PyQt6.QtGui import QKeyEvent
            from PyQt6.QtCore import Qt
            key_event: QKeyEvent = event  # type: ignore[assignment]
            if key_event.key() == Qt.Key.Key_Return:
                watched.focusNextChild()
                return True   # consume the event
        return super().eventFilter(watched, event)


# Usage
line_edit = QLineEdit()
enter_filter = EnterFilter(line_edit)
line_edit.installEventFilter(enter_filter)
```

:::tip[Subclassing vs event filters]
Prefer **event filters** when you need to intercept events on third-party or generated widgets. **Subclassing** and overriding `keyPressEvent()` / `mousePressEvent()` etc. is cleaner when you own the widget class.
:::

Python provides rich built-in support for file I/O. Whether you're reading plain text, processing CSV data, serialising to JSON, or handling binary formats, the standard library has you covered.

## Opening Files

Use the built-in `open()` function. Always prefer the `with` statement — it guarantees the file is closed even if an exception is raised.

```python
# Basic open — defaults to read mode, text encoding
with open('notes.txt', 'r', encoding='utf-8') as f:
    content = f.read()
```

### Mode Reference

| Mode | Meaning |
|------|---------|
| `'r'` | Read (default) — file must exist |
| `'w'` | Write — creates or truncates |
| `'a'` | Append — creates or appends |
| `'x'` | Exclusive create — fails if file exists |
| `'rb'` | Read binary |
| `'wb'` | Write binary |
| `'r+'` | Read + write (no truncate) |

```python
# Always specify encoding for text files
with open('data.txt', 'w', encoding='utf-8') as f:
    f.write('Hello, world!\n')

# Binary mode — no encoding parameter
with open('image.png', 'rb') as f:
    header = f.read(8)
```

<Aside type="tip">
  Always pass `encoding='utf-8'` explicitly. The default encoding is platform-dependent and can cause subtle bugs when moving code between Windows and Linux.
</Aside>

## Writing Files

```python
# write() — returns number of characters written
with open('output.txt', 'w', encoding='utf-8') as f:
    n = f.write('Line one\n')
    f.write('Line two\n')
print(n)  # 9

# writelines() — writes an iterable; no automatic newlines
lines = ['apple\n', 'banana\n', 'cherry\n']
with open('fruits.txt', 'w', encoding='utf-8') as f:
    f.writelines(lines)

# Appending to an existing file
with open('log.txt', 'a', encoding='utf-8') as f:
    f.write('New entry\n')
```

<Aside type="caution">
  `'w'` mode silently truncates the file if it already exists. Use `'x'` mode if you want an error when overwriting would be dangerous.
</Aside>

## Pathlib

`pathlib.Path` is the modern way to work with filesystem paths. It replaces most uses of `os.path` and provides an intuitive, object-oriented API.

```python
from pathlib import Path

# Constructing paths — '/' operator joins segments
base = Path('/home/user/projects')
config = base / 'myapp' / 'config.yaml'

print(config)           # /home/user/projects/myapp/config.yaml
print(config.name)      # config.yaml
print(config.stem)      # config
print(config.suffix)    # .yaml
print(config.parent)    # /home/user/projects/myapp
```

### Common Operations

```python
p = Path('data/results')

# Existence checks
p.exists()          # True/False
p.is_file()         # True if regular file
p.is_dir()          # True if directory

# Create directories
p.mkdir(parents=True, exist_ok=True)  # like mkdir -p

# Read/write text without open()
Path('hello.txt').write_text('Hello!\n', encoding='utf-8')
text = Path('hello.txt').read_text(encoding='utf-8')
print(text)  # Hello!

# Read/write bytes
Path('data.bin').write_bytes(b'\x00\xFF\xAB')
raw = Path('data.bin').read_bytes()

# Glob patterns
src = Path('src')
py_files = list(src.glob('**/*.py'))    # recursive
txt_files = list(src.glob('*.txt'))     # current dir only

for f in py_files:
    print(f)  # src/utils/helper.py, src/main.py, ...

# Rename and delete
Path('old.txt').rename('new.txt')
Path('temp.txt').unlink(missing_ok=True)  # delete file

# Iterate directory contents
for item in Path('.').iterdir():
    print(item.name, '(dir)' if item.is_dir() else '')
```

<Aside type="tip">
  Prefer `pathlib.Path` over `os.path` for all new code. It's more readable, cross-platform, and composable. `Path('/a') / 'b' / 'c'` beats `os.path.join('/a', 'b', 'c')` every time.
</Aside>

## os and os.path

When you need OS-level operations beyond what `pathlib` exposes:

```python
import os
import shutil

# Current directory and environment
print(os.getcwd())                  # /home/user/projects
print(os.environ.get('HOME', ''))   # /home/user

# List directory
entries = os.listdir('.')           # ['main.py', 'README.md', ...]

# Create nested directories (older codebases)
os.makedirs('a/b/c', exist_ok=True)

# File operations
os.remove('temp.txt')               # delete file
os.rename('old.txt', 'new.txt')     # rename / move
os.rmdir('empty_dir')               # remove empty directory

# shutil for copying and moving
shutil.copy('src.txt', 'dst.txt')           # copy file
shutil.copy2('src.txt', 'dst.txt')          # copy + metadata
shutil.move('file.txt', 'archive/')         # move
shutil.rmtree('build/', ignore_errors=True) # remove tree

# os.path — still common in older code
import os.path
print(os.path.join('/home', 'user', 'file.txt'))  # /home/user/file.txt
print(os.path.exists('config.yaml'))               # True/False
print(os.path.abspath('relative/path'))            # absolute path
print(os.path.splitext('archive.tar.gz'))          # ('archive.tar', '.gz')
```

## CSV Files

```python
import csv

# Writing CSV
rows = [
    ['Name', 'Age', 'City'],
    ['Alice', 30, 'London'],
    ['Bob', 25, 'Paris'],
]
with open('people.csv', 'w', newline='', encoding='utf-8') as f:
    writer = csv.writer(f)
    writer.writerows(rows)

# Reading CSV
with open('people.csv', 'r', newline='', encoding='utf-8') as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)
# ['Name', 'Age', 'City']
# ['Alice', '30', 'London']
# ['Bob', '25', 'Paris']
```

### DictReader / DictWriter

```python
# DictWriter — header row driven by fieldnames
fieldnames = ['name', 'score', 'grade']
with open('grades.csv', 'w', newline='', encoding='utf-8') as f:
    writer = csv.DictWriter(f, fieldnames=fieldnames)
    writer.writeheader()
    writer.writerows([
        {'name': 'Alice', 'score': 95, 'grade': 'A'},
        {'name': 'Bob',   'score': 82, 'grade': 'B'},
    ])

# DictReader — each row becomes a dict keyed by header
with open('grades.csv', 'r', newline='', encoding='utf-8') as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(f"{row['name']}: {row['grade']}")
# Alice: A
# Bob: B
```

<Aside type="caution">
  Always pass `newline=''` when opening CSV files. Without it, Python's universal newline handling can corrupt rows on Windows.
</Aside>

## JSON Files

```python
import json

data = {
    'name': 'Alice',
    'age': 30,
    'scores': [95, 88, 72],
    'active': True,
    'address': None,
}

# Serialise to file — indent makes it human-readable
with open('data.json', 'w', encoding='utf-8') as f:
    json.dump(data, f, indent=2)

# Deserialise from file
with open('data.json', 'r', encoding='utf-8') as f:
    loaded = json.load(f)
print(loaded['name'])  # Alice

# Working with strings (not files)
json_str = json.dumps(data, indent=2)       # dict  → str
parsed   = json.loads('{"x": 1, "y": 2}')  # str   → dict
print(parsed)  # {'x': 1, 'y': 2}
```

### Custom Encoders

```python
from datetime import datetime, date

class DateEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, (date, datetime)):
            return obj.isoformat()
        return super().default(obj)

event = {'name': 'Launch', 'date': date(2024, 6, 1)}
print(json.dumps(event, cls=DateEncoder))
# {"name": "Launch", "date": "2024-06-01"}
```

## Binary Files

```python
# Reading image header bytes
with open('photo.jpg', 'rb') as f:
    magic = f.read(3)
print(magic)  # b'\xff\xd8\xff'  (JPEG magic bytes)

# Using struct for fixed-width binary formats
import struct

# Pack: write two shorts and a float
record = struct.pack('>HHf', 100, 200, 3.14)
with open('record.bin', 'wb') as f:
    f.write(record)

# Unpack: read them back
with open('record.bin', 'rb') as f:
    raw = f.read(8)
x, y, z = struct.unpack('>HHf', raw)
print(x, y, round(z, 2))  # 100 200 3.14

# Seek to a position
with open('data.bin', 'rb') as f:
    f.seek(512)        # jump to byte 512
    chunk = f.read(64) # read 64 bytes from there
    pos = f.tell()     # current position → 576
```

## Temporary Files

```python
import tempfile
import os

# NamedTemporaryFile — auto-deleted on close
with tempfile.NamedTemporaryFile(mode='w', suffix='.txt',
                                  delete=True, encoding='utf-8') as tmp:
    tmp.write('temporary data')
    print(tmp.name)  # /tmp/tmpXXXXXX.txt
# File is deleted here

# Keep file after closing (delete=False)
tmp = tempfile.NamedTemporaryFile(delete=False, suffix='.csv')
tmp.close()
try:
    # process tmp.name ...
    pass
finally:
    os.unlink(tmp.name)  # clean up manually

# Temporary directory
with tempfile.TemporaryDirectory() as tmpdir:
    work = Path(tmpdir) / 'scratch.txt'
    work.write_text('processing...')
    print(work.exists())  # True
# Directory and all contents deleted here
```

## Common Patterns

### Reading Large Files Line by Line

```python
# Stream a multi-gigabyte log — constant memory usage
def count_errors(path: str) -> int:
    count = 0
    with open(path, 'r', encoding='utf-8') as f:
        for line in f:           # one line buffered at a time
            if 'ERROR' in line:
                count += 1
    return count
```

### Atomic Writes

Avoid partial writes by writing to a temp file then renaming:

```python
import os
from pathlib import Path

def atomic_write(path: str | Path, content: str) -> None:
    """Write content atomically — readers never see a partial file."""
    path = Path(path)
    tmp = path.with_suffix('.tmp')
    try:
        tmp.write_text(content, encoding='utf-8')
        tmp.replace(path)   # atomic on POSIX; near-atomic on Windows
    except Exception:
        tmp.unlink(missing_ok=True)
        raise
```

### File Locking

```python
import fcntl  # Unix only

with open('shared.txt', 'a', encoding='utf-8') as f:
    fcntl.flock(f, fcntl.LOCK_EX)   # exclusive lock
    f.write('safe concurrent write\n')
    fcntl.flock(f, fcntl.LOCK_UN)   # release

# Cross-platform option: use the 'filelock' third-party package
# pip install filelock
from filelock import FileLock

lock = FileLock('data.json.lock')
with lock:
    data = json.loads(Path('data.json').read_text())
    data['counter'] += 1
    atomic_write('data.json', json.dumps(data, indent=2))
```

<Aside type="tip">
  For cross-process file coordination in production code, prefer message queues or databases over file locking. File locking semantics vary between operating systems.
</Aside>

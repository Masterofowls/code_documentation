## os / sys

```python
import os
import sys

# Environment variables
home = os.environ.get('HOME', '/tmp')
db_url = os.environ['DATABASE_URL']           # raises KeyError if missing
os.environ['DEBUG'] = 'true'                  # set for current process

# Process info
print(os.getpid())     # 12345
print(os.getcwd())     # /home/user/projects

# Command-line arguments (sys.argv)
# python script.py --verbose output.txt
print(sys.argv)        # ['script.py', '--verbose', 'output.txt']
script_name = sys.argv[0]
args = sys.argv[1:]

# Python runtime info
print(sys.version)     # 3.12.0 (main, ...)
print(sys.platform)    # 'linux', 'win32', 'darwin'
print(sys.executable)  # /usr/bin/python3

# Exit the process
if '--help' in sys.argv:
    print('Usage: script.py [options]')
    sys.exit(0)   # 0 = success; non-zero = error

# Standard streams
sys.stdout.write('no newline')
print('error message', file=sys.stderr)
```

<Aside type="tip">
  Use `os.environ.get('KEY', 'default')` instead of `os.environ['KEY']` when the
  variable might be absent — it avoids a `KeyError` at startup.
</Aside>

## pathlib

```python
from pathlib import Path

# Build cross-platform paths with /
config = Path.home() / '.config' / 'myapp' / 'settings.yaml'

# Introspect
print(config.name)      # settings.yaml
print(config.stem)      # settings
print(config.suffix)    # .yaml
print(config.parent)    # ~/.config/myapp

# Create the directory tree
config.parent.mkdir(parents=True, exist_ok=True)

# Read and write without open()
config.write_text('debug: true\n', encoding='utf-8')
text = config.read_text(encoding='utf-8')
print(text)  # debug: true

# Glob — find all Python files recursively
src = Path('src')
for py in src.glob('**/*.py'):
    print(py)

# Stat and metadata
stat = config.stat()
print(stat.st_size)     # bytes
print(config.exists())  # True
print(config.is_file()) # True
print(config.is_dir())  # False

# Iterate a directory
for item in Path('.').iterdir():
    kind = 'dir' if item.is_dir() else 'file'
    print(f'{item.name:<30} {kind}')

# Delete
config.unlink(missing_ok=True)           # remove file
config.parent.rmdir()                    # remove empty dir
```

## datetime

```python
from datetime import datetime, date, time, timedelta, timezone

# Current moment
now = datetime.now()                       # local time
utc_now = datetime.now(tz=timezone.utc)   # UTC

print(now)          # 2024-06-01 14:32:10.123456
print(utc_now)      # 2024-06-01 12:32:10.123456+00:00

# Constructing specific dates
d = date(2024, 12, 25)
t = time(9, 30, 0)
dt = datetime(2024, 12, 25, 9, 30, 0)

# Arithmetic with timedelta
tomorrow = date.today() + timedelta(days=1)
one_hour_later = now + timedelta(hours=1)
diff = datetime(2025, 1, 1) - now
print(diff.days)    # days until new year

# Formatting — strftime
print(now.strftime('%Y-%m-%d'))          # 2024-06-01
print(now.strftime('%d/%m/%Y %H:%M'))    # 01/06/2024 14:32
print(now.strftime('%A, %B %d %Y'))      # Saturday, June 01 2024

# Parsing — strptime
parsed = datetime.strptime('2024-06-01 14:32', '%Y-%m-%d %H:%M')
print(parsed)  # 2024-06-01 14:32:00

# ISO format (round-trips cleanly)
iso = now.isoformat()                    # '2024-06-01T14:32:10.123456'
back = datetime.fromisoformat(iso)

# Timezone-aware operations
from zoneinfo import ZoneInfo            # Python 3.9+
tz_london = ZoneInfo('Europe/London')
tz_tokyo  = ZoneInfo('Asia/Tokyo')

london_now = datetime.now(tz=tz_london)
tokyo_now  = london_now.astimezone(tz_tokyo)
print(london_now.strftime('%H:%M %Z'))   # 14:32 BST
print(tokyo_now.strftime('%H:%M %Z'))    # 22:32 JST
```

## collections

### Counter

```python
from collections import Counter

words = ['apple', 'banana', 'apple', 'cherry', 'banana', 'apple']
c = Counter(words)
print(c)                    # Counter({'apple': 3, 'banana': 2, 'cherry': 1})
print(c['apple'])           # 3
print(c.most_common(2))     # [('apple', 3), ('banana', 2)]

# Arithmetic on counters
c2 = Counter(['apple', 'date'])
print(c + c2)               # adds counts
print(c - c2)               # subtracts (drops zero/negative)
```

### defaultdict

```python
from collections import defaultdict

# Group items without KeyError
grouped = defaultdict(list)
for name, dept in [('Alice', 'Eng'), ('Bob', 'HR'), ('Carol', 'Eng')]:
    grouped[dept].append(name)

print(dict(grouped))
# {'Eng': ['Alice', 'Carol'], 'HR': ['Bob']}

word_count = defaultdict(int)
for word in 'the cat sat on the mat'.split():
    word_count[word] += 1
```

### namedtuple

```python
from collections import namedtuple

Point = namedtuple('Point', ['x', 'y'])
p = Point(3, 4)
print(p.x, p.y)   # 3 4
print(p)           # Point(x=3, y=4)

# Unpack like a regular tuple
x, y = p

# _replace returns a new instance
p2 = p._replace(y=10)
print(p2)  # Point(x=3, y=10)
```

### deque

```python
from collections import deque

# O(1) appends and pops from both ends
dq = deque([1, 2, 3], maxlen=5)
dq.appendleft(0)    # [0, 1, 2, 3]
dq.append(4)        # [0, 1, 2, 3, 4]
dq.rotate(1)        # [4, 0, 1, 2, 3]
print(dq.popleft()) # 4
```

## itertools

```python
import itertools

# chain — flatten multiple iterables
nums = list(itertools.chain([1, 2], [3, 4], [5]))
print(nums)  # [1, 2, 3, 4, 5]

# islice — lazy slicing of any iterator
gen = (x**2 for x in itertools.count(1))  # infinite: 1, 4, 9, 16...
first_five = list(itertools.islice(gen, 5))
print(first_five)  # [1, 4, 9, 16, 25]

# groupby — consecutive groups (sort first!)
data = [('A', 1), ('A', 2), ('B', 3), ('B', 4), ('A', 5)]
data.sort(key=lambda x: x[0])
for key, group in itertools.groupby(data, key=lambda x: x[0]):
    print(key, list(group))
# A [('A', 1), ('A', 2)]
# B [('B', 3), ('B', 4)]
# A [('A', 5)]

# product — Cartesian product
for suit, rank in itertools.product('♠♥', ['A', 'K']):
    print(suit, rank, end='  ')
# ♠ A  ♠ K  ♥ A  ♥ K

# combinations and permutations
print(list(itertools.combinations('ABC', 2)))
# [('A', 'B'), ('A', 'C'), ('B', 'C')]

print(list(itertools.permutations('AB', 2)))
# [('A', 'B'), ('B', 'A')]

# accumulate — running totals
import operator
sales = [100, 200, 150, 300]
running = list(itertools.accumulate(sales))
print(running)  # [100, 300, 450, 750]
```

## functools

```python
import functools

# partial — fix some arguments
def power(base, exp):
    return base ** exp

square = functools.partial(power, exp=2)
cube   = functools.partial(power, exp=3)
print(square(5))  # 25
print(cube(3))    # 27

# lru_cache — memoize expensive calls
@functools.lru_cache(maxsize=128)
def fib(n: int) -> int:
    if n < 2:
        return n
    return fib(n - 1) + fib(n - 2)

print(fib(50))           # 12586269025
print(fib.cache_info())  # CacheInfo(hits=48, misses=51, ...)

# @cache (Python 3.9+) — unbounded cache, no maxsize
@functools.cache
def factorial(n: int) -> int:
    return 1 if n == 0 else n * factorial(n - 1)

# reduce — fold a sequence into a single value
product = functools.reduce(lambda acc, x: acc * x, [1, 2, 3, 4, 5])
print(product)  # 120

# wraps — preserve metadata when writing decorators
def log_calls(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f'Calling {func.__name__}')
        return func(*args, **kwargs)
    return wrapper

@log_calls
def greet(name: str) -> str:
    """Return a greeting."""
    return f'Hello, {name}!'

print(greet.__name__)  # greet  (not 'wrapper')
print(greet.__doc__)   # Return a greeting.
```

## re — Regular Expressions

```python
import re

text = 'Contact us at support@example.com or sales@company.org'

# re.search — first match anywhere in string
m = re.search(r'\b[\w.+-]+@[\w-]+\.\w+\b', text)
if m:
    print(m.group())  # support@example.com

# re.findall — all non-overlapping matches
emails = re.findall(r'\b[\w.+-]+@[\w-]+\.\w+\b', text)
print(emails)  # ['support@example.com', 'sales@company.org']

# re.match — only at the start of the string
m = re.match(r'\d{4}', '2024-06-01')
print(m.group() if m else None)  # 2024

# re.sub — replace
clean = re.sub(r'\s+', ' ', 'too   many    spaces')
print(clean)  # too many spaces

# Named groups
log_line = '2024-06-01 ERROR disk full'
pattern = r'(?P<date>\d{4}-\d{2}-\d{2}) (?P<level>\w+) (?P<msg>.+)'
m = re.match(pattern, log_line)
if m:
    print(m.group('date'))   # 2024-06-01
    print(m.group('level'))  # ERROR
    print(m.group('msg'))    # disk full
    print(m.groupdict())
    # {'date': '2024-06-01', 'level': 'ERROR', 'msg': 'disk full'}

# Compile for reuse in loops
email_re = re.compile(r'\b[\w.+-]+@[\w-]+\.\w+\b', re.IGNORECASE)
for address in email_re.finditer(text):
    print(address.start(), address.group())
# 15 support@example.com
# 37 sales@company.org
```

<Aside type="tip">
  Always use raw strings (`r'...'`) for regex patterns to avoid double-escaping
  backslashes. Use `re.compile()` when the same pattern is applied many times —
  the compiled object is reused.
</Aside>

## logging

```python
import logging
from pathlib import Path

# Basic configuration — call once at application startup
logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s [%(levelname)s] %(name)s: %(message)s',
    handlers=[
        logging.StreamHandler(),                     # console
        logging.FileHandler('app.log', encoding='utf-8'),  # file
    ],
)

logger = logging.getLogger(__name__)  # 'mymodule.submodule'

logger.debug('Connecting to %s:%d', 'localhost', 5432)
logger.info('Server started on port 8080')
logger.warning('Config file missing — using defaults')
logger.error('Failed to connect: %s', 'Connection refused')
logger.critical('Out of memory — shutting down')

# 2024-06-01 14:32:10,123 [DEBUG] __main__: Connecting to localhost:5432
# 2024-06-01 14:32:10,124 [INFO] __main__: Server started on port 8080
# ...
```

<Aside type="tip">
  Never use `print()` for application logging. `logging` gives you free
  timestamps, levels, module names, and the ability to route output to files,
  syslog, or external services — all configurable without code changes.
</Aside>

## subprocess

```python
import subprocess

# run() — preferred; captures output, raises on failure
result = subprocess.run(
    ['git', 'log', '--oneline', '-5'],
    capture_output=True,
    text=True,
    check=True,         # raises CalledProcessError on non-zero exit
)
print(result.stdout)
# abc1234 fix: handle empty input
# def5678 feat: add retry logic

# check_output — shorthand for capturing stdout
out = subprocess.check_output(['ls', '-la'], text=True)

# Pass input to stdin
result = subprocess.run(
    ['python3', '-c', 'import sys; print(sys.stdin.read().upper())'],
    input='hello world\n',
    capture_output=True,
    text=True,
)
print(result.stdout)  # HELLO WORLD

# Popen — for streaming or interactive processes
with subprocess.Popen(
    ['tail', '-f', 'app.log'],
    stdout=subprocess.PIPE,
    text=True,
) as proc:
    for line in proc.stdout:
        print(line, end='')
```

<Aside type="caution">
  Avoid `shell=True`. It spawns a shell (`/bin/sh -c ...`), which is vulnerable
  to shell injection if any part of the command contains user-supplied data.
  Pass a list of strings instead.
</Aside>

## threading / multiprocessing

### The GIL

Python's Global Interpreter Lock (GIL) allows only one thread to execute Python bytecode at a time. This means:

- **Threading** is effective for **I/O-bound** work (network, disk) — threads release the GIL during I/O waits.
- **Multiprocessing** is needed for **CPU-bound** work — each process has its own GIL-free interpreter.

```python
import threading
import multiprocessing
from queue import Queue

# --- Threading (I/O-bound) ---
results: Queue = Queue()

def fetch(url: str) -> None:
    import urllib.request
    with urllib.request.urlopen(url) as r:
        results.put((url, len(r.read())))

urls = ['https://python.org', 'https://pypi.org']
threads = [threading.Thread(target=fetch, args=(u,)) for u in urls]
for t in threads:
    t.start()
for t in threads:
    t.join()

while not results.empty():
    url, size = results.get()
    print(f'{url}: {size} bytes')

# --- Multiprocessing (CPU-bound) ---
def cpu_task(n: int) -> int:
    return sum(i * i for i in range(n))

with multiprocessing.Pool(processes=4) as pool:
    results_mp = pool.map(cpu_task, [10**6, 10**6, 10**6, 10**6])
print(results_mp)  # [333332833333500000, ...]

# Inter-process Queue
mp_queue: multiprocessing.Queue = multiprocessing.Queue()

def worker(q: multiprocessing.Queue, n: int) -> None:
    q.put(n * n)

procs = [multiprocessing.Process(target=worker, args=(mp_queue, i)) for i in range(4)]
for p in procs:
    p.start()
for p in procs:
    p.join()

while not mp_queue.empty():
    print(mp_queue.get())  # 0, 1, 4, 9 (order varies)
```

<Aside type="tip">
  For I/O-bound concurrency in modern Python, prefer `asyncio` or
  `concurrent.futures.ThreadPoolExecutor`. For CPU-bound work, use
  `concurrent.futures.ProcessPoolExecutor` — it wraps multiprocessing with a
  cleaner API.
</Aside>

## typing

```python
from typing import (
    List, Dict, Optional, Union, Callable,
    TypeVar, Generic, Tuple, Any,
)
from collections.abc import Iterator, Sequence

# Basic annotations (Python 3.9+ supports built-in generics: list[int])
def greet(name: str, times: int = 1) -> str:
    return (f'Hello, {name}!\n') * times

# Optional — value or None
def find_user(user_id: int) -> Optional[str]:  # same as str | None
    users = {1: 'Alice', 2: 'Bob'}
    return users.get(user_id)

# Union — multiple possible types (Python 3.10+ supports X | Y)
def stringify(value: Union[int, float, str]) -> str:
    return str(value)

# Callable
Transform = Callable[[str], str]

def apply(func: Transform, values: List[str]) -> List[str]:
    return [func(v) for v in values]

print(apply(str.upper, ['a', 'b', 'c']))  # ['A', 'B', 'C']

# TypeVar — generic functions
T = TypeVar('T')

def first(seq: Sequence[T]) -> Optional[T]:
    return seq[0] if seq else None

print(first([1, 2, 3]))      # 1   (inferred as int)
print(first(['a', 'b']))     # 'a' (inferred as str)

# Generic classes
class Stack(Generic[T]):
    def __init__(self) -> None:
        self._items: List[T] = []

    def push(self, item: T) -> None:
        self._items.append(item)

    def pop(self) -> T:
        return self._items.pop()

    def __len__(self) -> int:
        return len(self._items)

s: Stack[int] = Stack()
s.push(1)
s.push(2)
print(s.pop())  # 2
```

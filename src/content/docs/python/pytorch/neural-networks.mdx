`torch.nn` is PyTorch's neural network library. Every model is a subclass of `nn.Module`, which provides parameter tracking, device movement, serialization, and a clean `forward()` interface.

## nn.Module Basics

```python
import torch
import torch.nn as nn

class MyModel(nn.Module):
    def __init__(self):
        super().__init__()           # ALWAYS call super().__init__()
        # Layers defined here are automatically registered as sub-modules
        self.fc1  = nn.Linear(784, 256)
        self.fc2  = nn.Linear(256, 128)
        self.fc3  = nn.Linear(128, 10)
        self.relu = nn.ReLU()
        self.drop = nn.Dropout(p=0.3)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Define the computation here — called when you invoke model(x)
        x = self.relu(self.fc1(x))    # shape: (batch, 256)
        x = self.drop(x)              # randomly zeros 30% of neurons
        x = self.relu(self.fc2(x))    # shape: (batch, 128)
        return self.fc3(x)            # shape: (batch, 10)

model = MyModel()
print(model)
# MyModel(
#   (fc1): Linear(in_features=784, out_features=256, bias=True)
#   (fc2): Linear(in_features=256, out_features=128, bias=True)
#   (fc3): Linear(in_features=128, out_features=10, bias=True)
#   (relu): ReLU()
#   (drop): Dropout(p=0.3, inplace=False)
# )
```

:::tip[Always Call super().__init__()]
Forgetting `super().__init__()` in your `__init__` means parameters are not registered and `model.parameters()` returns nothing — your model will never learn.
:::

## Activation Functions

```python
import torch
import torch.nn as nn

x = torch.tensor([-2.0, -0.5, 0.0, 0.5, 2.0])

# Standard activations
print(torch.relu(x))         # tensor([0.0, 0.0, 0.0, 0.5, 2.0])
print(torch.sigmoid(x))      # tensor([0.1192, 0.3775, 0.5000, 0.6225, 0.8808])
print(torch.tanh(x))         # tensor([-0.9640, -0.4621,  0.0000,  0.4621,  0.9640])
print(nn.GELU()(x))           # Gaussian error linear unit — used in BERT/GPT
print(nn.SiLU()(x))           # Swish — used in EfficientNet / LLaMA
print(nn.Mish()(x))           # Mish — used in YOLOv4 / YOLOv5
print(nn.Softmax(dim=0)(x))   # Probabilities summing to 1 — use for output layer

# Leaky ReLU (fixes dying ReLU problem)
lrelu = nn.LeakyReLU(negative_slope=0.01)
print(lrelu(x))   # tensor([-0.0200, -0.0050,  0.0000,  0.5000,  2.0000])

# ELU
print(nn.ELU(alpha=1.0)(x))   # smooth negative region
```

## nn.Sequential

`nn.Sequential` chains layers without writing a custom `forward()`:

```python
import torch.nn as nn

# Simple MLP via Sequential
mlp = nn.Sequential(
    nn.Linear(784, 256),
    nn.BatchNorm1d(256),
    nn.ReLU(),
    nn.Dropout(0.3),
    nn.Linear(256, 128),
    nn.BatchNorm1d(128),
    nn.ReLU(),
    nn.Dropout(0.3),
    nn.Linear(128, 10)
)

# Equivalent using an OrderedDict (named layers)
from collections import OrderedDict
named_mlp = nn.Sequential(OrderedDict([
    ('flatten', nn.Flatten()),
    ('fc1',     nn.Linear(784, 256)),
    ('bn1',     nn.BatchNorm1d(256)),
    ('relu1',   nn.ReLU()),
    ('fc2',     nn.Linear(256, 10))
]))

# Access a layer by name
print(named_mlp.fc1)        # Linear(in_features=784, out_features=256, bias=True)
print(named_mlp.relu1)      # ReLU()

# nn.ModuleList — register a list of modules
import torch
class MultiHead(nn.Module):
    def __init__(self, n_heads, d_model):
        super().__init__()
        self.heads = nn.ModuleList([
            nn.Linear(d_model, d_model // n_heads) for _ in range(n_heads)
        ])
    def forward(self, x):
        return torch.cat([h(x) for h in self.heads], dim=-1)
```

## Counting Parameters

```python
import torch.nn as nn

def count_parameters(model: nn.Module, trainable_only: bool = True) -> int:
    """Count the number of (trainable) parameters in a model."""
    return sum(
        p.numel()
        for p in model.parameters()
        if (not trainable_only) or p.requires_grad
    )

mlp = nn.Sequential(nn.Linear(784, 256), nn.ReLU(), nn.Linear(256, 10))
total     = count_parameters(mlp, trainable_only=False)
trainable = count_parameters(mlp, trainable_only=True)
print(f'Total params      : {total:,}')       # Total params      : 203,530
print(f'Trainable params  : {trainable:,}')   # Trainable params  : 203,530
print(f'Size (float32 MB) : {total * 4 / 1e6:.2f}')  # 0.81 MB

# Per-layer breakdown
for name, p in mlp.named_parameters():
    print(f'  {name:20s}  {p.numel():>8,}  {tuple(p.shape)}')
```

## Weight Initialization

```python
import torch
import torch.nn as nn

class MyNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv = nn.Conv2d(3, 64, 3, padding=1)
        self.bn   = nn.BatchNorm2d(64)
        self.fc   = nn.Linear(64, 10)
        self._init_weights()

    def _init_weights(self):
        for m in self.modules():
            if isinstance(m, nn.Conv2d):
                # Kaiming (He) — best for ReLU networks
                nn.init.kaiming_normal_(m.weight, mode='fan_out',
                                        nonlinearity='relu')
                if m.bias is not None:
                    nn.init.zeros_(m.bias)

            elif isinstance(m, nn.Linear):
                # Xavier/Glorot — good for tanh/sigmoid
                nn.init.xavier_uniform_(m.weight)
                nn.init.zeros_(m.bias)

            elif isinstance(m, nn.BatchNorm2d):
                nn.init.ones_(m.weight)    # γ = 1
                nn.init.zeros_(m.bias)     # β = 0

    def forward(self, x):
        x = self.bn(self.conv(x)).relu()
        x = x.mean([2, 3])           # global average pool
        return self.fc(x)

# Other initialisation functions
w = torch.empty(64, 32)
nn.init.normal_(w, mean=0.0, std=0.02)        # N(0, 0.02²)
nn.init.uniform_(w, a=-0.1, b=0.1)            # U[-0.1, 0.1]
nn.init.constant_(w, val=0.0)                 # fill with constant
nn.init.orthogonal_(w)                        # orthogonal matrix
nn.init.eye_(torch.empty(32, 32))             # identity (square only)
nn.init.trunc_normal_(w, mean=0, std=0.02)    # truncated normal
```

## model.train() and model.eval()

```python
import torch
import torch.nn as nn

model = nn.Sequential(
    nn.Linear(128, 64),
    nn.BatchNorm1d(64),
    nn.ReLU(),
    nn.Dropout(0.5),
    nn.Linear(64, 10)
)

x = torch.randn(32, 128)

# ── Training mode ─────────────────────────────────────────────────────────────
model.train()
# BatchNorm: uses BATCH statistics (mean/var from current batch)
# Dropout: randomly zeros neurons with probability p
out_train = model(x)

# ── Evaluation mode ───────────────────────────────────────────────────────────
model.eval()
# BatchNorm: uses RUNNING statistics (accumulated during training)
# Dropout: passes all neurons (no dropping)
with torch.no_grad():   # always pair with no_grad during inference
    out_eval = model(x)

# Check mode of each sub-module
for name, m in model.named_modules():
    print(f'{name}: training={m.training}')
# : training=False
# 0: training=False
# ...
```

:::caution[Always Switch Modes]
Forgetting `model.eval()` during validation means BatchNorm uses batch statistics and Dropout randomly masks neurons — both cause incorrect, non-deterministic evaluation metrics. Always call `model.eval()` before validation and `model.train()` before the next training epoch.
:::

## Saving and Loading

```python
import torch
import torch.nn as nn
from pathlib import Path

model = nn.Linear(128, 10)

# ── Save / load state_dict (RECOMMENDED) ──────────────────────────────────────
# Saves only the parameters, not the model class definition
torch.save(model.state_dict(), 'model.pth')

# Load: must instantiate model first, then load weights
loaded_model = nn.Linear(128, 10)
loaded_model.load_state_dict(torch.load('model.pth', weights_only=True))
loaded_model.eval()

# ── Full checkpoint (model + optimiser + epoch) ───────────────────────────────
optimizer = torch.optim.Adam(model.parameters())
checkpoint = {
    'epoch'          : 42,
    'model_state'    : model.state_dict(),
    'optimizer_state': optimizer.state_dict(),
    'val_loss'       : 0.0234,
}
torch.save(checkpoint, 'checkpoint_epoch42.pth')

# Resume training from checkpoint
ckpt = torch.load('checkpoint_epoch42.pth', weights_only=True)
model.load_state_dict(ckpt['model_state'])
optimizer.load_state_dict(ckpt['optimizer_state'])
start_epoch = ckpt['epoch'] + 1
print(f'Resuming from epoch {start_epoch}')  # Resuming from epoch 43

# ── Partial load (fine-tuning with mismatched keys) ───────────────────────────
state = torch.load('pretrained.pth', weights_only=True)
model.load_state_dict(state, strict=False)   # ignore missing/unexpected keys
```

## Complete CNN for Image Classification

A production-quality CNN for CIFAR-10 with modern best practices:

```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class ConvBlock(nn.Module):
    """Conv → BN → ReLU → optional MaxPool."""
    def __init__(self, in_ch: int, out_ch: int, pool: bool = False):
        super().__init__()
        layers = [
            nn.Conv2d(in_ch, out_ch, kernel_size=3, padding=1, bias=False),
            nn.BatchNorm2d(out_ch),
            nn.ReLU(inplace=True),
        ]
        if pool:
            layers.append(nn.MaxPool2d(2, 2))   # halve spatial dims
        self.block = nn.Sequential(*layers)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.block(x)


class ResidualBlock(nn.Module):
    """Two 3×3 convolutions with a skip connection."""
    def __init__(self, channels: int):
        super().__init__()
        self.conv1 = nn.Conv2d(channels, channels, 3, padding=1, bias=False)
        self.bn1   = nn.BatchNorm2d(channels)
        self.conv2 = nn.Conv2d(channels, channels, 3, padding=1, bias=False)
        self.bn2   = nn.BatchNorm2d(channels)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        residual = x
        out = F.relu(self.bn1(self.conv1(x)), inplace=True)
        out = self.bn2(self.conv2(out))
        return F.relu(out + residual, inplace=True)   # skip connection


class CIFAR10Net(nn.Module):
    """
    ResNet-inspired CNN for CIFAR-10 (32×32 RGB → 10 classes).
    Achieves ~93% test accuracy with proper training.
    """
    def __init__(self, num_classes: int = 10, dropout: float = 0.3):
        super().__init__()

        # Stem — initial feature extraction
        self.stem = nn.Sequential(
            nn.Conv2d(3, 64, kernel_size=3, padding=1, bias=False),  # (B,3,32,32)→(B,64,32,32)
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
        )

        # Stage 1 — 64 channels, 32×32
        self.stage1 = nn.Sequential(
            ResidualBlock(64),
            ResidualBlock(64),
            nn.MaxPool2d(2, 2),              # (B,64,32,32)→(B,64,16,16)
        )

        # Stage 2 — 128 channels, 16×16
        self.stage2 = nn.Sequential(
            ConvBlock(64, 128, pool=True),   # (B,64,16,16)→(B,128,8,8)
            ResidualBlock(128),
            ResidualBlock(128),
        )

        # Stage 3 — 256 channels, 8×8
        self.stage3 = nn.Sequential(
            ConvBlock(128, 256, pool=True),  # (B,128,8,8)→(B,256,4,4)
            ResidualBlock(256),
        )

        # Classifier head
        self.head = nn.Sequential(
            nn.AdaptiveAvgPool2d((1, 1)),    # (B,256,4,4)→(B,256,1,1)
            nn.Flatten(),                    # (B,256)
            nn.Linear(256, 512),
            nn.ReLU(inplace=True),
            nn.Dropout(dropout),
            nn.Linear(512, num_classes),     # (B,10)
        )

        self._init_weights()

    def _init_weights(self):
        for m in self.modules():
            if isinstance(m, nn.Conv2d):
                nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
            elif isinstance(m, nn.BatchNorm2d):
                nn.init.ones_(m.weight)
                nn.init.zeros_(m.bias)
            elif isinstance(m, nn.Linear):
                nn.init.xavier_normal_(m.weight)
                nn.init.zeros_(m.bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.stem(x)     # (B, 64, 32, 32)
        x = self.stage1(x)   # (B, 64, 16, 16)
        x = self.stage2(x)   # (B, 128, 8, 8)
        x = self.stage3(x)   # (B, 256, 4, 4)
        return self.head(x)  # (B, 10)


# ── Instantiate and verify ────────────────────────────────────────────────────
model  = CIFAR10Net(num_classes=10)
device = 'cuda' if torch.cuda.is_available() else 'cpu'
model  = model.to(device)

# Shape smoke test
x   = torch.randn(4, 3, 32, 32).to(device)
out = model(x)
print(f'Input:  {x.shape}')       # torch.Size([4, 3, 32, 32])
print(f'Output: {out.shape}')     # torch.Size([4, 10])

# Parameter count
total = sum(p.numel() for p in model.parameters())
print(f'Parameters: {total:,}')   # ~1,200,000

# Trainable vs frozen
frozen = sum(p.numel() for p in model.parameters() if not p.requires_grad)
print(f'Frozen: {frozen:,}')      # 0
```

import torchvision.models as models
resnet = models.resnet50(weights='IMAGENET1K_V2')

    p.requires_grad = False          # freeze backbone

The App Router uses the file system to define routes and their behavior. Each file in the `app/` directory serves a specific purpose.

## File Conventions

| File | Purpose |
|---|---|
| `layout.tsx` | Shared UI for a segment and its children; persistent across navigations |
| `page.tsx` | Unique UI for the route; makes the segment publicly accessible |
| `loading.tsx` | Suspense boundary shown while the segment loads |
| `error.tsx` | Error boundary for unexpected errors |
| `not-found.tsx` | UI shown when `notFound()` is called |
| `template.tsx` | Like layout but re-mounted on each navigation |
| `route.ts` | API endpoint (HTTP handler); no UI |
| `default.tsx` | Fallback UI for parallel routes |

## layout.tsx

Layouts wrap pages and are **not** re-rendered when navigating between pages that share the same layout.

```tsx
// app/layout.tsx — Root layout (required)
import type { Metadata } from 'next';
import { Inter } from 'next/font/google';
import { Header } from '@/components/layout/Header';
import { Footer } from '@/components/layout/Footer';
import { Providers } from '@/components/Providers';
import './globals.css';

const inter = Inter({ subsets: ['latin'] });

export const metadata: Metadata = {
  title: { default: 'My App', template: '%s | My App' },
  description: 'My application description',
};

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body className={inter.className}>
        <Providers>
          <Header />
          <main>{children}</main>
          <Footer />
        </Providers>
      </body>
    </html>
  );
}
```

```tsx
// app/dashboard/layout.tsx — Nested layout (only for /dashboard/*)
import { Sidebar } from '@/components/layout/Sidebar';
import { requireAuth } from '@/lib/auth';

export default async function DashboardLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  await requireAuth(); // Protect all dashboard routes server-side

  return (
    <div className="flex h-screen">
      <Sidebar />
      <div className="flex-1 overflow-auto">
        {children}
      </div>
    </div>
  );
}
```

## page.tsx

Pages receive `params` (dynamic segments) and `searchParams` (query string) as props.

```tsx
// app/page.tsx — Home route (/)
export default function HomePage() {
  return <h1>Home</h1>;
}

// app/blog/[slug]/page.tsx — Dynamic route
interface BlogPageProps {
  params: Promise<{ slug: string }>;
  searchParams: Promise<{ page?: string }>;
}

export default async function BlogPage({ params, searchParams }: BlogPageProps) {
  const { slug } = await params;
  const { page = '1' } = await searchParams;

  const post = await fetchPost(slug);

  return (
    <article>
      <h1>{post.title}</h1>
      <p>Page: {page}</p>
    </article>
  );
}
```

:::tip[params is a Promise in Next.js 15]
In Next.js 15, `params` and `searchParams` are Promises that must be awaited. Always `await` them in new code to be forward-compatible.
:::

## loading.tsx

A `loading.tsx` file automatically wraps the page in a React Suspense boundary. It shows instantly while the page's async data fetches.

```tsx
// app/dashboard/loading.tsx
export default function DashboardLoading() {
  return (
    <div className="animate-pulse space-y-4 p-6">
      <div className="h-8 bg-gray-200 rounded w-1/4" />
      <div className="grid grid-cols-3 gap-4">
        {Array.from({ length: 3 }).map((_, i) => (
          <div key={i} className="h-32 bg-gray-200 rounded-lg" />
        ))}
      </div>
      <div className="space-y-2">
        {Array.from({ length: 5 }).map((_, i) => (
          <div key={i} className="h-16 bg-gray-200 rounded" />
        ))}
      </div>
    </div>
  );
}
```

## error.tsx

Error boundaries catch JavaScript errors during rendering. They **must** be Client Components.

```tsx
// app/dashboard/error.tsx
'use client';

import { useEffect } from 'react';

interface ErrorProps {
  error: Error & { digest?: string };
  reset: () => void;
}

export default function DashboardError({ error, reset }: ErrorProps) {
  useEffect(() => {
    console.error('Dashboard error:', error);
  }, [error]);

  return (
    <div className="flex flex-col items-center justify-center min-h-[400px] gap-4">
      <h2 className="text-xl font-semibold text-gray-900">Something went wrong</h2>
      <p className="text-gray-500 text-sm">{error.message}</p>
      {error.digest && (
        <p className="text-xs text-gray-400">Error ID: {error.digest}</p>
      )}
      <button
        onClick={reset}
        className="px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700"
      >
        Try again
      </button>
    </div>
  );
}
```

## not-found.tsx

```tsx
// app/not-found.tsx — Global 404 page
import Link from 'next/link';

export default function NotFound() {
  return (
    <div className="flex flex-col items-center justify-center min-h-screen gap-4">
      <h1 className="text-6xl font-bold text-gray-200">404</h1>
      <h2 className="text-2xl font-semibold">Page not found</h2>
      <p className="text-gray-500">The page you are looking for does not exist.</p>
      <Link href="/" className="mt-4 text-blue-600 hover:underline">
        Go home
      </Link>
    </div>
  );
}
```

```tsx
// Trigger 404 from a page
import { notFound } from 'next/navigation';

export default async function PostPage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = await params;
  const post = await fetchPost(id);
  if (!post) notFound(); // Renders the nearest not-found.tsx
  return <article>{post.title}</article>;
}
```

## template.tsx

Unlike layouts, templates create a **new instance** for each navigation. Use for per-page animations or analytics.

```tsx
// app/template.tsx
'use client';

import { motion } from 'framer-motion';

export default function Template({ children }: { children: React.ReactNode }) {
  return (
    <motion.div
      initial={{ opacity: 0, y: 8 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.2 }}
    >
      {children}
    </motion.div>
  );
}
```

## Route Groups

Route groups use `(folderName)` syntax. They do **not** affect the URL but let you organize files and share different layouts.

```
app/
├── (marketing)/           # No URL segment
│   ├── layout.tsx         # Marketing layout (no sidebar)
│   ├── page.tsx           # / (home)
│   ├── about/
│   │   └── page.tsx       # /about
│   └── pricing/
│       └── page.tsx       # /pricing
│
└── (app)/                 # No URL segment
    ├── layout.tsx          # App layout (sidebar + auth)
    ├── dashboard/
    │   └── page.tsx       # /dashboard
    └── settings/
        └── page.tsx       # /settings
```

:::tip[Multiple Root Layouts]
You can have multiple root layouts using route groups. Each group can have its own `layout.tsx` with a different `<html>` and `<body>` — useful for having a marketing site and app with completely different structures.
:::

## Private Folders

Prefix with `_` to exclude from routing entirely:

```
app/
├── _components/        # Shared UI (not a route)
├── _hooks/             # Custom hooks (not a route)
├── dashboard/
│   ├── _tests/         # Co-located tests (not a route)
│   └── page.tsx
```

## Parallel Routes

Render multiple pages in the same layout simultaneously using `@slot` folders:

```
app/
├── layout.tsx           # Receives @team and @analytics as props
├── page.tsx
├── @team/
│   └── page.tsx         # Rendered in the @team slot
└── @analytics/
    └── page.tsx         # Rendered in the @analytics slot
```

```tsx
// app/layout.tsx
export default function Layout({
  children,
  team,
  analytics,
}: {
  children: React.ReactNode;
  team: React.ReactNode;
  analytics: React.ReactNode;
}) {
  return (
    <div>
      {children}
      <div className="grid grid-cols-2 gap-4">
        {team}
        {analytics}
      </div>
    </div>
  );
}
```

## Intercepting Routes

Show a route in a modal without losing the current context. The URL updates so the modal is shareable/bookmarkable.

```
app/
├── feed/
│   └── page.tsx          # Full feed list
├── photo/
│   └── [id]/
│       └── page.tsx      # Full photo page (direct navigation)
└── @modal/
    ├── default.tsx        # null — no modal by default
    └── (.)photo/
        └── [id]/
            └── page.tsx  # Intercepted: photo in modal when from feed
```

```tsx
// app/@modal/default.tsx
export default function Default() {
  return null;  // No modal by default
}

// app/@modal/(.)photo/[id]/page.tsx
import { Modal } from '@/components/Modal';

export default async function PhotoModal({
  params,
}: {
  params: Promise<{ id: string }>;
}) {
  const { id } = await params;
  const photo = await fetchPhoto(id);

  return (
    <Modal>
      <img src={photo.url} alt={photo.title} className="max-w-2xl rounded-lg" />
      <h2 className="mt-4 text-xl font-semibold">{photo.title}</h2>
    </Modal>
  );
}
```

## Metadata API

### Static Metadata

```tsx
// app/about/page.tsx
import type { Metadata } from 'next';

export const metadata: Metadata = {
  title: 'About Us',
  description: 'Learn about our team and mission.',
  openGraph: {
    title: 'About Us',
    description: 'Learn about our team and mission.',
    images: [{ url: '/og/about.png', width: 1200, height: 630 }],
  },
  twitter: {
    card: 'summary_large_image',
    title: 'About Us',
    images: ['/og/about.png'],
  },
  keywords: ['about', 'team', 'mission'],
  alternates: {
    canonical: 'https://myapp.com/about',
  },
};
```

### Dynamic Metadata with generateMetadata

```tsx
// app/blog/[slug]/page.tsx
import type { Metadata, ResolvingMetadata } from 'next';

interface Props {
  params: Promise<{ slug: string }>;
}

export async function generateMetadata(
  { params }: Props,
  parent: ResolvingMetadata
): Promise<Metadata> {
  const { slug } = await params;
  const post = await fetchPost(slug);

  const parentImages = (await parent).openGraph?.images ?? [];

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      title: post.title,
      description: post.excerpt,
      type: 'article',
      publishedTime: post.publishedAt,
      authors: [post.author.name],
      images: [
        {
          url: post.coverImage,
          width: 1200,
          height: 630,
          alt: post.title,
        },
        ...parentImages,
      ],
    },
  };
}
```

## generateStaticParams

Pre-generate static pages at build time:

```tsx
// app/blog/[slug]/page.tsx
export async function generateStaticParams() {
  const posts = await fetchAllPosts();
  return posts.map(post => ({ slug: post.slug }));
}

// Nested dynamic segments
// app/[lang]/blog/[slug]/page.tsx
export async function generateStaticParams() {
  const languages = ['en', 'fr', 'de'];
  const posts = await fetchAllPosts();

  return languages.flatMap(lang =>
    posts.map(post => ({ lang, slug: post.slug }))
  );
}

// Return 404 for unknown paths instead of dynamic rendering
export const dynamicParams = false;
```

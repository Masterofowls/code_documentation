The App Router, introduced in Next.js 13 and the current standard, uses a file-system convention where **special files** define the UI for each segment of a route. This page covers every convention and the complete metadata API.

## File Conventions Overview

Every folder in `app/` can contain these special files:

| File            | Description                                                      |
| --------------- | ---------------------------------------------------------------- |
| `page.tsx`      | Unique UI for a route — makes the segment publicly accessible    |
| `layout.tsx`    | Shared UI that wraps child routes and persists across navigation |
| `loading.tsx`   | Instant loading UI using React Suspense                          |
| `error.tsx`     | Error boundary for the segment and its children                  |
| `not-found.tsx` | UI rendered when `notFound()` is called                          |
| `template.tsx`  | Like layout, but creates a new instance on navigation            |
| `route.ts`      | API endpoint (no UI)                                             |
| `default.tsx`   | Fallback UI for parallel routes                                  |

## Root Layout

The root layout is the top-most layout and **must** define the `<html>` and `<body>` tags:

```tsx
// app/layout.tsx
import type { Metadata } from 'next';
import { Inter } from 'next/font/google';
import './globals.css';

const inter = Inter({
  subsets: ['latin'],
  variable: '--font-inter',
  display: 'swap',
});

export const metadata: Metadata = {
  metadataBase: new URL('https://example.com'),
  title: {
    default: 'My App',
    template: '%s | My App',
  },
  description: 'The best app on the internet.',
  openGraph: {
    title: 'My App',
    description: 'The best app on the internet.',
    url: 'https://example.com',
    siteName: 'My App',
    images: [
      {
        url: '/og-image.png',
        width: 1200,
        height: 630,
        alt: 'My App',
      },
    ],
    type: 'website',
  },
  twitter: {
    card: 'summary_large_image',
    title: 'My App',
    description: 'The best app on the internet.',
    images: ['/og-image.png'],
  },
  robots: {
    index: true,
    follow: true,
  },
};

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en" className={inter.variable}>
      <body className="min-h-screen bg-background font-sans antialiased">
        <Header />
        <main>{children}</main>
        <Footer />
      </body>
    </html>
  );
}
```

## page.tsx

A `page.tsx` file makes a route segment publicly accessible and can receive `params` and `searchParams` as props:

```tsx
// app/blog/[slug]/page.tsx
import type { Metadata } from 'next';
import { notFound } from 'next/navigation';
import { getPost } from '@/lib/posts';

interface PageProps {
  params: Promise<{ slug: string }>;
  searchParams: Promise<{ [key: string]: string | string[] | undefined }>;
}

// Generate metadata dynamically based on route params
export async function generateMetadata({
  params,
}: PageProps): Promise<Metadata> {
  const { slug } = await params;
  const post = await getPost(slug);

  if (!post) {
    return { title: 'Post Not Found' };
  }

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      title: post.title,
      description: post.excerpt,
      type: 'article',
      publishedTime: post.publishedAt,
      authors: [post.author.name],
    },
  };
}

export default async function PostPage({ params, searchParams }: PageProps) {
  const { slug } = await params;
  const { preview } = await searchParams;

  const post = await getPost(slug, { preview: preview === 'true' });

  if (!post) {
    notFound(); // Renders not-found.tsx
  }

  return (
    <article className="prose max-w-2xl mx-auto">
      <h1>{post.title}</h1>
      <time>{new Date(post.publishedAt).toLocaleDateString()}</time>
      <div dangerouslySetInnerHTML={{ __html: post.content }} />
    </article>
  );
}
```

:::tip[Params are Now Promises]
In Next.js 15, `params` and `searchParams` are now **async Promises**. Always `await` them before accessing their values. This change enables more performance optimisations in the framework.
:::

## Nested Layouts

Layouts wrap their children and persist across navigations within their segment. Nesting layouts creates a hierarchy:

```tsx
// app/blog/layout.tsx — wraps all /blog/* pages
import { Suspense } from 'react';
import { BlogSidebar } from '@/components/BlogSidebar';
import { RecentPostsSkeleton } from '@/components/skeletons';

export default function BlogLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <div className="flex gap-8 max-w-6xl mx-auto px-4 py-8">
      <div className="flex-1 min-w-0">{children}</div>
      <aside className="w-72 shrink-0 hidden lg:block">
        {/* Sidebar loads independently with its own Suspense boundary */}
        <Suspense fallback={<RecentPostsSkeleton />}>
          <BlogSidebar />
        </Suspense>
      </aside>
    </div>
  );
}
```

## loading.tsx

`loading.tsx` wraps the page in a React `Suspense` boundary automatically. It shows instantly while the page loads:

```tsx
// app/blog/loading.tsx
export default function BlogLoading() {
  return (
    <div className="space-y-6">
      {Array.from({ length: 5 }).map((_, i) => (
        <div key={i} className="animate-pulse">
          <div className="h-6 bg-gray-200 rounded w-3/4 mb-2" />
          <div className="h-4 bg-gray-200 rounded w-full mb-1" />
          <div className="h-4 bg-gray-200 rounded w-5/6" />
        </div>
      ))}
    </div>
  );
}
```

## error.tsx

`error.tsx` must be a Client Component and acts as an error boundary for the segment:

```tsx
// app/blog/error.tsx
'use client';

import { useEffect } from 'react';

interface ErrorProps {
  error: Error & { digest?: string };
  reset: () => void;
}

export default function BlogError({ error, reset }: ErrorProps) {
  useEffect(() => {
    // Log the error to an error reporting service
    console.error(error);
  }, [error]);

  return (
    <div className="flex flex-col items-center justify-center min-h-64 gap-4">
      <h2 className="text-xl font-semibold">Something went wrong!</h2>
      <p className="text-gray-500 text-sm">{error.message}</p>
      <button
        onClick={reset}
        className="px-4 py-2 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700"
      >
        Try again
      </button>
    </div>
  );
}
```

## not-found.tsx

```tsx
// app/not-found.tsx — global 404 page
import Link from 'next/link';

export default function NotFound() {
  return (
    <div className="flex flex-col items-center justify-center min-h-screen gap-4">
      <h1 className="text-6xl font-bold text-gray-200">404</h1>
      <h2 className="text-2xl font-semibold">Page Not Found</h2>
      <p className="text-gray-500">
        The page you are looking for does not exist.
      </p>
      <Link
        href="/"
        className="px-6 py-3 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700"
      >
        Go Home
      </Link>
    </div>
  );
}
```

## template.tsx

`template.tsx` is similar to `layout.tsx`, but it creates a **new instance** on every navigation. This means state is not preserved and effects re-run. Use it when you need this behaviour (e.g., page transition animations):

```tsx
// app/template.tsx
'use client';

import { motion } from 'framer-motion';

export default function Template({ children }: { children: React.ReactNode }) {
  return (
    <motion.div
      initial={{ opacity: 0, y: 8 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -8 }}
      transition={{ duration: 0.2 }}
    >
      {children}
    </motion.div>
  );
}
```

## Route Groups

Route groups `(folderName)` organise routes without affecting the URL structure:

```
app/
├── (marketing)/       # Route group — no URL impact
│   ├── layout.tsx     # Shared layout for marketing pages only
│   ├── about/
│   │   └── page.tsx   # /about
│   └── pricing/
│       └── page.tsx   # /pricing
├── (auth)/
│   ├── layout.tsx     # Auth-specific layout (centered form)
│   ├── login/
│   │   └── page.tsx   # /login
│   └── signup/
│       └── page.tsx   # /signup
└── (app)/
    ├── layout.tsx     # App layout (sidebar + header)
    ├── dashboard/
    │   └── page.tsx   # /dashboard
    └── settings/
        └── page.tsx   # /settings
```

## Private Folders

Folders prefixed with `_` are excluded from routing entirely:

```
app/
├── _components/       # Not a route — colocated components
├── _lib/              # Not a route — colocated utilities
└── blog/
    ├── _utils.ts      # Not a route — blog-specific utility
    └── page.tsx       # /blog
```

## Parallel Routes

Parallel routes render multiple pages simultaneously in the same layout using named slots (`@folder`):

```
app/
├── layout.tsx          # Receives @team and @analytics as separate props
├── page.tsx
├── @team/
│   └── page.tsx        # Renders in the "team" slot
└── @analytics/
    └── page.tsx        # Renders in the "analytics" slot
```

```tsx
// app/layout.tsx — dashboard with parallel routes
export default function DashboardLayout({
  children,
  team,
  analytics,
}: {
  children: React.ReactNode;
  team: React.ReactNode;
  analytics: React.ReactNode;
}) {
  return (
    <div>
      <div>{children}</div>
      <div className="grid grid-cols-2 gap-4">
        <div>{team}</div>
        <div>{analytics}</div>
      </div>
    </div>
  );
}
```

## Intercepting Routes

Intercepting routes render a route inside the current layout (e.g., an image modal) while navigating to the full route on direct access:

```
app/
├── photos/
│   ├── page.tsx         # /photos — gallery grid
│   └── [id]/
│       └── page.tsx     # /photos/123 — full photo page
└── @modal/
    └── (.)photos/       # (.) intercepts from same level
        └── [id]/
            └── page.tsx # Shows as modal when navigated from gallery
```

## Metadata API

### Static Metadata

```tsx
// app/about/page.tsx
import type { Metadata } from 'next';

export const metadata: Metadata = {
  title: 'About Us', // Becomes "About Us | My App" with title template
  description: 'Learn about our company and mission.',
  keywords: ['company', 'about', 'team'],
  authors: [{ name: 'My Team' }],
  openGraph: {
    title: 'About Us | My App',
    description: 'Learn about our company and mission.',
    images: '/about-og.png',
  },
  alternates: {
    canonical: 'https://example.com/about',
  },
};
```

### Dynamic Metadata

```tsx
// app/products/[id]/page.tsx
export async function generateMetadata({
  params,
}: {
  params: Promise<{ id: string }>;
}): Promise<Metadata> {
  const { id } = await params;
  const product = await getProduct(id);

  return {
    title: product.name,
    description: product.description,
    openGraph: {
      images: [{ url: product.imageUrl, width: 800, height: 600 }],
    },
    other: {
      'product:price:amount': product.price.toString(),
      'product:price:currency': 'USD',
    },
  };
}
```

## generateStaticParams

Pre-generate dynamic routes at build time for static generation:

```tsx
// app/blog/[slug]/page.tsx

// Tell Next.js which slugs to pre-render
export async function generateStaticParams() {
  const posts = await getAllPosts();

  return posts.map((post) => ({
    slug: post.slug,
  }));
}

// Optional: control what happens for slugs not in generateStaticParams
// 'force-static' | 'dynamic' | 'error'
export const dynamicParams = true; // default — renders unknown slugs on-demand
```

For nested dynamic segments:

```tsx
// app/[lang]/blog/[slug]/page.tsx
export async function generateStaticParams() {
  const languages = ['en', 'fr', 'de'];
  const posts = await getAllPosts();

  // Generate all combinations
  return languages.flatMap((lang) =>
    posts.map((post) => ({
      lang,
      slug: post.slug,
    })),
  );
}
```

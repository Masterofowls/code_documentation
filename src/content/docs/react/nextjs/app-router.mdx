The App Router is Next.js's modern routing system, built on React Server Components. It uses a file-system-based convention where special files define the UI for each route segment.

## File Conventions

Every folder in `app/` that contains a `page.tsx` becomes a route. Special filenames have specific roles:

| File | Purpose |
|---|---|
| `layout.tsx` | Shared UI that wraps child pages and persists across navigation |
| `page.tsx` | The unique UI for a route (makes the route publicly accessible) |
| `loading.tsx` | Instant loading state shown while a page or layout is loading |
| `error.tsx` | Error boundary for a route segment and its children |
| `not-found.tsx` | UI shown when `notFound()` is called |
| `template.tsx` | Like layout, but creates a new instance on navigation (resets state) |
| `route.ts` | API endpoint for the route (no UI, returns data) |
| `default.tsx` | Fallback for parallel routes when no active state exists |

```
app/
├── layout.tsx          # Root layout — wraps EVERYTHING
├── page.tsx            # Home page (/)
├── loading.tsx         # Root loading state
├── error.tsx           # Root error boundary
├── not-found.tsx       # 404 page
├── blog/
│   ├── layout.tsx      # Blog section layout
│   ├── page.tsx        # /blog
│   └── [slug]/
│       ├── page.tsx    # /blog/my-post
│       └── loading.tsx # Loading state for individual posts
└── dashboard/
    ├── layout.tsx      # Dashboard layout (sidebar, nav)
    └── page.tsx        # /dashboard
```

## Root Layout

The root layout is required and must include `<html>` and `<body>` tags. It's the outermost shared UI.

```tsx
// app/layout.tsx
import type { Metadata } from 'next';
import { Inter } from 'next/font/google';
import './globals.css';

const inter = Inter({
  subsets: ['latin'],
  variable: '--font-inter',
  display: 'swap',
});

export const metadata: Metadata = {
  metadataBase: new URL('https://myapp.com'),
  title: {
    default: 'MyApp — Build something great',
    template: '%s | MyApp',
  },
  description: 'The best app for doing great things.',
  openGraph: {
    type: 'website',
    locale: 'en_US',
    url: 'https://myapp.com',
    siteName: 'MyApp',
    images: [{ url: '/og-image.png', width: 1200, height: 630 }],
  },
  twitter: {
    card: 'summary_large_image',
    site: '@myapp',
  },
  robots: {
    index: true,
    follow: true,
  },
};

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en" className={inter.variable}>
      <body className="bg-white text-gray-900 antialiased">
        {children}
      </body>
    </html>
  );
}
```

## Nested Layouts

Layouts wrap their child segments. The blog layout only applies to `/blog/**` routes:

```tsx
// app/blog/layout.tsx
export default function BlogLayout({ children }: { children: React.ReactNode }) {
  return (
    <div className="max-w-4xl mx-auto px-4 py-8">
      <header className="mb-8">
        <h1 className="text-4xl font-bold">The Blog</h1>
        <nav>
          <a href="/blog">All Posts</a>
          <a href="/blog/categories">Categories</a>
        </nav>
      </header>
      <main>{children}</main>
    </div>
  );
}
```

:::tip[Layout Persistence]
Layouts do not unmount during navigation between child routes. This means state (scroll position, form values, fetch results) persists as users navigate between `/blog` and `/blog/my-post`. Use `template.tsx` if you need a fresh mount on each navigation.
:::

## Page Component

```tsx
// app/blog/[slug]/page.tsx
import { notFound } from 'next/navigation';
import type { Metadata } from 'next';

interface Props {
  params: Promise<{ slug: string }>;
  searchParams: Promise<{ preview?: string }>;
}

// Dynamic metadata for this page
export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const { slug } = await params;
  const post = await getPost(slug);

  if (!post) return { title: 'Post Not Found' };

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      title: post.title,
      description: post.excerpt,
      images: [{ url: post.coverImage, width: 1200, height: 630 }],
      type: 'article',
      publishedTime: post.publishedAt,
    },
  };
}

// Pre-render known slugs at build time
export async function generateStaticParams() {
  const posts = await getAllPosts();
  return posts.map(post => ({ slug: post.slug }));
}

export default async function BlogPostPage({ params, searchParams }: Props) {
  const { slug } = await params;
  const { preview } = await searchParams;

  const post = await getPost(slug);
  if (!post) notFound();

  return (
    <article>
      <header>
        <h1 className="text-3xl font-bold">{post.title}</h1>
        <time dateTime={post.publishedAt}>{formatDate(post.publishedAt)}</time>
      </header>
      <div dangerouslySetInnerHTML={{ __html: post.contentHtml }} />
    </article>
  );
}
```

## Loading UI

`loading.tsx` uses React Suspense under the hood. It shows immediately while the page's async work completes — no client-side JavaScript required.

```tsx
// app/blog/loading.tsx
export default function BlogLoading() {
  return (
    <div className="space-y-6">
      {[...Array(3)].map((_, i) => (
        <div key={i} className="animate-pulse">
          <div className="h-6 bg-gray-200 rounded w-3/4 mb-2" />
          <div className="h-4 bg-gray-100 rounded w-1/2 mb-1" />
          <div className="h-4 bg-gray-100 rounded w-2/3" />
        </div>
      ))}
    </div>
  );
}
```

## Error Boundary

`error.tsx` must be a Client Component. It catches errors thrown in the route segment and its children.

```tsx
// app/blog/error.tsx
'use client';

import { useEffect } from 'react';

interface ErrorProps {
  error: Error & { digest?: string };
  reset: () => void;
}

export default function BlogError({ error, reset }: ErrorProps) {
  useEffect(() => {
    // Log to error reporting service
    console.error('[Blog Error]', error);
  }, [error]);

  return (
    <div className="text-center py-16">
      <h2 className="text-2xl font-bold text-gray-900 mb-4">Something went wrong</h2>
      <p className="text-gray-600 mb-8">
        {error.message || 'An unexpected error occurred.'}
      </p>
      <button
        onClick={reset}
        className="px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700"
      >
        Try again
      </button>
    </div>
  );
}
```

## Not Found Page

```tsx
// app/not-found.tsx
import Link from 'next/link';

export default function NotFoundPage() {
  return (
    <div className="min-h-screen flex flex-col items-center justify-center">
      <h1 className="text-6xl font-bold text-gray-200 mb-4">404</h1>
      <h2 className="text-2xl font-semibold text-gray-900 mb-2">Page Not Found</h2>
      <p className="text-gray-500 mb-8">
        The page you are looking for does not exist.
      </p>
      <Link href="/" className="text-blue-600 hover:underline">
        Go back home
      </Link>
    </div>
  );
}
```

Trigger it programmatically in a page or layout:

```tsx
import { notFound } from 'next/navigation';

export default async function ProductPage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = await params;
  const product = await db.product.findUnique({ where: { id } });

  if (!product) notFound(); // renders app/not-found.tsx (or nearest not-found.tsx)

  return <Product product={product} />;
}
```

## Route Groups

Wrap a folder name in parentheses to create a **route group** — it organizes routes without affecting the URL structure.

```
app/
├── (auth)/              # Group: no URL segment
│   ├── layout.tsx       # Shared auth layout (centered card)
│   ├── login/
│   │   └── page.tsx     # /login
│   └── register/
│       └── page.tsx     # /register
├── (marketing)/         # Group: marketing pages with full-width layout
│   ├── layout.tsx
│   ├── page.tsx         # /
│   ├── about/
│   │   └── page.tsx     # /about
│   └── pricing/
│       └── page.tsx     # /pricing
└── (app)/               # Group: authenticated app with sidebar
    ├── layout.tsx
    └── dashboard/
        └── page.tsx     # /dashboard
```

## Private Folders

Prefix a folder with `_` to exclude it from routing. Use for component files, utilities, or test fixtures that live inside the `app/` directory:

```
app/
├── blog/
│   ├── page.tsx
│   └── _components/        # Not a route — private to blog feature
│       ├── PostCard.tsx
│       └── PostList.tsx
├── _lib/                   # Shared utilities — not a route
│   └── formatDate.ts
└── page.tsx
```

## Parallel Routes

Parallel routes render multiple pages in the same layout simultaneously, using `@slot` syntax:

```
app/
├── layout.tsx               # Has @analytics and @team slots
├── page.tsx                 # Default page
├── @analytics/
│   ├── page.tsx             # Analytics panel
│   └── default.tsx          # Default (when not active)
└── @team/
    ├── page.tsx             # Team panel
    └── default.tsx
```

```tsx
// app/layout.tsx
export default function DashboardLayout({
  children,
  analytics,
  team,
}: {
  children: React.ReactNode;
  analytics: React.ReactNode;
  team: React.ReactNode;
}) {
  return (
    <div className="grid grid-cols-12 gap-4">
      <main className="col-span-8">{children}</main>
      <aside className="col-span-4 space-y-4">
        {analytics}
        {team}
      </aside>
    </div>
  );
}
```

:::tip[Parallel Routes for Modals]
Parallel routes combined with intercepting routes enable URL-addressable modals — the modal opens as an overlay but has its own URL, and refreshing shows the full page at that URL.
:::

## Intercepting Routes

Intercepting routes let you load a route within the current context while keeping the current page visible — perfect for modals.

```
app/
├── feed/
│   └── page.tsx                   # /feed — the main feed
├── photo/
│   └── [id]/
│       └── page.tsx               # /photo/123 — full photo page
└── @modal/
    ├── default.tsx                # null — no modal by default
    └── (.)photo/                  # Intercept /photo routes from same level
        └── [id]/
            └── page.tsx           # Renders as modal over /feed
```

Interception syntax:
- `(.)` — same level
- `(..)` — one level up
- `(..)(..)` — two levels up
- `(...)` — from app root

## Metadata API

Next.js provides a first-class Metadata API for `<head>` tags:

```tsx
// Static metadata
export const metadata: Metadata = {
  title: 'My Page',
  description: 'Page description',
};

// Dynamic metadata (async, access params/data)
export async function generateMetadata({
  params,
}: {
  params: Promise<{ slug: string }>;
}): Promise<Metadata> {
  const { slug } = await params;
  const post = await getPost(slug);

  return {
    title: post.title,
    description: post.excerpt,
    alternates: {
      canonical: `/blog/${slug}`,
    },
    openGraph: {
      title: post.title,
      images: [post.coverImage],
    },
  };
}
```

### Title Template

```tsx
// Root layout sets the template
export const metadata: Metadata = {
  title: {
    default: 'MyApp',          // Shown when no child sets title
    template: '%s | MyApp',    // %s = child's title
    absolute: 'Override All',  // Ignores template entirely
  },
};

// Child page — title becomes "About Us | MyApp"
export const metadata: Metadata = {
  title: 'About Us',
};

// Child page using absolute — title is just "Our Blog" (no template)
export const metadata: Metadata = {
  title: { absolute: 'Our Blog' },
};
```

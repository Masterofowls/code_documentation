Next.js extends the native `fetch` API with caching and revalidation options, and React Server Components enable data fetching directly in the component tree — no `useEffect`, no API roundtrip from the browser.

## fetch in Server Components

Server Components are `async` by default. You can `await` any async call directly in the component body.

```tsx
// app/posts/page.tsx — Server Component (no 'use client')
interface Post {
  id: number;
  title: string;
  body: string;
  userId: number;
}

async function getPosts(): Promise<Post[]> {
  const res = await fetch('https://jsonplaceholder.typicode.com/posts', {
    next: { revalidate: 3600 }, // ISR: regenerate at most once per hour
  });

  if (!res.ok) {
    throw new Error(`Failed to fetch posts: ${res.status}`);
  }

  return res.json();
}

export default async function PostsPage() {
  const posts = await getPosts();

  return (
    <ul>
      {posts.slice(0, 10).map(post => (
        <li key={post.id}>
          <h2>{post.title}</h2>
          <p>{post.body}</p>
        </li>
      ))}
    </ul>
  );
}
```

## Cache Options

The `cache` and `next` options on `fetch` control how responses are cached:

```typescript
// SSG — cache forever, regenerate at build time
const data = await fetch('/api/data', {
  cache: 'force-cache', // default behavior
});

// SSR — never cache, always fresh
const data = await fetch('/api/data', {
  cache: 'no-store',
});

// ISR — cache and revalidate every N seconds (time-based)
const data = await fetch('/api/data', {
  next: { revalidate: 60 }, // regenerate after 60 seconds
});

// Tag-based revalidation (on-demand)
const data = await fetch('/api/data', {
  next: {
    revalidate: 3600,
    tags: ['products', 'product-123'],
  },
});
```

### On-Demand Revalidation

```typescript
// app/api/revalidate/route.ts
import { revalidateTag, revalidatePath } from 'next/cache';
import { NextRequest, NextResponse } from 'next/server';

export async function POST(request: NextRequest) {
  const secret = request.headers.get('x-revalidation-secret');

  if (secret !== process.env.REVALIDATION_SECRET) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const body = await request.json();

  if (body.tag) {
    revalidateTag(body.tag);           // Revalidate all fetch() with this tag
  }

  if (body.path) {
    revalidatePath(body.path, 'page'); // Revalidate a specific page
  }

  return NextResponse.json({ revalidated: true, now: Date.now() });
}
```

Call this from your CMS webhook:
```bash
curl -X POST https://myapp.com/api/revalidate \
  -H "x-revalidation-secret: your-secret" \
  -H "Content-Type: application/json" \
  -d '{"tag": "posts"}'
```

## React cache() for Deduplication

When multiple components in the same render tree fetch the same data, `cache()` ensures the request is made only once.

```typescript
// src/lib/data.ts
import { cache } from 'react';
import { db } from './db';

// This function is deduped per request — even if called from 10 components,
// the database query only runs once
export const getUser = cache(async (userId: string) => {
  return db.user.findUnique({
    where: { id: userId },
    include: { profile: true },
  });
});

export const getPost = cache(async (postId: string) => {
  return db.post.findUnique({
    where: { id: postId },
    include: { author: true, tags: true },
  });
});
```

```tsx
// Used in multiple Server Components in the same render — no duplicate queries:
// app/posts/[id]/page.tsx
export default async function PostPage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = await params;
  const post = await getPost(id);         // Query 1 — DB hit
  const author = await getUser(post!.authorId); // Query 2 — DB hit
  return <PostView post={post!} author={author!} />;
}

// app/posts/[id]/layout.tsx — also calls getPost(id):
export default async function PostLayout({ params, children }: LayoutProps) {
  const { id } = await params;
  const post = await getPost(id);        // Returns CACHED result — no extra DB query
  return (
    <>
      <PostHeader post={post!} />
      {children}
    </>
  );
}
```

:::tip[cache() vs fetch() Deduplication]
`fetch()` requests with the same URL are automatically deduplicated within a render. Use `React.cache()` for database queries, ORM calls, or any async function that isn't a `fetch()`.
:::

## Parallel Data Fetching

Fetch multiple independent resources simultaneously with `Promise.all`:

```tsx
// app/dashboard/page.tsx
async function getDashboardData() {
  // Sequential — slow (each waits for the previous)
  const user = await getUser();
  const stats = await getStats();       // waits for getUser
  const notifications = await getNotifications(); // waits for getStats

  // Parallel — fast (all start simultaneously)
  const [userP, statsP, notificationsP] = await Promise.all([
    getUser(),
    getStats(),
    getNotifications(),
  ]);

  return { user: userP, stats: statsP, notifications: notificationsP };
}

export default async function DashboardPage() {
  const { user, stats, notifications } = await getDashboardData();
  return (
    <div>
      <UserGreeting user={user} />
      <StatsGrid stats={stats} />
      <NotificationsList notifications={notifications} />
    </div>
  );
}
```

## Suspense Boundaries for Streaming

Wrap slow components in `Suspense` to stream content progressively — fast parts load immediately, slow parts stream in as they complete.

```tsx
// app/dashboard/page.tsx
import { Suspense } from 'react';

// This component fetches data independently
async function RevenueChart() {
  const data = await getRevenueData(); // Slow — 1.5s
  return <Chart data={data} />;
}

async function LatestInvoices() {
  const invoices = await getLatestInvoices(); // Medium — 500ms
  return <InvoiceList invoices={invoices} />;
}

async function UserCard() {
  const user = await getUser(); // Fast — 50ms
  return <UserProfile user={user} />;
}

// Skeleton loading components
function ChartSkeleton() {
  return <div className="h-64 bg-gray-100 rounded-xl animate-pulse" />;
}

function InvoiceSkeleton() {
  return (
    <div className="space-y-3">
      {[...Array(5)].map((_, i) => (
        <div key={i} className="h-12 bg-gray-100 rounded animate-pulse" />
      ))}
    </div>
  );
}

export default function DashboardPage() {
  return (
    <div className="space-y-6">
      {/* Fast component — renders immediately */}
      <Suspense fallback={<p>Loading user...</p>}>
        <UserCard />
      </Suspense>

      {/* Medium component */}
      <Suspense fallback={<InvoiceSkeleton />}>
        <LatestInvoices />
      </Suspense>

      {/* Slow component — renders last but doesn't block others */}
      <Suspense fallback={<ChartSkeleton />}>
        <RevenueChart />
      </Suspense>
    </div>
  );
}
```

## unstable_cache

For caching non-fetch async functions (database queries, expensive computations) with tag support:

```typescript
import { unstable_cache } from 'next/cache';
import { db } from '@/lib/db';

export const getCachedPosts = unstable_cache(
  async (category: string, limit: number) => {
    return db.post.findMany({
      where: { category, published: true },
      orderBy: { createdAt: 'desc' },
      take: limit,
      include: { author: { select: { name: true, avatarUrl: true } } },
    });
  },
  ['posts-by-category'],            // Cache key prefix
  {
    revalidate: 300,                // 5 minutes
    tags: ['posts'],                // revalidateTag('posts') clears this
  }
);

// Usage:
const posts = await getCachedPosts('tutorials', 10);
```

## Server Actions

Server Actions are async functions that run on the server, called from the client (forms, event handlers) or other Server Components.

```tsx
// app/actions/posts.ts
'use server';

import { revalidatePath, revalidateTag } from 'next/cache';
import { redirect } from 'next/navigation';
import { db } from '@/lib/db';
import { z } from 'zod';
import { getSession } from '@/lib/auth';

const createPostSchema = z.object({
  title: z.string().min(1).max(200),
  content: z.string().min(10),
  category: z.enum(['tech', 'design', 'business']),
});

export async function createPost(formData: FormData) {
  const session = await getSession();
  if (!session) throw new Error('Not authenticated');

  const parsed = createPostSchema.safeParse({
    title: formData.get('title'),
    content: formData.get('content'),
    category: formData.get('category'),
  });

  if (!parsed.success) {
    return { error: parsed.error.flatten() };
  }

  const post = await db.post.create({
    data: { ...parsed.data, authorId: session.userId },
  });

  revalidateTag('posts');
  revalidatePath('/blog');
  redirect(`/blog/${post.slug}`);
}

export async function deletePost(postId: string) {
  const session = await getSession();
  if (!session) throw new Error('Not authenticated');

  const post = await db.post.findUnique({ where: { id: postId } });
  if (!post || post.authorId !== session.userId) {
    throw new Error('Not authorized');
  }

  await db.post.delete({ where: { id: postId } });
  revalidateTag('posts');
  revalidatePath('/blog');
}
```

### Using Server Actions in Forms

```tsx
// app/blog/new/page.tsx
import { createPost } from '@/app/actions/posts';

export default function NewPostPage() {
  return (
    <form action={createPost}>
      <input name="title" placeholder="Post title" required />

        <option value="tech">Tech</option>

      </select>

```

### useActionState for Feedback

```tsx
'use client';
import { useActionState } from 'react';
import { createPost } from '@/app/actions/posts';

interface FormState {
  error?: { title?: string[]; content?: string[] };
  message?: string;
}

const initialState: FormState = {};

export function CreatePostForm() {
  const [state, formAction, isPending] = useActionState(createPost, initialState);

  return (
    <form action={formAction}>
      {state.message && <p className="text-green-600">{state.message}</p>}

      <div>
        <input name="title" placeholder="Title" />
        {state.error?.title && (
          <p className="text-red-500 text-sm">{state.error.title[0]}</p>
        )}
      </div>

      <div>
        <textarea name="content" placeholder="Content" />
        {state.error?.content && (
          <p className="text-red-500 text-sm">{state.error.content[0]}</p>
        )}
      </div>

      <button type="submit" disabled={isPending}>
        {isPending ? 'Publishing...' : 'Publish Post'}
      </button>
    </form>
  );
}
```

## Client-Side Fetching with SWR

For data that must update without a page reload, use SWR or TanStack Query:

```bash
npm install swr
```

```tsx
'use client';
import useSWR, { SWRConfig } from 'swr';

const fetcher = (url: string) => fetch(url).then(r => {
  if (!r.ok) throw new Error(`HTTP ${r.status}`);
  return r.json();
});

// Global SWR config in a provider:
export function SWRProvider({ children }: { children: React.ReactNode }) {
  return (
    <SWRConfig
      value={{
        fetcher,
        revalidateOnFocus: false,
        revalidateOnReconnect: true,
        dedupingInterval: 5000,
        errorRetryCount: 3,
      }}
    >
      {children}
    </SWRConfig>
  );
}

// Usage:
export function UserNotifications() {
  const { data, error, isLoading, mutate } = useSWR<Notification[]>(
    '/api/notifications',
    { refreshInterval: 30000 } // poll every 30s
  );

  if (isLoading) return <NotificationSkeleton />;
  if (error) return <p>Failed to load notifications.</p>;

  return (
    <ul>
      {data?.map(n => (
        <li key={n.id}>{n.message}</li>
      ))}
    </ul>
  );
}
```

:::caution[Server vs Client Fetching]
Avoid duplicating data-fetching logic. Fetch in Server Components when possible — it runs on the server, doesn't bloat the JS bundle, and has direct database access. Use client-side fetching only for data that must be live, user-specific, or requires interactivity.
:::

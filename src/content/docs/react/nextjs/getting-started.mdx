This guide walks through creating a production-ready Next.js project from scratch, configuring TypeScript, setting up the folder structure, and running the development server.

## Prerequisites

- **Node.js** 18.17 or later
- **npm**, **yarn**, **pnpm**, or **bun**

## Creating a New Project

```bash
# Interactive setup (recommended)
npx create-next-app@latest my-app

# Non-interactive with common flags
npx create-next-app@latest my-app \
  --typescript \
  --tailwind \
  --eslint \
  --app \
  --src-dir \
  --import-alias "@/*"
```

The interactive prompt will ask:
1. TypeScript? → **Yes**
2. ESLint? → **Yes**
3. Tailwind CSS? → **Yes** (optional)
4. `src/` directory? → **Yes** (keeps root clean)
5. App Router? → **Yes** (recommended)
6. Turbopack? → **Yes** (faster dev server, Next.js 15+)
7. Import alias? → **`@/*`** (standard)

## Folder Structure

```
my-app/
├── src/
│   ├── app/                    # App Router — routes and layouts
│   │   ├── layout.tsx          # Root layout (html, body)
│   │   ├── page.tsx            # Home route (/)
│   │   ├── globals.css         # Global styles
│   │   ├── favicon.ico
│   │   └── (dashboard)/        # Route group (no URL segment)
│   │       ├── layout.tsx
│   │       ├── page.tsx        # /dashboard
│   │       └── settings/
│   │           └── page.tsx    # /settings
│   ├── components/             # Reusable UI components
│   │   ├── ui/                 # Design system primitives
│   │   └── layout/             # Header, Footer, Nav
│   ├── lib/                    # Utilities, helpers, configs
│   │   ├── db.ts               # Database connection
│   │   ├── auth.ts             # Auth helpers
│   │   └── utils.ts            # General utilities
│   ├── hooks/                  # Custom React hooks
│   ├── types/                  # TypeScript types and interfaces
│   └── server/                 # Server-only code
│       └── db/
├── public/                     # Static assets (served from /)
│   ├── images/
│   └── fonts/
├── next.config.ts
├── tsconfig.json
├── tailwind.config.ts
├── eslint.config.mjs
├── .env.local                  # Local env vars (gitignored)
├── .env.example                # Template (commit this)
└── package.json
```

## next.config.ts

```typescript
// next.config.ts
import type { NextConfig } from 'next';

const nextConfig: NextConfig = {
  reactStrictMode: true,

  images: {
    remotePatterns: [
      {
        protocol: 'https',
        hostname: 'images.unsplash.com',
      },
      {
        protocol: 'https',
        hostname: '**.githubusercontent.com',
      },
    ],
    formats: ['image/avif', 'image/webp'],
  },

  experimental: {
    ppr: 'incremental',
    typedRoutes: true,
    serverActions: {
      allowedOrigins: ['localhost:3000', 'myapp.com'],
    },
  },

  async redirects() {
    return [
      {
        source: '/old-blog/:slug',
        destination: '/blog/:slug',
        permanent: true,
      },
    ];
  },

  async headers() {
    return [
      {
        source: '/(.*)',
        headers: [
          { key: 'X-Frame-Options', value: 'DENY' },
          { key: 'X-Content-Type-Options', value: 'nosniff' },
          { key: 'Referrer-Policy', value: 'strict-origin-when-cross-origin' },
        ],
      },
    ];
  },
};

export default nextConfig;
```

### Using Path Aliases

```typescript
// Without alias (verbose and fragile)
import { Button } from '../../../components/ui/Button';

// With @/* alias (clean and refactor-safe)
import { Button } from '@/components/ui/Button';
import { db } from '@/lib/db';
import { useAuth } from '@/hooks/useAuth';
import type { User } from '@/types/user';
```

## ESLint Configuration

```javascript
// eslint.config.mjs (flat config, Next.js 15+)
import { dirname } from 'path';
import { fileURLToPath } from 'url';
import { FlatCompat } from '@eslint/eslintrc';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

const compat = new FlatCompat({ baseDirectory: __dirname });

const eslintConfig = [
  ...compat.extends('next/core-web-vitals', 'next/typescript'),
  {
    rules: {
      'no-console': ['warn', { allow: ['warn', 'error'] }],
      '@typescript-eslint/no-unused-vars': ['error', { argsIgnorePattern: '^_' }],
      '@typescript-eslint/no-explicit-any': 'error',
      'prefer-const': 'error',
    },
  },
];

export default eslintConfig;
```

## Environment Variables

```bash
# .env.local — development secrets (never commit)
DATABASE_URL=postgresql://localhost:5432/mydb
NEXTAUTH_SECRET=a-very-long-random-secret-string
STRIPE_SECRET_KEY=sk_test_...

# .env.example — document all required variables (commit this)
DATABASE_URL=
NEXTAUTH_SECRET=
STRIPE_SECRET_KEY=
NEXT_PUBLIC_STRIPE_PUBLISHABLE_KEY=
NEXT_PUBLIC_APP_URL=http://localhost:3000
```

```typescript
// lib/env.ts — validate at startup with zod
import { z } from 'zod';

const envSchema = z.object({
  DATABASE_URL: z.string().url(),
  NEXTAUTH_SECRET: z.string().min(32),
  STRIPE_SECRET_KEY: z.string().startsWith('sk_'),
  NEXT_PUBLIC_APP_URL: z.string().url(),
});

export const env = envSchema.parse(process.env);
```

:::caution[NEXT_PUBLIC_ Prefix]
Variables prefixed with `NEXT_PUBLIC_` are embedded in the client JavaScript bundle at build time. Never put secrets in `NEXT_PUBLIC_` variables — they are visible to all users.
:::

Next.js uses a file-system-based router where every file in the `app/` directory maps to a URL segment. This page covers every routing mechanism from basic pages to advanced middleware and internationalisation.

## File-Based Routing Mechanics

The folder structure under `app/` directly maps to URL paths:

```
app/
├── page.tsx                    →  /
├── about/
│   └── page.tsx                →  /about
├── blog/
│   ├── page.tsx                →  /blog
│   └── [slug]/
│       └── page.tsx            →  /blog/:slug
├── shop/
│   ├── page.tsx                →  /shop
│   ├── [category]/
│   │   ├── page.tsx            →  /shop/:category
│   │   └── [productId]/
│   │       └── page.tsx        →  /shop/:category/:productId
│   └── cart/
│       └── page.tsx            →  /shop/cart
└── api/
    └── users/
        └── route.ts            →  /api/users (no page rendered)
```

## Dynamic Segments

### `[slug]` — Single Dynamic Segment

```tsx
// app/blog/[slug]/page.tsx
interface PageProps {
  params: Promise<{ slug: string }>;
}

export default async function PostPage({ params }: PageProps) {
  const { slug } = await params;
  const post = await getPost(slug);

  return <article>{post.content}</article>;
}
```

### `[...slug]` — Catch-All Segment

Matches any number of path segments. `/docs/a/b/c` → `slug: ['a', 'b', 'c']`.

```tsx
// app/docs/[...slug]/page.tsx
interface PageProps {
  params: Promise<{ slug: string[] }>;
}

export default async function DocsPage({ params }: PageProps) {
  const { slug } = await params;
  // /docs/getting-started/installation → slug: ['getting-started', 'installation']
  const path = slug.join('/');
  const doc = await getDoc(path);

  return <div>{doc.content}</div>;
}
```

### `[[...slug]]` — Optional Catch-All

Like catch-all, but also matches the root path. `/docs` → `slug: undefined`. `/docs/a/b` → `slug: ['a', 'b']`.

```tsx
// app/docs/[[...slug]]/page.tsx
interface PageProps {
  params: Promise<{ slug?: string[] }>;
}

export default async function DocsPage({ params }: PageProps) {
  const { slug } = await params;
  const path = slug?.join('/') ?? 'index';
  const doc = await getDoc(path);

  return <div>{doc.content}</div>;
}
```

## The Link Component

`Link` is Next.js's client-side navigation component. It prefetches linked pages automatically:

```tsx
import Link from 'next/link';

export function Navigation() {
  return (
    <nav>
      {/* Basic navigation */}
      <Link href="/">Home</Link>
      <Link href="/about">About</Link>

      {/* Dynamic route */}
      <Link href={`/blog/${post.slug}`}>Read Post</Link>

      {/* With query params */}
      <Link href={{ pathname: '/search', query: { q: 'nextjs', page: '2' } }}>
        Search Results
      </Link>

      {/* Replace current history entry (no back button) */}
      <Link href="/login" replace>
        Log In
      </Link>

      {/* Disable scroll to top on navigation */}
      <Link href="/blog#comments" scroll={false}>
        Jump to Comments
      </Link>

      {/* Disable prefetching (useful for expensive pages) */}
      <Link href="/admin" prefetch={false}>
        Admin Panel
      </Link>
    </nav>
  );
}
```

:::tip[Active Link Styling]
To highlight the active link, use `usePathname()` and compare with the href:

```tsx
'use client';
import Link from 'next/link';
import { usePathname } from 'next/navigation';

export function NavLink({
  href,
  children,
}: {
  href: string;
  children: React.ReactNode;
}) {
  const pathname = usePathname();
  const isActive = pathname === href || pathname.startsWith(href + '/');

  return (
    <Link
      href={href}
      className={
        isActive
          ? 'text-indigo-600 font-semibold'
          : 'text-gray-600 hover:text-gray-900'
      }
    >
      {children}
    </Link>
  );
}
```

:::

## Client-Side Navigation Hooks

These hooks are available in Client Components (`'use client'`):

### useRouter

```tsx
'use client';
import { useRouter } from 'next/navigation';

export function SearchForm() {
  const router = useRouter();

  const handleSearch = (query: string) => {
    // Push a new history entry
    router.push(`/search?q=${encodeURIComponent(query)}`);

    // Replace current history entry
    router.replace('/dashboard');

    // Go back
    router.back();

    // Go forward
    router.forward();

    // Refresh server components without full page reload
    router.refresh();

    // Prefetch a route
    router.prefetch('/heavy-page');
  };

  return (
    <form
      onSubmit={(e) => {
        e.preventDefault();
        handleSearch('nextjs');
      }}
    >
      ...
    </form>
  );
}
```

### usePathname and useSearchParams

```tsx
'use client';
import { usePathname, useSearchParams } from 'next/navigation';
import { useCallback } from 'react';
import { useRouter } from 'next/navigation';

export function FilterBar() {
  const router = useRouter();
  const pathname = usePathname();
  const searchParams = useSearchParams();

  // Read individual params
  const currentPage = searchParams.get('page') ?? '1';
  const currentCategory = searchParams.get('category');

  // Create a new searchParams string with modifications
  const createQueryString = useCallback(
    (name: string, value: string) => {
      const params = new URLSearchParams(searchParams.toString());
      params.set(name, value);
      return params.toString();
    },
    [searchParams],
  );

  const setCategory = (category: string) => {
    router.push(pathname + '?' + createQueryString('category', category));
  };

  return (
    <div>
      <p>Current path: {pathname}</p>
      <p>Page: {currentPage}</p>
      {['all', 'tech', 'design', 'business'].map((cat) => (
        <button
          key={cat}
          onClick={() => setCategory(cat)}
          className={currentCategory === cat ? 'font-bold' : ''}
        >
          {cat}
        </button>
      ))}
    </div>
  );
}
```

:::caution[Suspense Boundary Required for useSearchParams]
`useSearchParams()` must be used inside a `<Suspense>` boundary when used in a component that is part of static generation. Wrap your component in `<Suspense fallback={...}>` to avoid build errors.
:::

## Server-Side Redirect Functions

Use these in Server Components, Server Actions, and Route Handlers:

```tsx
// app/dashboard/page.tsx
import { redirect, permanentRedirect } from 'next/navigation';
import { auth } from '@/lib/auth';

export default async function DashboardPage() {
  const session = await auth();

  if (!session) {
    // Temporary redirect (307) — user must log in
    redirect('/login?callbackUrl=/dashboard');
  }

  if (session.user.role !== 'admin') {
    // Permanent redirect (308) — this path has moved forever
    permanentRedirect('/not-authorized');
  }

  return <div>Welcome to the dashboard!</div>;
}
```

## notFound()

Call `notFound()` in a Server Component to render the nearest `not-found.tsx`:

```tsx
import { notFound } from 'next/navigation';

export default async function ProductPage({
  params,
}: {
  params: Promise<{ id: string }>;
}) {
  const { id } = await params;
  const product = await getProduct(id);

  if (!product) {
    notFound(); // Renders app/not-found.tsx (or nearest not-found.tsx)
  }

  return <div>{product.name}</div>;
}
```

## Middleware

Middleware runs **before** every request and can rewrite, redirect, or modify headers. It runs at the edge (closest to the user).

```typescript
// middleware.ts (root of project, not in app/)
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;
  const token = request.cookies.get('auth-token')?.value;

  // Protect dashboard routes
  if (pathname.startsWith('/dashboard')) {
    if (!token) {
      const loginUrl = new URL('/login', request.url);
      loginUrl.searchParams.set('callbackUrl', pathname);
      return NextResponse.redirect(loginUrl);
    }
  }

  // A/B testing: route 50% of users to a different version
  if (pathname === '/') {
    const isVariant = Math.random() < 0.5;
    if (isVariant) {
      return NextResponse.rewrite(new URL('/home-v2', request.url));
    }
  }

  // Add security headers to all responses
  const response = NextResponse.next();
  response.headers.set('X-Frame-Options', 'DENY');
  response.headers.set('X-Robots-Tag', 'index, follow');

  return response;
}

// Configure which paths the middleware applies to
export const config = {
  matcher: [
    // Match all except static files and Next.js internals
    '/((?!_next/static|_next/image|favicon.ico|public/).*)',
    // Always run for API routes
    '/api/(.*)',
  ],
};
```

:::tip[Middleware vs Route Handlers]
Use **middleware** for auth checks, redirects, rewrites, and response header modification that needs to run before the request reaches the page. Use **Route Handlers** for building API endpoints. Middleware cannot read the request body or return JSON — use Route Handlers for that.
:::

## Internationalisation with next-intl

A common pattern for internationalised apps using the `next-intl` library:

<Steps>
1. **Install next-intl**

```bash
npm install next-intl
```

2. **Create message files**

   ```json
   // messages/en.json
   {
     "home": {
       "title": "Welcome to My App",
       "subtitle": "The best app on the internet"
     }
   }
   ```

   ```json
   // messages/fr.json
   {
     "home": {
       "title": "Bienvenue dans Mon Application",
       "subtitle": "La meilleure application sur Internet"
     }
   }
   ```

3. **Configure middleware for locale detection**

   ```typescript
   // middleware.ts
   import createMiddleware from 'next-intl/middleware';

   export default createMiddleware({
     locales: ['en', 'fr', 'de'],
     defaultLocale: 'en',
     localePrefix: 'as-needed', // /en/... for non-default, /... for default
   });

   export const config = {
     matcher: ['/((?!api|_next|_vercel|.*\\..*).*)'],
   };
   ```

4. **Use translations in components**

   ```tsx
   // app/[locale]/page.tsx
   import { useTranslations } from 'next-intl';

   export default function Home() {
     const t = useTranslations('home');

     return (
       <div>
         <h1>{t('title')}</h1>
         <p>{t('subtitle')}</p>
       </div>
     );
   }
   ```

   </Steps>

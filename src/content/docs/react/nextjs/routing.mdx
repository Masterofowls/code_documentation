Next.js uses a file-system-based router where every folder in the `app/` directory maps to a URL segment. Understanding the routing mechanics — especially dynamic routes, the Link component, and middleware — is fundamental to building any Next.js application.

## File-Based Routing Mechanics

Every folder containing a `page.tsx` file becomes a publicly accessible URL:

```
app/
├── page.tsx              → /
├── about/
│   └── page.tsx          → /about
├── blog/
│   ├── page.tsx          → /blog
│   └── tags/
│       └── page.tsx      → /blog/tags
└── dashboard/
    ├── page.tsx          → /dashboard
    ├── settings/
    │   └── page.tsx      → /dashboard/settings
    └── users/
        └── page.tsx      → /dashboard/users
```

URL segments map directly to folder names. Folders without a `page.tsx` are not publicly accessible (useful for organizing files).

## Dynamic Segments

### Single Dynamic Segment

```
app/
└── blog/
    └── [slug]/
        └── page.tsx     → /blog/hello-world, /blog/my-post
```

```tsx
// app/blog/[slug]/page.tsx
interface Props {
  params: Promise<{ slug: string }>;
}

export default async function BlogPost({ params }: Props) {
  const { slug } = await params;
  // slug = 'hello-world' for /blog/hello-world
  return <h1>Post: {slug}</h1>;
}
```

### Catch-All Segments

```
app/
└── docs/
    └── [...slug]/
        └── page.tsx     → /docs/react, /docs/react/hooks, /docs/react/hooks/use-state
```

```tsx
// app/docs/[...slug]/page.tsx
interface Props {
  params: Promise<{ slug: string[] }>;
}

export default async function DocsPage({ params }: Props) {
  const { slug } = await params;
  // /docs/react/hooks → slug = ['react', 'hooks']
  const path = slug.join('/');
  return <Docs path={path} />;
}
```

### Optional Catch-All Segments

```
app/
└── [[...slug]]/
    └── page.tsx         → /, /anything, /any/path/here
```

The double brackets make the segment optional — it also matches the root.

```tsx
// app/[[...slug]]/page.tsx
interface Props {
  params: Promise<{ slug?: string[] }>;
}

export default async function Page({ params }: Props) {
  const { slug } = await params;
  // / → slug = undefined
  // /docs → slug = ['docs']
  // /docs/react → slug = ['docs', 'react']

  if (!slug) return <Homepage />;
  return <ContentPage segments={slug} />;
}
```

## Link Component

`Link` is the primary navigation component. It prefetches the destination page automatically on hover.

```tsx
import Link from 'next/link';

export function Navigation() {
  return (
    <nav>
      {/* Simple link */}
      <Link href="/about">About</Link>

      {/* Dynamic route */}
      <Link href={`/blog/${post.slug}`}>Read Post</Link>

      {/* Object form with query params */}
      <Link
        href={{
          pathname: '/search',
          query: { q: 'react', category: 'tutorials' },
        }}
      >
        Search React Tutorials
      </Link>

      {/* Disable prefetch (useful for links behind auth) */}
      <Link href="/dashboard" prefetch={false}>
        Dashboard
      </Link>

      {/* Replace current history entry (no back button) */}
      <Link href="/login" replace>
        Login
      </Link>

      {/* Scroll to top on navigation */}
      <Link href="/blog" scroll={true}>
        Blog
      </Link>

      {/* External link (still ok to use, but consider <a> for clarity) */}
      <Link href="https://github.com" target="_blank" rel="noopener noreferrer">
        GitHub
      </Link>
    </nav>
  );
}
```

### Active Link Styling

```tsx
'use client';
import Link from 'next/link';
import { usePathname } from 'next/navigation';

const navItems = [
  { href: '/', label: 'Home' },
  { href: '/blog', label: 'Blog' },
  { href: '/about', label: 'About' },
];

export function NavBar() {
  const pathname = usePathname();

  return (
    <nav className="flex gap-4">
      {navItems.map(({ href, label }) => {
        const isActive = pathname === href || pathname.startsWith(`${href}/`);
        return (
          <Link
            key={href}
            href={href}
            className={`px-3 py-2 rounded-md text-sm font-medium transition-colors ${
              isActive
                ? 'bg-blue-100 text-blue-700'
                : 'text-gray-600 hover:text-gray-900 hover:bg-gray-100'
            }`}
            aria-current={isActive ? 'page' : undefined}
          >
            {label}
          </Link>
        );
      })}
    </nav>
  );
}
```

## useRouter

`useRouter` from `next/navigation` is for programmatic navigation in Client Components.

```tsx
'use client';
import { useRouter } from 'next/navigation';

export function LoginForm() {
  const router = useRouter();

  const handleSubmit = async (data: FormData) => {
    await signIn(data);

    // Navigate to dashboard
    router.push('/dashboard');

    // Replace (no back button)
    router.replace('/dashboard');

    // Go back
    router.back();

    // Go forward
    router.forward();

    // Refresh current route (re-fetch server data without full reload)
    router.refresh();

    // Prefetch a route
    router.prefetch('/dashboard');
  };

  return <form onSubmit={handleSubmit}>{/* ... */}</form>;
}
```

:::caution[useRouter from next/navigation]
Always import `useRouter` from `next/navigation`, not from `next/router`. The `next/router` package is for the Pages Router only.
:::

## usePathname and useSearchParams

```tsx
'use client';
import { usePathname, useSearchParams } from 'next/navigation';
import { useCallback } from 'react';

export function SearchFilters() {
  const pathname = usePathname();
  const searchParams = useSearchParams();
  const router = useRouter();

  // Current search param value
  const currentCategory = searchParams.get('category') ?? 'all';
  const currentPage = Number(searchParams.get('page') ?? '1');

  // Update search params without losing others
  const setCategory = useCallback(
    (category: string) => {
      const params = new URLSearchParams(searchParams.toString());
      params.set('category', category);
      params.set('page', '1'); // reset pagination on filter change
      router.push(`${pathname}?${params.toString()}`);
    },
    [pathname, router, searchParams]
  );

  return (
    <div>
      <p>Current path: {pathname}</p>
      <p>Current category: {currentCategory}</p>
      <button onClick={() => setCategory('tutorials')}>Tutorials</button>
      <button onClick={() => setCategory('news')}>News</button>
    </div>
  );
}
```

## Server-Side Redirect Functions

In Server Components, layouts, and server actions:

```tsx
import { redirect, permanentRedirect } from 'next/navigation';

export default async function ProfilePage() {
  const session = await getSession();

  if (!session) {
    redirect('/login'); // 307 temporary redirect
  }

  if (session.user.isDeleted) {
    permanentRedirect('/account-deleted'); // 308 permanent redirect
  }

  return <Profile user={session.user} />;
}
```

:::tip[redirect() Throws]
`redirect()` works by throwing a special error. Never call it inside a `try/catch` block without re-throwing, or the redirect won't happen.
:::

## Middleware

Middleware runs before every request and can modify responses, redirect, rewrite, or set headers. It runs on the Edge Runtime.

```typescript
// src/middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';
import { verifyJWT } from '@/lib/auth';

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // Auth protection
  if (pathname.startsWith('/dashboard')) {
    const token = request.cookies.get('auth-token')?.value;

    if (!token) {
      return NextResponse.redirect(new URL('/login', request.url));
    }

    try {
      await verifyJWT(token);
    } catch {
      return NextResponse.redirect(new URL('/login', request.url));
    }
  }

  // Redirect old blog URLs
  if (pathname.startsWith('/posts/')) {
    const newPath = pathname.replace('/posts/', '/blog/');
    return NextResponse.redirect(new URL(newPath, request.url), { status: 301 });
  }

  // Rewrite (proxy without URL change)
  if (pathname.startsWith('/api/v1/')) {
    const rewriteUrl = new URL(
      pathname.replace('/api/v1/', 'https://legacy-api.example.com/'),
    );
    return NextResponse.rewrite(rewriteUrl);
  }

  // Modify request/response headers
  const response = NextResponse.next();
  response.headers.set('X-Current-Path', pathname);

  return response;
}

// Limit middleware to specific paths (skip static files automatically)
export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|public/).*)',
  ],
};
```

### Middleware Matcher Patterns

```typescript
export const config = {
  matcher: [
    // Match all routes except static files and API routes
    '/((?!api|_next/static|_next/image|favicon.ico).*)',

    // Match specific paths
    '/dashboard/:path*',
    '/admin/:path*',

    // Match specific file extensions
    '/(.*)\\.(?:jpg|jpeg|gif|png|svg|ico)$',
  ],
};
```

## Internationalization with next-intl

```bash
npm install next-intl
```

```
app/
├── [locale]/                  # Locale segment
│   ├── layout.tsx             # Sets locale context
│   ├── page.tsx               # / and /en/ and /fr/
│   └── about/
│       └── page.tsx           # /about, /en/about, /fr/about
└── messages/
    ├── en.json
    └── fr.json
```

```typescript
// middleware.ts — redirect to default locale
import createMiddleware from 'next-intl/middleware';

export default createMiddleware({
  locales: ['en', 'fr', 'de', 'es'],
  defaultLocale: 'en',
  localeDetection: true,
});

export const config = {
  matcher: ['/((?!api|_next|_vercel|.*\\..*).*)'],
};
```

```tsx
// app/[locale]/layout.tsx
import { NextIntlClientProvider } from 'next-intl';
import { getMessages } from 'next-intl/server';
import { notFound } from 'next/navigation';

const locales = ['en', 'fr', 'de', 'es'];

export default async function LocaleLayout({
  children,
  params,
}: {
  children: React.ReactNode;
  params: Promise<{ locale: string }>;
}) {
  const { locale } = await params;

  if (!locales.includes(locale)) notFound();

  const messages = await getMessages();

  return (
    <html lang={locale}>
      <body>
        <NextIntlClientProvider messages={messages}>
          {children}
        </NextIntlClientProvider>
      </body>
    </html>
  );
}
```

import { useTranslations } from 'next-intl';

export function HomePage() {

  );
}

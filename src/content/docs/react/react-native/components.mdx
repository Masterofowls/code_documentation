React Native provides a set of core components that map directly to native platform UI elements. Unlike the web, there are no `<div>`, `<span>`, or `<input>` tags — everything is a purpose-built native component.

## View

`View` is the fundamental building block — the equivalent of a `<div>`. It supports flexbox layout, styling, touch handling, and accessibility.

```tsx
import { View, StyleSheet } from 'react-native';

export function Card({ children }: { children: React.ReactNode }) {
  return (
    <View style={styles.card}>
      <View style={styles.cardHeader}>{children}</View>
    </View>
  );
}

const styles = StyleSheet.create({
  card: {
    backgroundColor: '#fff',
    borderRadius: 12,
    padding: 16,
    shadowColor: '#000',
    shadowOffset: { width: 0, height: 2 },
    shadowOpacity: 0.1,
    shadowRadius: 4,
    elevation: 3,            // Android shadow
  },
  cardHeader: {
    borderBottomWidth: 1,
    borderBottomColor: '#eee',
    paddingBottom: 12,
    marginBottom: 12,
  },
});
```

## Text

`Text` is the only component that can render text strings. All text must be inside a `<Text>` component.

```tsx
import { Text, StyleSheet } from 'react-native';

export function TruncatedText({ content }: { content: string }) {
  return (
    <>
      {/* Single line with ellipsis */}
      <Text
        numberOfLines={1}
        ellipsizeMode="tail"
        style={styles.title}
      >
        {content}
      </Text>

      {/* Multi-line clamped */}
      <Text
        numberOfLines={3}
        ellipsizeMode="tail"
        style={styles.body}
      >
        {content}
      </Text>

      {/* Middle truncation — useful for file paths */}
      <Text
        numberOfLines={1}
        ellipsizeMode="middle"
        style={styles.path}
      >
        /very/long/file/path/that/needs/truncation/somewhere/in/the/middle
      </Text>
    </>
  );
}

const styles = StyleSheet.create({
  title: { fontSize: 18, fontWeight: '700', color: '#111' },
  body: { fontSize: 14, color: '#555', lineHeight: 20 },
  path: { fontSize: 12, color: '#888', fontFamily: 'monospace' },
});
```

### Text Nesting

```tsx
<Text style={{ fontSize: 16 }}>
  Normal text with{' '}
  <Text style={{ fontWeight: 'bold' }}>bold</Text>
  {' '}and{' '}
  <Text style={{ color: 'blue', textDecorationLine: 'underline' }}>
    linked
  </Text>
  {' '}portions.
</Text>
```

:::tip[Text Performance]
For long lists of text content, use `allowFontScaling={false}` to prevent accessibility font scaling from breaking your layout, and set explicit `maxFontSizeMultiplier` to cap scaling.
:::

## TextInput

`TextInput` is the core text entry component. It maps to `UITextField`/`UITextView` on iOS and `EditText` on Android.

```tsx
import React, { useState, useRef } from 'react';
import { TextInput, View, StyleSheet, TextInputProps } from 'react-native';

interface InputProps extends TextInputProps {
  label?: string;
}

export function FormInput({ label, ...props }: InputProps) {
  const [isFocused, setIsFocused] = useState(false);

  return (
    <TextInput
      style={[styles.input, isFocused && styles.inputFocused]}
      onFocus={() => setIsFocused(true)}
      onBlur={() => setIsFocused(false)}
      placeholderTextColor="#999"
      {...props}
    />
  );
}

export function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const passwordRef = useRef<TextInput>(null);

  return (
    <View>
      <FormInput
        value={email}
        onChangeText={setEmail}
        placeholder="Email address"
        keyboardType="email-address"
        autoCapitalize="none"
        autoCorrect={false}
        autoComplete="email"
        returnKeyType="next"
        onSubmitEditing={() => passwordRef.current?.focus()}
      />

      <FormInput
        ref={passwordRef}
        value={password}
        onChangeText={setPassword}
        placeholder="Password"
        secureTextEntry
        autoComplete="current-password"
        returnKeyType="done"
        onSubmitEditing={() => console.log('Submit!')}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  input: {
    height: 48,
    borderWidth: 1,
    borderColor: '#ddd',
    borderRadius: 8,
    paddingHorizontal: 16,
    fontSize: 16,
    marginBottom: 12,
  },
  inputFocused: {
    borderColor: '#007AFF',
    borderWidth: 2,
  },
});
```

### TextInput Key Props

| Prop | Type | Description |
|---|---|---|
| `value` | `string` | Controlled value |
| `onChangeText` | `(text: string) => void` | Called on every keystroke |
| `onBlur` | `() => void` | Lost focus |
| `onFocus` | `() => void` | Gained focus |
| `keyboardType` | `KeyboardTypeOptions` | `default`, `email-address`, `numeric`, `phone-pad`, `decimal-pad` |
| `returnKeyType` | `ReturnKeyTypeOptions` | `done`, `next`, `search`, `send`, `go` |
| `secureTextEntry` | `boolean` | Hides characters (password) |
| `multiline` | `boolean` | Allows multiple lines |
| `autoCapitalize` | `'none'` \| `'sentences'` \| `'words'` \| `'characters'` | Auto-capitalization |
| `autoComplete` | `AutoCompleteType` | OS autofill hints |
| `editable` | `boolean` | Disable editing |
| `maxLength` | `number` | Max character count |

## Image

```tsx
import { Image, StyleSheet } from 'react-native';

export function AvatarImage({ uri }: { uri: string }) {
  return (
    <>
      {/* Remote image */}
      <Image
        source={{ uri }}
        style={styles.avatar}
        resizeMode="cover"
        defaultSource={require('./assets/placeholder.png')}
      />

      {/* Local bundled image */}
      <Image
        source={require('./assets/logo.png')}
        style={styles.logo}
        resizeMode="contain"
      />
    </>
  );
}

const styles = StyleSheet.create({
  avatar: {
    width: 64,
    height: 64,
    borderRadius: 32,
  },
  logo: {
    width: 200,
    height: 60,
  },
});
```

`resizeMode` options: `cover`, `contain`, `stretch`, `repeat`, `center`

:::tip[Expo Image]
For better performance (caching, blur hash placeholders, transitions), use `expo-image` instead of the built-in `Image`. It has a near-identical API but significantly better performance.
:::

## ScrollView

```tsx
import { ScrollView, View, Text, RefreshControl, useState } from 'react-native';

export function ContentList() {
  const [refreshing, setRefreshing] = useState(false);

  const onRefresh = async () => {
    setRefreshing(true);
    await fetchData();
    setRefreshing(false);
  };

  return (
    <ScrollView
      showsVerticalScrollIndicator={false}
      contentContainerStyle={{ padding: 16 }}
      refreshControl={
        <RefreshControl refreshing={refreshing} onRefresh={onRefresh} />
      }
      keyboardShouldPersistTaps="handled"
    >
      {items.map(item => (
        <View key={item.id} style={{ marginBottom: 12 }}>
          <Text>{item.title}</Text>
        </View>
      ))}
    </ScrollView>
  );
}
```

:::caution[ScrollView vs FlatList]
`ScrollView` renders all children at once. Never use it for long or dynamic lists — it will degrade performance. Use `FlatList` or `SectionList` for any list that could have more than ~20 items.
:::

## FlatList

`FlatList` is the performant virtualized list component. It only renders items visible on screen.

```tsx
import { FlatList, View, Text, ActivityIndicator, StyleSheet } from 'react-native';

interface Post {
  id: string;
  title: string;
  excerpt: string;
}

interface Props {
  posts: Post[];
  loading: boolean;
  onLoadMore: () => void;
  onRefresh: () => void;
  refreshing: boolean;
}

export function PostList({ posts, loading, onLoadMore, onRefresh, refreshing }: Props) {
  return (
    <FlatList
      data={posts}
      keyExtractor={item => item.id}
      renderItem={({ item, index }) => (
        <View style={styles.item}>
          <Text style={styles.title}>{item.title}</Text>
          <Text style={styles.excerpt} numberOfLines={2}>
            {item.excerpt}
          </Text>
        </View>
      )}
      // Header rendered above first item
      ListHeaderComponent={() => (
        <Text style={styles.header}>Latest Posts</Text>
      )}
      // Footer rendered below last item
      ListFooterComponent={() =>
        loading ? <ActivityIndicator size="large" style={{ padding: 20 }} /> : null
      }
      // Shown when data is empty
      ListEmptyComponent={() => (
        <View style={styles.empty}>
          <Text>No posts yet.</Text>
        </View>
      )}
      // Separator between items
      ItemSeparatorComponent={() => <View style={styles.separator} />}
      // Pull-to-refresh
      refreshing={refreshing}
      onRefresh={onRefresh}
      // Infinite scroll — triggers when 20% from the bottom
      onEndReached={onLoadMore}
      onEndReachedThreshold={0.2}
      // Performance
      initialNumToRender={10}
      maxToRenderPerBatch={10}
      windowSize={5}
      removeClippedSubviews
      getItemLayout={(_, index) => ({
        length: ITEM_HEIGHT,
        offset: ITEM_HEIGHT * index,
        index,
      })}
    />
  );
}

const ITEM_HEIGHT = 80;

const styles = StyleSheet.create({
  item: { padding: 16, minHeight: ITEM_HEIGHT },
  title: { fontSize: 16, fontWeight: '600' },
  excerpt: { fontSize: 14, color: '#666', marginTop: 4 },
  header: { fontSize: 22, fontWeight: 'bold', padding: 16 },
  separator: { height: 1, backgroundColor: '#eee' },
  empty: { flex: 1, alignItems: 'center', padding: 40 },
});
```

:::tip[getItemLayout for Performance]
If your list items have a fixed height, implement `getItemLayout`. This allows FlatList to skip measuring each item, dramatically improving scroll-to-index performance.
:::

## SectionList

`SectionList` groups items into sections with section headers — ideal for contacts, settings screens, and grouped data.

```tsx
import { SectionList, Text, View } from 'react-native';

const DATA = [
  {
    title: 'Fruits',
    data: ['Apple', 'Banana', 'Cherry'],
  },
  {
    title: 'Vegetables',
    data: ['Carrot', 'Broccoli', 'Spinach'],
  },
];

export function GroupedList() {
  return (
    <SectionList
      sections={DATA}
      keyExtractor={(item, index) => `${item}-${index}`}
      renderItem={({ item }) => (
        <View style={{ padding: 12, paddingLeft: 24 }}>
          <Text>{item}</Text>
        </View>
      )}
      renderSectionHeader={({ section: { title } }) => (
        <View style={{ backgroundColor: '#f0f0f0', padding: 8 }}>
          <Text style={{ fontWeight: 'bold', fontSize: 13, color: '#555' }}>
            {title.toUpperCase()}
          </Text>
        </View>
      )}
      stickySectionHeadersEnabled
    />
  );
}
```

## TouchableOpacity vs Pressable

Both components handle touch input. **Pressable** is the modern replacement.

```tsx
import { TouchableOpacity, Pressable, Text, StyleSheet } from 'react-native';

// Legacy — still works fine, simpler API
function OldButton({ onPress, label }: { onPress: () => void; label: string }) {
  return (
    <TouchableOpacity
      onPress={onPress}
      onLongPress={() => console.log('Long pressed')}
      disabled={false}
      activeOpacity={0.7}
      style={styles.button}
    >
      <Text style={styles.label}>{label}</Text>
    </TouchableOpacity>
  );
}

// Modern — more flexible, supports pressed state in style
function NewButton({ onPress, label }: { onPress: () => void; label: string }) {
  return (
    <Pressable
      onPress={onPress}
      onLongPress={() => console.log('Long pressed')}
      disabled={false}
      style={({ pressed }) => [
        styles.button,
        pressed && styles.buttonPressed,
      ]}
      android_ripple={{ color: 'rgba(0,0,0,0.1)', borderless: false }}
    >
      {({ pressed }) => (
        <Text style={[styles.label, pressed && styles.labelPressed]}>
          {label}
        </Text>
      )}
    </Pressable>
  );
}

const styles = StyleSheet.create({
  button: {
    backgroundColor: '#007AFF',
    borderRadius: 10,
    paddingVertical: 14,
    paddingHorizontal: 24,
    alignItems: 'center',
  },
  buttonPressed: {
    backgroundColor: '#0056b3',
    transform: [{ scale: 0.97 }],
  },
  label: { color: '#fff', fontSize: 16, fontWeight: '600' },
  labelPressed: { opacity: 0.8 },
});
```

## Modal

```tsx
import { Modal, View, Text, Pressable, StyleSheet } from 'react-native';

interface ConfirmModalProps {
  visible: boolean;
  title: string;
  message: string;
  onConfirm: () => void;
  onDismiss: () => void;
}

export function ConfirmModal({
  visible,
  title,
  message,
  onConfirm,
  onDismiss,
}: ConfirmModalProps) {
  return (
    <Modal
      visible={visible}
      animationType="fade"        // 'none' | 'slide' | 'fade'
      transparent
      onRequestClose={onDismiss}  // Required on Android (back button)
      statusBarTranslucent        // Android: render over status bar
    >
      {/* Backdrop */}
      <Pressable style={styles.backdrop} onPress={onDismiss}>
        {/* Stop propagation so tapping the dialog doesn't close it */}
        <Pressable style={styles.dialog} onPress={() => {}}>
          <Text style={styles.dialogTitle}>{title}</Text>
          <Text style={styles.dialogMessage}>{message}</Text>
          <View style={styles.dialogActions}>
            <Pressable style={styles.cancelBtn} onPress={onDismiss}>
              <Text>Cancel</Text>
            </Pressable>
            <Pressable style={styles.confirmBtn} onPress={onConfirm}>
              <Text style={{ color: '#fff' }}>Confirm</Text>
            </Pressable>
          </View>
        </Pressable>
      </Pressable>
    </Modal>
  );
}

const styles = StyleSheet.create({
  backdrop: {
    flex: 1,
    backgroundColor: 'rgba(0,0,0,0.5)',
    justifyContent: 'center',
    alignItems: 'center',
  },
  dialog: {
    backgroundColor: '#fff',
    borderRadius: 16,
    padding: 24,
    width: '85%',
    maxWidth: 400,
  },
  dialogTitle: { fontSize: 18, fontWeight: 'bold', marginBottom: 8 },
  dialogMessage: { fontSize: 14, color: '#555', marginBottom: 24 },
  dialogActions: { flexDirection: 'row', justifyContent: 'flex-end', gap: 12 },
  cancelBtn: { padding: 10 },
  confirmBtn: { backgroundColor: '#007AFF', borderRadius: 8, padding: 10, paddingHorizontal: 20 },
});
```

## Platform-Specific Components

### SafeAreaView

```tsx
import { SafeAreaView } from 'react-native-safe-area-context';

// Renders content below the notch, dynamic island, and above home indicator
export function ScreenWrapper({ children }: { children: React.ReactNode }) {
  return (
    <SafeAreaView style={{ flex: 1, backgroundColor: '#fff' }} edges={['top', 'bottom']}>
      {children}
    </SafeAreaView>
  );
}
```

:::tip[Use the Community SafeAreaView]
Use `SafeAreaView` from `react-native-safe-area-context` (not from `react-native`). The community version provides hooks (`useSafeAreaInsets`) and supports more options.
:::

### KeyboardAvoidingView

```tsx
import { KeyboardAvoidingView, Platform, ScrollView, StyleSheet } from 'react-native';

export function KeyboardAwareForm({ children }: { children: React.ReactNode }) {
  return (
    <KeyboardAvoidingView
      style={{ flex: 1 }}
      behavior={Platform.OS === 'ios' ? 'padding' : 'height'}
      keyboardVerticalOffset={Platform.OS === 'ios' ? 88 : 0}
    >
      <ScrollView
        keyboardShouldPersistTaps="handled"
        contentContainerStyle={{ padding: 16, flexGrow: 1 }}
      >
        {children}
      </ScrollView>
    </KeyboardAvoidingView>
  );
}
```

## RefreshControl

```tsx
import { FlatList, RefreshControl, useState } from 'react-native';

export function RefreshableList() {
  const [refreshing, setRefreshing] = useState(false);

  const handleRefresh = async () => {
    setRefreshing(true);
    try {
      await refetchData();
    } finally {
      setRefreshing(false);
    }
  };

  return (
    <FlatList
      data={items}
      renderItem={renderItem}
      refreshControl={
        <RefreshControl
          refreshing={refreshing}
          onRefresh={handleRefresh}
          tintColor="#007AFF"        // iOS spinner color
          colors={['#007AFF']}       // Android spinner colors
          title="Pull to refresh"   // iOS only
        />
      }
    />
  );
}
```

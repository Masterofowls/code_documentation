React Native provides a set of built-in components that map directly to native UI elements on each platform. Understanding these core building blocks is essential for building any React Native app.

## View

`View` is the fundamental building block — analogous to `<div>` in HTML. It supports Flexbox layout, styling, touch handling, and accessibility.

```tsx
import { View, StyleSheet } from 'react-native';

export function Card({ children }: { children: React.ReactNode }) {
  return <View style={styles.card}>{children}</View>;
}

const styles = StyleSheet.create({
  card: {
    backgroundColor: '#fff',
    borderRadius: 12,
    padding: 16,
    marginHorizontal: 16,
    marginVertical: 8,
    // iOS shadow
    shadowColor: '#000',
    shadowOffset: { width: 0, height: 2 },
    shadowOpacity: 0.1,
    shadowRadius: 8,
    // Android shadow
    elevation: 4,
  },
});
```

:::tip[View vs ScrollView]
`View` does not scroll. If your content might overflow the screen, wrap it in a `ScrollView` or use `FlatList` for lists. Using `ScrollView` with many items will render all of them at once; use `FlatList` for performance with large datasets.
:::

## Text

`Text` renders a native text element. Unlike HTML, **only `Text` components can contain string children** in React Native — you cannot put raw strings inside `View`.

```tsx
import { Text, StyleSheet } from 'react-native';

export function ArticleText() {
  return (
    <>
      {/* numberOfLines truncates overflowing text */}
      <Text numberOfLines={2} ellipsizeMode="tail" style={styles.title}>
        This is a long title that will be truncated after two lines of text
      </Text>

      {/* Nested Text for inline styling */}
      <Text style={styles.body}>
        This paragraph has <Text style={styles.bold}>bold</Text> and{' '}
        <Text style={styles.italic}>italic</Text> inline text.
      </Text>

      {/* Selectable text for copy/paste */}
      <Text selectable style={styles.code}>
        npx expo start
      </Text>
    </>
  );
}

const styles = StyleSheet.create({
  title: { fontSize: 20, fontWeight: '700', color: '#111' },
  body: { fontSize: 16, lineHeight: 24, color: '#444' },
  bold: { fontWeight: '700' },
  italic: { fontStyle: 'italic' },
  code: { fontFamily: 'monospace', backgroundColor: '#f5f5f5', padding: 4 },
});
```

| Prop                   | Type                                     | Description                     |
| ---------------------- | ---------------------------------------- | ------------------------------- |
| `numberOfLines`        | `number`                                 | Maximum lines before truncation |
| `ellipsizeMode`        | `'head' \| 'middle' \| 'tail' \| 'clip'` | Where to truncate               |
| `selectable`           | `boolean`                                | Allow text selection            |
| `onPress`              | `() => void`                             | Make text tappable              |
| `adjustsFontSizeToFit` | `boolean`                                | Shrink font to fit container    |

## TextInput

`TextInput` renders a native text field. It's highly configurable for keyboard types, return key behaviour, and secure entry.

```tsx
import React, { useState, useRef } from 'react';
import { TextInput, View, Text, StyleSheet } from 'react-native';

export function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [emailError, setEmailError] = useState('');
  const passwordRef = useRef<TextInput>(null);

  const validateEmail = (value: string) => {
    const isValid = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
    setEmailError(isValid ? '' : 'Invalid email address');
  };

  return (
    <View style={styles.form}>
      <TextInput
        style={[styles.input, emailError ? styles.inputError : null]}
        placeholder="Email address"
        placeholderTextColor="#999"
        value={email}
        onChangeText={setEmail}
        onBlur={() => validateEmail(email)}
        keyboardType="email-address"
        autoCapitalize="none"
        autoCorrect={false}
        autoComplete="email"
        textContentType="emailAddress"
        returnKeyType="next"
        // Move focus to password field on "Next"
        onSubmitEditing={() => passwordRef.current?.focus()}
        blurOnSubmit={false}
      />
      {emailError ? <Text style={styles.error}>{emailError}</Text> : null}

      <TextInput
        ref={passwordRef}
        style={styles.input}
        placeholder="Password"
        placeholderTextColor="#999"
        value={password}
        onChangeText={setPassword}
        secureTextEntry={true}
        autoComplete="password"
        textContentType="password"
        returnKeyType="done"
      />
    </View>
  );
}

const styles = StyleSheet.create({
  form: { gap: 12, paddingHorizontal: 16 },
  input: {
    height: 48,
    borderWidth: 1,
    borderColor: '#ddd',
    borderRadius: 8,
    paddingHorizontal: 12,
    fontSize: 16,
    backgroundColor: '#fff',
  },
  inputError: { borderColor: '#e53e3e' },
  error: { color: '#e53e3e', fontSize: 12, marginTop: -8 },
});
```

| Prop              | Values                                                                               | Use Case                        |
| ----------------- | ------------------------------------------------------------------------------------ | ------------------------------- |
| `keyboardType`    | `'default'`, `'email-address'`, `'numeric'`, `'phone-pad'`, `'decimal-pad'`, `'url'` | Matching keyboard to input type |
| `returnKeyType`   | `'done'`, `'next'`, `'search'`, `'send'`, `'go'`                                     | Action button on keyboard       |
| `secureTextEntry` | `boolean`                                                                            | Password fields                 |
| `autoCapitalize`  | `'none'`, `'sentences'`, `'words'`, `'characters'`                                   | Text capitalisation             |
| `textContentType` | `'emailAddress'`, `'password'`, `'newPassword'`, `'name'`, etc.                      | iOS autofill hints              |
| `autoComplete`    | `'email'`, `'password'`, `'name'`, etc.                                              | Android autofill                |

## Image

`Image` displays a static or remote image with native performance.

```tsx
import { Image, View, StyleSheet } from 'react-native';

export function ProfileCard({ avatarUrl }: { avatarUrl: string }) {
  return (
    <View style={styles.card}>
      {/* Remote image */}
      <Image
        source={{ uri: avatarUrl }}
        style={styles.avatar}
        resizeMode="cover"
        // Fade in on load
        fadeDuration={300}
        // Placeholder during loading
        defaultSource={require('../assets/placeholder-avatar.png')}
      />

      {/* Local asset */}
      <Image
        source={require('../assets/logo.png')}
        style={styles.logo}
        resizeMode="contain"
      />
    </View>
  );
}

const styles = StyleSheet.create({
  card: { alignItems: 'center', padding: 16 },
  avatar: {
    width: 80,
    height: 80,
    borderRadius: 40, // Perfect circle
  },
  logo: {
    width: 120,
    height: 40,
  },
});
```

| `resizeMode` | Description                                                |
| ------------ | ---------------------------------------------------------- |
| `cover`      | Fill the frame, crop if necessary (aspect-ratio preserved) |
| `contain`    | Fit within the frame, letterboxed if necessary             |
| `stretch`    | Stretch to fill; aspect ratio not preserved                |
| `repeat`     | Tile the image                                             |
| `center`     | Center without scaling                                     |

:::tip[Use expo-image for Better Performance]
The `expo-image` package (`import { Image } from 'expo-image'`) provides superior caching, blur hash placeholders, and better memory management compared to the built-in `Image` component. Recommended for production apps.
:::

## ScrollView

`ScrollView` renders all children at once. Use it for short content (forms, settings screens). For lists with many items, use `FlatList`.

```tsx
import {
  ScrollView,
  View,
  Text,
  StyleSheet,
  RefreshControl,
} from 'react-native';
import { useState, useCallback } from 'react';

export function SettingsScreen() {
  const [refreshing, setRefreshing] = useState(false);

  const onRefresh = useCallback(async () => {
    setRefreshing(true);
    await fetchSettings(); // Your data fetch
    setRefreshing(false);
  }, []);

  return (
    <ScrollView
      style={styles.container}
      contentContainerStyle={styles.content}
      showsVerticalScrollIndicator={false}
      bounces={true} // iOS rubber-band effect
      keyboardShouldPersistTaps="handled" // Dismiss keyboard on tap outside input
      refreshControl={
        <RefreshControl refreshing={refreshing} onRefresh={onRefresh} />
      }
    >
      <Text style={styles.section}>Account</Text>
      {/* Settings items... */}
    </ScrollView>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: '#f5f5f5' },
  content: { paddingBottom: 40 },
  section: {
    fontSize: 13,
    fontWeight: '600',
    color: '#888',
    padding: 16,
    textTransform: 'uppercase',
  },
});
```

## FlatList

`FlatList` is a performant scrollable list. It uses windowing to only render visible items, making it ideal for long lists.

```tsx
import {
  FlatList,
  View,
  Text,
  ActivityIndicator,
  StyleSheet,
} from 'react-native';

interface Post {
  id: string;
  title: string;
  body: string;
}

interface FeedProps {
  posts: Post[];
  loading: boolean;
  onEndReached: () => void;
  onRefresh: () => void;
  refreshing: boolean;
}

export function Feed({
  posts,
  loading,
  onEndReached,
  onRefresh,
  refreshing,
}: FeedProps) {
  return (
    <FlatList
      data={posts}
      keyExtractor={(item) => item.id}
      renderItem={({ item, index }) => (
        <View style={styles.item}>
          <Text style={styles.title}>{item.title}</Text>
          <Text style={styles.body} numberOfLines={3}>
            {item.body}
          </Text>
        </View>
      )}
      // Performance tuning
      initialNumToRender={10}
      maxToRenderPerBatch={10}
      windowSize={5}
      removeClippedSubviews={true}
      // Header and Footer
      ListHeaderComponent={
        <View style={styles.header}>
          <Text style={styles.headerText}>Latest Posts</Text>
        </View>
      }
      ListFooterComponent={
        loading ? <ActivityIndicator style={styles.loader} /> : null
      }
      ListEmptyComponent={
        <View style={styles.empty}>
          <Text>No posts found.</Text>
        </View>
      }
      // Infinite scroll
      onEndReached={onEndReached}
      onEndReachedThreshold={0.5}
      // Pull to refresh
      onRefresh={onRefresh}
      refreshing={refreshing}
      // Layout
      ItemSeparatorComponent={() => <View style={styles.separator} />}
      contentContainerStyle={styles.list}
    />
  );
}

const styles = StyleSheet.create({
  list: { paddingVertical: 8 },
  item: { padding: 16, backgroundColor: '#fff' },
  title: { fontSize: 16, fontWeight: '600', marginBottom: 4 },
  body: { fontSize: 14, color: '#666' },
  header: { padding: 16, borderBottomWidth: 1, borderBottomColor: '#eee' },
  headerText: { fontSize: 20, fontWeight: '700' },
  loader: { padding: 16 },
  empty: { flex: 1, alignItems: 'center', padding: 40 },
  separator: { height: 1, backgroundColor: '#f0f0f0' },
});
```

:::caution[Never Use ScrollView for Long Lists]
Wrapping a large dataset in `ScrollView` will render all items immediately, causing slow startup and excessive memory usage. Always use `FlatList` (or `@shopify/flash-list` for maximum performance) for dynamic lists.
:::

## SectionList

`SectionList` is like `FlatList` but groups items into sections, each with its own header.

```tsx
import { SectionList, Text, View, StyleSheet } from 'react-native';

const contacts = [
  {
    title: 'A',
    data: [
      { id: '1', name: 'Alice Johnson' },
      { id: '2', name: 'Andrew Smith' },
    ],
  },
  {
    title: 'B',
    data: [{ id: '3', name: 'Bob Williams' }],
  },
];

export function ContactList() {
  return (
    <SectionList
      sections={contacts}
      keyExtractor={(item) => item.id}
      renderItem={({ item }) => <Text style={styles.name}>{item.name}</Text>}
      renderSectionHeader={({ section }) => (
        <View style={styles.sectionHeader}>
          <Text style={styles.sectionTitle}>{section.title}</Text>
        </View>
      )}
      stickySectionHeadersEnabled={true}
    />
  );
}

const styles = StyleSheet.create({
  name: { fontSize: 16, padding: 16, backgroundColor: '#fff' },
  sectionHeader: {
    backgroundColor: '#f5f5f5',
    paddingHorizontal: 16,
    paddingVertical: 8,
  },
  sectionTitle: {
    fontSize: 13,
    fontWeight: '700',
    color: '#888',
    textTransform: 'uppercase',
  },
});
```

## TouchableOpacity vs Pressable

`Pressable` is the modern replacement for `TouchableOpacity`. It offers finer control over press states.

```tsx
import { Pressable, TouchableOpacity, Text, StyleSheet } from 'react-native';

// TouchableOpacity — simple, opaque feedback
export function LegacyButton({
  onPress,
  title,
}: {
  onPress: () => void;
  title: string;
}) {
  return (
    <TouchableOpacity
      style={styles.button}
      onPress={onPress}
      onLongPress={() => console.log('Long pressed')}
      activeOpacity={0.7}
      disabled={false}
    >
      <Text style={styles.label}>{title}</Text>
    </TouchableOpacity>
  );
}

// Pressable — modern, flexible, style-based feedback
export function ModernButton({
  onPress,
  title,
  disabled = false,
}: {
  onPress: () => void;
  title: string;
  disabled?: boolean;
}) {
  return (
    <Pressable
      style={({ pressed }) => [
        styles.button,
        pressed && styles.buttonPressed,
        disabled && styles.buttonDisabled,
      ]}
      onPress={onPress}
      onLongPress={() => console.log('Long pressed')}
      delayLongPress={500}
      disabled={disabled}
      // Hitbox extension for small targets
      hitSlop={{ top: 8, bottom: 8, left: 8, right: 8 }}
    >
      {({ pressed }) => (
        <Text style={[styles.label, pressed && styles.labelPressed]}>
          {title}
        </Text>
      )}
    </Pressable>
  );
}

const styles = StyleSheet.create({
  button: {
    backgroundColor: '#6366f1',
    paddingVertical: 12,
    paddingHorizontal: 24,
    borderRadius: 8,
    alignItems: 'center',
  },
  buttonPressed: { backgroundColor: '#4f46e5', transform: [{ scale: 0.98 }] },
  buttonDisabled: { backgroundColor: '#a5b4fc', opacity: 0.6 },
  label: { color: '#fff', fontSize: 16, fontWeight: '600' },
  labelPressed: { opacity: 0.9 },
});
```

## Platform-Specific Components

### SafeAreaView

`SafeAreaView` automatically insets content away from device notches, home indicators, and rounded corners. Always wrap screen-level content with it.

```tsx
import { SafeAreaView, StyleSheet } from 'react-native';

export function Screen({ children }: { children: React.ReactNode }) {
  return <SafeAreaView style={styles.safe}>{children}</SafeAreaView>;
}

const styles = StyleSheet.create({
  safe: { flex: 1, backgroundColor: '#fff' },
});
```

:::tip[react-native-safe-area-context]
For more control (e.g., applying insets to specific edges only), use the `useSafeAreaInsets()` hook from `react-native-safe-area-context` (included in Expo SDK).
:::

### KeyboardAvoidingView

Adjusts layout when the software keyboard appears, preventing it from covering input fields.

```tsx
import {
  KeyboardAvoidingView,
  Platform,
  ScrollView,
  StyleSheet,
} from 'react-native';

export function FormScreen({ children }: { children: React.ReactNode }) {
  return (
    <KeyboardAvoidingView
      style={styles.container}
      behavior={Platform.OS === 'ios' ? 'padding' : 'height'}
      keyboardVerticalOffset={Platform.OS === 'ios' ? 90 : 0}
    >
      <ScrollView
        keyboardShouldPersistTaps="handled"
        contentContainerStyle={styles.scrollContent}
      >
        {children}
      </ScrollView>
    </KeyboardAvoidingView>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1 },
  scrollContent: { flexGrow: 1 },
});
```

### ActivityIndicator and Switch

```tsx
import {
  ActivityIndicator,
  Switch,
  View,
  Text,
  useState,
  StyleSheet,
} from 'react-native';

export function LoadingAndToggle() {
  const [isEnabled, setIsEnabled] = useState(false);

  return (
    <View style={styles.row}>
      {/* Loading spinner */}
      <ActivityIndicator size="large" color="#6366f1" />
      <ActivityIndicator size="small" color="#999" />

      {/* Toggle switch */}
      <View style={styles.toggle}>
        <Text>Notifications</Text>
        <Switch
          value={isEnabled}
          onValueChange={setIsEnabled}
          trackColor={{ false: '#ddd', true: '#6366f1' }}
          thumbColor={isEnabled ? '#fff' : '#f4f4f4'}
          ios_backgroundColor="#ddd"
        />
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  row: { gap: 16, alignItems: 'center', padding: 16 },
  toggle: {
    flexDirection: 'row',
    justifyContent: 'space-between',
    alignItems: 'center',
    width: '100%',
  },
});
```

## RefreshControl

Use `RefreshControl` inside `ScrollView` or `FlatList` to add pull-to-refresh:

```tsx
import { FlatList, RefreshControl } from 'react-native';
import { useState, useCallback } from 'react';

export function RefreshableFeed() {
  const [refreshing, setRefreshing] = useState(false);

  const onRefresh = useCallback(async () => {
    setRefreshing(true);
    try {
      await refetchData();
    } finally {
      setRefreshing(false);
    }
  }, []);

  return (
    <FlatList
      data={items}
      renderItem={renderItem}
      keyExtractor={(item) => item.id}
      refreshControl={
        <RefreshControl
          refreshing={refreshing}
          onRefresh={onRefresh}
          tintColor="#6366f1" // iOS spinner color
          colors={['#6366f1']} // Android spinner colors
          title="Refreshing..." // iOS loading text
        />
      }
    />
  );
}
```

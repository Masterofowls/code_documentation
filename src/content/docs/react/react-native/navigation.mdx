Navigation in React Native is handled by community libraries rather than built-in platform APIs. **React Navigation** is the most widely used solution, providing Stack, Tab, Drawer navigators, deep linking, and authentication flows all with a React-centric API.

## Installation

<Steps>

1. **Install React Navigation core**

   ```bash
   npm install @react-navigation/native
   ```

2. **Install peer dependencies**

   ```bash
   # Expo:
   npx expo install react-native-screens react-native-safe-area-context

   # Bare React Native:
   npm install react-native-screens react-native-safe-area-context
   cd ios && pod install
   ```

3. **Install the navigators you need**

   ```bash
   # Native Stack (recommended):
   npm install @react-navigation/native-stack

   # Bottom Tab:
   npm install @react-navigation/bottom-tabs

   # Drawer:
   npm install @react-navigation/drawer react-native-gesture-handler react-native-reanimated
   ```

4. **Enable screens (Bare RN only)**

   Add to the top of your `App.tsx` or `index.js`:

   ```typescript
   import 'react-native-gesture-handler'; // Must be first import
   import { enableScreens } from 'react-native-screens';
   enableScreens();
   ```

</Steps>

## NavigationContainer

Wrap your entire app in `NavigationContainer`. It manages the navigation tree and contains the navigation state.

```tsx
// App.tsx
import React from 'react';
import { NavigationContainer } from '@react-navigation/native';
import { SafeAreaProvider } from 'react-native-safe-area-context';
import { RootNavigator } from './src/navigation/RootNavigator';

export default function App() {
  return (
    <SafeAreaProvider>
      <NavigationContainer
        // Optional: persist navigation state across app restarts
        // onStateChange={(state) => saveNavigationState(state)}
        // initialState={savedState}

        // Optional: linking config for deep links
        linking={{
          prefixes: ['myapp://', 'https://myapp.com'],
          config: {
            screens: {
              Home: 'home',
              Profile: 'user/:userId',
              Settings: 'settings',
            },
          },
        }}
      >
        <RootNavigator />
      </NavigationContainer>
    </SafeAreaProvider>
  );
}
```

## Stack Navigator

`createNativeStackNavigator` uses native navigation primitives — `UINavigationController` on iOS and Fragment transitions on Android — for optimal performance.

```tsx
import React from 'react';
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import { HomeScreen } from '../screens/HomeScreen';
import { ProductDetailScreen } from '../screens/ProductDetailScreen';
import { CartScreen } from '../screens/CartScreen';

// Define your param list for type safety:
export type RootStackParamList = {
  Home: undefined;
  ProductDetail: { productId: string; title: string };
  Cart: undefined;
};

const Stack = createNativeStackNavigator<RootStackParamList>();

export function RootNavigator() {
  return (
    <Stack.Navigator
      initialRouteName="Home"
      screenOptions={{
        // Applied to all screens in this navigator:
        headerStyle: { backgroundColor: '#fff' },
        headerTintColor: '#1a202c',
        headerTitleStyle: { fontWeight: '700' },
        headerShadowVisible: false,
        contentStyle: { backgroundColor: '#f8fafc' },
        animation: 'slide_from_right', // iOS default; can use 'fade', 'none', etc.
        gestureEnabled: true,
      }}
    >
      <Stack.Screen
        name="Home"
        component={HomeScreen}
        options={{ title: 'Products', headerLargeTitle: true }}
      />
      <Stack.Screen
        name="ProductDetail"
        component={ProductDetailScreen}
        options={({ route }) => ({
          title: route.params.title,
          headerShown: true,
          headerBackTitle: 'Back',
        })}
      />
      <Stack.Screen
        name="Cart"
        component={CartScreen}
        options={{
          presentation: 'modal',         // slide up from bottom
          headerShown: false,
        }}
      />
    </Stack.Navigator>
  );
}
```

### Dynamic Screen Options

```tsx
// Inside a screen component, set options dynamically:
import { useLayoutEffect } from 'react';
import { useNavigation } from '@react-navigation/native';

function ProductDetailScreen() {
  const navigation = useNavigation();
  const [isFavorite, setIsFavorite] = useState(false);

  useLayoutEffect(() => {
    navigation.setOptions({
      headerRight: () => (
        <Pressable onPress={() => setIsFavorite(f => !f)}>
          <Text>{isFavorite ? '❤️' : '🤍'}</Text>
        </Pressable>
      ),
    });
  }, [navigation, isFavorite]);

  return <View>{/* ... */}</View>;
}
```

## Tab Navigator

`createBottomTabNavigator` renders a tab bar at the bottom of the screen, similar to native iOS tab bars and Android bottom navigation.

```tsx
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import { Ionicons } from '@expo/vector-icons'; // or any icon library
import { View, Text, StyleSheet } from 'react-native';

export type TabParamList = {
  Feed: undefined;
  Search: undefined;
  Notifications: undefined;
  Profile: undefined;
};

const Tab = createBottomTabNavigator<TabParamList>();

function TabBadge({ count }: { count: number }) {
  if (count === 0) return null;
  return (
    <View style={styles.badge}>
      <Text style={styles.badgeText}>{count > 99 ? '99+' : count}</Text>
    </View>
  );
}

export function MainTabNavigator() {
  const notificationCount = 3; // from your state/store

  return (
    <Tab.Navigator
      screenOptions={({ route }) => ({
        tabBarIcon: ({ focused, color, size }) => {
          const icons: Record<string, { active: string; inactive: string }> = {
            Feed: { active: 'home', inactive: 'home-outline' },
            Search: { active: 'search', inactive: 'search-outline' },
            Notifications: { active: 'notifications', inactive: 'notifications-outline' },
            Profile: { active: 'person', inactive: 'person-outline' },
          };
          const iconName = focused
            ? icons[route.name].active
            : icons[route.name].inactive;
          return <Ionicons name={iconName as any} size={size} color={color} />;
        },
        tabBarActiveTintColor: '#4f46e5',
        tabBarInactiveTintColor: '#94a3b8',
        tabBarStyle: {
          backgroundColor: '#fff',
          borderTopColor: '#e2e8f0',
          height: 60,
          paddingBottom: 8,
        },
        tabBarLabelStyle: { fontSize: 12, fontWeight: '600' },
        headerShown: false,
      })}
    >
      <Tab.Screen name="Feed" component={FeedScreen} />
      <Tab.Screen name="Search" component={SearchScreen} />
      <Tab.Screen
        name="Notifications"
        component={NotificationsScreen}
        options={{
          tabBarLabel: 'Alerts',
          tabBarBadge: notificationCount > 0 ? notificationCount : undefined,
        }}
      />
      <Tab.Screen name="Profile" component={ProfileScreen} />
    </Tab.Navigator>
  );
}

const styles = StyleSheet.create({
  badge: {
    position: 'absolute',
    right: -6,
    top: -4,
    backgroundColor: '#ef4444',
    borderRadius: 10,
    minWidth: 18,
    height: 18,
    justifyContent: 'center',
    alignItems: 'center',
    paddingHorizontal: 4,
  },
  badgeText: { color: '#fff', fontSize: 11, fontWeight: '700' },
});
```

## Drawer Navigator

```tsx
import { createDrawerNavigator } from '@react-navigation/drawer';
import { GestureHandlerRootView } from 'react-native-gesture-handler';

const Drawer = createDrawerNavigator();

// Wrap your app root in GestureHandlerRootView:
export default function App() {
  return (
    <GestureHandlerRootView style={{ flex: 1 }}>
      <NavigationContainer>
        <DrawerNavigator />
      </NavigationContainer>
    </GestureHandlerRootView>
  );
}

function DrawerNavigator() {
  return (
    <Drawer.Navigator
      screenOptions={{
        drawerStyle: { backgroundColor: '#1a202c', width: 280 },
        drawerLabelStyle: { color: '#fff', fontWeight: '600' },
        drawerActiveTintColor: '#a5b4fc',
        drawerType: 'front',             // front | back | slide | permanent
        swipeEdgeWidth: 50,
      }}
    >
      <Drawer.Screen name="Home" component={HomeScreen} />
      <Drawer.Screen name="Settings" component={SettingsScreen} />
    </Drawer.Navigator>
  );
}
```

## Nested Navigators

Nest navigators to create complex navigation structures (e.g., a tab navigator inside a stack):

```tsx
// A common pattern: Stack (auth) → Tabs → Stack (detail)
function RootNavigator() {
  const { isAuthenticated } = useAuthStore();

  return (
    <Stack.Navigator screenOptions={{ headerShown: false }}>
      {isAuthenticated ? (
        // Authenticated: show tabs
        <Stack.Screen name="Main" component={MainTabNavigator} />
      ) : (
        // Unauthenticated: show auth screens
        <Stack.Screen name="Auth" component={AuthNavigator} />
      )}
    </Stack.Navigator>
  );
}

// MainTabNavigator renders each tab, which can themselves be Stack navigators:
function MainTabNavigator() {
  return (
    <Tab.Navigator>
      <Tab.Screen name="Feed" component={FeedStackNavigator} />
      <Tab.Screen name="Profile" component={ProfileStackNavigator} />
    </Tab.Navigator>
  );
}

function FeedStackNavigator() {
  return (
    <Stack.Navigator>
      <Stack.Screen name="FeedList" component={FeedListScreen} />
      <Stack.Screen name="PostDetail" component={PostDetailScreen} />
    </Stack.Navigator>
  );
}
```

## Passing Params to Screens

```tsx
import { NativeStackScreenProps } from '@react-navigation/native-stack';

type Props = NativeStackScreenProps<RootStackParamList, 'ProductDetail'>;

function ProductDetailScreen({ route, navigation }: Props) {
  // route.params is fully typed:
  const { productId, title } = route.params;

  return (
    <View>
      <Text>Product: {productId}</Text>
      <Text>Title: {title}</Text>
    </View>
  );
}

// Navigate and pass params:
navigation.navigate('ProductDetail', { productId: '123', title: 'Sneakers' });

// Update params on the current screen:
navigation.setParams({ title: 'Updated Title' });
```

## Navigation Methods

```tsx
// Navigate to a screen (goes back if already in stack):
navigation.navigate('Home');

// Push always adds a new screen (even if it already exists in stack):
navigation.push('ProductDetail', { productId: '456', title: 'Boots' });

// Replace current screen without adding to history:
navigation.replace('Home');

// Go back one screen:
navigation.goBack();

// Check if we can go back:
if (navigation.canGoBack()) {
  navigation.goBack();
}

// Navigate back to a specific screen:
navigation.navigate('Home');

// Reset the navigation state entirely (useful after login):
navigation.reset({
  index: 0,
  routes: [{ name: 'Home' }],
});
```

:::tip[navigate vs push]
Use `navigate` for most cases — it deduplicates screens. Use `push` when you need multiple instances of the same screen (e.g., a file browser where each folder is the same screen).
:::

## Authentication Flow

The recommended pattern for conditional navigation based on auth state:

```tsx
// src/navigation/RootNavigator.tsx
import { useEffect } from 'react';
import { useAuthStore } from '../store/authStore';

function RootNavigator() {
  const { isAuthenticated, isLoading } = useAuthStore();

  if (isLoading) {
    return <SplashScreen />;
  }

  return (
    <Stack.Navigator screenOptions={{ headerShown: false }}>
      {isAuthenticated ? (
        <>
          <Stack.Screen name="Main" component={MainTabNavigator} />
          <Stack.Screen name="EditProfile" component={EditProfileScreen} options={{ presentation: 'modal' }} />
        </>
      ) : (
        <>
          <Stack.Screen name="Welcome" component={WelcomeScreen} />
          <Stack.Screen name="Login" component={LoginScreen} />
          <Stack.Screen name="Register" component={RegisterScreen} />
          <Stack.Screen name="ForgotPassword" component={ForgotPasswordScreen} />
        </>
      )}
    </Stack.Navigator>
  );
}
```

When `isAuthenticated` changes from `false` to `true`, React Navigation automatically unmounts the auth screens and mounts the app screens. No manual navigation call needed.

:::caution[Don't Navigate on Auth Change]
Never call `navigation.navigate('Home')` after a login action. Instead, simply update your auth state (`isAuthenticated = true`) and let React Navigation's conditional rendering handle the transition automatically.
:::

## Deep Linking

```tsx
// App.tsx — configure linking
const linking = {
  prefixes: ['myapp://', 'https://myapp.com'],
  config: {
    screens: {
      // Matches: myapp://home
      Home: 'home',
      // Matches: myapp://product/abc-123
      ProductDetail: 'product/:productId',
      // Matches: myapp://user/456/posts
      Profile: {
        path: 'user/:userId',
        screens: {
          PostsList: 'posts',
          PostDetail: 'posts/:postId',
        },
      },
    },
  },
};

<NavigationContainer linking={linking}>
  {/* ... */}
</NavigationContainer>
```

### Linking Module

Programmatically open URLs and deep links:

```tsx
import { Linking } from 'react-native';

// Open a URL in the default browser:
await Linking.openURL('https://example.com');

// Open app settings:
await Linking.openSettings();

// Check if a URL can be opened:
const supported = await Linking.canOpenURL('maps://');
if (supported) {
  await Linking.openURL('maps://0,0?q=Tokyo');
}

// Listen for incoming deep links while app is open:
useEffect(() => {
  const subscription = Linking.addEventListener('url', ({ url }) => {
    console.log('Deep link received:', url);
  });
  return () => subscription.remove();
}, []);

// Get the initial URL (app opened via deep link):
const initialUrl = await Linking.getInitialURL();
```

## TypeScript Navigation Types

Create a central type declaration file for all your navigation param lists:

```typescript
// src/navigation/types.ts
import type { NativeStackScreenProps } from '@react-navigation/native-stack';
import type { BottomTabScreenProps } from '@react-navigation/bottom-tabs';
import type { CompositeScreenProps } from '@react-navigation/native';

export type RootStackParamList = {
  Splash: undefined;
  Auth: undefined;
  Main: undefined;
};

export type AuthStackParamList = {
  Welcome: undefined;
  Login: { redirect?: string };
  Register: undefined;
  ForgotPassword: { email?: string };
};

export type TabParamList = {
  Feed: undefined;
  Search: undefined;
  Notifications: undefined;
  Profile: { userId?: string };
};

// Composite type for screens deep inside nested navigators:
export type FeedScreenProps = CompositeScreenProps<
  BottomTabScreenProps<TabParamList, 'Feed'>,
  NativeStackScreenProps<RootStackParamList>
>;

// Declare globally for useNavigation without generic:
declare global {
  namespace ReactNavigation {
    interface RootParamList extends RootStackParamList {}
  }
}
```

Navigation in React Native is handled through the community library **React Navigation** — the most widely used and recommended navigation solution for React Native apps.

## Installation

<Steps>
1. **Install React Navigation core**

```bash
npm install @react-navigation/native
```

2. **Install dependencies**

   ```bash
   # Expo
   npx expo install react-native-screens react-native-safe-area-context

   # Bare React Native
   npm install react-native-screens react-native-safe-area-context
   cd ios && pod install
   ```

3. **Install navigator packages**

   ```bash
   # Stack Navigator (native stack — recommended)
   npm install @react-navigation/native-stack

   # Bottom Tab Navigator
   npm install @react-navigation/bottom-tabs

   # Drawer Navigator
   npm install @react-navigation/drawer react-native-reanimated react-native-gesture-handler
   ```

</Steps>

## NavigationContainer

Wrap your entire app (or at minimum the navigation tree) in `NavigationContainer`. This is the root of all navigation state.

```tsx
// App.tsx
import React from 'react';
import { NavigationContainer } from '@react-navigation/native';
import { RootNavigator } from './src/navigation/RootNavigator';

export default function App() {
  return (
    <NavigationContainer>
      <RootNavigator />
    </NavigationContainer>
  );
}
```

## Stack Navigator

The Stack Navigator presents screens as a stack — each new screen slides in on top, and going back pops it off.

```tsx
// src/navigation/HomeStack.tsx
import React from 'react';
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import { HomeScreen } from '../screens/HomeScreen';
import { DetailScreen } from '../screens/DetailScreen';
import { ProfileScreen } from '../screens/ProfileScreen';

export type HomeStackParamList = {
  Home: undefined;
  Detail: { postId: string; title: string };
  Profile: { userId: string };
};

const Stack = createNativeStackNavigator<HomeStackParamList>();

export function HomeStack() {
  return (
    <Stack.Navigator
      initialRouteName="Home"
      screenOptions={{
        headerStyle: { backgroundColor: '#6366f1' },
        headerTintColor: '#fff',
        headerTitleStyle: { fontWeight: '700' },
        animation: 'slide_from_right',
        headerBackTitleVisible: false,
      }}
    >
      <Stack.Screen
        name="Home"
        component={HomeScreen}
        options={{ title: 'Feed', headerShown: true }}
      />
      <Stack.Screen
        name="Detail"
        component={DetailScreen}
        options={({ route }) => ({
          title: route.params.title,
          animation: 'slide_from_bottom',
        })}
      />
      <Stack.Screen
        name="Profile"
        component={ProfileScreen}
        options={{
          title: 'Profile',
          headerShown: false, // Full-screen profile
          presentation: 'modal',
        }}
      />
    </Stack.Navigator>
  );
}
```

### Screen Options Reference

| Option           | Values                                                                       | Description                                |
| ---------------- | ---------------------------------------------------------------------------- | ------------------------------------------ |
| `headerShown`    | `boolean`                                                                    | Show or hide the header                    |
| `title`          | `string`                                                                     | Header title                               |
| `headerRight`    | `() => ReactNode`                                                            | Right header button                        |
| `headerLeft`     | `() => ReactNode`                                                            | Left header button (overrides back button) |
| `animation`      | `'default'`, `'slide_from_right'`, `'slide_from_bottom'`, `'fade'`, `'none'` | Screen transition                          |
| `presentation`   | `'card'`, `'modal'`, `'fullScreenModal'`, `'transparentModal'`               | How the screen appears                     |
| `gestureEnabled` | `boolean`                                                                    | Allow swipe-to-go-back                     |

## Tab Navigator

The Bottom Tab Navigator renders a tab bar at the bottom of the screen with persistent navigation between top-level sections.

```tsx
// src/navigation/TabNavigator.tsx
import React from 'react';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import { View } from 'react-native';
import Ionicons from '@expo/vector-icons/Ionicons';
import { HomeStack } from './HomeStack';
import { SearchScreen } from '../screens/SearchScreen';
import { NotificationsScreen } from '../screens/NotificationsScreen';
import { ProfileScreen } from '../screens/ProfileScreen';

export type TabParamList = {
  HomeTab: undefined;
  Search: undefined;
  Notifications: undefined;
  Profile: undefined;
};

const Tab = createBottomTabNavigator<TabParamList>();

export function TabNavigator() {
  const unreadCount = 3; // From your notification state

  return (
    <Tab.Navigator
      screenOptions={({ route }) => ({
        headerShown: false,
        tabBarIcon: ({ focused, color, size }) => {
          let iconName: keyof typeof Ionicons.glyphMap = 'home';

          if (route.name === 'HomeTab') {
            iconName = focused ? 'home' : 'home-outline';
          } else if (route.name === 'Search') {
            iconName = focused ? 'search' : 'search-outline';
          } else if (route.name === 'Notifications') {
            iconName = focused ? 'notifications' : 'notifications-outline';
          } else if (route.name === 'Profile') {
            iconName = focused ? 'person' : 'person-outline';
          }

          return <Ionicons name={iconName} size={size} color={color} />;
        },
        tabBarActiveTintColor: '#6366f1',
        tabBarInactiveTintColor: '#9ca3af',
        tabBarStyle: {
          backgroundColor: '#fff',
          borderTopColor: '#f0f0f0',
          height: 60,
          paddingBottom: 8,
        },
        tabBarLabelStyle: { fontSize: 12, fontWeight: '500' },
      })}
    >
      <Tab.Screen
        name="HomeTab"
        component={HomeStack}
        options={{ tabBarLabel: 'Home' }}
      />
      <Tab.Screen
        name="Search"
        component={SearchScreen}
        options={{ tabBarLabel: 'Search' }}
      />
      <Tab.Screen
        name="Notifications"
        component={NotificationsScreen}
        options={{
          tabBarLabel: 'Notifications',
          tabBarBadge: unreadCount > 0 ? unreadCount : undefined,
          tabBarBadgeStyle: { backgroundColor: '#ef4444' },
        }}
      />
      <Tab.Screen
        name="Profile"
        component={ProfileScreen}
        options={{ tabBarLabel: 'Profile' }}
      />
    </Tab.Navigator>
  );
}
```

## Drawer Navigator

A side drawer (hamburger menu) for apps with many top-level sections.

```tsx
import { createDrawerNavigator } from '@react-navigation/drawer';
import { HomeScreen } from '../screens/HomeScreen';
import { SettingsScreen } from '../screens/SettingsScreen';
import { HelpScreen } from '../screens/HelpScreen';

const Drawer = createDrawerNavigator();

export function DrawerNavigator() {
  return (
    <Drawer.Navigator
      initialRouteName="Home"
      screenOptions={{
        drawerPosition: 'left',
        drawerType: 'front',
        drawerStyle: { backgroundColor: '#1a1a2e', width: 280 },
        drawerActiveTintColor: '#6366f1',
        drawerInactiveTintColor: '#9ca3af',
        drawerLabelStyle: { fontSize: 16 },
        headerStyle: { backgroundColor: '#6366f1' },
        headerTintColor: '#fff',
      }}
    >
      <Drawer.Screen name="Home" component={HomeScreen} />
      <Drawer.Screen name="Settings" component={SettingsScreen} />
      <Drawer.Screen name="Help" component={HelpScreen} />
    </Drawer.Navigator>
  );
}
```

## Nested Navigators

Real apps combine multiple navigators. A common pattern is a Tab navigator inside a Stack navigator:

```tsx
// src/navigation/RootNavigator.tsx
import React from 'react';
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import { TabNavigator } from './TabNavigator';
import { AuthStack } from './AuthStack';
import { useAuthStore } from '../store/auth';

export type RootStackParamList = {
  Auth: undefined;
  Main: undefined;
  Modal: { screen: string };
};

const Root = createNativeStackNavigator<RootStackParamList>();

export function RootNavigator() {
  const isAuthenticated = useAuthStore((state) => state.isAuthenticated);

  return (
    <Root.Navigator screenOptions={{ headerShown: false }}>
      {isAuthenticated ? (
        <Root.Screen name="Main" component={TabNavigator} />
      ) : (
        <Root.Screen name="Auth" component={AuthStack} />
      )}
    </Root.Navigator>
  );
}
```

:::tip[Type Safety with Nested Navigators]
Use the `CompositeNavigationProp` type from React Navigation to get full type safety when navigating across nested navigator boundaries.
:::

## Passing Params to Screens

Pass data between screens using route params:

```tsx
// Navigating with params
navigation.navigate('Detail', { postId: '123', title: 'Hello World' });

// Receiving params in the target screen
import { useRoute } from '@react-navigation/native';
import type { NativeStackScreenProps } from '@react-navigation/native-stack';
import type { HomeStackParamList } from '../navigation/HomeStack';

type DetailScreenProps = NativeStackScreenProps<HomeStackParamList, 'Detail'>;

export function DetailScreen({ route, navigation }: DetailScreenProps) {
  const { postId, title } = route.params;

  return (
    <View>
      <Text>Post ID: {postId}</Text>
      <Text>Title: {title}</Text>
    </View>
  );
}
```

:::caution[Params Are Not Global State]
Route params are for navigation data, not global application state. Don't pass large objects, functions, or class instances as params — they won't survive serialisation and can cause issues with deep linking. Pass IDs and fetch the full data in the target screen.
:::

## Navigation Hooks

### useNavigation

```tsx
import { useNavigation } from '@react-navigation/native';
import type { NativeStackNavigationProp } from '@react-navigation/native-stack';
import type { HomeStackParamList } from '../navigation/HomeStack';

type HomeNavProp = NativeStackNavigationProp<HomeStackParamList>;

export function PostCard({ post }: { post: Post }) {
  const navigation = useNavigation<HomeNavProp>();

  return (
    <Pressable
      onPress={() =>
        navigation.navigate('Detail', { postId: post.id, title: post.title })
      }
    >
      <Text>{post.title}</Text>
    </Pressable>
  );
}
```

### useRoute

```tsx
import { useRoute } from '@react-navigation/native';
import type { RouteProp } from '@react-navigation/native';
import type { HomeStackParamList } from '../navigation/HomeStack';

type DetailRouteProp = RouteProp<HomeStackParamList, 'Detail'>;

export function DetailScreen() {
  const route = useRoute<DetailRouteProp>();
  const { postId } = route.params;
  // ...
}
```

### Other Useful Hooks

```tsx
import {
  useIsFocused, // true when screen is focused
  useNavigation,
  useRoute,
  useFocusEffect, // runs effect when screen is focused
  useNavigationState, // read navigation state
} from '@react-navigation/native';
import { useCallback } from 'react';

export function MyScreen() {
  const isFocused = useIsFocused();

  // Runs when screen comes into focus
  useFocusEffect(
    useCallback(() => {
      // Fetch fresh data when screen is focused
      fetchData();

      return () => {
        // Cleanup when screen loses focus
      };
    }, []),
  );

  return <Text>{isFocused ? 'Screen is visible' : 'Screen is hidden'}</Text>;
}
```

## Navigation Methods

```tsx
const navigation = useNavigation();

// Navigate to a screen (smart — won't add duplicate if already there)
navigation.navigate('Profile', { userId: '123' });

// Push always adds a new screen onto the stack
navigation.push('Profile', { userId: '456' });

// Replace current screen with a new one (no back button)
navigation.replace('Home');

// Go back one screen
navigation.goBack();

// Pop back to first screen in the stack
navigation.popToTop();

// Reset navigation state entirely
navigation.reset({
  index: 0,
  routes: [{ name: 'Home' }],
});

// Prefetch a screen (loads it before the user navigates)
navigation.preload('Detail', { postId: '789' });
```

## Deep Linking

Deep links allow URLs to open specific screens in your app.

```tsx
// App.tsx — linking configuration
import { NavigationContainer } from '@react-navigation/native';
import { Linking } from 'react-native';

const linking = {
  prefixes: ['myapp://', 'https://myapp.com'],
  config: {
    screens: {
      Main: {
        screens: {
          HomeTab: {
            screens: {
              Home: 'feed',
              Detail: 'post/:postId',
            },
          },
          Profile: 'profile/:userId',
        },
      },
      Auth: {
        screens: {
          Login: 'login',
          Reset: 'reset-password',
        },
      },
    },
  },
};

export default function App() {
  return (
    <NavigationContainer linking={linking} fallback={<LoadingScreen />}>
      <RootNavigator />
    </NavigationContainer>
  );
}
```


## Authentication Flow

Implement conditional navigation based on authentication state — a pattern where the navigator tree itself changes based on whether the user is logged in:

```tsx
// src/navigation/RootNavigator.tsx
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import { useAuthStore } from '../store/auth';
import { LoginScreen } from '../screens/LoginScreen';
import { SignUpScreen } from '../screens/SignUpScreen';
import { MainTabs } from './MainTabs';
import { SplashScreen } from '../screens/SplashScreen';

const Stack = createNativeStackNavigator();

export function RootNavigator() {
  const { isAuthenticated, isLoading } = useAuthStore();

  if (isLoading) {
    return <SplashScreen />;
  }

  return (
    <Stack.Navigator screenOptions={{ headerShown: false, animation: 'fade' }}>
      {isAuthenticated ? (
        // Authenticated screens
        <Stack.Screen name="Main" component={MainTabs} />
      ) : (
        // Auth screens — separate navigator group
        <Stack.Group>
          <Stack.Screen name="Login" component={LoginScreen} />
          <Stack.Screen name="SignUp" component={SignUpScreen} />
        </Stack.Group>
      )}
    </Stack.Navigator>
  );
}
```

When `isAuthenticated` changes (user logs in or out), React Navigation **automatically transitions** to the correct screen group. The auth screens are unmounted when the user is logged in, preventing the back button from going back to the login screen.

:::tip[Token Persistence]
On app startup, check AsyncStorage or SecureStore for a saved auth token before rendering the navigator. Set `isLoading = true` during this check and show a splash screen to avoid flashing the login screen.
:::

## Linking Module

Use the `Linking` API to open external URLs, email clients, phone diallers, or other apps:

```tsx
import { Linking, Pressable, Text } from 'react-native';

export function ExternalLinks() {
  const openLink = async (url: string) => {
    const supported = await Linking.canOpenURL(url);
    if (supported) {
      await Linking.openURL(url);
    } else {
      Alert.alert('Cannot open URL', url);
    }
  };

  return (
    <View>
      <Pressable onPress={() => openLink('https://reactnative.dev')}>
        <Text>Open Website</Text>
      </Pressable>
      <Pressable onPress={() => openLink('mailto:support@example.com')}>
        <Text>Send Email</Text>
      </Pressable>
      <Pressable onPress={() => openLink('tel:+15551234567')}>
        <Text>Call Us</Text>
      </Pressable>
      <Pressable onPress={() => openLink('sms:+15551234567')}>
        <Text>Send SMS</Text>
      </Pressable>
    </View>
  );
}
```

React Native apps need to manage several types of state: local UI state, shared global state, server-fetched data, and persisted data. This guide covers the full spectrum of options — from built-in hooks to production-ready libraries.

## Local State: useState and useReducer

For simple component-level state, `useState` is all you need. For more complex logic with multiple related values, reach for `useReducer`.

```tsx
import React, { useState, useReducer } from 'react';
import { View, TextInput, Pressable, Text, StyleSheet } from 'react-native';

// Simple useState example
export function SearchBar() {
  const [query, setQuery] = useState('');
  const [isFocused, setIsFocused] = useState(false);

  return (
    <View style={[styles.bar, isFocused && styles.barFocused]}>
      <TextInput
        value={query}
        onChangeText={setQuery}
        onFocus={() => setIsFocused(true)}
        onBlur={() => setIsFocused(false)}
        placeholder="Search..."
      />
      {query.length > 0 && (
        <Pressable onPress={() => setQuery('')}>
          <Text>✕</Text>
        </Pressable>
      )}
    </View>
  );
}

// useReducer for complex form state
type FormState = {
  email: string;
  password: string;
  loading: boolean;
  error: string | null;
};

type FormAction =
  | { type: 'SET_EMAIL'; payload: string }
  | { type: 'SET_PASSWORD'; payload: string }
  | { type: 'SUBMIT_START' }
  | { type: 'SUBMIT_SUCCESS' }
  | { type: 'SUBMIT_ERROR'; payload: string };

const initialState: FormState = {
  email: '',
  password: '',
  loading: false,
  error: null,
};

function formReducer(state: FormState, action: FormAction): FormState {
  switch (action.type) {
    case 'SET_EMAIL':
      return { ...state, email: action.payload, error: null };
    case 'SET_PASSWORD':
      return { ...state, password: action.payload, error: null };
    case 'SUBMIT_START':
      return { ...state, loading: true, error: null };
    case 'SUBMIT_SUCCESS':
      return { ...state, loading: false };
    case 'SUBMIT_ERROR':
      return { ...state, loading: false, error: action.payload };
    default:
      return state;
  }
}

export function LoginForm() {
  const [state, dispatch] = useReducer(formReducer, initialState);

  const handleLogin = async () => {
    dispatch({ type: 'SUBMIT_START' });
    try {
      await loginUser(state.email, state.password);
      dispatch({ type: 'SUBMIT_SUCCESS' });
    } catch (error) {
      dispatch({ type: 'SUBMIT_ERROR', payload: (error as Error).message });
    }
  };

  return (
    <View style={styles.form}>
      <TextInput
        value={state.email}
        onChangeText={(text) => dispatch({ type: 'SET_EMAIL', payload: text })}
        keyboardType="email-address"
        autoCapitalize="none"
      />
      <TextInput
        value={state.password}
        onChangeText={(text) =>
          dispatch({ type: 'SET_PASSWORD', payload: text })
        }
        secureTextEntry
      />
      {state.error && <Text style={styles.error}>{state.error}</Text>}
      <Pressable onPress={handleLogin} disabled={state.loading}>
        <Text>{state.loading ? 'Signing in...' : 'Sign In'}</Text>
      </Pressable>
    </View>
  );
}

const styles = StyleSheet.create({
  bar: {
    flexDirection: 'row',
    borderWidth: 1,
    borderColor: '#ddd',
    borderRadius: 8,
    padding: 10,
  },
  barFocused: { borderColor: '#6366f1' },
  form: { gap: 12, padding: 16 },
  error: { color: '#e53e3e', fontSize: 13 },
});
```

## Context API for Global State

React Context is built in and works well for low-frequency global state (theme, auth, locale). Avoid using it for high-frequency updates (it re-renders all consumers on every change).

```tsx
// src/context/ThemeContext.tsx
import React, { createContext, useContext, useState, useCallback } from 'react';

type Theme = 'light' | 'dark';

interface ThemeContextValue {
  theme: Theme;
  toggleTheme: () => void;
  colors: typeof lightColors | typeof darkColors;
}

const lightColors = {
  background: '#ffffff',
  text: '#111111',
  primary: '#6366f1',
  card: '#f9fafb',
  border: '#e5e7eb',
};

const darkColors = {
  background: '#111827',
  text: '#f9fafb',
  primary: '#818cf8',
  card: '#1f2937',
  border: '#374151',
};

const ThemeContext = createContext<ThemeContextValue | null>(null);

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<Theme>('light');

  const toggleTheme = useCallback(() => {
    setTheme((t) => (t === 'light' ? 'dark' : 'light'));
  }, []);

  const colors = theme === 'light' ? lightColors : darkColors;

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme, colors }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme(): ThemeContextValue {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within a ThemeProvider');
  }
  return context;
}
```

## Zustand for Global State

Zustand is the recommended lightweight state manager for React Native. It's simple, fast, and avoids the boilerplate of Redux.

```bash
npm install zustand
```

```tsx
// src/store/auth.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';

interface User {
  id: string;
  email: string;
  name: string;
  avatar: string | null;
}

interface AuthState {
  user: User | null;
  token: string | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  login: (email: string, password: string) => Promise<void>;
  logout: () => void;
  updateProfile: (data: Partial<User>) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      token: null,
      isAuthenticated: false,
      isLoading: false,

      login: async (email, password) => {
        set({ isLoading: true });
        try {
          const { user, token } = await apiLogin(email, password);
          set({ user, token, isAuthenticated: true, isLoading: false });
        } catch (error) {
          set({ isLoading: false });
          throw error;
        }
      },

      logout: () => {
        set({ user: null, token: null, isAuthenticated: false });
      },

      updateProfile: (data) => {
        const currentUser = get().user;
        if (currentUser) {
          set({ user: { ...currentUser, ...data } });
        }
      },
    }),
    {
      name: 'auth-storage',
      storage: createJSONStorage(() => AsyncStorage),
      // Only persist specific fields
      partialize: (state) => ({
        token: state.token,
        user: state.user,
        isAuthenticated: state.isAuthenticated,
      }),
    },
  ),
);

// Usage in a component
function ProfileHeader() {
  const { user, logout } = useAuthStore();

  return (
    <View>
      <Text>{user?.name}</Text>
      <Pressable onPress={logout}>
        <Text>Sign Out</Text>
      </Pressable>
    </View>
  );
}
```

## TanStack Query (React Query)

TanStack Query manages server state — fetching, caching, background updates, and synchronisation with the server.

```bash
npm install @tanstack/react-query
```

<Steps>
1. **Set up the QueryClient**

```tsx
// App.tsx
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60 * 5, // 5 minutes
      gcTime: 1000 * 60 * 30, // 30 minutes
      retry: 2,
      retryDelay: (attempt) => Math.min(1000 * 2 ** attempt, 30000),
    },
  },
});

export default function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <NavigationContainer>
        <RootNavigator />
      </NavigationContainer>
    </QueryClientProvider>
  );
}
```

2. **Use useQuery for data fetching**

   ```tsx
   // src/hooks/usePosts.ts
   import { useQuery, useInfiniteQuery } from '@tanstack/react-query';

   export function usePosts() {
     return useQuery({
       queryKey: ['posts'],
       queryFn: async () => {
         const response = await fetch('https://api.example.com/posts');
         if (!response.ok) throw new Error('Failed to fetch posts');
         return response.json() as Promise<Post[]>;
       },
     });
   }

   // Infinite scrolling feed
   export function useInfinitePosts() {
     return useInfiniteQuery({
       queryKey: ['posts', 'infinite'],
       queryFn: async ({ pageParam = 0 }) => {
         const response = await fetch(
           `https://api.example.com/posts?cursor=${pageParam}&limit=20`,
         );
         return response.json() as Promise<{
           posts: Post[];
           nextCursor: string | null;
         }>;
       },
       getNextPageParam: (lastPage) => lastPage.nextCursor ?? undefined,
       initialPageParam: 0,
     });
   }
   ```

3. **Use useMutation for writes**

   ```tsx
   import { useMutation, useQueryClient } from '@tanstack/react-query';

   export function useCreatePost() {
     const queryClient = useQueryClient();

     return useMutation({
       mutationFn: async (newPost: { title: string; body: string }) => {
         const response = await fetch('https://api.example.com/posts', {
           method: 'POST',
           headers: { 'Content-Type': 'application/json' },
           body: JSON.stringify(newPost),
         });
         if (!response.ok) throw new Error('Failed to create post');
         return response.json() as Promise<Post>;
       },
       onSuccess: (data) => {
         // Optimistically add new post to the list
         queryClient.setQueryData<Post[]>(['posts'], (old) =>
           old ? [data, ...old] : [data],
         );
         // Refetch in background to ensure consistency
         queryClient.invalidateQueries({ queryKey: ['posts'] });
       },
     });
   }
   ```

</Steps>

## MMKV: Fast Key-Value Storage

MMKV is a C++ key-value storage library offering read/write speeds ~10x faster than AsyncStorage. Ideal for preferences, tokens, and small cached values.

```bash
npm install react-native-mmkv
```

```tsx
// src/lib/storage.ts
import { MMKV } from 'react-native-mmkv';

export const storage = new MMKV({
  id: 'app-storage',
  encryptionKey: 'my-secret-key', // Optional AES encryption
});

// Typed storage helpers
export const StorageKeys = {
  AUTH_TOKEN: 'auth_token',
  USER_ID: 'user_id',
  THEME: 'theme',
  ONBOARDED: 'onboarded',
} as const;

export const secureStorage = {
  setToken: (token: string) => storage.set(StorageKeys.AUTH_TOKEN, token),
  getToken: () => storage.getString(StorageKeys.AUTH_TOKEN) ?? null,
  clearToken: () => storage.delete(StorageKeys.AUTH_TOKEN),

  setUserId: (id: string) => storage.set(StorageKeys.USER_ID, id),
  getUserId: () => storage.getString(StorageKeys.USER_ID) ?? null,

  setOnboarded: (value: boolean) => storage.set(StorageKeys.ONBOARDED, value),
  isOnboarded: () => storage.getBoolean(StorageKeys.ONBOARDED) ?? false,

  clear: () => storage.clearAll(),
};

// MMKV supports listeners for reactive state
storage.addOnValueChangedListener((changedKey) => {
  if (changedKey === StorageKeys.THEME) {
    // React to theme changes
  }
});
```

## AsyncStorage (Legacy)

AsyncStorage is the built-in asynchronous key-value store. Slower than MMKV but widely supported and built into Expo.

:::caution[MMKV Preferred]
For new projects, prefer `react-native-mmkv` for its performance advantages. AsyncStorage is still suitable for simple use cases and has zero native setup with Expo.
:::

```bash
npx expo install @react-native-async-storage/async-storage
```

```tsx
import AsyncStorage from '@react-native-async-storage/async-storage';

// Helper for typed JSON storage
export const Storage = {
  async get<T>(key: string): Promise<T | null> {
    try {
      const value = await AsyncStorage.getItem(key);
      return value ? (JSON.parse(value) as T) : null;
    } catch {
      return null;
    }
  },

  async set(key: string, value: unknown): Promise<void> {
    await AsyncStorage.setItem(key, JSON.stringify(value));
  },

  async remove(key: string): Promise<void> {
    await AsyncStorage.removeItem(key);
  },

  async clear(): Promise<void> {
    await AsyncStorage.clear();
  },
};
```

## SQLite with expo-sqlite

For structured, relational data that needs complex queries.

```bash
npx expo install expo-sqlite
```

```tsx
import * as SQLite from 'expo-sqlite';

// Open (or create) a database
const db = SQLite.openDatabaseSync('app.db');

// Create tables
db.execSync(`
  CREATE TABLE IF NOT EXISTS notes (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    title TEXT NOT NULL,
    body TEXT,
    created_at INTEGER NOT NULL DEFAULT (strftime('%s', 'now')),
    updated_at INTEGER NOT NULL DEFAULT (strftime('%s', 'now'))
  );

  CREATE INDEX IF NOT EXISTS idx_notes_created ON notes(created_at DESC);
`);

// CRUD operations with the async API
export async function createNote(title: string, body: string): Promise<number> {
  const result = await db.runAsync(
    'INSERT INTO notes (title, body) VALUES (?, ?)',
    [title, body],
  );
  return result.lastInsertRowId;
}

export async function getNotes(): Promise<Note[]> {
  return db.getAllAsync<Note>('SELECT * FROM notes ORDER BY created_at DESC');
}

export async function updateNote(
  id: number,
  title: string,
  body: string,
): Promise<void> {
  await db.runAsync(
    "UPDATE notes SET title = ?, body = ?, updated_at = strftime('%s', 'now') WHERE id = ?",
    [title, body, id],
  );
}

export async function deleteNote(id: number): Promise<void> {
  await db.runAsync('DELETE FROM notes WHERE id = ?', [id]);
}
```

## HTTP with fetch and axios

```tsx
// src/lib/api.ts
import axios, { type AxiosInstance } from 'axios';
import { secureStorage } from './storage';

// Create an axios instance with auth interceptor
export const api: AxiosInstance = axios.create({
  baseURL: 'https://api.example.com',
  timeout: 10000,
  headers: { 'Content-Type': 'application/json' },
});

// Attach auth token to every request
api.interceptors.request.use((config) => {
  const token = secureStorage.getToken();
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Handle 401 by redirecting to login
api.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      secureStorage.clearToken();
      // Navigate to login screen
    }
    return Promise.reject(error);
  },
);
```

## SecureStore for Sensitive Data

Use `expo-secure-store` for tokens, passwords, and other sensitive values. Data is encrypted using the platform's secure enclave (Keychain on iOS, Keystore on Android).

```bash
npx expo install expo-secure-store
```

```tsx
import * as SecureStore from 'expo-secure-store';

export const TokenStorage = {
  async save(token: string): Promise<void> {
    await SecureStore.setItemAsync('auth_token', token, {
      keychainAccessible: SecureStore.WHEN_UNLOCKED,
    });
  },

  async get(): Promise<string | null> {
    return SecureStore.getItemAsync('auth_token');
  },

  async delete(): Promise<void> {
    await SecureStore.deleteItemAsync('auth_token');
  },
};
```

## Offline-First Patterns

:::tip[Offline-First Strategy]
Design your app to work without a network connection by default. Cache data locally, queue mutations, and sync when connectivity is restored.
:::

```tsx
// src/hooks/useOfflineSync.ts
import NetInfo from '@react-native-community/netinfo';
import { useEffect } from 'react';
import { useQueryClient } from '@tanstack/react-query';

export function useOfflineSync() {
  const queryClient = useQueryClient();

  useEffect(() => {
    // TanStack Query v5 has built-in online/offline support
    const unsubscribe = NetInfo.addEventListener((state) => {
      const isOnline = state.isConnected && state.isInternetReachable;

      if (isOnline) {
        // Refetch stale queries when coming back online
        queryClient.resumePausedMutations();
        queryClient.invalidateQueries();
      }
    });

    return unsubscribe;
  }, [queryClient]);
}
```

```tsx
// Persist query cache to AsyncStorage for offline access
import { PersistQueryClientProvider } from '@tanstack/react-query-persist-client';
import { createAsyncStoragePersister } from '@tanstack/query-async-storage-persister';
import AsyncStorage from '@react-native-async-storage/async-storage';

const persister = createAsyncStoragePersister({
  storage: AsyncStorage,
  throttleTime: 3000,
});

export default function App() {
  return (
    <PersistQueryClientProvider
      client={queryClient}
      persistOptions={{ persister, maxAge: 1000 * 60 * 60 * 24 }} // 24 hours
    >
      <NavigationContainer>
        <RootNavigator />
      </NavigationContainer>
    </PersistQueryClientProvider>
  );
}
```

## State Management Decision Guide

| Use Case                                | Recommended Solution |
| --------------------------------------- | -------------------- |
| UI toggle, input value, scroll position | `useState`           |
| Complex form or multi-step flow         | `useReducer`         |
| Theme, auth status (low frequency)      | Context API          |
| Global app state (cart, settings)       | Zustand              |
| Server data (lists, profiles, feeds)    | TanStack Query       |
| Fast preferences, tokens                | MMKV                 |
| Auth tokens, passwords                  | expo-secure-store    |
| Structured relational data              | expo-sqlite          |
| Large file cache                        | expo-file-system     |

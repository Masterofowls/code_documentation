React Native uses the same React state primitives as the web, but mobile apps have unique requirements: offline-first data, fast local storage, secure credential storage, and sync with remote APIs.

## Local Component State

### useState

For simple, independent component state:

```tsx
import React, { useState } from 'react';
import { View, Text, Pressable } from 'react-native';

export function Counter() {
  const [count, setCount] = useState(0);
  const [error, setError] = useState<string | null>(null);

  // Functional updates avoid stale closure issues
  const increment = () => setCount(prev => prev + 1);
  const decrement = () => setCount(prev => Math.max(0, prev - 1));

  return (
    <View>
      {error && <Text style={{ color: 'red' }}>{error}</Text>}
      <Text>{count}</Text>
      <Pressable onPress={increment}><Text>+</Text></Pressable>
      <Pressable onPress={decrement}><Text>-</Text></Pressable>
    </View>
  );
}
```

### useReducer

For complex state with multiple related values or complex update logic:

```tsx
import React, { useReducer } from 'react';

interface FormState {
  email: string;
  password: string;
  isSubmitting: boolean;
  error: string | null;
}

type FormAction =
  | { type: 'SET_FIELD'; field: keyof Pick<FormState, 'email' | 'password'>; value: string }
  | { type: 'SUBMIT_START' }
  | { type: 'SUBMIT_SUCCESS' }
  | { type: 'SUBMIT_ERROR'; error: string };

const initialState: FormState = {
  email: '',
  password: '',
  isSubmitting: false,
  error: null,
};

function formReducer(state: FormState, action: FormAction): FormState {
  switch (action.type) {
    case 'SET_FIELD':
      return { ...state, [action.field]: action.value, error: null };
    case 'SUBMIT_START':
      return { ...state, isSubmitting: true, error: null };
    case 'SUBMIT_SUCCESS':
      return { ...state, isSubmitting: false };
    case 'SUBMIT_ERROR':
      return { ...state, isSubmitting: false, error: action.error };
  }
}

export function LoginForm() {
  const [state, dispatch] = useReducer(formReducer, initialState);

  const handleSubmit = async () => {
    dispatch({ type: 'SUBMIT_START' });
    try {
      await signIn(state.email, state.password);
      dispatch({ type: 'SUBMIT_SUCCESS' });
    } catch (err) {
      dispatch({ type: 'SUBMIT_ERROR', error: 'Invalid credentials' });
    }
  };

  return (/* form JSX */);
}
```

## Context API

Context is best for low-frequency global data like theme, locale, or auth state. Avoid putting frequently-updated data in context — every consumer re-renders.

```tsx
// src/context/ThemeContext.tsx
import React, { createContext, useContext, useState, useCallback } from 'react';

type Theme = 'light' | 'dark';

interface ThemeContextValue {
  theme: Theme;
  toggleTheme: () => void;
  colors: typeof lightColors;
}

const lightColors = {
  background: '#ffffff',
  surface: '#f5f5f5',
  text: '#111111',
  subtext: '#666666',
  primary: '#007AFF',
  border: '#e0e0e0',
};

const darkColors = {
  background: '#000000',
  surface: '#1c1c1e',
  text: '#ffffff',
  subtext: '#aeaeb2',
  primary: '#0a84ff',
  border: '#38383a',
};

const ThemeContext = createContext<ThemeContextValue | undefined>(undefined);

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<Theme>('light');
  const toggleTheme = useCallback(() => setTheme(t => t === 'light' ? 'dark' : 'light'), []);
  const colors = theme === 'light' ? lightColors : darkColors;

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme, colors }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const ctx = useContext(ThemeContext);
  if (!ctx) throw new Error('useTheme must be used within ThemeProvider');
  return ctx;
}
```

## Zustand

Zustand is a minimal, fast state management library — no boilerplate, no providers needed.

```bash
npm install zustand
```

```typescript
// src/store/authStore.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';

interface User {
  id: string;
  email: string;
  name: string;
  avatarUrl?: string;
}

interface AuthState {
  user: User | null;
  token: string | null;
  isLoading: boolean;
  signIn: (email: string, password: string) => Promise<void>;
  signOut: () => void;
  updateUser: (updates: Partial<User>) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      token: null,
      isLoading: false,

      signIn: async (email, password) => {
        set({ isLoading: true });
        try {
          const { user, token } = await authApi.login(email, password);
          set({ user, token, isLoading: false });
        } catch (error) {
          set({ isLoading: false });
          throw error;
        }
      },

      signOut: () => {
        set({ user: null, token: null });
      },

      updateUser: (updates) => {
        const current = get().user;
        if (current) {
          set({ user: { ...current, ...updates } });
        }
      },
    }),
    {
      name: 'auth-storage',
      storage: createJSONStorage(() => AsyncStorage),
      // Only persist these keys:
      partialize: (state) => ({ user: state.user, token: state.token }),
    }
  )
);
```

```typescript
// src/store/uiStore.ts — example of a non-persisted store
import { create } from 'zustand';

interface UIState {
  bottomSheetOpen: boolean;
  activeModal: string | null;
  openModal: (id: string) => void;
  closeModal: () => void;
  toggleBottomSheet: () => void;
}

export const useUIStore = create<UIState>((set) => ({
  bottomSheetOpen: false,
  activeModal: null,
  openModal: (id) => set({ activeModal: id }),
  closeModal: () => set({ activeModal: null }),
  toggleBottomSheet: () => set(s => ({ bottomSheetOpen: !s.bottomSheetOpen })),
}));
```

:::tip[Zustand Selectors]
Always select only the state you need to prevent unnecessary re-renders:
```typescript
// Bad — re-renders when ANY state changes:
const { user, token, signIn } = useAuthStore();

// Good — re-renders only when `user` changes:
const user = useAuthStore(s => s.user);
const signIn = useAuthStore(s => s.signIn);
```
:::

## TanStack Query (React Query)

TanStack Query handles all your server state: fetching, caching, background refetching, error handling, and pagination.

```bash
npm install @tanstack/react-query
```

```tsx
// App.tsx — set up QueryClient
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60 * 5,     // 5 minutes
      gcTime: 1000 * 60 * 30,       // 30 minutes
      retry: 2,
      retryDelay: attemptIndex => Math.min(1000 * 2 ** attemptIndex, 30000),
      refetchOnWindowFocus: false,   // Not useful on mobile
      refetchOnReconnect: true,      // Refetch when back online
    },
    mutations: {
      retry: 0,
    },
  },
});

export default function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <NavigationContainer>
        <RootNavigator />
      </NavigationContainer>
    </QueryClientProvider>
  );
}
```

```typescript
// src/api/posts.ts — API functions
export const postsApi = {
  list: async (page: number): Promise<{ posts: Post[]; hasMore: boolean }> => {
    const res = await fetch(`/api/posts?page=${page}`);
    if (!res.ok) throw new Error('Failed to fetch posts');
    return res.json();
  },

  get: async (id: string): Promise<Post> => {
    const res = await fetch(`/api/posts/${id}`);
    if (!res.ok) throw new Error('Post not found');
    return res.json();
  },

  create: async (data: CreatePostDto): Promise<Post> => {
    const res = await fetch('/api/posts', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });
    if (!res.ok) throw new Error('Failed to create post');
    return res.json();
  },

  delete: async (id: string): Promise<void> => {
    await fetch(`/api/posts/${id}`, { method: 'DELETE' });
  },
};
```

```typescript
// src/hooks/usePosts.ts — query hooks
import { useQuery, useMutation, useInfiniteQuery, useQueryClient } from '@tanstack/react-query';

// Query keys factory for consistency
export const postKeys = {
  all: ['posts'] as const,
  lists: () => [...postKeys.all, 'list'] as const,
  list: (page: number) => [...postKeys.lists(), { page }] as const,
  details: () => [...postKeys.all, 'detail'] as const,
  detail: (id: string) => [...postKeys.details(), id] as const,
};

// Basic query
export function usePost(id: string) {
  return useQuery({
    queryKey: postKeys.detail(id),
    queryFn: () => postsApi.get(id),
    enabled: !!id,
    staleTime: 1000 * 60 * 2,
  });
}

// Infinite scroll query
export function useInfinitePosts() {
  return useInfiniteQuery({
    queryKey: postKeys.lists(),
    queryFn: ({ pageParam }) => postsApi.list(pageParam),
    initialPageParam: 1,
    getNextPageParam: (lastPage, _, lastPageParam) =>
      lastPage.hasMore ? lastPageParam + 1 : undefined,
  });
}

// Create mutation
export function useCreatePost() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: postsApi.create,
    onSuccess: (newPost) => {
      // Invalidate and refetch list
      queryClient.invalidateQueries({ queryKey: postKeys.lists() });
      // Optimistically add to cache
      queryClient.setQueryData(postKeys.detail(newPost.id), newPost);
    },
    onError: (error) => {
      console.error('Failed to create post:', error);
    },
  });
}

// Delete mutation
export function useDeletePost() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: postsApi.delete,
    onMutate: async (deletedId) => {
      // Optimistic update
      await queryClient.cancelQueries({ queryKey: postKeys.lists() });
      const previousData = queryClient.getQueryData(postKeys.lists());
      queryClient.setQueriesData({ queryKey: postKeys.lists() }, (old: any) =>
        old ? { ...old, posts: old.posts.filter((p: Post) => p.id !== deletedId) } : old
      );
      return { previousData };
    },
    onError: (_, __, context) => {
      // Roll back on error
      if (context?.previousData) {
        queryClient.setQueryData(postKeys.lists(), context.previousData);
      }
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: postKeys.lists() });
    },
  });
}
```

## MMKV (Fast Local Storage)

MMKV is a key-value storage built on WeChat's MMKV framework. It is 10–30x faster than AsyncStorage for reads and writes.

```bash
# Expo (requires development build, not Expo Go)
npx expo install react-native-mmkv

# Run pod install for iOS
cd ios && pod install
```

```typescript
// src/storage/mmkv.ts
import { MMKV } from 'react-native-mmkv';

// Default instance
export const storage = new MMKV();

// Typed storage helper
export const appStorage = {
  getString: (key: string) => storage.getString(key),
  setString: (key: string, value: string) => storage.set(key, value),
  getNumber: (key: string) => storage.getNumber(key),
  setNumber: (key: string, value: number) => storage.set(key, value),
  getBool: (key: string) => storage.getBoolean(key),
  setBool: (key: string, value: boolean) => storage.set(key, value),
  delete: (key: string) => storage.delete(key),
  clear: () => storage.clearAll(),

  // JSON helper
  getObject: <T>(key: string): T | undefined => {
    const raw = storage.getString(key);
    return raw ? (JSON.parse(raw) as T) : undefined;
  },
  setObject: <T>(key: string, value: T) => {
    storage.set(key, JSON.stringify(value));
  },
};

// Use MMKV as Zustand storage backend
import { StateStorage } from 'zustand/middleware';
export const mmkvZustandStorage: StateStorage = {
  getItem: (name) => storage.getString(name) ?? null,
  setItem: (name, value) => storage.set(name, value),
  removeItem: (name) => storage.delete(name),
};
```

## AsyncStorage (Legacy)

AsyncStorage is the legacy solution — still widely used but slower than MMKV. Use it if you can't use MMKV (e.g., in Expo Go during development).

```bash
npx expo install @react-native-async-storage/async-storage
```

```typescript
import AsyncStorage from '@react-native-async-storage/async-storage';

// Always wrap in try/catch — it's async and can fail
const setItem = async (key: string, value: string) => {
  try {
    await AsyncStorage.setItem(key, value);
  } catch (error) {
    console.error('AsyncStorage write error:', error);
  }
};

const getItem = async (key: string): Promise<string | null> => {
  try {
    return await AsyncStorage.getItem(key);
  } catch (error) {
    console.error('AsyncStorage read error:', error);
    return null;
  }
};

// Multi-set/get for batch operations
await AsyncStorage.multiSet([
  ['@user_id', '123'],
  ['@settings', JSON.stringify({ darkMode: true })],
]);

const [[, userId], [, settingsRaw]] = await AsyncStorage.multiGet([
  '@user_id',
  '@settings',
]);
```

## SQLite with expo-sqlite

For relational data, full-text search, or complex queries:

```bash
npx expo install expo-sqlite
```

```typescript
// src/db/database.ts
import * as SQLite from 'expo-sqlite';

const db = SQLite.openDatabaseSync('myapp.db');

// Create tables (run on app start)
export function initDatabase() {
  db.execSync(`
    PRAGMA journal_mode = WAL;
    PRAGMA foreign_keys = ON;

    CREATE TABLE IF NOT EXISTS notes (
      id TEXT PRIMARY KEY,
      title TEXT NOT NULL,
      content TEXT NOT NULL,
      created_at INTEGER NOT NULL,
      updated_at INTEGER NOT NULL,
      synced INTEGER NOT NULL DEFAULT 0
    );

    CREATE INDEX IF NOT EXISTS idx_notes_updated_at ON notes(updated_at);
  `);
}

// CRUD operations
export const notesDb = {
  insert: (note: Note) => {
    db.runSync(
      'INSERT INTO notes (id, title, content, created_at, updated_at) VALUES (?, ?, ?, ?, ?)',
      [note.id, note.title, note.content, Date.now(), Date.now()]
    );
  },

  getAll: (): Note[] => {
    return db.getAllSync('SELECT * FROM notes ORDER BY updated_at DESC');
  },

  getById: (id: string): Note | null => {
    return db.getFirstSync('SELECT * FROM notes WHERE id = ?', [id]);
  },

  update: (id: string, updates: Partial<Note>) => {
    db.runSync(
      'UPDATE notes SET title = ?, content = ?, updated_at = ? WHERE id = ?',
      [updates.title!, updates.content!, Date.now(), id]
    );
  },

  delete: (id: string) => {
    db.runSync('DELETE FROM notes WHERE id = ?', [id]);
  },

  getUnsynced: (): Note[] => {
    return db.getAllSync('SELECT * FROM notes WHERE synced = 0');
  },

  markSynced: (ids: string[]) => {
    const placeholders = ids.map(() => '?').join(',');
    db.runSync(`UPDATE notes SET synced = 1 WHERE id IN (${placeholders})`, ids);
  },
};
```

```typescript
// src/storage/secureStorage.ts
import * as SecureStore from 'expo-secure-store';

export const secureStorage = {
  // Stored in iOS Keychain / Android Keystore
  setToken: async (token: string) => {
    await SecureStore.setItemAsync('auth_token', token, {
      keychainAccessible: SecureStore.WHEN_UNLOCKED_THIS_DEVICE_ONLY,
    });
  },

  getToken: async (): Promise<string | null> => {
    return SecureStore.getItemAsync('auth_token');
  },

  deleteToken: async () => {
    await SecureStore.deleteItemAsync('auth_token');
  },
};
```

## HTTP with Axios

```typescript
// src/api/client.ts
import axios from 'axios';
import { secureStorage } from '../storage/secureStorage';

const apiClient = axios.create({
  baseURL: process.env.EXPO_PUBLIC_API_URL ?? 'https://api.example.com',
  timeout: 10000,
  headers: { 'Content-Type': 'application/json' },
});

// Request interceptor — attach auth token
apiClient.interceptors.request.use(async (config) => {
  const token = await secureStorage.getToken();
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Response interceptor — handle 401
apiClient.interceptors.response.use(
  response => response,
  async (error) => {
    if (error.response?.status === 401) {
      await secureStorage.deleteToken();
      // Redirect to login (use navigation ref)
      navigationRef.reset({ index: 0, routes: [{ name: 'Login' }] });
    }
    return Promise.reject(error);
  }
);

export { apiClient };
```

## Offline-First Patterns

:::tip[Offline-First Strategy]
Design your app to work without a network connection. Store data locally first, then sync in the background. Users expect mobile apps to work on the subway.
:::

```typescript
// src/hooks/useOfflineSync.ts
import NetInfo from '@react-native-community/netinfo';
import { useEffect } from 'react';
import { notesDb } from '../db/database';
import { apiClient } from '../api/client';

export function useOfflineSync() {
  useEffect(() => {
    // Subscribe to network state changes
    const unsubscribe = NetInfo.addEventListener(async (state) => {
      if (state.isConnected && state.isInternetReachable) {
        await syncPendingChanges();
      }
    });
    return unsubscribe;
  }, []);

  const syncPendingChanges = async () => {
    const unsyncedNotes = notesDb.getUnsynced();
    if (!unsyncedNotes.length) return;

    try {
      const { data } = await apiClient.post('/notes/batch', { notes: unsyncedNotes });
      notesDb.markSynced(unsyncedNotes.map(n => n.id));
      console.log(`Synced ${unsyncedNotes.length} notes`);
    } catch (error) {
      console.warn('Sync failed, will retry when online:', error);
    }
  };
}
```

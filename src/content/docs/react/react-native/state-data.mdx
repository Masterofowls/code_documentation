React Native apps have the same state management needs as React web apps — plus the additional concerns of offline-first design, device storage, and secure credential handling. This guide covers the full stack from local component state to server state caching.

## Local State: useState and useReducer

For state that lives within a single component or a small subtree, React's built-in hooks are all you need.

```tsx
import React, { useState, useReducer } from 'react';
import { View, TextInput, Pressable, Text } from 'react-native';

// useState for simple values:
function Counter() {
  const [count, setCount] = useState(0);
  return (
    <Pressable onPress={() => setCount(c => c + 1)}>
      <Text>Count: {count}</Text>
    </Pressable>
  );
}

// useReducer for complex state with multiple sub-values:
type CartState = {
  items: CartItem[];
  total: number;
  isLoading: boolean;
};

type CartAction =
  | { type: 'ADD_ITEM'; payload: CartItem }
  | { type: 'REMOVE_ITEM'; payload: string }
  | { type: 'CLEAR_CART' }
  | { type: 'SET_LOADING'; payload: boolean };

function cartReducer(state: CartState, action: CartAction): CartState {
  switch (action.type) {
    case 'ADD_ITEM': {
      const items = [...state.items, action.payload];
      return { ...state, items, total: items.reduce((s, i) => s + i.price, 0) };
    }
    case 'REMOVE_ITEM': {
      const items = state.items.filter(i => i.id !== action.payload);
      return { ...state, items, total: items.reduce((s, i) => s + i.price, 0) };
    }
    case 'CLEAR_CART':
      return { items: [], total: 0, isLoading: false };
    case 'SET_LOADING':
      return { ...state, isLoading: action.payload };
    default:
      return state;
  }
}

function CartScreen() {
  const [state, dispatch] = useReducer(cartReducer, {
    items: [],
    total: 0,
    isLoading: false,
  });

  return <View>{/* ... */}</View>;
}
```

## Context API for Global State

The Context API is ideal for app-wide state like theme, locale, or auth user — values that change infrequently.

```tsx
// src/context/ThemeContext.tsx
import React, { createContext, useContext, useState, useMemo } from 'react';

type Theme = 'light' | 'dark';

interface ThemeContextValue {
  theme: Theme;
  colors: typeof lightColors;
  toggleTheme: () => void;
}

const lightColors = {
  background: '#f8fafc',
  surface: '#ffffff',
  text: '#1a202c',
  textSecondary: '#718096',
  primary: '#4f46e5',
  border: '#e2e8f0',
};

const darkColors: typeof lightColors = {
  background: '#0f172a',
  surface: '#1e293b',
  text: '#f1f5f9',
  textSecondary: '#94a3b8',
  primary: '#818cf8',
  border: '#334155',
};

const ThemeContext = createContext<ThemeContextValue | undefined>(undefined);

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<Theme>('light');

  const value = useMemo<ThemeContextValue>(
    () => ({
      theme,
      colors: theme === 'light' ? lightColors : darkColors,
      toggleTheme: () => setTheme(t => (t === 'light' ? 'dark' : 'light')),
    }),
    [theme],
  );

  return <ThemeContext.Provider value={value}>{children}</ThemeContext.Provider>;
}

export function useTheme() {
  const ctx = useContext(ThemeContext);
  if (!ctx) throw new Error('useTheme must be used within ThemeProvider');
  return ctx;
}

// Usage:
function Card() {
  const { colors } = useTheme();
  return (
    <View style={{ backgroundColor: colors.surface, borderColor: colors.border }}>
      <Text style={{ color: colors.text }}>Content</Text>
    </View>
  );
}
```

:::tip[Context is Not a State Manager]
Context is great for static-ish global values. For high-frequency updates (animations, form state, live data), Context re-renders every consumer on every change. Use Zustand or Jotai for that.
:::

## Zustand for Global State

Zustand is a minimal, hook-based state manager with zero boilerplate. It shines for client-side global state like user profile, cart, and UI preferences.

```bash
npm install zustand
```

```typescript
// src/store/cartStore.ts
import { create } from 'zustand';
import { devtools, persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';

interface CartItem {
  id: string;
  name: string;
  price: number;
  quantity: number;
}

interface CartStore {
  items: CartItem[];
  addItem: (item: Omit<CartItem, 'quantity'>) => void;
  removeItem: (id: string) => void;
  updateQuantity: (id: string, quantity: number) => void;
  clearCart: () => void;
  total: () => number;
  itemCount: () => number;
}

export const useCartStore = create<CartStore>()(
  devtools(
    persist(
      (set, get) => ({
        items: [],

        addItem: (newItem) =>
          set((state) => {
            const existing = state.items.find(i => i.id === newItem.id);
            if (existing) {
              return {
                items: state.items.map(i =>
                  i.id === newItem.id ? { ...i, quantity: i.quantity + 1 } : i,
                ),
              };
            }
            return { items: [...state.items, { ...newItem, quantity: 1 }] };
          }),

        removeItem: (id) =>
          set((state) => ({ items: state.items.filter(i => i.id !== id) })),

        updateQuantity: (id, quantity) =>
          set((state) => ({
            items: quantity <= 0
              ? state.items.filter(i => i.id !== id)
              : state.items.map(i => i.id === id ? { ...i, quantity } : i),
          })),

        clearCart: () => set({ items: [] }),

        total: () => get().items.reduce((sum, i) => sum + i.price * i.quantity, 0),

        itemCount: () => get().items.reduce((sum, i) => sum + i.quantity, 0),
      }),
      {
        name: 'cart-storage',
        storage: createJSONStorage(() => AsyncStorage),
        // Only persist the items array, not computed functions:
        partialize: (state) => ({ items: state.items }),
      },
    ),
    { name: 'CartStore' },
  ),
);

// Usage in a component:
function CartIcon() {
  const itemCount = useCartStore(state => state.itemCount());
  return <Text>{itemCount}</Text>;
}
```

:::caution[Selector Granularity]
Always select only the slice of state you need: `useCartStore(state => state.items)` — not the whole store. Selecting the entire store object causes unnecessary re-renders on every state change.
:::

## TanStack Query for Server State

TanStack Query (formerly React Query) is the gold standard for managing async server data: fetching, caching, background refetching, and mutations.

```bash
npm install @tanstack/react-query
```

### Setup

```tsx
// App.tsx
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 5 * 60 * 1000,   // Data is fresh for 5 minutes
      gcTime: 10 * 60 * 1000,     // Keep unused data in cache for 10 minutes
      retry: 2,
      refetchOnWindowFocus: false, // Disable — mobile apps don't have "focus"
      refetchOnReconnect: true,    // Refetch when network reconnects
    },
  },
});

export default function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <NavigationContainer>{/* ... */}</NavigationContainer>
    </QueryClientProvider>
  );
}
```

### useQuery

```typescript
// src/hooks/useProducts.ts
import { useQuery } from '@tanstack/react-query';
import { fetchProducts, fetchProductById } from '../api/products';

// Query keys as typed tuples prevent typos and enable precise invalidation:
export const productKeys = {
  all: ['products'] as const,
  lists: () => [...productKeys.all, 'list'] as const,
  list: (filters: Record<string, unknown>) => [...productKeys.lists(), filters] as const,
  details: () => [...productKeys.all, 'detail'] as const,
  detail: (id: string) => [...productKeys.details(), id] as const,
};

export function useProducts(category?: string) {
  return useQuery({
    queryKey: productKeys.list({ category }),
    queryFn: () => fetchProducts({ category }),
    select: (data) => data.sort((a, b) => a.name.localeCompare(b.name)),
    placeholderData: (previousData) => previousData, // keep old data while fetching new
  });
}

export function useProduct(id: string) {
  return useQuery({
    queryKey: productKeys.detail(id),
    queryFn: () => fetchProductById(id),
    enabled: Boolean(id), // don't fetch if id is empty
    staleTime: 10 * 60 * 1000, // product details stay fresh longer
  });
}

// Usage:
function ProductListScreen() {
  const { data: products, isLoading, isError, error, refetch } = useProducts();

  if (isLoading) return <ActivityIndicator />;
  if (isError) return <ErrorView message={error.message} onRetry={refetch} />;

  return (
    <FlatList
      data={products}
      renderItem={({ item }) => <ProductCard product={item} />}
      keyExtractor={item => item.id}
      refreshControl={<RefreshControl refreshing={false} onRefresh={refetch} />}
    />
  );
}
```

### useMutation

```typescript
// src/hooks/useCreateOrder.ts
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { createOrder } from '../api/orders';

export function useCreateOrder() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: createOrder,
    onSuccess: (newOrder) => {
      // Invalidate order lists so they refetch:
      queryClient.invalidateQueries({ queryKey: ['orders'] });
      // Optimistically add the new order to cache:
      queryClient.setQueryData(['orders', newOrder.id], newOrder);
    },
    onError: (error) => {
      Alert.alert('Order Failed', error.message);
    },
  });
}

// Usage:
function CheckoutButton() {
  const { mutate: createOrder, isPending } = useCreateOrder();

  return (
    <Pressable
      onPress={() => createOrder({ items: cart.items, address })}
      disabled={isPending}
    >
      {isPending ? <ActivityIndicator color="#fff" /> : <Text>Place Order</Text>}
    </Pressable>
  );
}
```

## MMKV for Fast Local Storage

MMKV is a key-value storage library from WeChat, **10× faster than AsyncStorage**. It's synchronous, type-safe, and supports encryption.

```bash
npm install react-native-mmkv
cd ios && pod install
```

```typescript
// src/storage/mmkv.ts
import { MMKV } from 'react-native-mmkv';

// Default storage (unencrypted):
export const storage = new MMKV();

// Encrypted storage (for sensitive data):
export const secureStorage = new MMKV({
  id: 'secure-storage',
  encryptionKey: 'your-encryption-key', // In production, use Keychain/Keystore
});

// TypeScript-safe wrappers:
export const StorageKeys = {
  AUTH_TOKEN: 'auth_token',
  USER_ID: 'user_id',
  ONBOARDING_COMPLETE: 'onboarding_complete',
  THEME: 'theme',
} as const;

export function getStorageItem<T>(key: string): T | undefined {
  const json = storage.getString(key);
  if (!json) return undefined;
  try {
    return JSON.parse(json) as T;
  } catch {
    return undefined;
  }
}

export function setStorageItem<T>(key: string, value: T): void {
  storage.set(key, JSON.stringify(value));
}

// Zustand storage adapter using MMKV:
export const mmkvStorage = {
  getItem: (name: string) => storage.getString(name) ?? null,
  setItem: (name: string, value: string) => storage.set(name, value),
  removeItem: (name: string) => storage.delete(name),
};
```

## AsyncStorage (Legacy)

AsyncStorage is the older, async key-value store. Prefer MMKV for new projects, but AsyncStorage is fine for simple cases.

```bash
npm install @react-native-async-storage/async-storage
```

```typescript
import AsyncStorage from '@react-native-async-storage/async-storage';

// Store a value:
await AsyncStorage.setItem('@user_token', token);

// Retrieve a value:
const token = await AsyncStorage.getItem('@user_token');

// Remove a value:
await AsyncStorage.removeItem('@user_token');

// Store an object (must JSON.stringify):
await AsyncStorage.setItem('@user_profile', JSON.stringify(profile));
const raw = await AsyncStorage.getItem('@user_profile');
const profile = raw ? JSON.parse(raw) : null;

// Multi-operation batch:
await AsyncStorage.multiSet([
  ['@user_id', userId],
  ['@user_email', email],
]);
```

:::caution[AsyncStorage is Slow]
AsyncStorage uses a disk-based SQLite store on Android and a flat files store on iOS. For anything read on app startup or in a render loop, use MMKV instead.
:::

## SQLite with expo-sqlite

For relational data, structured queries, or offline-first apps with complex data models:

```bash
npx expo install expo-sqlite
```

```typescript
// src/db/database.ts
import * as SQLite from 'expo-sqlite';

const db = SQLite.openDatabaseSync('myapp.db');

// Run migrations synchronously on startup:
export function initializeDatabase() {
  db.execSync(`
    PRAGMA journal_mode = WAL;
    PRAGMA foreign_keys = ON;

    CREATE TABLE IF NOT EXISTS products (
      id TEXT PRIMARY KEY,
      name TEXT NOT NULL,
      price REAL NOT NULL,
      category TEXT,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE INDEX IF NOT EXISTS idx_products_category ON products(category);
  `);
}

// Typed queries:
interface Product {
  id: string;
  name: string;
  price: number;
  category: string | null;
  created_at: string;
}

export function getProductsByCategory(category: string): Product[] {
  return db.getAllSync<Product>(
    'SELECT * FROM products WHERE category = ? ORDER BY name ASC',
    [category],
  );
}

export function upsertProduct(product: Omit<Product, 'created_at'>): void {
  db.runSync(
    `INSERT OR REPLACE INTO products (id, name, price, category) VALUES (?, ?, ?, ?)`,
    [product.id, product.name, product.price, product.category],
  );
}
```

## HTTP with fetch and axios

```typescript
// src/api/client.ts
import axios from 'axios';
import { storage, StorageKeys } from '../storage/mmkv';

export const apiClient = axios.create({
  baseURL: process.env.EXPO_PUBLIC_API_URL ?? 'https://api.example.com',
  timeout: 10_000,
  headers: { 'Content-Type': 'application/json' },
});

// Attach auth token to every request:
apiClient.interceptors.request.use((config) => {
  const token = storage.getString(StorageKeys.AUTH_TOKEN);
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Handle 401 globally (token refresh or logout):
apiClient.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      await signOut(); // clear auth state
    }
    return Promise.reject(error);
  },
);
```

## SecureStore for Sensitive Data

For credentials, tokens, and PII — store in the device's secure enclave (Keychain on iOS, Keystore on Android):

```bash
npx expo install expo-secure-store
```

```typescript
import * as SecureStore from 'expo-secure-store';

// Store a secret:
await SecureStore.setItemAsync('user_token', token, {
  keychainAccessible: SecureStore.WHEN_UNLOCKED, // require device unlock
  requireAuthentication: false, // set true to require biometrics
});

// Retrieve a secret:
const token = await SecureStore.getItemAsync('user_token');

// Delete:
await SecureStore.deleteItemAsync('user_token');
```

:::tip[Never Store Tokens in MMKV or AsyncStorage]
Use `expo-secure-store` for all authentication tokens, refresh tokens, API keys, and any user PII. MMKV and AsyncStorage are not encrypted by default and can be read from rooted/jailbroken devices.
:::

## Offline-First Patterns

```typescript
// Pattern: Optimistic updates with rollback
function useUpdateProfile() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: updateProfileAPI,

    onMutate: async (newProfile) => {
      // Cancel any in-flight refetches:
      await queryClient.cancelQueries({ queryKey: ['profile'] });

      // Snapshot the current value for rollback:
      const previous = queryClient.getQueryData(['profile']);

      // Optimistically update the cache:
      queryClient.setQueryData(['profile'], newProfile);

      // Return the snapshot so we can rollback on error:
      return { previous };
    },

    onError: (err, variables, context) => {
      // Rollback to the snapshot on failure:
      if (context?.previous) {
        queryClient.setQueryData(['profile'], context.previous);
      }
      Alert.alert('Update failed', 'Your changes could not be saved.');
    },

    onSettled: () => {
      // Always refetch after success or failure:
      queryClient.invalidateQueries({ queryKey: ['profile'] });
    },
  });
}
```

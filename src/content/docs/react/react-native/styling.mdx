React Native's styling system is inspired by CSS but runs natively — styles are applied as inline styles on native components, not the DOM. Understanding the key differences from web CSS is essential for writing predictable layouts.

## StyleSheet.create

`StyleSheet.create` is the primary API for defining styles. It validates your styles at development time and may optimise them for production.

```tsx
import { View, Text, StyleSheet } from 'react-native';

export function ProfileCard() {
  return (
    <View style={styles.card}>
      <Text style={styles.name}>Alice Johnson</Text>
      <Text style={[styles.tag, styles.tagBlue]}>Developer</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  // The object must use camelCase property names, not kebab-case
  card: {
    backgroundColor: '#ffffff',
    borderRadius: 12,
    padding: 16,
    marginHorizontal: 16,
    marginVertical: 8,
  },
  name: {
    fontSize: 18,
    fontWeight: '600',
    color: '#111827',
  },
  // Style composition via arrays
  tag: {
    display: 'flex',
    paddingHorizontal: 10,
    paddingVertical: 4,
    borderRadius: 99,
    fontSize: 12,
    fontWeight: '500',
    alignSelf: 'flex-start',
    overflow: 'hidden',
  },
  tagBlue: {
    backgroundColor: '#ede9fe',
    color: '#6d28d9',
  },
});
```

:::tip[Why StyleSheet.create?]

- **Validation**: Warns on invalid style properties in development
- **Performance**: Styles are registered once and referenced by ID, reducing overhead on re-renders
- **Autocomplete**: TypeScript provides type checking and IntelliSense for all style properties
- **Readability**: Separates styles from JSX, making components easier to read

Use `StyleSheet.create` over plain objects wherever possible.
:::

## Flexbox Layout

React Native uses Flexbox for all layout. Key differences from web CSS:

| Property        | Web Default | React Native Default |
| --------------- | ----------- | -------------------- |
| `flexDirection` | `row`       | **`column`**         |
| `alignContent`  | `stretch`   | `flex-start`         |
| `flexShrink`    | `1`         | **`0`**              |
| `position`      | `static`    | `relative`           |

```tsx
import { View, Text, StyleSheet } from 'react-native';

export function LayoutExamples() {
  return (
    <View style={styles.container}>
      {/* Column layout (default) — items stack vertically */}
      <View style={styles.column}>
        <View style={styles.box} />
        <View style={styles.box} />
        <View style={styles.box} />
      </View>

      {/* Row layout — items sit side by side */}
      <View style={styles.row}>
        <View style={[styles.box, styles.flex1]} />
        <View style={[styles.box, styles.flex2]} />
        <View style={[styles.box, styles.flex1]} />
      </View>

      {/* Center content */}
      <View style={styles.centered}>
        <Text>Perfectly centred</Text>
      </View>

      {/* Space between items */}
      <View style={styles.spaceBetween}>
        <Text>Left</Text>
        <Text>Centre</Text>
        <Text>Right</Text>
      </View>

      {/* Wrap items to next line */}
      <View style={styles.wrap}>
        {['React', 'Native', 'Flexbox', 'Layout', 'Wrap'].map((tag) => (
          <View key={tag} style={styles.chip}>
            <Text style={styles.chipText}>{tag}</Text>
          </View>
        ))}
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, padding: 16, gap: 16 },
  column: { flexDirection: 'column', gap: 8 },
  row: { flexDirection: 'row', gap: 8 },
  box: { height: 50, backgroundColor: '#e0e7ff', borderRadius: 4 },
  flex1: { flex: 1 },
  flex2: { flex: 2 },
  centered: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    backgroundColor: '#f0fdf4',
    borderRadius: 8,
    height: 80,
  },
  spaceBetween: {
    flexDirection: 'row',
    justifyContent: 'space-between',
    alignItems: 'center',
    backgroundColor: '#fff7ed',
    padding: 12,
    borderRadius: 8,
  },
  wrap: { flexDirection: 'row', flexWrap: 'wrap', gap: 8 },
  chip: {
    backgroundColor: '#ede9fe',
    borderRadius: 99,
    paddingHorizontal: 12,
    paddingVertical: 6,
  },
  chipText: { color: '#6d28d9', fontSize: 13, fontWeight: '500' },
});
```

## Dimensions and useWindowDimensions

Get the screen size to create responsive layouts.

```tsx
import {
  View,
  Text,
  StyleSheet,
  Dimensions,
  useWindowDimensions,
} from 'react-native';

// Static — won't update on orientation change
const { width: SCREEN_WIDTH, height: SCREEN_HEIGHT } = Dimensions.get('window');
const IS_SMALL_DEVICE = SCREEN_WIDTH < 375;

// Dynamic — updates when orientation or window size changes (preferred)
export function ResponsiveGrid() {
  const { width, height, fontScale } = useWindowDimensions();
  const isLandscape = width > height;
  const columns = isLandscape ? 3 : 2;
  const itemWidth = (width - (columns + 1) * 16) / columns;

  return (
    <View style={styles.grid}>
      {items.map((item) => (
        <View key={item.id} style={[styles.item, { width: itemWidth }]}>
          <Text style={{ fontSize: 14 * fontScale }}>{item.title}</Text>
        </View>
      ))}
    </View>
  );
}

const styles = StyleSheet.create({
  grid: { flexDirection: 'row', flexWrap: 'wrap', padding: 16, gap: 16 },
  item: { aspectRatio: 1, backgroundColor: '#f3f4f6', borderRadius: 8 },
});
```

## PixelRatio

Mobile screens have varying pixel densities. `PixelRatio` lets you specify values in physical pixels.

```tsx
import { PixelRatio, StyleSheet } from 'react-native';

const pixelRatio = PixelRatio.get(); // 1, 2, or 3 (device pixel density)
const fontScale = PixelRatio.getFontScale(); // User's font size preference

// Convert logical pixels to physical pixels
const physicalPixels = PixelRatio.getPixelSizeForLayoutSize(1); // e.g., 3 on @3x

// Hair-thin border that's exactly 1 physical pixel on all devices
const hairline = StyleSheet.hairlineWidth; // 0.333... on @3x

const styles = StyleSheet.create({
  hairlineBorder: {
    borderBottomWidth: StyleSheet.hairlineWidth,
    borderBottomColor: '#e5e7eb',
  },
  // Scale an icon based on screen density
  icon: {
    width: PixelRatio.roundToNearestPixel(24),
    height: PixelRatio.roundToNearestPixel(24),
  },
});
```

## Platform-Specific Styles

Use `Platform.OS` or `Platform.select` to apply styles that differ between iOS and Android.

```tsx
import { Platform, StyleSheet, View, Text } from 'react-native';

export function PlatformCard() {
  return (
    <View style={styles.card}>
      <Text style={styles.label}>Platform: {Platform.OS}</Text>
      <Text style={styles.version}>Version: {Platform.Version}</Text>
    </View>
  );
}

const styles = StyleSheet.create({
  card: {
    backgroundColor: '#fff',
    borderRadius: 12,
    padding: 16,
    // Platform.select returns different values per platform
    ...Platform.select({
      ios: {
        shadowColor: '#000',
        shadowOffset: { width: 0, height: 4 },
        shadowOpacity: 0.12,
        shadowRadius: 12,
      },
      android: {
        elevation: 6,
      },
      // Web fallback (if using React Native Web)
      web: {
        boxShadow: '0 4px 12px rgba(0,0,0,0.12)',
      },
    }),
  },
  label: {
    fontSize: Platform.OS === 'ios' ? 16 : 15,
    fontWeight: '600',
  },
  version: {
    fontSize: 13,
    color: '#9ca3af',
    // Inline Platform.OS check
    marginTop: Platform.OS === 'android' ? 4 : 2,
  },
});
```

### Platform-Specific Files

For larger platform differences, use platform-specific file extensions:

```
Button.ios.tsx      # Used on iOS only
Button.android.tsx  # Used on Android only
Button.native.tsx   # Used on all native (iOS + Android); overridden by .ios/.android
Button.tsx          # Fallback / web
```

```tsx
// Button.ios.tsx
import { TouchableOpacity, Text } from 'react-native';
export function Button({ title, onPress }: ButtonProps) {
  return (
    <TouchableOpacity style={iosStyles.btn} onPress={onPress}>
      <Text style={iosStyles.text}>{title}</Text>
    </TouchableOpacity>
  );
}

// Button.android.tsx
import { Pressable, Text } from 'react-native';
export function Button({ title, onPress }: ButtonProps) {
  return (
    <Pressable android_ripple={{ color: '#6366f1' }} onPress={onPress}>
      <Text>{title}</Text>
    </Pressable>
  );
}
```

## Shadows

Shadows work differently on iOS and Android:

```tsx
const styles = StyleSheet.create({
  card: {
    backgroundColor: '#fff',
    borderRadius: 12,

    // ✅ iOS shadow — all four properties required
    shadowColor: '#6366f1',
    shadowOffset: { width: 0, height: 8 },
    shadowOpacity: 0.15,
    shadowRadius: 16,

    // ✅ Android shadow — single elevation value
    elevation: 8,
  },
});
```

:::caution[Shadows on Android]
`elevation` adds both a shadow and material-style theming on Android. The shadow color is always black on Android and cannot be customised (unlike iOS). For full cross-platform shadow control, consider `react-native-shadow-2`.
:::

## Absolute Positioning

```tsx
const styles = StyleSheet.create({
  container: {
    position: 'relative', // default — establishes a containing block
    width: 200,
    height: 200,
```

## NativeWind (Tailwind for React Native)

NativeWind compiles Tailwind CSS utility classes to React Native `StyleSheet` objects, letting you use familiar Tailwind syntax on native components.

```bash
npm install nativewind tailwindcss
npx tailwindcss init
```

```javascript
// tailwind.config.js
/** @type {import('tailwindcss').Config} */
module.exports = {
  content: ['./app/**/*.{js,jsx,ts,tsx}', './src/**/*.{js,jsx,ts,tsx}'],
  presets: [require('nativewind/preset')],
  theme: {
    extend: {
      colors: {
        brand: { 500: '#6366f1', 600: '#4f46e5' },
      },
    },
  },
  plugins: [],
};
```

```tsx
// With NativeWind v4 — uses className prop
import { View, Text, Pressable } from 'react-native';

export function NativeWindCard({ title, description }: CardProps) {
  return (
    <View className="bg-white rounded-xl p-4 mx-4 my-2 shadow-md">
      <Text className="text-gray-900 text-lg font-bold mb-1">{title}</Text>
      <Text className="text-gray-500 text-sm leading-5">{description}</Text>
      <Pressable className="mt-4 bg-brand-500 rounded-lg py-3 items-center active:bg-brand-600">
        <Text className="text-white font-semibold">Read More</Text>
      </Pressable>
    </View>
  );
}
```

## Safe Area Insets

Respect device notches, home indicators, and rounded corners:

```tsx
import { useSafeAreaInsets } from 'react-native-safe-area-context';
import { View, StyleSheet } from 'react-native';

export function Screen({ children }: { children: React.ReactNode }) {
  const insets = useSafeAreaInsets();

  return (
    <View
      style={[
        styles.container,
        {
          paddingTop: insets.top,
          paddingBottom: insets.bottom,
          paddingLeft: insets.left,
          paddingRight: insets.right,
        },
      ]}
    >
      {children}
    </View>
  );
}

// Or apply insets selectively — e.g., only bottom padding for tab bar content
export function TabContent({ children }: { children: React.ReactNode }) {
  const { bottom } = useSafeAreaInsets();

  return (
    <View style={[styles.container, { paddingBottom: bottom + 16 }]}>
      {children}
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: '#fff' },
});
```

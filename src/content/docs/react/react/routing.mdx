React Router v6 is the most widely used client-side routing library for React. Version 6 introduced a redesigned API with nested routes, relative paths, and a data router pattern inspired by Remix.

## Installation

```bash
npm install react-router-dom
```

## Basic Setup

```tsx
// main.tsx
import { createRoot } from 'react-dom/client';
import { BrowserRouter } from 'react-router-dom';
import App from './App';

createRoot(document.getElementById('root')!).render(
  <BrowserRouter>
    <App />
  </BrowserRouter>
);
```

```tsx
// App.tsx
import { Routes, Route, Navigate } from 'react-router-dom';
import { Layout } from './components/Layout';
import { Home }     from './pages/Home';
import { About }    from './pages/About';
import { NotFound } from './pages/NotFound';

function App() {
  return (
    <Routes>
      <Route path="/" element={<Layout />}>
        <Route index element={<Home />} />
        <Route path="about" element={<About />} />
        <Route path="*" element={<NotFound />} />
      </Route>
    </Routes>
  );
}
```

## Navigation Components

### Link and NavLink

```tsx
import { Link, NavLink } from 'react-router-dom';

function Navigation() {
  return (
    <nav>
      {/* Basic link */}
      <Link to="/">Home</Link>
      <Link to="/about">About</Link>

        to="/settings"
        style={({ isActive }) => ({
          fontWeight: isActive ? 700 : 400,
          color:      isActive ? '#0070f3' : 'inherit',
        })}
      >
        Settings
      </NavLink>
    </nav>
  );
}
```

## Hooks

### useParams

```tsx
import { useParams } from 'react-router-dom';

// Route: <Route path="/users/:userId/posts/:postId" element={<PostDetail />} />
function PostDetail() {
  const { userId, postId } = useParams<{ userId: string; postId: string }>();

  return (

      <p>Post: {postId}</p>
    </div>
  );

```

### useNavigate

```tsx
import { useNavigate } from 'react-router-dom';

function LoginForm() {
  const navigate = useNavigate();

  const handleSubmit = async (credentials: Credentials) => {
    await login(credentials);
    navigate('/dashboard');                    // push
    navigate('/dashboard', { replace: true }); // replace (no back button history)
    navigate(-1);                              // go back one step
    navigate(-2);                             // go back two steps
  };

  return <form onSubmit={handleSubmit}>{/* ... */}</form>;
}
```

### useSearchParams

```tsx
import { useSearchParams } from 'react-router-dom';

function ProductList() {
  const [searchParams, setSearchParams] = useSearchParams();

  const page    = Number(searchParams.get('page') ?? '1');
  const sort    = searchParams.get('sort') ?? 'name';
  const filter  = searchParams.get('filter') ?? '';

  const updateFilter = (value: string) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      if (value) {
        next.set('filter', value);
      } else {
        next.delete('filter');
      }
      next.set('page', '1'); // reset page on filter change
      return next;
    });
  };

  return (
    <div>
      <input
        value={filter}
        onChange={e => updateFilter(e.target.value)}
        placeholder="Filter products…"
      />
      <select value={sort} onChange={e => setSearchParams({ sort: e.target.value, page: '1' })}>
        <option value="name">Name</option>
        <option value="price">Price</option>
        <option value="rating">Rating</option>
      </select>
      {/* URL: /products?filter=shoe&sort=price&page=1 */}
    </div>
  );
}
```

## Nested Routes with Outlet

`Outlet` renders the matched child route inside a layout:

```tsx
// Layout component with shared navigation and footer
function RootLayout() {
  return (
    <div className="app">
      <header>
        <Navigation />
      </header>
      <main>
        {/* Child routes render here */}
        <Outlet />
      </main>
      <footer>© 2025 My App</footer>
    </div>
  );
}

// Dashboard with its own sub-layout
function DashboardLayout() {
  return (
    <div className="dashboard">
      <aside>
        <DashboardNav />
      </aside>
      <div className="dashboard__content">
        <Outlet />   {/* Dashboard sub-routes render here */}
      </div>
    </div>
  );
}

// Router configuration
function App() {
  return (
    <Routes>
      {/* Root layout wraps all routes */}
      <Route path="/" element={<RootLayout />}>
        <Route index element={<HomePage />} />
        <Route path="about" element={<AboutPage />} />

        {/* Dashboard sub-routes */}
        <Route path="dashboard" element={<DashboardLayout />}>
          <Route index element={<DashboardHome />} />
          <Route path="analytics" element={<Analytics />} />
          <Route path="users" element={<UserList />} />
          <Route path="users/:id" element={<UserDetail />} />
          <Route path="settings" element={<Settings />} />
        </Route>

        <Route path="*" element={<NotFound />} />
      </Route>
    </Routes>
  );
}
```

### Outlet with Context

Pass data from a layout to its child routes:

```tsx
function DashboardLayout() {
  const [sidebarOpen, setSidebarOpen] = useState(true);

  return (
    <div className="dashboard">
      <aside className={sidebarOpen ? 'open' : 'closed'}>
        <button onClick={() => setSidebarOpen(o => !o)}>Toggle</button>
        <DashboardNav />
      </aside>
      <div className="dashboard__content">
        {/* Pass context to child routes */}
        <Outlet context={{ sidebarOpen }} />
      </div>
    </div>
  );
}

// Child route consumes the context
function Analytics() {
  const { sidebarOpen } = useOutletContext<{ sidebarOpen: boolean }>();
  return <div style={{ marginLeft: sidebarOpen ? '240px' : '0' }}>Analytics content</div>;
}
```

## Data Router Pattern

React Router v6.4+ includes loaders and actions for co-locating data fetching with routes:

```tsx
import {
  createBrowserRouter,
  RouterProvider,
  useLoaderData,
  useActionData,
  Form,
  redirect,
} from 'react-router-dom';

// --- Loader: runs before the component renders ---
async function postsLoader({ request }: { request: Request }) {
  const url    = new URL(request.url);
  const page   = url.searchParams.get('page') ?? '1';
  const res    = await fetch(`/api/posts?page=${page}`);
  if (!res.ok) throw new Response('Failed to load posts', { status: res.status });
  return res.json();
}

// --- Component uses loader data ---
function PostsPage() {
  const posts = useLoaderData() as Post[];

  return (
    <ul>
      {posts.map(post => (
        <li key={post.id}>
          <Link to={`/posts/${post.id}`}>{post.title}</Link>
        </li>
      ))}
    </ul>
  );
}

// --- Action: handles form submissions ---
async function createPostAction({ request }: { request: Request }) {
  const formData = await request.formData();
  const title    = formData.get('title') as string;
  const body     = formData.get('body') as string;

  if (!title) return { error: 'Title is required' };

  const res = await fetch('/api/posts', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ title, body }),
  });
  const post = await res.json();
  return redirect(`/posts/${post.id}`);
}

function NewPostPage() {
  const actionData = useActionData() as { error?: string } | undefined;

  return (
    <Form method="post">
      {actionData?.error && <p className="error">{actionData.error}</p>}
      <input name="title" placeholder="Title" />
      <textarea name="body" placeholder="Body" />
      <button type="submit">Publish</button>
    </Form>
  );
}

// --- Router definition ---
const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    errorElement: <GlobalError />,
    children: [
      { index: true, element: <Home /> },
      {
        path: 'posts',
        element: <PostsPage />,
        loader: postsLoader,
        errorElement: <PostsError />,
        children: [
          {
            path: 'new',
            element: <NewPostPage />,
            action: createPostAction,
          },
        ],
      },
    ],
  },
]);

function App() {
  return <RouterProvider router={router} />;
}
```

### errorElement

```tsx
import { useRouteError, isRouteErrorResponse } from 'react-router-dom';

function RouteErrorPage() {
  const error = useRouteError();

  if (isRouteErrorResponse(error)) {
    return (
      <div>
        <h1>{error.status} {error.statusText}</h1>
        <p>{error.data}</p>
      </div>
    );
  }

  if (error instanceof Error) {
    return (
      <div>
        <h1>Unexpected Error</h1>
        <pre>{error.message}</pre>
      </div>
    );
  }

  return <h1>Unknown error</h1>;
}
```

## Protected Routes

```tsx
import { Navigate, Outlet, useLocation } from 'react-router-dom';
import { useAuth } from '../context/AuthContext';

interface ProtectedRouteProps {
  allowedRoles?: ('admin' | 'user')[];
}

function ProtectedRoute({ allowedRoles }: ProtectedRouteProps) {
  const { user, isAuthenticated } = useAuth();
  const location = useLocation();

  if (!isAuthenticated) {
    // Redirect to login, saving intended destination
    return <Navigate to="/login" state={{ from: location.pathname }} replace />;
  }

  if (allowedRoles && user && !allowedRoles.includes(user.role)) {
    return <Navigate to="/403" replace />;
  }

  return <Outlet />;
}

// Usage in router
function App() {
  return (
    <Routes>
      <Route path="/login"   element={<LoginPage />} />
      <Route path="/register" element={<RegisterPage />} />

      {/* All authenticated routes */}
      <Route element={<ProtectedRoute />}>
        <Route path="/dashboard" element={<Dashboard />} />
        <Route path="/profile"   element={<Profile />} />

        {/* Admin-only routes */}
        <Route element={<ProtectedRoute allowedRoles={['admin']} />}>
          <Route path="/admin" element={<AdminPanel />} />
          <Route path="/admin/users" element={<UserManagement />} />
        </Route>
      </Route>

      <Route path="/403" element={<Forbidden />} />
      <Route path="*"    element={<NotFound />} />
    </Routes>
  );
}
```

:::tip[Redirect After Login]
After a successful login, redirect the user to where they originally wanted to go:

```tsx
function LoginPage() {
  const navigate = useNavigate();
  const location = useLocation();
  const state    = location.state as { from?: string };
  const from     = state?.from ?? '/dashboard';

  const handleLogin = async (credentials: Credentials) => {
    await login(credentials);
    navigate(from, { replace: true });
  };
}
```
:::

## Lazy Routes with Suspense

```tsx
import { lazy, Suspense } from 'react';
import { Routes, Route } from 'react-router-dom';

// Each page is a separate JS chunk
const Dashboard  = lazy(() => import('./pages/Dashboard'));
const Profile    = lazy(() => import('./pages/Profile'));
const Analytics  = lazy(() => import('./pages/Analytics'));
const Settings   = lazy(() => import('./pages/Settings'));

function App() {
  return (
    <Suspense fallback={<PageLoader />}>
      <Routes>
        <Route path="/" element={<Layout />}>
          <Route path="dashboard"  element={<Dashboard />} />
          <Route path="profile"    element={<Profile />} />
          <Route path="analytics"  element={<Analytics />} />
          <Route path="settings"   element={<Settings />} />
        </Route>
      </Routes>
    </Suspense>
  );
}
```

### With createBrowserRouter (Data Router)

```tsx
const router = createBrowserRouter([

    path: '/',
    element: <RootLayout />,
    children: [
      {
        path: 'analytics',
        lazy: async () => {
          // Dynamic import returns both the loader and the component
          const { loader, default: Component } = await import('./pages/Analytics');
          return { loader, Component };
        },
      },
    ],
  },
]);
```

## Route-Based Code Splitting Best Practices

:::caution[Avoid Lazy Loading Small Components]
Only lazy-load route-level components (pages). Lazy-loading small shared components adds network waterfalls with minimal bundle savings. Aim for chunks > 30kb uncompressed to justify the split.
:::

```tsx
// ✅ Good: lazy-load pages (each is likely 20–200kb)
const DashboardPage = lazy(() => import('./pages/DashboardPage'));

// ❌ Unnecessary: lazy-loading a small shared component
const Button = lazy(() => import('./components/Button'));
```

## Navigation Guards with useBlocker

```tsx
import { useBlocker } from 'react-router-dom';

function EditForm() {
  const [isDirty, setIsDirty] = useState(false);

  // Block navigation when form has unsaved changes
  const blocker = useBlocker(
    ({ currentLocation, nextLocation }) =>
      isDirty && currentLocation.pathname !== nextLocation.pathname
  );

  return (
    <form onChange={() => setIsDirty(true)}>
      {/* form fields */}

      {blocker.state === 'blocked' && (
        <dialog open>
          <p>You have unsaved changes. Leave anyway?</p>
          <button onClick={() => blocker.proceed()}>Leave</button>
          <button onClick={() => blocker.reset()}>Stay</button>
        </dialog>
      )}
    </form>
  );
}
```

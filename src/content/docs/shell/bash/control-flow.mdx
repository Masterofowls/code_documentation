# Bash Control Flow

## if / elif / else

```bash
# Basic if
if [[ "$name" == "Alice" ]]; then
  echo "Hello, Alice!"
fi

# if / elif / else
if [[ $score -ge 90 ]]; then
  echo "A"
elif [[ $score -ge 80 ]]; then
  echo "B"
elif [[ $score -ge 70 ]]; then
  echo "C"
else
  echo "F"
fi

# Inline (for simple true/false)
[[ -f "$file" ]] && echo "exists" || echo "missing"
command -v git &>/dev/null && echo "git found" || echo "git not found"

# The if command tests exit status (0 = true, non-zero = false)
if grep -q "pattern" file.txt; then
  echo "found"
fi

if ! command -v docker &>/dev/null; then
  echo "Docker is not installed"
  exit 1
fi
```

## test vs [ ] vs [[ ]]

```bash
# test command and [ ] are equivalent (POSIX)
test -f file.txt && echo "exists"
[ -f file.txt ] && echo "exists"

# [[ ]] is a Bash keyword (not a command)
# Preferred for Bash scripts — safer and more powerful
[[ -f file.txt ]] && echo "exists"
```

### String Tests

```bash
str="hello"
other="world"

[[ -z "$str" ]]      # true if string is empty (zero length)
[[ -n "$str" ]]      # true if string is non-empty
[[ "$str" == "hello" ]]   # string equality
[[ "$str" != "world" ]]   # string inequality
[[ "$str" < "$other" ]]   # lexicographic less-than (in [[]])
[[ "$str" > "$other" ]]   # lexicographic greater-than
[[ "$str" == hello* ]]    # glob pattern matching (no quotes on pattern!)
[[ "$str" =~ ^h.*o$ ]]   # regex matching (POSIX ERE)
                           # capture groups in BASH_REMATCH array
```

### Numeric Tests

```bash
a=10
b=20

[[ $a -eq $b ]]   # equal
[[ $a -ne $b ]]   # not equal
[[ $a -lt $b ]]   # less than
[[ $a -le $b ]]   # less than or equal
[[ $a -gt $b ]]   # greater than
[[ $a -ge $b ]]   # greater than or equal

# Or use arithmetic evaluation
if (( a > b )); then echo "a is greater"; fi
if (( a == 0 )); then echo "zero"; fi
```

### File Tests

```bash
file="/etc/passwd"
dir="/tmp"

[[ -e "$file" ]]    # exists (any type)
[[ -f "$file" ]]    # exists and is a regular file
[[ -d "$dir" ]]     # exists and is a directory
[[ -L "$file" ]]    # exists and is a symbolic link
[[ -r "$file" ]]    # readable by current user
[[ -w "$file" ]]    # writable by current user
[[ -x "$file" ]]    # executable by current user
[[ -s "$file" ]]    # exists and is non-empty (size > 0)
[[ -p "$file" ]]    # is a named pipe (FIFO)
[[ -S "$file" ]]    # is a socket
[[ -b "$file" ]]    # is a block device
[[ -c "$file" ]]    # is a character device
[[ -N "$file" ]]    # has been modified since last read
[[ "$f1" -nt "$f2" ]]  # f1 is newer than f2 (modification time)
[[ "$f1" -ot "$f2" ]]  # f1 is older than f2
[[ "$f1" -ef "$f2" ]]  # f1 and f2 are the same file (hard links)

# Practical examples
[[ -d ~/.config ]] || mkdir -p ~/.config
[[ -r "$config_file" ]] || { echo "Cannot read config"; exit 1; }
[[ -x "$binary" ]] || { echo "$binary is not executable"; exit 1; }
```

### Compound Tests

```bash
# && and || inside [[ ]]
[[ -f "$file" && -r "$file" ]]    # file exists AND is readable
[[ -z "$a" || -z "$b" ]]          # either a or b is empty

# ! negation
[[ ! -f "$file" ]]   # file does NOT exist

# Short-circuit operators outside conditionals
[[ -d /tmp/mydir ]] || mkdir /tmp/mydir
cmd1 && cmd2    # run cmd2 only if cmd1 succeeds
cmd1 || cmd2    # run cmd2 only if cmd1 fails
```

## case / esac

```bash
# Basic case statement
case "$OS" in
  linux*)
    echo "Linux variant"
    ;;
  darwin*)
    echo "macOS"
    ;;
  cygwin*|msys*|mingw*)
    echo "Windows (Cygwin/MSYS)"
    ;;
  *)
    echo "Unknown OS: $OS"
    ;;
esac

# Pattern types in case
case "$input" in
  y|Y|yes|YES)   echo "affirmative" ;;
  n|N|no|NO)     echo "negative" ;;
  [0-9])         echo "single digit" ;;
  [0-9]*)        echo "starts with digit" ;;
  *.txt)         echo "text file" ;;
  *.sh)          echo "shell script" ;;
  "")            echo "empty" ;;
  *)             echo "other: $input" ;;
esac

# Fall-through with ;;&
case "$char" in
  [[:upper:]])
    echo "uppercase"
    ;;&   # continue checking (fall through to next pattern)
  [[:alpha:]])
    echo "alphabetic"
    ;;
esac
# For 'A': prints both "uppercase" and "alphabetic"

# ;& falls through unconditionally (executes next action)
case "$num" in
  1)
    echo "one"
    ;&   # fall through unconditionally
  2)
    echo "one-or-two"
    ;;
esac
```

## for Loops

```bash
# Iterate over a list
for name in Alice Bob Carol Dave; do
  echo "Hello, $name"
done

# Iterate over an array
fruits=("apple" "banana" "cherry")
for fruit in "${fruits[@]}"; do
  echo "Fruit: $fruit"
done

# Iterate over files
for file in *.log; do
  [[ -f "$file" ]] || continue   # skip if no .log files (glob unexpanded)
  echo "Processing: $file"
done

# C-style for loop
for (( i = 0; i < 10; i++ )); do
  printf "%d " "$i"
done
echo

# Brace expansion range
for i in {1..10}; do
  echo "Item $i"
done

# With step
for i in {0..20..5}; do
  echo "$i"   # 0, 5, 10, 15, 20
done

# seq for more control
for i in $(seq 1 5); do echo "$i"; done
for i in $(seq 0 0.1 1.0); do echo "$i"; done  # floats!

# Loop over command output (careful with word splitting)
while IFS= read -r line; do
  echo "Line: $line"
done < <(find . -name "*.sh")   # process substitution avoids subshell
```

## while Loop

```bash
# Basic while
count=0
while (( count < 5 )); do
  echo "count=$count"
  (( count++ ))
done

# Read lines from file
while IFS= read -r line; do
  echo "[$line]"
done < file.txt

# Read from command output
while IFS= read -r line; do
  echo "Found: $line"
done < <(grep -r "TODO" src/)

# Read fields with IFS
while IFS=: read -r user _ uid gid _ home shell; do
  [[ "$shell" != "/sbin/nologin" ]] || continue
  printf "%-15s uid=%-5s home=%s\n" "$user" "$uid" "$home"
done < /etc/passwd

# Infinite loop with break
while true; do
  read -r -p "Enter command (q to quit): " cmd
  case "$cmd" in
    q|quit) break ;;
    *) echo "Got: $cmd" ;;
  esac
done

# Process in batches
i=0
while IFS= read -r item; do
  batch+=("$item")
  (( i++ ))
  if (( i % 100 == 0 )); then
    process_batch "${batch[@]}"
    batch=()
  fi
done < items.txt
[[ ${#batch[@]} -gt 0 ]] && process_batch "${batch[@]}"
```

## until Loop

```bash
# until: loop UNTIL condition is true (opposite of while)
attempt=0
until ping -c1 -W1 "$host" &>/dev/null; do
  (( attempt++ ))
  if (( attempt >= 10 )); then
    echo "Host unreachable after 10 attempts"
    exit 1
  fi
  echo "Attempt $attempt: waiting for $host..."
  sleep 2
done
echo "Host $host is reachable"
```

## break and continue

```bash
# break: exit innermost loop
for i in {1..10}; do
  (( i == 5 )) && break
  echo "$i"
done   # prints 1 2 3 4

# continue: skip to next iteration
for i in {1..10}; do
  (( i % 2 == 0 )) && continue
  echo "$i"
done   # prints odd numbers: 1 3 5 7 9

# break/continue with label (N = loop level)
outer: for i in {1..3}; do
  for j in {1..3}; do
    if (( i == 2 && j == 2 )); then
      break 2   # break out of 2 levels (both loops)
    fi
    echo "$i,$j"
  done
done
```

## select Statement (Menu)

```bash
# select automatically creates a numbered menu
PS3="Choose an option: "   # prompt for select (default is #?)
options=("Start server" "Stop server" "Show status" "Quit")

select choice in "${options[@]}"; do
  case "$REPLY" in   # REPLY contains the number entered
    1) start_server ;;
    2) stop_server ;;
    3) show_status ;;
    4) break ;;
    *) echo "Invalid option: $REPLY" ;;
  esac
done
```

## getopts — Option Parsing

```bash
#!/usr/bin/env bash
# getopts parses short options: -v -f file -n 5

VERBOSE=false
OUTPUT_FILE=""
COUNT=1

while getopts ":vf:n:h" opt; do
  case "$opt" in
    v)
      VERBOSE=true
      ;;
    f)
      OUTPUT_FILE="$OPTARG"    # OPTARG holds the argument value
      ;;
    n)
      if ! [[ "$OPTARG" =~ ^[0-9]+$ ]]; then
        echo "Error: -n requires a positive integer" >&2
        exit 1
      fi
      COUNT="$OPTARG"
      ;;
    h)
      usage
      exit 0
      ;;
    :)
      echo "Error: -${OPTARG} requires an argument" >&2
      exit 1
      ;;
    \?)
      echo "Error: unknown option -${OPTARG}" >&2
      exit 1
      ;;
  esac
done

# After getopts, OPTIND is the index of the first non-option argument
shift $(( OPTIND - 1 ))
remaining_args=("$@")

echo "Verbose: $VERBOSE"
echo "Output: ${OUTPUT_FILE:-stdout}"
echo "Count: $COUNT"
echo "Remaining args: ${remaining_args[*]}"
```

:::tip[getopts vs getopt]
`getopts` is a shell builtin and handles short options (`-v`, `-f file`). `getopt` (external command) handles long options (`--verbose`, `--file=path`). For long options in bash, use manual parsing or `getopt`. The `getopts` builtin is simpler and more portable.
:::

## mapfile / readarray

```bash
# Read lines into an array
mapfile -t lines < file.txt       # -t strips trailing newlines
readarray -t lines < file.txt     # synonym for mapfile

echo "${lines[0]}"    # first line
echo "${#lines[@]}"   # number of lines

# Read from command output
mapfile -t users < <(awk -F: '$3 >= 1000 {print $1}' /etc/passwd)

# Read with a callback (called every N lines)
mapfile -t -c 1 -C process_line data.txt

# Filter while reading
mapfile -t nonblank < <(grep -v '^\s*$' file.txt)

# Read with delimiter
mapfile -t -d $'\0' files < <(find . -name "*.sh" -print0)
# -d sets the line delimiter (null byte here, safe for filenames with spaces)
```

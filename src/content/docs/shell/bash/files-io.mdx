# Bash Files and I/O

## File Descriptors

Every process has three standard file descriptors (FDs):

| FD | Name | Default | Symbol |
|----|------|---------|--------|
| 0 | stdin | Keyboard | `<` |
| 1 | stdout | Terminal | `>` |
| 2 | stderr | Terminal | `2>` |

```bash
# Custom file descriptors (3-9 available by default)
exec 3> output.log           # open FD 3 for writing
exec 4< input.txt            # open FD 4 for reading
exec 5<> bidirectional.txt   # open FD 5 for reading AND writing

echo "Writing to FD 3" >&3   # write to output.log
read -r line <&4             # read from input.txt
echo "Line: $line"

# Close file descriptors
exec 3>&-    # close FD 3
exec 4<&-    # close FD 4

# Duplicate FDs
exec 6>&1    # FD 6 is a copy of stdout
exec 1>logfile.txt  # redirect stdout to file
echo "Goes to logfile"
exec 1>&6    # restore stdout from FD 6
exec 6>&-    # close FD 6
```

## Redirection

```bash
# Output redirection
command > file.txt       # redirect stdout to file (overwrite)
command >> file.txt      # redirect stdout to file (append)
command 2> errors.txt    # redirect stderr to file
command 2>> errors.txt   # append stderr to file

# Both stdout and stderr
command > out.txt 2>&1   # redirect stdout, then stderr to same place
command &> out.txt       # bash shorthand: same as above
command &>> out.txt      # append both stdout and stderr
command > out.txt 2> err.txt  # separate files

# Discard output
command > /dev/null      # discard stdout
command 2> /dev/null     # discard stderr
command &> /dev/null     # discard both (silent)
command > /dev/null 2>&1 # equivalent, POSIX-compatible form

# Input redirection
command < input.txt      # read stdin from file

# Write stderr to stdout (redirect error messages to log tools)
something_that_errors 2>&1 | logger  # pipe both to syslog

# tee — write to file AND stdout
command | tee output.txt        # stdout goes to both terminal and file
command | tee -a output.txt     # append mode
command 2>&1 | tee combined.log # both stdout and stderr
```

:::tip[Order matters for 2>&1]
`command > file.txt 2>&1` means "redirect stdout to file, then redirect stderr to wherever stdout is now (file)". The order matters: `command 2>&1 > file.txt` would redirect stderr to the terminal (old stdout), then stdout to the file — wrong!
:::

## Pipes

```bash
# Basic pipe: stdout of cmd1 → stdin of cmd2
ls -la | grep ".sh"
cat access.log | grep "404" | wc -l

# Long pipelines
cat /etc/passwd |
  grep -v "^#" |
  awk -F: '{print $1, $3}' |
  sort -k2 -n |
  tail -10

# tee in pipeline — split the pipe
cat data.txt | tee raw.txt | sort | tee sorted.txt | uniq > unique.txt

# Named pipes (FIFOs)
mkfifo /tmp/mypipe
command_producer > /tmp/mypipe &   # must background
command_consumer < /tmp/mypipe
rm /tmp/mypipe

# Multiple writers to a named pipe (log aggregation)
mkfifo /tmp/log_pipe
cat /tmp/log_pipe >> /var/log/combined.log &
./server_a 2>&1 > /tmp/log_pipe &
./server_b 2>&1 > /tmp/log_pipe &
```

## Process Substitution

Process substitution creates a temporary file descriptor connected to a command:

```bash
# <(command) — treat command output as a file (for reading)
diff <(sort file1.txt) <(sort file2.txt)    # compare sorted versions
comm -23 <(sort a.txt) <(sort b.txt)        # lines in a but not b
wc -l <(find . -name "*.sh")               # count found scripts

# >(command) — treat command input as a file (for writing)
tee >(gzip > backup.tar.gz) >(sha256sum > backup.sha256) > /dev/null

# Compare live command output
diff <(ssh server1 "ls /app/bin") <(ssh server2 "ls /app/bin")

# Avoid subshell in while loops (vs piping)
count=0
while IFS= read -r line; do
  (( count++ ))
  process_line "$line"
done < <(find . -name "*.log" -newer checkpoint.txt)
echo "Processed $count files"
```

## exec for Permanent Redirection

```bash
# Redirect script's stdout to a log file for the rest of execution
exec 1> >(tee -a "$LOG_FILE")   # stdout to tee (file + terminal)
exec 2>&1                        # stderr → stdout → tee

echo "This goes to log"
ls /nonexistent   # error also goes to log

# Redirect entire script output
exec > /tmp/script-output.log 2>&1
echo "All output captured"

# Reopen terminal for interactive prompts even when stdout is redirected
exec 3> /tmp/output.log
echo "to file" >&3
read -r -p "Enter value: " val </dev/tty  # read from real terminal
echo "You entered: $val" >&3
```

## File Operations

```bash
# cat — concatenate files
cat file.txt
cat file1.txt file2.txt > combined.txt
cat -n file.txt    # with line numbers
cat -A file.txt    # show non-printing characters

# head and tail
head -n 20 file.txt         # first 20 lines
tail -n 20 file.txt         # last 20 lines
tail -f /var/log/syslog     # follow (live view)
tail -n +5 file.txt         # from line 5 onwards

# wc — word count
wc -l file.txt    # line count
wc -w file.txt    # word count
wc -c file.txt    # byte count
wc -m file.txt    # character count (handles multibyte)

# cut — extract fields
cut -d: -f1 /etc/passwd          # first field, colon delimiter
cut -d, -f2,4 data.csv           # fields 2 and 4
cut -c1-10 file.txt              # characters 1-10
echo "hello world" | cut -d' ' -f2  # world

# sort
sort file.txt                   # alphabetical
sort -n numbers.txt             # numeric sort
sort -rn scores.txt             # reverse numeric
sort -t: -k3 -n /etc/passwd     # sort by 3rd field (UID)
sort -u file.txt                # sort + deduplicate
sort -k1,1 -k2,2n data.txt     # primary sort col1, secondary numeric col2

# uniq — remove adjacent duplicates (sort first!)
sort file.txt | uniq             # unique lines
sort file.txt | uniq -c          # count occurrences
sort file.txt | uniq -d          # only show duplicates
sort file.txt | uniq -u          # only show unique (non-duplicated)

# tr — translate characters
echo "hello" | tr 'a-z' 'A-Z'   # HELLO
echo "hello world" | tr ' ' '\n' # one word per line
cat file.txt | tr -d '\r'        # remove carriage returns (Windows line endings)
echo "aabbccdd" | tr -s 'a-z'   # squeeze: aabcd (single consecutive occurrences)
```

## find

```bash
# Basic find
find /path -name "*.sh"           # by name (glob)
find /path -iname "*.SH"          # case-insensitive name
find /path -type f                # files only
find /path -type d                # directories only
find /path -type l                # symlinks only

# By modification time
find . -mtime -7                  # modified in last 7 days
find . -mtime +30                 # modified more than 30 days ago
find . -newer reference.txt       # newer than reference.txt
find . -mmin -60                  # modified in last 60 minutes

# By size
find . -size +1M                  # larger than 1 MB
find . -size -100k                # smaller than 100 KB
find . -size +10M -size -100M     # between 10 and 100 MB

# By permissions
find . -perm 644                  # exact permissions
find . -perm /u+x                 # any executable bit set for user
find . -perm -0600                # at least owner r+w

# Execute command on found files
find . -name "*.log" -exec rm {} \;      # delete each file
find . -name "*.log" -exec rm {} +       # faster: batch multiple args
find . -name "*.sh" -exec chmod +x {} +  # make all .sh executable

# Complex conditions
find . -type f \( -name "*.jpg" -o -name "*.png" \)   # OR
find . -type f -not -name "*.bak"                       # NOT
find . -type f -name "*.log" -newer checkpoint.log \
       -size +1k -not -name "debug.log"                 # combined

# Prune directories
find . -path ./node_modules -prune -o -name "*.js" -print

# Find empty files/dirs
find . -empty -type f
find . -empty -type d

# Print0 for safe handling of filenames with spaces
find . -name "*.txt" -print0 | xargs -0 wc -l
```

## xargs

```bash
# Basic xargs
find . -name "*.log" | xargs rm         # remove found files
find . -name "*.sh" | xargs chmod +x

# -I {}: replace pattern
find . -name "*.txt" | xargs -I{} cp {} /backup/

# -P: parallel execution
find . -name "*.jpg" | xargs -P4 -I{} convert {} -resize 800x {}-thumb.jpg

# -0: null-delimited (safe for filenames with spaces)
find . -name "*.txt" -print0 | xargs -0 grep -l "pattern"

# Limit args per command (-n)
echo {1..100} | xargs -n10 echo  # 10 args per command

# With confirmation (-p)
find . -name "*.bak" | xargs -p rm

# Building commands
ls *.txt | xargs -I{} bash -c 'wc -l "$1"; echo "---"' _ {}
```

## awk

```bash
# Basic awk: process field-separated data
awk '{print $1}' file.txt           # first field (space-separated)
awk -F: '{print $1}' /etc/passwd    # first field, colon-separated
awk '{print NR, $0}' file.txt       # line numbers + content

# Key variables
# NR — current record (line) number
# NF — number of fields in current record
# $0 — entire current line
# $1, $2 ... — individual fields
# FS — input field separator (default: space/tab)
# OFS — output field separator
# RS — record separator (default: newline)
# ORS — output record separator

# Patterns and actions
awk 'NR > 5 { print $1 }' file.txt          # skip first 5 lines
awk '/error/ { print NR, $0 }' app.log      # lines matching pattern
awk 'NF > 3 { print }' data.txt            # lines with > 3 fields
awk '$3 > 1000 { print $1, $3 }' data.txt  # condition on field 3

# BEGIN and END blocks
awk 'BEGIN { print "Start"; total=0 }
     { total += $2 }
     END { print "Total:", total }' sales.txt

# printf formatting
awk -F: 'NR <= 5 { printf "%-20s uid=%-5d home=%s\n", $1, $3, $6 }' /etc/passwd

# Aggregation
awk '{sum+=$1; count++} END {printf "avg=%.2f\n", sum/count}' numbers.txt
awk '{counts[$1]++} END {for (k in counts) print counts[k], k}' words.txt | sort -rn

# Multi-file processing
awk 'FNR==1 {print "File:", FILENAME}' file1.txt file2.txt
```

## sed

```bash
# Basic substitution: s/pattern/replacement/flags
sed 's/old/new/' file.txt          # replace first on each line
sed 's/old/new/g' file.txt         # replace all (global)
sed 's/old/new/i' file.txt         # case-insensitive
sed 's/old/new/2' file.txt         # replace 2nd occurrence

# In-place editing
sed -i 's/localhost/db.example.com/g' config.txt
sed -i.bak 's/foo/bar/g' file.txt  # with backup

# Delete lines
sed '/pattern/d' file.txt          # delete matching lines
sed '1d' file.txt                  # delete first line
sed '1,5d' file.txt               # delete lines 1-5
sed '$d' file.txt                  # delete last line

# Print specific lines
sed -n '5p' file.txt               # print line 5
sed -n '5,10p' file.txt           # print lines 5-10
sed -n '/pattern/p' file.txt       # print matching lines

# Insert/append
sed '3i\New line before 3' file.txt  # insert before line 3
sed '3a\New line after 3' file.txt   # append after line 3
sed '/pattern/i\Before match' file.txt

# Ranges
sed '/start/,/end/ s/foo/bar/g' file.txt   # substitute between markers
sed '/^#/d' config.txt             # delete comment lines

# Multi-command
sed -e 's/foo/bar/g' -e 's/baz/qux/g' file.txt
```

## Process Substitution — Advanced Uses

```bash
# Log splitting
./application 2> >(grep -v "DEBUG" >> errors.log) | \
              tee >(grep "WARN" >> warnings.log) \
                  >(grep "ERROR" >> errors.log) > /dev/null

# Multi-way diff
diff3 <(cat v1.txt) <(cat v2.txt) <(cat v3.txt)

# Sort and join
join <(sort -k1 users.txt) <(sort -k1 orders.txt)

# Parallel compression
tar -cf - large_directory/ | \
  tee >(gzip > backup.tar.gz) >(bzip2 > backup.tar.bz2) > /dev/null
```

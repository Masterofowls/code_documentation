# Getting Started with Bash

## Running Scripts

```bash
# Method 1: Pass to bash interpreter
bash script.sh
bash -x script.sh   # with xtrace (prints each command)

# Method 2: Make executable and run directly (requires shebang line)
chmod +x script.sh
./script.sh          # ./ required because . (current dir) is not in PATH

# Method 3: Source — runs in CURRENT shell (affects current environment)
source script.sh
. script.sh         # same thing — POSIX form

# When to use source:
# - Loading environment variables into current shell
# - Activating virtual environments (source venv/bin/activate)
# - Reloading shell configuration (source ~/.bashrc)
# When NOT to source: most scripts — they modify your current shell state
```

## Script Structure

```bash
#!/usr/bin/env bash
# ============================================================
# script-name.sh — Brief description
# Usage: ./script-name.sh [options] <arguments>
# Author: Your Name
# ============================================================
set -euo pipefail

# Constants (uppercase by convention)
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly SCRIPT_NAME="$(basename "$0")"
readonly LOG_FILE="/tmp/${SCRIPT_NAME}.log"

# Global variables
VERBOSE=false
DRY_RUN=false

# Functions defined before main()
usage() {
  cat <<EOF
Usage: ${SCRIPT_NAME} [OPTIONS] <input-file>

Options:
  -v, --verbose    Enable verbose output
  -n, --dry-run    Show what would be done without doing it
  -h, --help       Show this help message

Examples:
  ${SCRIPT_NAME} data.csv
  ${SCRIPT_NAME} -v --dry-run input.txt
EOF
}

main() {
  parse_args "$@"
  validate_inputs
  do_the_work
}

# Entry point
main "$@"
```

## Comments

```bash
# Single-line comment — everything after # is ignored

echo "hello"  # inline comment

# Multi-line comment (using : command + heredoc trick)
: <<'COMMENT'
This is a multi-line comment.
None of this is executed.
Use this sparingly — prefer single # on each line.
COMMENT

# Block of commented-out code
# echo "disabled line 1"
# echo "disabled line 2"
```

## echo and printf

```bash
# echo — simple, but behavior varies across systems
echo "Hello, World!"
echo -n "no newline"      # -n flag: no trailing newline (not POSIX)
echo -e "tab:\there"      # -e flag: interpret escape sequences (not POSIX)

# printf — portable, consistent, preferred in scripts
printf "Hello, %s!\n" "World"
printf "%-20s %5d\n" "item" 42     # left-aligned, right-aligned with width
printf "%05.2f\n" 3.1415           # 03.14 — zero-padded, 2 decimal places
printf "Line 1\nLine 2\n"
printf "%s\n" "$@"                  # print each argument on its own line

# Writing to stderr
echo "Error: file not found" >&2
printf "Error: %s\n" "file not found" >&2
```

## Variables

```bash
# Assignment: no spaces around =
name="Alice"
count=42
pi=3.14159

# Reading variables
echo "$name"          # always quote variables to prevent word splitting
echo "${name}"        # braces explicit: useful before letters/numbers
echo "Hello, ${name}!"

# Avoid:
echo $name            # dangerous: word splits on spaces, glob expands
echo Hello, $name!    # ! at end starts history expansion in interactive shell

# Variable naming conventions
MY_CONSTANT="readonly value"
local_variable="mutable"
_internal_var="underscore prefix for private-ish"

# Unset a variable
unset name

# Check if set
if [[ -v name ]]; then echo "set"; fi       # bash 4.2+
if [[ -n "${name+x}" ]]; then echo "set"; fi  # portable
```

## Command Substitution

```bash
# Modern form: $() — preferred (nestable, readable)
today=$(date +%Y-%m-%d)
files=$(ls -1 *.txt 2>/dev/null)
count=$(wc -l < /etc/passwd)
user_home=$(eval echo "~$username")

# Legacy form: backticks `` — avoid (hard to read, hard to nest)
today=`date +%Y-%m-%d`   # works but prefer $()

# Nested command substitution
archive_name=$(date +%Y%m%d)_$(hostname)_backup.tar.gz
lines=$(wc -l < "$(find . -name '*.log' | head -1)")

# Capturing exit code
output=$(some-command); exit_code=$?
```

## Arithmetic

```bash
# $(( )) — arithmetic expansion (integer only)
result=$((2 + 3))        # 5
result=$((10 / 3))       # 3 (integer division, truncates)
result=$((10 % 3))       # 1 (modulus)
result=$((2 ** 8))       # 256 (exponentiation)

# Variable arithmetic
x=10
y=3
echo $((x * y))          # 30
echo $((x > y ? x : y))  # 10 (ternary operator)

# Increment/decrement
((count++))
((count--))
((count += 5))
((count *= 2))

# let command (older style)
let result=2+3
let "result = 10 / 3"

# Floating-point arithmetic with bc
pi=$(echo "scale=4; 4*a(1)" | bc -l)   # 3.1415
result=$(echo "scale=2; 10 / 3" | bc)  # 3.33
area=$(echo "scale=4; 3.14159 * 5^2" | bc)  # 78.5397

# Check if number
if [[ "$var" =~ ^-?[0-9]+$ ]]; then echo "integer"; fi
if [[ "$var" =~ ^-?[0-9]*\.?[0-9]+$ ]]; then echo "number"; fi
```

## Quoting

```bash
# No quotes — word splitting + glob expansion happen
files=*.txt
echo $files          # expands glob
echo "hello world"   # two args to echo without quotes

# Single quotes — literal string, no expansion at all
echo 'Hello $USER'   # prints: Hello $USER
echo '*.txt'         # prints: *.txt
echo 'it'\''s'       # workaround for single quote in single-quoted string

# Double quotes — variable/command expansion, no word splitting or glob
echo "Hello $USER"   # expands $USER
echo "Today: $(date)"  # expands command substitution
echo "*.txt"         # prints: *.txt (glob NOT expanded)
echo "$files"        # expands variable, preserves spaces in value

# ANSI-C quoting $'...' — interprets escape sequences
echo $'line1\nline2'     # actual newline
echo $'tab\there'        # actual tab
echo $'\x41\x42\x43'    # ABC (hex)
newline=$'\n'            # store newline in variable

# When to quote:
# - ALWAYS quote variables: "$var" not $var
# - ALWAYS quote command substitution: "$(cmd)" not $(cmd)
# - EXCEPT inside [[ ]] and (( )) — no word splitting there
# - EXCEPT when you WANT word splitting (passing array as multiple args)
```

## Special Variables

```bash
# Positional parameters
$0          # script name (or bash if sourced)
$1, $2...   # first, second argument
${10}       # 10th argument (braces required for double digits)
$#          # number of arguments
$@          # all arguments as separate quoted strings (preferred)
$*          # all arguments as a single word (IFS-joined)

# Process variables
$$          # PID of current shell
$!          # PID of most recently backgrounded process
$?          # exit status of last command
$-          # current shell options (e.g., "himBHs" for interactive)
$_          # last argument of previous command

# Script location (important for finding files relative to script)
${BASH_SOURCE[0]}   # path to current script (even when sourced)
SCRIPT_DIR=$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)

# Arguments usage examples
process_files() {
  local count=$#
  echo "Processing $count files"

  for file in "$@"; do   # always use $@ for argument lists
    echo "  → $file"
  done
}

# Shift: consume positional parameters
while [[ $# -gt 0 ]]; do
  case "$1" in
    -v) VERBOSE=true ;;
    *)  break ;;
  esac
  shift   # consume $1, shift $2 to $1, $3 to $2, etc.
done
```

## PATH Management

```bash
# Add directory to PATH (prepend = higher priority)
export PATH="$HOME/.local/bin:$PATH"
export PATH="$HOME/bin:$PATH"

# Add to end of PATH (lower priority)
export PATH="$PATH:/opt/custom/bin"

# View current PATH (one directory per line)
echo "$PATH" | tr ':' '\n'

# Find where a command lives
which bash          # /bin/bash (finds first in PATH)
which -a python     # shows all matches
type bash           # bash is /bin/bash (also handles builtins/functions)
type ll             # ll is aliased to `ls -la` (shows if it's an alias)
command -v git      # portable: returns path if found, empty if not
```

## readonly and export

```bash
# readonly: prevent variable from being modified or unset
readonly API_KEY="secret-key-123"
readonly MAX_RETRIES=3
# API_KEY="other"   # would fail: readonly variable

# export: make variable available to child processes
export DATABASE_URL="postgres://localhost/mydb"
export NODE_ENV="production"

# Combined
readonly export VERSION="1.0.0"
declare -rx VERSION="1.0.0"   # same thing with declare

# printenv: show exported variables
printenv | grep "^MY_"

# env: run a command with modified environment
env DATABASE_URL="postgres://test/mydb" ./run-tests.sh
env -i PATH="$PATH" HOME="$HOME" bash --noprofile --norc  # clean environment
```

## The env Command

```bash
# Display all environment variables
env

# Run command with specific environment
env VAR1=val1 VAR2=val2 command

# Clear environment and run command
env -i command

# Used in shebang to find interpreter in PATH
#!/usr/bin/env python3     # finds python3 in PATH
#!/usr/bin/env node        # finds node in PATH

# Useful for testing scripts with different environments
env DATABASE_URL="postgres://test/db" bash ./my-script.sh
```

# Getting Started with Bash Scripting

## Running Scripts

```bash
# Method 1: Pass script to bash explicitly
bash script.sh
bash -x script.sh    # Debug mode — print each command
bash -n script.sh    # Syntax check without running

# Method 2: Make executable and run directly
chmod +x script.sh
./script.sh          # ./ required — current directory not in PATH by default

# Method 3: source (run in current shell — shares environment)
source script.sh
. script.sh          # Same as source (POSIX dot command)
# Use source when the script needs to modify the current shell's environment
# (e.g., setting variables, changing directory)

# Run with specific options
bash -euo pipefail script.sh       # Override set flags
bash -- script.sh --arg1 --arg2    # -- separates bash options from script args
```

## Script Structure

```bash
#!/usr/bin/env bash
# ^^^ Shebang: tells the OS which interpreter to use
# Always use /usr/bin/env bash for portability

# --- Strict mode ---
set -euo pipefail

# --- Script metadata ---
readonly SCRIPT_NAME="$(basename "${BASH_SOURCE[0]}")"
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly SCRIPT_VERSION="1.0.0"

# --- Constants ---
readonly LOG_FILE="/var/log/myscript.log"
readonly DEFAULT_TIMEOUT=30

# --- Cleanup trap ---
cleanup() {
  local exit_code=$?
  # Remove temporary files
  [[ -d "${TMPDIR:-}" ]] && rm -rf "$TMPDIR"
  exit $exit_code
}
trap cleanup EXIT INT TERM HUP

# --- Helper functions ---
log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*" >&2; }
die() { log "ERROR: $*"; exit 1; }

# --- Main function (best practice: all logic in main) ---
main() {
  log "Starting ${SCRIPT_NAME}"

  # Check dependencies
  command -v jq >/dev/null 2>&1 || die "jq is required but not installed"
  command -v curl >/dev/null 2>&1 || die "curl is required but not installed"

  # Your script logic here
  log "Done."
}

# --- Entry point ---
main "$@"
```

## Comments

```bash
# Single line comment

Multi-line comment using the colon no-op command.
This is a heredoc passed to : which ignores it.
Useful for block comments in scripts.
'

echo "Code here"   # Inline comment
```

## echo and printf

```bash
# echo — simple but with portability quirks
echo "Hello World"
echo -n "No newline"   # -n suppresses newline (not POSIX, but works in Bash)
echo -e "Tab:\there"   # -e enables escape sequences (\t, \n, etc.)

# printf — portable, precise formatting (preferred for scripts)
printf "Hello %s\n" "World"
printf "%-20s %5d\n" "Alice" 42     # Left-aligned, right-aligned
printf "%.2f\n" 3.14159              # 2 decimal places: 3.14
printf "%05d\n" 42                   # Zero-padded: 00042
printf "%b\n" "Line1\nLine2"        # %b interprets escape sequences

# Write to stderr (use for error/log messages)
printf "Error: %s\n" "file not found" >&2

# Write to a variable (printf capture trick)
printf -v result "Value is: %d" 42
echo "$result"
```

:::tip[Prefer printf Over echo]
`echo` behavior varies across shells and systems (`echo -e` may not work, `echo -n` may print `-n` literally). `printf` is defined by POSIX and behaves consistently. Use `printf` in scripts for reliable output.
:::

## Variables

```bash
# Assignment — no spaces around =
name="Alice"
age=30
file_path="/home/user/document.txt"

# Reading variables — always quote when using
echo "Hello, $name"
echo "Hello, ${name}"    # Braces disambiguate: ${name}s vs $names

# NEVER do this (word splitting and globbing):
echo $name              # Works but fragile
# ALWAYS do this:
echo "$name"            # Quoted: safe

# Readonly variables (cannot be changed)
readonly API_KEY="secret123"
declare -r BASE_URL="https://api.example.com"

# Exported variables (available to child processes)
export DATABASE_URL="postgres://localhost/mydb"
declare -x LOG_LEVEL="info"   # -x is same as export

# Unset a variable
unset name

# Check if variable is set
[[ -v name ]] && echo "name is set" || echo "name is not set"
[[ -n "${name:-}" ]] && echo "name is non-empty"
```

## Command Substitution

```bash
# Modern syntax: $(command) — recommended
current_date=$(date '+%Y-%m-%d')
hostname=$(hostname -f)
git_branch=$(git rev-parse --abbrev-ref HEAD 2>/dev/null || echo "not-git")

# Old syntax: `command` — avoid (harder to nest, escaping issues)
old_way=`date`

# Nested command substitution
nested=$(echo "$(date '+%Y')-$(date '+%m')-$(date '+%d')")
user_home=$(getent passwd "$(whoami)" | cut -d: -f6)

# Capture exit code along with output
output=$(some_command 2>&1) && echo "Success: $output" || echo "Failed: $output"

# Multi-line output preserves newlines in variable, but echo removes them
lines=$(ls -la)
echo "$lines"     # Preserves newlines (quoted)
echo $lines       # Word splits — loses newlines (unquoted!)

# Process substitution (not command substitution, but related)
diff <(sort file1.txt) <(sort file2.txt)   # Compare sorted outputs
```

## Arithmetic

```bash
# $(( )) — arithmetic expansion (integer only)
a=10
b=3
echo $((a + b))        # 13
echo $((a - b))        # 7
echo $((a * b))        # 30
echo $((a / b))        # 3 (integer division)
echo $((a % b))        # 1 (modulo)
echo $((a ** b))       # 1000 (exponentiation, Bash only)
echo $((a++))          # 10 (post-increment)
echo $((++a))          # 12 (pre-increment)
echo $((a > b ? a : b))  # Ternary: 12 (max)

# let command (same as $(()))
let "result = a * b"
let a++

# Compound assignment
(( total += price * quantity ))
(( count++ ))
(( i = i < max ? i + 1 : 0 ))

# Comparison in arithmetic context (returns 0=true, 1=false for if)
if (( a > b )); then echo "a is greater"; fi

# bc — floating point arithmetic (bc is an external calculator)
result=$(echo "scale=4; 355 / 113" | bc)    # 3.1415
result=$(bc -l <<< "sqrt(2)")                # 1.41421356...
result=$(python3 -c "print(3.14 * 2**2)")   # For complex math
```

## Quoting

```bash
# Single quotes: literal, no expansion whatsoever
echo '$HOME'           # $HOME (literally)
echo 'It'\''s a test'  # Escape single quote: close, add \', reopen

# Double quotes: expand $variables, $(commands), `backticks`, \escapes
echo "Home: $HOME"     # Home: /home/user
echo "Date: $(date)"   # Date: Mon Jan 15 14:30:00 EST 2024
echo "Tab:\there"      # Tab:    here (with actual tab)

# ANSI-C quoting: $'...' — interprets escape sequences
echo $'Line1\nLine2'   # Two lines
echo $'\t\tindented'   # Two tabs
echo $'\x41'           # A (hex escape)

# No quotes: word splitting and globbing apply
files=$(ls *.txt)
echo $files             # DANGEROUS: word splits on spaces in filenames!
echo "$files"           # Safe: preserves the string as-is

# Summary of when to quote:
# Variables: always "$var" unless you want word splitting
# Command substitution: always "$(cmd)" unless you want word splitting
# Glob patterns: no quotes (you WANT globbing)
# Heredoc delimiter: quote to prevent expansion in the body
cat << 'EOF'
$HOME is not expanded here
EOF
```

## Word Splitting and Globbing

```bash
# Word splitting happens on $IFS characters (default: space, tab, newline)
files="file1.txt file2.txt file3.txt"
for f in $files; do   # Splits on spaces — works here
  echo "$f"
done
# But breaks if filenames have spaces!

# Safe approach: use arrays
files=(file1.txt "file with spaces.txt" file3.txt)
for f in "${files[@]}"; do   # "${arr[@]}" — each element is a separate word
  echo "$f"
done

# Globbing with word splitting
for f in /tmp/*.log; do
  echo "$f"
done
# If no files match, $f is literally "/tmp/*.log" — check with -e or nullglob
shopt -s nullglob    # Make non-matching globs expand to nothing
shopt -s failglob    # Make non-matching globs error
```

## Special Variables

```bash
# Script/function context
echo "$0"           # Script name or -bash if interactive
echo "$1" "$2" "$3" # Positional parameters 1, 2, 3
echo "${10}"        # Positional parameters > 9 need braces
echo "$#"           # Number of positional parameters
echo "$@"           # All positional parameters (each as separate word) — use this!
echo "$*"           # All positional parameters (as one word joined by $IFS[0])

# Process
echo "$$"           # Current process PID
echo "$!"           # PID of last background job
echo "$?"           # Exit code of last command (0=success, non-zero=failure)

# Shell state
echo "$-"           # Current shell options (e.g., "himBHs")
echo "$_"           # Last argument of last command

# Script path (reliable way to get script directory)
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
# ${BASH_SOURCE[0]} = path to current file (works even when sourced)
# $0 can be "bash" or "-bash" when sourced, so use BASH_SOURCE

# "$@" vs "$*" difference:
set -- "arg one" "arg two"
printf '"%s" ' "$@"    # "arg one" "arg two"  (preserves separation)
printf '"%s" ' "$*"    # "arg one arg two"    (joins into one string)
```

## PATH Management

```bash
# Add directory to PATH (at beginning for priority)
export PATH="/usr/local/bin:$PATH"

# Add directory to PATH (at end)
export PATH="$PATH:$HOME/.local/bin"

# Add only if directory exists and is not already in PATH
add_to_path() {
  local dir="$1"
  if [[ -d "$dir" ]] && [[ ":$PATH:" != *":$dir:"* ]]; then
    export PATH="$dir:$PATH"
  fi
}

add_to_path "$HOME/.cargo/bin"
add_to_path "$HOME/go/bin"

# Find which command would run
which bash           # /bin/bash (searches PATH)
type bash            # bash is /bin/bash (also shows builtins, functions)
type -a python3      # Shows all matching locations
command -v git       # Like which, POSIX-compatible, returns path

# PATH debugging
echo "${PATH//:/$'\n'}"   # Print each PATH entry on a new line
IFS=: read -ra paths <<< "$PATH"
printf '%s\n' "${paths[@]}"
```

## env Command

```bash
# Print all environment variables
env
printenv

# Run command with modified environment (without affecting current shell)
env KEY=value command
env HOME=/tmp command    # Override HOME for this command only

# Run command with clean environment (only specified vars)
env -i PATH=/usr/bin HOME=/tmp bash    # Minimal environment

# Unset a variable for one command
env -u SOME_VAR command

# Useful for testing scripts in clean environments
env -i HOME="$HOME" PATH="$PATH" bash script.sh
```

:::caution[Never Put Passwords in env Variables in Shell Scripts]
While environment variables are commonly used for configuration, they are visible to all processes running as the same user via `/proc/PID/environ`. For sensitive values, use secret management tools (HashiCorp Vault, AWS Secrets Manager) or read from a restricted file.
:::

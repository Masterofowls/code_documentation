# Bash Getting Started

## Running Scripts

```bash
# Method 1: Pass script to bash interpreter
bash script.sh
bash -x script.sh          # debug mode: print each command

# Method 2: Make executable and run directly
chmod +x script.sh
./script.sh                # ./ is required (not just script.sh)

# Method 3: Source the script (runs in current shell, not subshell)
source script.sh           # changes affect current shell
. script.sh                # POSIX equivalent (dot command)

# Method 4: Run with explicit path
/path/to/script.sh

# Check exit code of last command
echo $?                    # 0 = success, non-zero = error
```

:::tip[Why ./script.sh and not script.sh?]
The current directory `.` is NOT in `$PATH` by default (for security). You must explicitly reference the path. Without `./`, bash searches `$PATH` and won't find your script unless it's in a PATH directory.
:::

## Script Structure

```bash
#!/usr/bin/env bash
# ────────────────────────────────────────────────
# deploy.sh — Deploy application to target environment
# Usage: ./deploy.sh [--env ENVIRONMENT] [--dry-run]
# Author: Your Name
# ────────────────────────────────────────────────

set -euo pipefail

# ── Constants ──────────────────────────────────
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly SCRIPT_NAME="$(basename "$0")"
readonly LOG_FILE="/var/log/deploy.log"

# ── Default configuration ──────────────────────
ENV="production"
DRY_RUN=false

# ── Logging ────────────────────────────────────
log_info()  { echo "[INFO]  $(date '+%Y-%m-%d %H:%M:%S') $*" | tee -a "$LOG_FILE"; }
log_error() { echo "[ERROR] $(date '+%Y-%m-%d %H:%M:%S') $*" | tee -a "$LOG_FILE" >&2; }

# ── Cleanup ────────────────────────────────────
cleanup() {
  local exit_code=$?
  if [[ $exit_code -ne 0 ]]; then
    log_error "Script failed with exit code $exit_code"
  fi
  # Remove temp files, release locks, etc.
  rm -f /tmp/deploy.lock
}
trap cleanup EXIT

# ── Functions ──────────────────────────────────
usage() {
  cat <<EOF
Usage: $SCRIPT_NAME [OPTIONS]

Options:
  --env ENV       Target environment (default: production)
  --dry-run       Print actions without executing
  -h, --help      Show this help
EOF
}

# ── Main ───────────────────────────────────────
main() {
  parse_args "$@"
  validate_environment
  deploy_application
  log_info "Deployment complete"
}

# Entry point
main "$@"
```

## echo and printf

```bash
# echo — simple output
echo "Hello, World!"
echo -n "No newline"            # -n suppresses newline
echo -e "Tab:\there\nNewline"   # -e enables escape sequences
echo Hello World                # works without quotes (word split)

# printf — portable and more powerful
printf "Hello, %s!\n" "World"
printf "%-20s %5d\n" "item" 42      # left-align string, right-align int
printf "%05.2f\n" 3.14159           # 03.14
printf "Hex: %x  Oct: %o\n" 255 255 # ff  377

# printf to variable (no subshell)
printf -v my_var "Value is %d" 42
echo "$my_var"   # Value is 42

# Print to stderr
echo "Error message" >&2
printf "Error: %s\n" "something failed" >&2
```

:::tip[printf over echo]
Prefer `printf` in scripts for predictable behavior. `echo` behavior differs between systems for options like `-e` and `-n`. `printf` is POSIX-defined and consistent.
:::

## Variables

```bash
# Assignment — NO spaces around = (spaces are syntax errors)
name="Alice"
count=42
pi=3.14159
empty=""

# Wrong: creates three separate commands
# name = "Alice"  ← syntax error

# Reading a variable
echo "$name"          # Always quote variable expansions!
echo "Hello, ${name}!"  # Braces for disambiguation

# Unset variable
unset name
echo "${name}"        # empty string (with set -u: would error)
```

### readonly and export

```bash
# readonly: cannot be changed or unset
readonly MAX_RETRIES=3
readonly DB_NAME="myapp"
# MAX_RETRIES=5  ← error: readonly variable

# export: make available to child processes (environment variables)
export PATH="$HOME/.local/bin:$PATH"
export EDITOR=vim
export -n EDITOR    # unexport (remove from environment, keep as shell var)

# readonly + export combined
readonly -x API_URL="https://api.example.com"

# List all exported variables
export
env
printenv
```

## Command Substitution

```bash
# Modern syntax (preferred): $()
current_date=$(date +%Y-%m-%d)
file_count=$(ls -1 | wc -l)
hostname=$(hostname -f)

# Nested command substitution
parent_dir=$(dirname "$(readlink -f "$0")")

# Legacy syntax: backticks (avoid)
old_style=`date`   # hard to nest, hard to read

# Capture both stdout and exit code
output=$(some_command 2>/dev/null) || handle_error
```

## Arithmetic

```bash
# $(( )) — arithmetic expansion (integers only)
sum=$(( 3 + 5 ))
echo $(( 10 * 5 + 2 ))   # 52
echo $(( 2 ** 8 ))        # 256 (exponentiation)
echo $(( 17 / 3 ))        # 5 (integer division)
echo $(( 17 % 3 ))        # 2 (modulo)

# Increment/decrement
(( count++ ))
(( count-- ))
(( count += 5 ))

# Bitwise
echo $(( 0xFF & 0x0F ))    # 15
echo $(( 1 << 4 ))         # 16

# let command (similar to (( )))
let "a = 5 + 3"
let "b = a * 2"

# expr — POSIX, slower, awkward (use only for POSIX portability)
sum=$(expr 3 + 5)

# Floating point — requires bc or awk
result=$(echo "scale=4; 22/7" | bc)         # 3.1428
result=$(awk 'BEGIN { printf "%.4f\n", 22/7 }')  # 3.1429
```

## Quoting

```bash
# Single quotes: everything literal, no expansion
echo '$PATH'            # outputs: $PATH
echo 'Hello\nWorld'     # outputs: Hello\nWorld (no newline)
echo '$(date)'          # outputs: $(date)

# Double quotes: expand $variables, $(commands), \escapes; not * or ?
name="Alice"
echo "Hello, $name!"    # outputs: Hello, Alice!
echo "Home: $HOME"      # expands variable
echo "Date: $(date)"    # expands command substitution
echo "Literal: \$name"  # escape the dollar sign

# No quotes: subject to word splitting and glob expansion
files=*.txt             # becomes array of matching files (or literal *.txt if none)
echo $files             # word splits spaces in filenames

# ANSI-C quoting: $'' (interpret escape sequences)
echo $'Tab:\there'      # actual tab character
echo $'Newline:\nHere'  # actual newline
echo $'\033[31mRed\033[0m'  # ANSI color codes

# The quoting golden rule: ALWAYS double-quote variable expansions
# "$var" not $var
# "$@" not $@
cp "$source" "$destination"   # ✅ handles spaces in paths
cp $source $destination        # ❌ breaks on spaces
```

## Word Splitting and Globbing

```bash
# Word splitting: unquoted variables are split on IFS characters (default: space tab newline)
files="file1.txt file2.txt file3.txt"
for f in $files; do         # splits on spaces — 3 iterations
  echo "$f"
done
for f in "$files"; do       # quoted — 1 iteration (entire string)
  echo "$f"
done

# Globbing: unquoted * ? [...] are expanded by the shell
for f in *.txt; do          # expands to matching files
  echo "$f"
done

# Disable globbing (for pattern strings)
set -f                      # disable glob expansion
result=$(command --pattern "*.txt")   # * not expanded
set +f                      # re-enable

# Glob patterns
echo *.txt                  # all .txt files
echo file?.txt              # file1.txt, fileA.txt, etc.
echo file[123].txt          # file1.txt, file2.txt, file3.txt
echo **/*.sh                # all .sh files recursively (requires shopt -s globstar)
shopt -s globstar
for f in **/*.py; do echo "$f"; done
```

## Special Variables

```bash
$0                    # script name (or shell name if interactive)
$1 $2 ... $9          # positional parameters (arguments)
${10} ${11}           # 10th, 11th parameter (braces required for >9)
$#                    # number of positional parameters
$@                    # all positional parameters as separate words
$*                    # all positional parameters as one word (IFS-joined)
"$@"                  # all params, each properly quoted — ALWAYS use this
"$*"                  # all params as single quoted string
$?                    # exit code of last command
$$                    # PID of current shell process
$!                    # PID of last background process
$-                    # current shell option flags (e.g., himBHs)
$_                    # last argument of previous command
$BASH_SOURCE          # array of source file names in call stack
${BASH_SOURCE[0]}     # current script's path (more reliable than $0)
$LINENO               # current line number
$FUNCNAME             # array of function names in call stack
$BASH_VERSION         # version string
$BASH_PID             # PID of current bash process (4.0+)
$PPID                 # PID of parent process
$HOSTNAME             # system hostname
$PWD                  # current directory (same as `pwd`)
$OLDPWD               # previous directory (cd -)
$HOME                 # home directory
$USER / $LOGNAME      # current username
$IFS                  # Internal Field Separator
$PATH                 # command search path
$PS1                  # primary prompt
$RANDOM               # random integer 0-32767
$SECONDS              # seconds since shell started
$LINENO               # current line number in script
```

```bash
# Practical usage
script_dir="${BASH_SOURCE[0]%/*}"   # directory containing the script
script_name="${BASH_SOURCE[0]##*/}" # filename only

# Shift positional parameters
echo "First: $1"
shift            # remove $1, shift others down ($2 becomes $1, etc.)
echo "Was second, now first: $1"
shift 2          # shift by 2
```

## PATH Management

```bash
# Add directory to PATH (append)
export PATH="$PATH:/usr/local/myapp/bin"

# Add directory to PATH (prepend — takes priority)
export PATH="/usr/local/myapp/bin:$PATH"

# Add only if not already in PATH (idempotent)
add_to_path() {
  local dir="$1"
  case ":$PATH:" in
    *":$dir:"*) ;;           # already present
    *)  export PATH="${dir}:${PATH}" ;;
  esac
}

# Find executable location
which bash          # shows first in PATH
type bash           # shows type (builtin, function, alias, or file)
type -a ls          # show all locations of ls
command -v python3  # POSIX-compatible, returns path or empty

# Check if command exists
if command -v docker &>/dev/null; then
  echo "Docker is installed"
fi
```

## env Command

```bash
# Print all environment variables
env
printenv

# Print specific variable
printenv HOME
printenv PATH

# Run command with modified environment
env DATABASE_URL="postgres://..." node server.js

# Run command with clean environment
env -i PATH=/usr/bin:/bin bash   # only PATH set

# Run command with variable set
DATABASE_URL="postgres://..." node server.js  # same as env but shorter syntax
```

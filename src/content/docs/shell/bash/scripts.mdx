# Bash Scripts

## Script Best Practices Checklist

Every production Bash script should include:

```bash
#!/usr/bin/env bash
# ─────────────────────────────────────────────────────────────
# Script: backup.sh
# Description: Backup application data to S3
# Usage: ./backup.sh [--env staging|production] [--dry-run]
# Requirements: aws CLI, tar, gzip
# ─────────────────────────────────────────────────────────────

# 1. STRICT MODE — fail fast and loudly
set -euo pipefail

# 2. SCRIPT DIRECTORY — reliable script-relative paths
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly SCRIPT_NAME="$(basename "${BASH_SOURCE[0]}")"

# 3. CHECK REQUIRED COMMANDS
require_commands() {
  local cmd
  for cmd in "$@"; do
    if ! command -v "$cmd" &>/dev/null; then
      echo "Error: Required command not found: $cmd" >&2
      exit 1
    fi
  done
}
require_commands aws tar gzip

# 4. CONSTANTS (readonly to prevent accidental modification)
readonly DEFAULT_ENV="staging"
readonly TIMESTAMP="$(date +%Y%m%d_%H%M%S)"
readonly LOG_FILE="/var/log/backup_${TIMESTAMP}.log"
readonly MAX_BACKUPS=7

# 5. VALIDATE REQUIRED ENV VARS
: "${AWS_BUCKET:?'AWS_BUCKET environment variable is required'}"
: "${DATABASE_URL:?'DATABASE_URL environment variable is required'}"

# 6. CLEANUP TRAP — runs on any exit
TMPDIR_PATH=""
cleanup() {
  local exit_code=$?
  [[ -n "$TMPDIR_PATH" ]] && rm -rf "$TMPDIR_PATH"
  exit "$exit_code"
}
trap cleanup EXIT INT TERM HUP

# 7. LOGGING FUNCTIONS
log_info()  { printf '[INFO]  %s %s\n' "$(date '+%H:%M:%S')" "$*"; }
log_warn()  { printf '[WARN]  %s %s\n' "$(date '+%H:%M:%S')" "$*" >&2; }
log_error() { printf '[ERROR] %s %s\n' "$(date '+%H:%M:%S')" "$*" >&2; }

# 8. MAIN FUNCTION — entry point
main() {
  parse_args "$@"
  validate_inputs
  create_backup
  upload_backup
  cleanup_old_backups
  log_info "Backup complete!"
}

main "$@"
```

## Argument Parsing

### getopts (Short Options)

```bash
parse_args() {
  local opt
  while getopts ":e:n:dvh" opt; do
    case "$opt" in
      e) ENV="$OPTARG"
         if [[ "$ENV" != "staging" && "$ENV" != "production" ]]; then
           log_error "Invalid environment: $ENV (use staging or production)"
           exit 1
         fi
         ;;
      n) COUNT="$OPTARG"
         if ! [[ "$COUNT" =~ ^[1-9][0-9]*$ ]]; then
           log_error "-n must be a positive integer"
           exit 1
         fi
         ;;
      d) DRY_RUN=true ;;
      v) VERBOSE=true ;;
      h) usage; exit 0 ;;
      :) log_error "Option -$OPTARG requires an argument"; exit 1 ;;
      \?) log_error "Unknown option: -$OPTARG"; exit 1 ;;
    esac
  done
  shift $(( OPTIND - 1 ))
  POSITIONAL_ARGS=("$@")
}
```

### Manual Long Options Parsing

```bash
parse_args_long() {
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --env|-e)
        [[ -z "${2:-}" ]] && { log_error "--env requires a value"; exit 1; }
        ENV="$2"; shift 2 ;;
      --env=*)
        ENV="${1#*=}"; shift ;;
      --dry-run)
        DRY_RUN=true; shift ;;
      --verbose|-v)
        VERBOSE=true; shift ;;
      --count=*)
        COUNT="${1#*=}"; shift ;;
      --help|-h)
        usage; exit 0 ;;
      --)
        shift; POSITIONAL_ARGS+=("$@"); break ;;
      -*)
        log_error "Unknown option: $1"
        usage
        exit 1 ;;
      *)
        POSITIONAL_ARGS+=("$1"); shift ;;
    esac
  done
}
```

## Input Validation

```bash
validate_inputs() {
  # Require non-empty
  [[ -z "${DB_HOST:-}" ]] && { log_error "DB_HOST is required"; exit 1; }

  # Validate file exists
  [[ -f "$CONFIG_FILE" ]] || { log_error "Config not found: $CONFIG_FILE"; exit 1; }

  # Validate directory and writable
  if [[ ! -d "$OUTPUT_DIR" ]]; then
    log_info "Creating output directory: $OUTPUT_DIR"
    mkdir -p "$OUTPUT_DIR" || { log_error "Cannot create $OUTPUT_DIR"; exit 1; }
  fi
  [[ -w "$OUTPUT_DIR" ]] || { log_error "Output dir not writable: $OUTPUT_DIR"; exit 1; }

  # Validate numeric
  if ! [[ "$PORT" =~ ^[0-9]+$ ]] || (( PORT < 1 || PORT > 65535 )); then
    log_error "Invalid port: $PORT (must be 1-65535)"
    exit 1
  fi

  # Validate enum value
  local valid_envs=("staging" "production" "development")
  local valid=false
  for env in "${valid_envs[@]}"; do
    [[ "$ENV" == "$env" ]] && { valid=true; break; }
  done
  "$valid" || { log_error "Invalid env: $ENV. Must be one of: ${valid_envs[*]}"; exit 1; }
}
```

## Color Output

```bash
# ANSI escape codes
setup_colors() {
  # Only use colors when stdout is a terminal
  if [[ -t 1 ]]; then
    RED='\033[0;31m'
    GREEN='\033[0;32m'
    YELLOW='\033[1;33m'
    BLUE='\033[0;34m'
    MAGENTA='\033[0;35m'
    CYAN='\033[0;36m'
    WHITE='\033[1;37m'
    BOLD='\033[1m'
    DIM='\033[2m'
    UNDERLINE='\033[4m'
    RESET='\033[0m'
  else
    RED='' GREEN='' YELLOW='' BLUE='' MAGENTA='' CYAN=''
    WHITE='' BOLD='' DIM='' UNDERLINE='' RESET=''
  fi
}
setup_colors

# Or use tput (more portable, respects TERM variable)
setup_colors_tput() {
  if [[ -t 1 ]] && command -v tput &>/dev/null; then
    RED=$(tput setaf 1)
    GREEN=$(tput setaf 2)
    YELLOW=$(tput setaf 3)
    BLUE=$(tput setaf 4)
    BOLD=$(tput bold)
    RESET=$(tput sgr0)
  else
    RED='' GREEN='' YELLOW='' BLUE='' BOLD='' RESET=''
  fi
}

# Colored log functions
success() { echo -e "${GREEN}✓${RESET} $*"; }
failure() { echo -e "${RED}✗${RESET} $*" >&2; }
info()    { echo -e "${BLUE}ℹ${RESET} $*"; }
warn()    { echo -e "${YELLOW}⚠${RESET} $*" >&2; }
step()    { echo -e "${BOLD}${BLUE}=>${RESET}${BOLD} $*${RESET}"; }
```

## Logging with Timestamps

```bash
# Logging to both file and stdout
LOG_FILE="/var/log/myscript.log"

log() {
  local level="$1"; shift
  local message="$*"
  local timestamp
  timestamp=$(date '+%Y-%m-%d %H:%M:%S')
  local line="${timestamp} [${level}] ${message}"

  # Write to log file
  echo "$line" >> "$LOG_FILE"

  # Write to terminal with color
  case "$level" in
    INFO)  echo -e "${BLUE}${line}${RESET}" ;;
    WARN)  echo -e "${YELLOW}${line}${RESET}" >&2 ;;
    ERROR) echo -e "${RED}${line}${RESET}" >&2 ;;
    DEBUG) ${VERBOSE:-false} && echo -e "${DIM}${line}${RESET}" ;;
  esac
}

log_info()  { log INFO  "$@"; }
log_warn()  { log WARN  "$@"; }
log_error() { log ERROR "$@"; }
log_debug() { log DEBUG "$@"; }

# Redirect all output to log (and screen via tee)
exec > >(tee -a "$LOG_FILE") 2>&1
```

## Dry-Run Mode

```bash
# DRY_RUN mode: show what would happen without doing it
DRY_RUN=false

# Wrapper: run command or show it
run() {
  if "$DRY_RUN"; then
    echo "[DRY-RUN] $*"
  else
    "$@"
  fi
}

# Usage:
run rm -f /tmp/stale_file.txt
run mv /tmp/backup.tar.gz /var/backups/
run systemctl restart nginx
```

## Idempotent Scripts

```bash
# Idempotent: running multiple times produces the same result

# Check before creating
if [[ ! -d /var/app/data ]]; then
  mkdir -p /var/app/data
  log_info "Created /var/app/data"
else
  log_debug "/var/app/data already exists"
fi

# Write only if changed
write_if_changed() {
  local target="$1"
  local content="$2"
  if [[ ! -f "$target" ]] || [[ "$(cat "$target")" != "$content" ]]; then
    echo "$content" > "$target"
    log_info "Updated $target"
    return 0
  fi
  log_debug "$target unchanged"
  return 1
}

# Idempotent systemd service enable
if ! systemctl is-enabled myapp.service &>/dev/null; then
  systemctl enable myapp.service
fi
```

## Bash Strict Mode Pitfalls

```bash
# ─── set -e pitfalls ─────────────────────────────────────

# 1. Commands in if/while/until conditions are EXEMPT from -e
set -e
if grep -q "missing" file.txt; then   # grep failure doesn't exit
  echo "found"
fi

# 2. Commands followed by || or && are exempt
set -e
cat missing_file || true   # doesn't exit — OK
command_that_fails || handle_error  # doesn't exit

# 3. The last command in a pipeline might fail even with pipefail
set -eo pipefail
echo "test" | false | echo "after"   # exits because false failed

# ─── set -u pitfalls ─────────────────────────────────────

# Arrays with set -u
set -u
arr=()
echo "${arr[@]}"    # ERROR: unbound variable (empty array)
echo "${arr[@]+"${arr[@]}"}  # Safe: uses ${var+value} idiom

# Safer empty array access
[[ "${#arr[@]}" -gt 0 ]] && echo "${arr[@]}"

# ─── pipefail pitfalls ───────────────────────────────────

# grep returns 1 when no match found — triggers pipefail!
set -eo pipefail
cat file | grep "maybe_missing"  # EXIT if no match!

# Fix: handle grep's exit code
set -eo pipefail
result=$(grep "pattern" file || true)   # OR true if not found
matches=$(grep -c "pattern" file || echo 0)  # count, 0 if none
```

## shellcheck

shellcheck is a static analysis tool for shell scripts:

```bash
# Install
sudo apt install shellcheck    # Ubuntu/Debian
brew install shellcheck        # macOS

# Run
shellcheck script.sh
shellcheck -S warning *.sh     # warning level and above
shellcheck -f json script.sh   # JSON output for IDE integration

# Inline disable (use sparingly, with comment explaining why)
# shellcheck disable=SC2086   # intentional word splitting
ls $pattern
```

### Common shellcheck Warnings

| Code   | Warning                                 | Fix                                |
| ------ | --------------------------------------- | ---------------------------------- | ----- | -------- | --- | ------- |
| SC2086 | Double quote to prevent globbing        | Use `"$var"`                       |
| SC2048 | Double quote array expansion            | Use `"${arr[@]}"`                  |
| SC2068 | Double quote array expansion in command | Use `"${arr[@]}"`                  |
| SC2206 | Quote to prevent word splitting         | Use `mapfile` or `read -ra`        |
| SC2207 | Prefer mapfile or read -ra              | Replace `arr=($())` with `mapfile` |
| SC2155 | Declare and assign separately           | `local var; var=$(cmd)`            |
| SC2164 | Use `cd ...                             |                                    | exit` | `cd /dir |     | exit 1` |
| SC2044 | Find with -exec replaces variable       | Use `while read` loop              |

## Testing with bats-core

```bash
# Install bats-core
git clone https://github.com/bats-core/bats-core.git
cd bats-core && ./install.sh /usr/local

# Or via npm: npm install -g bats
# Or via brew: brew install bats-core
```

```bash
#!/usr/bin/env bats
# tests/backup.bats

# Setup: run before each test
setup() {
  # Create temp directory for test files
  TEST_TMPDIR=$(mktemp -d)
  export TEST_TMPDIR
  # Source the script under test (without running main)
  source "${BATS_TEST_DIRNAME}/../lib/functions.sh"
}

# Teardown: run after each test
teardown() {
  rm -rf "$TEST_TMPDIR"
}

# Test syntax
@test "script has valid syntax" {
  run bash -n backup.sh
  [ "$status" -eq 0 ]
}

@test "backup creates output file" {
  run backup_to_dir "$TEST_TMPDIR"
  [ "$status" -eq 0 ]
  [ -f "$TEST_TMPDIR/backup.tar.gz" ]
}

@test "backup fails with missing source" {
  run backup_to_dir "/nonexistent/path"
  [ "$status" -ne 0 ]
  [[ "$output" == *"not found"* ]]
}

@test "validate_email accepts valid email" {
  run validate_email "alice@example.com"
  [ "$status" -eq 0 ]
}

@test "validate_email rejects invalid email" {
  run validate_email "notanemail"
  [ "$status" -ne 0 ]
}

@test "get_config returns value for known key" {
  echo "DB_HOST=localhost" > "$TEST_TMPDIR/.env"
  run get_config "$TEST_TMPDIR/.env" "DB_HOST"
  [ "$status" -eq 0 ]
  [ "$output" = "localhost" ]
}

@test "script requires AWS_BUCKET env var" {
  unset AWS_BUCKET
  run bash backup.sh
  [ "$status" -ne 0 ]
  [[ "$output" == *"AWS_BUCKET"* ]]
}
```

```bash
# Run tests
bats tests/backup.bats
bats tests/                    # all test files in directory
bats -t tests/backup.bats      # show timing
bats --filter "backup" tests/  # run matching test names

# Output:
# ✓ script has valid syntax
# ✓ backup creates output file
# ✓ backup fails with missing source
# ✓ validate_email accepts valid email
# ✓ validate_email rejects invalid email
# 5 tests, 0 failures
```

:::tip[Test Shell Scripts Like Any Code]
Shell scripts that run in production deserve the same testing discipline as application code. Use bats-core for unit tests, test helper libraries (bats-support, bats-assert, bats-file) for richer assertions, and run shellcheck in your CI pipeline.
:::

# Bash Variables & Parameter Expansion

## Variable Types

```bash
# Scalar (regular variable)
name="Alice"
count=42

# Integer (declare -i enforces integer type)
declare -i num=10
num="5+3"    # Evaluated as arithmetic: num=8
num="hello"  # Becomes 0

# Lowercase/Uppercase enforcement (Bash 4.0+)
declare -l lower_name="HELLO"   # Converts to lowercase: "hello"
declare -u upper_name="hello"   # Converts to uppercase: "HELLO"

# Readonly
declare -r CONSTANT="immutable"
readonly MAX_RETRIES=3

# Export to environment
declare -x DB_HOST="localhost"  # Same as: export DB_HOST="localhost"

# Nameref (reference to another variable, Bash 4.3+)
declare -n ref=name
ref="Bob"    # Sets name="Bob"

# Show variable type and attributes
declare -p name      # declare -- name="Bob"
declare -p CONSTANT  # declare -r CONSTANT="immutable"
```

## Arrays

```bash
# Indexed array creation
fruits=("apple" "banana" "cherry" "date")
declare -a colors   # Declare without initializing

# Sparse array (gaps are allowed)
arr[0]="first"
arr[5]="sixth"    # arr[1..4] don't exist

# Access elements
echo "${fruits[0]}"    # apple (0-indexed)
echo "${fruits[1]}"    # banana
echo "${fruits[-1]}"   # date (last element, Bash 4.2+)
echo "${fruits[-2]}"   # cherry (second to last)

# Get all elements
echo "${fruits[@]}"    # apple banana cherry date (each as separate word)
echo "${fruits[*]}"    # "apple banana cherry date" (joined by IFS[0])
# Use [@] — preserves elements with spaces

# Number of elements
echo "${#fruits[@]}"   # 4

# Indices (useful for sparse arrays)
echo "${!fruits[@]}"   # 0 1 2 3

# Slicing: ${arr[@]:start:length}
echo "${fruits[@]:1:2}"  # banana cherry (2 elements starting at index 1)
echo "${fruits[@]:1}"    # banana cherry date (from index 1 to end)

# Append to array
fruits+=("elderberry")
fruits+=("fig" "grape")

# Remove an element (leaves gap in indexed array)
unset 'fruits[1]'        # Removes banana; index 1 is now empty

# Re-index after deletion
fruits=("${fruits[@]}")  # Recreate without gaps

# Iterate over array
for fruit in "${fruits[@]}"; do
  echo "Fruit: $fruit"
done

# Iterate with index
for i in "${!fruits[@]}"; do
  echo "$i: ${fruits[$i]}"
done

# Array length and element length
echo "${#fruits[@]}"      # Number of elements
echo "${#fruits[0]}"      # Length of first element string

# Array as function argument (pass by serialization)
process_items() {
  local -a items=("$@")   # Receive as "$@"
  for item in "${items[@]}"; do
    echo "Processing: $item"
  done
}
process_items "${fruits[@]}"

# mapfile/readarray — read lines from file/stdin into array
mapfile -t lines < /etc/hosts           # Read file into array (strip newlines with -t)
readarray -t words <<< "one two three"  # From string (splits on newlines, not spaces)
mapfile -t files < <(find . -name "*.sh" -type f)  # From command output
```

## Associative Arrays (Hash Maps)

Requires Bash 4.0+.

```bash
# Declare (REQUIRED for associative arrays)
declare -A user
declare -A config=([host]="localhost" [port]="5432" [name]="mydb")

# Assign values
user[name]="Alice"
user[email]="alice@example.com"
user[age]=30

# Read values
echo "${user[name]}"          # Alice
echo "${user[nonexistent]}"   # (empty string, no error)

# All keys
echo "${!user[@]}"            # name email age (order not guaranteed)

# All values
echo "${user[@]}"             # Alice alice@example.com 30

# Number of pairs
echo "${#user[@]}"            # 3

# Check if key exists
[[ -v user[email] ]] && echo "email key exists"

# Iterate over key-value pairs
for key in "${!user[@]}"; do
  echo "$key = ${user[$key]}"
done

# Remove a key
unset 'user[age]'

# Merge two associative arrays
declare -A merged
for k in "${!config[@]}"; do merged[$k]="${config[$k]}"; done
for k in "${!user[@]}";   do merged[$k]="${user[$k]}";   done
```

:::tip[Associative Arrays Need declare -A]
You MUST use `declare -A` before using an associative array. Without it, Bash treats `arr[key]` as an indexed array with key as an arithmetic expression — `arr[one]` becomes `arr[0]` since "one" evaluates to 0.
:::

## Parameter Expansion

### Default and Assignment Values

```bash
name=""
unset title

# ${var:-default} — use default if var is unset or empty
echo "${name:-Alice}"      # Alice (name is empty, use default)
echo "${title:-Untitled}"  # Untitled (title is unset, use default)
echo "${name}"             # (still empty — didn't assign)

# ${var:=default} — use AND ASSIGN default if var is unset or empty
echo "${name:=Alice}"      # Alice (assigned to name)
echo "${name}"             # Alice (now set!)

# ${var:+alternate} — use alternate if var IS set and non-empty
echo "${name:+Hello, $name}"  # Hello, Alice (name is set)
echo "${title:+set}"          # (empty — title is unset)

# ${var:?error_message} — error and exit if var is unset or empty
: "${REQUIRED_VAR:?REQUIRED_VAR must be set}"   # Exits with error if unset
: "${API_KEY:?Please set API_KEY in .env}"
```

### Substring Operations

```bash
str="Hello, World!"

# ${var:offset:length} — substring extraction
echo "${str:7}"        # World!   (from position 7 to end)
echo "${str:7:5}"      # World    (5 chars from position 7)
echo "${str: -6}"      # World!   (last 6 chars — space required before -)
echo "${str: -6:5}"    # World    (5 chars from position -6)

# String length
echo "${#str}"         # 13

# Array length
arr=(a b c d)
echo "${#arr[@]}"      # 4
echo "${#arr[0]}"      # 1 (length of first element)
```

### Pattern Removal

```bash
path="/home/user/docs/file.txt"

# ${var#pattern} — remove shortest prefix matching pattern
echo "${path#*/}"      # home/user/docs/file.txt

# ${var##pattern} — remove longest prefix matching pattern
echo "${path##*/}"     # file.txt  (basename equivalent)

# ${var%pattern} — remove shortest suffix matching pattern
echo "${path%.*}"      # /home/user/docs/file (remove extension)

# ${var%%pattern} — remove longest suffix matching pattern
echo "${path%%/*}"     # (empty — removes everything from first /)

# Practical examples
filename="${path##*/}"           # file.txt
basename="${filename%.*}"        # file
extension="${filename##*.}"      # txt
directory="${path%/*}"           # /home/user/docs
```

### Pattern Substitution

```bash
str="Hello Hello Hello"

# ${var/search/replace} — replace first match
echo "${str/Hello/Hi}"     # Hi Hello Hello

# ${var//search/replace} — replace ALL matches
echo "${str//Hello/Hi}"    # Hi Hi Hi

# ${var/#search/replace} — replace if at start (prefix)
echo "${str/#Hello/Hi}"    # Hi Hello Hello

# ${var/%search/replace} — replace if at end (suffix)
echo "${str/%Hello/Hi}"    # Hello Hello Hi

# Remove (replace with empty)
echo "${str// /}"          # HelloHelloHello (remove all spaces)

# With glob patterns
files="a.txt b.log c.txt d.log"
echo "${files//*.log/}"    # Remove all .log files (pattern matching)

# Change extension
for f in *.txt; do
  newname="${f%.txt}.md"   # Replace .txt with .md
  mv "$f" "$newname"
done
```

### Case Conversion (Bash 4.0+)

```bash
str="Hello World"

# ${var^} — uppercase first character
echo "${str^}"    # Hello World (already capitalized)

# ${var^^} — uppercase all characters
echo "${str^^}"   # HELLO WORLD

# ${var,} — lowercase first character
echo "${str,}"    # hello World

# ${var,,} — lowercase all characters
echo "${str,,}"   # hello world

# ${var~} — toggle case of first char (Bash 5.1+)
echo "${str~}"    # hELLO wORLD (Bash 5.1+ only)

# With pattern (apply only to matching chars)
echo "${str^^[aeiou]}"  # HEllO WOrld (uppercase vowels only)
```

### Transformation Operators (Bash 4.4+)

```bash
name="Alice O'Brien"

# ${var@Q} — shell-quoted form (safe for re-use in shell)
echo "${name@Q}"     # 'Alice O'\''Brien' (properly quoted)

# ${var@E} — expand escape sequences like $'...'
str=$'Hello\tWorld'
echo "${str@E}"      # Hello	World (tab expanded)

# ${var@P} — expand as PS1 prompt
echo "${PS1@P}"      # Current prompt string (resolved)

# ${var@A} — declare command to recreate variable
arr=(1 2 3)
echo "${arr[@]@A}"   # declare -a arr=([0]="1" [1]="2" [2]="3")

# ${var@a} — variable attributes (flags)
declare -r CONST="x"
echo "${CONST@a}"    # r (readonly)
```

## Here-Documents

```bash
# Basic heredoc — content is expanded
name="World"
cat << EOF
Hello, $name!
Today is $(date '+%Y-%m-%d')
EOF

# Quoted delimiter — NO expansion (literal content)
cat << 'EOF'
Hello, $name!    # Not expanded
$(date)          # Not expanded
EOF

# Indented heredoc (<<- strips leading TABS only, not spaces)
if true; then
  cat <<- EOF
	This line has a tab prefix that is stripped
	So does this line
	EOF
fi

# Heredoc to a variable
sql_query=$(cat << 'EOF'
SELECT u.id, u.name, u.email
FROM users u
WHERE u.active = 1
ORDER BY u.name;
EOF
)

# Heredoc with a command
mysql -u root -p mydb << 'SQL'
DROP TABLE IF EXISTS temp_table;
CREATE TABLE temp_table AS SELECT * FROM users WHERE created_at < '2020-01-01';
SQL
```

## Here-Strings

```bash
# <<< "string" — pass string as stdin
grep "pattern" <<< "This is a test string with pattern in it"

# Practical uses
read -r first_word <<< "hello world"   # first_word="hello"
IFS='.' read -ra parts <<< "1.2.3.4"  # parts=(1 2 3 4)

# Equivalent to echo "string" | command, but in current process (no subshell)
# Better for performance with built-ins
```

## IFS — Internal Field Separator

```bash
# IFS is used for word splitting and read
# Default IFS contains: space, tab, newline
echo "${IFS@Q}"   # Bash 4.4+: ' \t\n'

# Change IFS for splitting
data="Alice:30:Engineer"
IFS=':' read -ra fields <<< "$data"
echo "${fields[0]}"   # Alice
echo "${fields[1]}"   # 30
echo "${fields[2]}"   # Engineer

# Restore IFS after changing
old_ifs="$IFS"
IFS=','
# ... use comma splitting ...
IFS="$old_ifs"

# Or use a subshell to avoid pollution
(
  IFS=':'
  read -ra parts <<< "a:b:c"
  echo "${parts[@]}"   # a b c
)  # IFS restored when subshell exits

# IFS for joining arrays
arr=(a b c d)
IFS=','; joined="${arr[*]}"; IFS=$' \t\n'  # joined="a,b,c,d"
# Or with printf:
printf '%s,' "${arr[@]}" | sed 's/,$//'    # a,b,c,d (remove trailing comma)
# Or with a loop:
joined=$(IFS=','; echo "${arr[*]}")         # Clean one-liner

# Read loop with IFS
while IFS=':' read -r user _ uid gid _ home shell; do
  echo "User: $user, Home: $home, Shell: $shell"
done < /etc/passwd

# Disable word splitting (useful with set -u)
IFS=''  # No splitting — each character is its own field
```

## Nameref (Variable References)

```bash
# Nameref — declare -n creates a reference to another variable (Bash 4.3+)
declare -n ref=name     # ref is now an alias for the 'name' variable
name="Alice"
echo "$ref"             # Alice
ref="Bob"
echo "$name"            # Bob (modified through nameref)

# Use case: pass array by reference to a function
append_to_array() {
  declare -n _arr="$1"   # _arr is a reference to the array named by $1
  local item="$2"
  _arr+=("$item")
}

my_list=()
append_to_array my_list "apple"
append_to_array my_list "banana"
echo "${my_list[@]}"    # apple banana

# Return value via nameref
get_config() {
  declare -n _result="$1"
  _result="value from function"
}

get_config output
echo "$output"   # value from function
```

## Indirect Reference

```bash
# Old-style indirect reference (Bash 3+)
var_name="greeting"
greeting="Hello, World!"

echo "${!var_name}"    # Hello, World! (dereference through var_name)

# With arrays
declare -A config=([host]="localhost" [port]="5432")
key="host"
echo "${config[$key]}"  # localhost

# Dynamic variable names (use with care — nameref is better in Bash 4.3+)
for color in red green blue; do
  declare "color_${color}=255"
done
echo "${!color_red}"    # Won't work — need eval for old-style
eval "echo \$color_${color}"   # Works but eval is dangerous
# Better: use associative arrays instead
```

# Bash Variables

## Variable Types

```bash
# Scalar (regular) variable
name="Alice"

# Integer variable (arithmetic operations work without $(()))
declare -i count=10
count+=5     # count is now 15 (integer addition, not string concatenation)

# Readonly variable
declare -r MAX=100
readonly MIN=0

# Export variable (to environment)
declare -x HOME_DIR="$HOME"

# Array
declare -a fruits=("apple" "banana" "cherry")

# Associative array (bash 4+)
declare -A colors=([red]="#FF0000" [green]="#00FF00" [blue]="#0000FF")

# Nameref (reference to another variable)
declare -n ref=name   # ref now points to 'name'
ref="Bob"             # modifies 'name'
echo "$name"          # Bob
```

## Indexed Arrays

```bash
# Declaration
declare -a fruits
fruits=("apple" "banana" "cherry")

# Or simply:
animals=("cat" "dog" "bird")

# Accessing elements
echo "${animals[0]}"      # cat
echo "${animals[1]}"      # dog
echo "${animals[-1]}"     # bird (last element, bash 4.3+)

# All elements
echo "${animals[@]}"      # cat dog bird (separate words)
echo "${animals[*]}"      # cat dog bird (single word, IFS-joined)

# Number of elements
echo "${#animals[@]}"     # 3

# All indexes
echo "${!animals[@]}"     # 0 1 2

# Slicing: ${arr[@]:offset:length}
echo "${animals[@]:1:2}"  # dog bird (start at index 1, take 2)
echo "${animals[@]:1}"    # dog bird (from index 1 to end)

# Appending elements
animals+=("fish" "snake")
animals[10]="owl"          # sparse array — index 10, skips 4-9

# Removing an element (leaves gap in sparse array)
unset 'animals[1]'         # removes "dog"; array becomes sparse

# Iterating
for animal in "${animals[@]}"; do
  echo "Animal: $animal"
done

# Iterate with index
for i in "${!animals[@]}"; do
  echo "Index $i: ${animals[$i]}"
done

# Copy an array
copy=("${animals[@]}")

# Sort an array
IFS=$'\n' sorted=($(sort <<< "${animals[*]}")); unset IFS

# Array length tricks
files=(*.txt)
echo "${#files[@]}"  # number of .txt files
```

:::tip[Always quote array expansions]
Use `"${arr[@]}"` (with double quotes and `@`) when passing array elements as separate arguments. `${arr[*]}` joins elements with the first character of IFS, losing boundaries. `${arr[@]}` without quotes suffers word splitting.
:::

## Associative Arrays (bash 4+)

```bash
# Must declare with -A
declare -A user
user[name]="Alice"
user[email]="alice@example.com"
user[age]=30

# Or initialize at once
declare -A config=(
  [host]="localhost"
  [port]="5432"
  [database]="myapp"
  [user]="postgres"
)

# Access
echo "${config[host]}"     # localhost
echo "${config[port]}"     # 5432

# All keys
echo "${!config[@]}"       # host port database user

# All values
echo "${config[@]}"        # localhost 5432 myapp postgres

# Number of entries
echo "${#config[@]}"       # 4

# Check if key exists
if [[ -v config[host] ]]; then
  echo "host is set"
fi

# Iterate
for key in "${!config[@]}"; do
  printf "%-15s = %s\n" "$key" "${config[$key]}"
done

# Delete a key
unset 'config[port]'

# Build from command output
declare -A word_count
while read -r word; do
  word_count[$word]=$(( ${word_count[$word]:-0} + 1 ))
done < <(tr ' ' '\n' < text.txt)
```

## Parameter Expansion

Parameter expansion is Bash's powerful string manipulation mechanism.

### Default Values

```bash
name=""
host="localhost"

# ${var:-default} — use default if var is unset or empty
echo "${name:-Anonymous}"        # Anonymous (name is empty)
echo "${host:-127.0.0.1}"        # localhost (host is set)

# ${var:=default} — assign default if unset or empty (modifies var)
echo "${DATABASE_URL:=postgres://localhost/myapp}"
echo "$DATABASE_URL"   # now set to postgres://localhost/myapp

# ${var:+alternate} — use alternate if var IS set and non-empty
echo "${DEBUG:+--verbose}"   # prints --verbose only if DEBUG is set
echo "${file:+File: $file}"  # conditional message

# ${var:?error_msg} — exit with error if unset or empty
: "${API_KEY:?API_KEY environment variable is required}"
# Script exits: bash: API_KEY: API_KEY environment variable is required
```

### Substring Operations

```bash
str="Hello, World!"

# ${var:offset:length}
echo "${str:0:5}"     # Hello
echo "${str:7}"       # World!
echo "${str:7:5}"     # World
echo "${str: -6}"     # orld! (negative offset from end)
echo "${str: -6:5}"   # orld (5 chars from 6 before end)

# String length
echo "${#str}"        # 13
```

### Pattern Removal

```bash
filename="archive.tar.gz"
path="/home/user/documents/report.pdf"

# ${var#pattern} — remove shortest match from FRONT
echo "${filename#*.}"    # tar.gz

# ${var##pattern} — remove longest match from FRONT
echo "${filename##*.}"   # gz

# ${var%pattern} — remove shortest match from END
echo "${filename%.*}"    # archive.tar

# ${var%%pattern} — remove longest match from END
echo "${filename%%.*}"   # archive

# Practical examples
basename="${path##*/}"    # report.pdf (like basename command)
dirname="${path%/*}"      # /home/user/documents (like dirname command)
extension="${filename##*.}"  # gz
no_ext="${filename%.*}"     # archive.tar
```

### Search and Replace

```bash
str="Hello World World"

# ${var/search/replace} — replace first match
echo "${str/World/Bash}"     # Hello Bash World

# ${var//search/replace} — replace ALL matches
echo "${str//World/Bash}"    # Hello Bash Bash

# ${var/#pattern/replace} — replace if match at START
echo "${str/#Hello/Hi}"      # Hi World World

# ${var/%pattern/replace} — replace if match at END
echo "${str/%World/Bash}"    # Hello World Bash

# Delete (replace with empty)
echo "${str// /}"            # HelloWorldWorld
echo "${str/World/}"         # Hello  World (first occurrence removed)
```

### Case Modification (bash 4+)

```bash
str="Hello World"

echo "${str^}"     # Hello World (capitalize first char)
echo "${str^^}"    # HELLO WORLD (uppercase all)
echo "${str,}"     # hello World (lowercase first char)
echo "${str,,}"    # hello world (lowercase all)

# Case toggle
echo "${str~~}"    # hELLO wORLD (toggle case)

# Practical: normalize input
input="  ALICE  "
normalized="${input,,}"    # "  alice  "
trimmed="${normalized#"${normalized%%[! ]*}"}"  # left trim... (complex)
# Or use xargs for trimming:
trimmed=$(echo "$input" | xargs)
normalized="${trimmed,,}"  # alice
```

## Heredocs

```bash
# Basic heredoc
cat <<EOF
This is a heredoc.
Variables expand: $USER
Command substitution: $(date)
EOF

# Quoted heredoc (no expansion)
cat <<'EOF'
No expansion here: $USER
This is a literal dollar sign.
EOF

# Indented heredoc (<<-)
# strips leading TABS (not spaces) from the body
if true; then
  cat <<-EOF
	This line has a leading tab (stripped).
	$USER is still expanded.
	EOF
fi

# Heredoc to a variable
readme=$(cat <<EOF
# My Project
Version: ${VERSION:-1.0.0}
Author: ${AUTHOR:-Unknown}
EOF
)

# Heredoc as function input
while IFS=: read -r user _ uid gid name home shell; do
  printf "%-20s uid=%-5s home=%s\n" "$user" "$uid" "$home"
done <<EOF
root:x:0:0:root:/root:/bin/bash
alice:x:1000:1000:Alice:/home/alice:/bin/bash
EOF

# Heredoc for SQL
mysql -u root -p mydb <<EOF
SELECT id, name, email
FROM users
WHERE created_at > '2024-01-01'
ORDER BY name
LIMIT 10;
EOF
```

## Here-Strings

```bash
# <<< "string" — pass string as stdin
grep "pattern" <<< "text to search"
wc -w <<< "count these words"
read -r first rest <<< "hello world foo"
echo "$first"   # hello
echo "$rest"    # world foo

# Useful for passing multi-line strings
while IFS= read -r line; do
  echo "Line: $line"
done <<< "line one
line two
line three"
```

## IFS — Internal Field Separator

IFS controls how Bash splits strings into words and how `read` splits input into fields.

```bash
# Default IFS is space, tab, newline
echo "$IFS" | cat -A   # shows $ ^ I $ (space, tab, newline)

# Change IFS for a specific read
IFS=: read -r user pass uid gid gecos home shell <<< "root:x:0:0:root:/root:/bin/bash"
echo "$user $uid $home"   # root 0 /root

# Change IFS for a loop (then restore)
old_IFS="$IFS"
IFS=,
for field in "a,b,c,d"; do
  echo "Field: $field"
done
IFS="$old_IFS"

# Better pattern: use subshell or local IFS
(IFS=,; for f in $csv_line; do echo "$f"; done)

# IFS='' (empty) preserves exact line content in read
while IFS= read -r line; do    # preserves leading/trailing whitespace
  echo "[$line]"
done < file.txt

# Split string to array using IFS
IFS=',' read -ra parts <<< "one,two,three"
echo "${parts[0]}"   # one
echo "${parts[@]}"   # one two three
```

:::caution[Changing global IFS is dangerous]
Modifying IFS globally can break many shell features. Always restore it after use, or use a subshell, or use the `local IFS=...` pattern inside functions.
:::

## Indirect References

```bash
# nameref (bash 4.3+) — most readable
declare -n ptr=my_variable
ptr="new value"
echo "$my_variable"   # new value

# Use in functions to return arrays
fill_array() {
  declare -n result=$1  # result is a nameref to the variable named by $1
  result=("one" "two" "three")
}

fill_array my_array
echo "${my_array[@]}"   # one two three

# Old-style indirect reference: ${!varname}
varname="username"
username="alice"
echo "${!varname}"    # alice (value of variable whose name is in $varname)
```

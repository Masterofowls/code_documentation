# Bash Variables

## Variable Types

```bash
# Scalar (string by default)
name="Alice"
count=42

# Integer (declare -i)
declare -i num=42
declare -i result
result=num*2+1    # arithmetic context, no $(( )) needed

# Readonly
declare -r MAX=100
readonly MIN=0

# Export (environment variable)
declare -x PATH="/usr/local/bin:$PATH"
export EDITOR=vim

# Lowercase conversion (Bash 4+)
declare -l lowercase_var="HELLO"
echo "$lowercase_var"    # hello

# Uppercase conversion (Bash 4+)
declare -u uppercase_var="hello"
echo "$uppercase_var"    # HELLO

# Nameref — reference another variable by name (Bash 4.3+)
declare -n ref=name      # ref is a reference to variable 'name'
echo "$ref"              # same as echo "$name"
ref="Bob"                # modifies $name

# Print variable attributes
declare -p name           # declare -- name="Alice"
declare -p count          # declare -- count="42"
```

## Arrays (Indexed)

```bash
# Declaration and initialization
declare -a fruits               # declare empty array
fruits=("apple" "banana" "cherry" "date")
numbers=(1 2 3 4 5)

# Create from command output
lines=( $(ls *.txt) )           # ❌ word splits, glob expands, breaks on spaces
readarray -t lines < <(ls *.txt)  # ✅ correct way (mapfile/readarray)

# Access elements
echo "${fruits[0]}"             # apple (0-indexed)
echo "${fruits[1]}"             # banana
echo "${fruits[-1]}"            # date (last element, Bash 4.3+)
echo "${fruits[-2]}"            # cherry (second to last)

# All elements
echo "${fruits[@]}"             # apple banana cherry date (separate words)
echo "${fruits[*]}"             # apple banana cherry date (joined by IFS)
# ALWAYS use [@] over [*] to preserve element boundaries:
for item in "${fruits[@]}"; do
  echo "$item"
done

# Array length
echo "${#fruits[@]}"            # 4 (number of elements)
echo "${#fruits[0]}"            # 5 (length of first element)

# Indices
echo "${!fruits[@]}"            # 0 1 2 3 (all indices)

# Append to array
fruits+=("elderberry")
fruits+=("fig" "grape")         # append multiple

# Modify element
fruits[0]="avocado"

# Delete element (creates sparse array — index still exists but is empty)
unset 'fruits[1]'               # remove index 1
echo "${fruits[@]}"             # avocado cherry date elderberry (no banana)
echo "${!fruits[@]}"            # 0 2 3 4 (gap at 1)

# Re-index after deletion
fruits=("${fruits[@]}")         # re-create contiguous array

# Slicing: ${array[@]:offset:length}
echo "${fruits[@]:1:2}"         # elements 1 and 2
echo "${fruits[@]:2}"           # elements from index 2 onwards

# Copy an array
copy=("${fruits[@]}")

# Iterate with index
for i in "${!fruits[@]}"; do
  echo "Index $i: ${fruits[$i]}"
done
```

## Associative Arrays (Hash Maps)

Associative arrays require Bash 4.0+.

```bash
# Declare (MUST use declare -A for associative arrays)
declare -A user
user=([name]="Alice" [email]="alice@example.com" [role]="admin")

# Create incrementally
declare -A config
config[database]="postgres://localhost/myapp"
config[redis]="redis://localhost:6379"
config[max_workers]=8

# Access
echo "${config[database]}"

# Check if key exists
if [[ -v config[database] ]]; then
  echo "database key exists"
fi

# All keys
echo "${!config[@]}"            # database redis max_workers

# All values
echo "${config[@]}"

# Number of entries
echo "${#config[@]}"            # 3

# Delete a key
unset 'config[redis]'

# Iterate over keys and values
for key in "${!config[@]}"; do
  echo "$key = ${config[$key]}"
done

# Build from output
declare -A colors
while IFS='=' read -r name hex; do
  colors["$name"]="$hex"
done < colors.txt               # file: red=#FF0000
```

## Parameter Expansion

Parameter expansion is Bash's powerful string manipulation mechanism.

### Default Values

```bash
name=""
dir=""

# ${var:-default}  — use default if var is unset or empty
echo "${name:-anonymous}"       # anonymous (name is empty)
echo "${missing:-fallback}"     # fallback (missing is unset)
echo "$name"                    # still empty (doesn't modify)

# ${var:=default}  — assign default if var is unset or empty
echo "${dir:=/tmp}"             # /tmp (assigns to dir)
echo "$dir"                     # /tmp (now set!)

# ${var:+alternate}  — use alternate if var IS set (opposite of :-)
echo "${name:+User: $name}"     # empty (name is empty)
name="Alice"
echo "${name:+User: $name}"     # User: Alice

# ${var:?error_msg}  — error and exit if var is unset or empty
: "${DATABASE_URL:?DATABASE_URL environment variable is required}"
# ↑ exits with error if DATABASE_URL is unset

# Without colon: only tests for unset (not empty)
# ${var-default}  ${var=default}  ${var+alt}  ${var?err}
```

### String Length and Substring

```bash
str="Hello, World!"

# Length
echo "${#str}"                  # 13

# Substring: ${var:offset:length}
echo "${str:7}"                 # World!
echo "${str:7:5}"               # World
echo "${str: -6}"               # orld! (space before - avoids ambiguity)
echo "${str: -6:5}"             # orld!
```

### Prefix/Suffix Removal

```bash
path="/usr/local/bin/python3.11"
filename="report_2024_01.csv"

# Remove shortest prefix: ${var#pattern}
echo "${path#/usr/}"            # local/bin/python3.11
echo "${filename#*_}"           # 2024_01.csv  (removes up to first _)

# Remove longest prefix: ${var##pattern}
echo "${path##*/}"              # python3.11  (basename)
echo "${filename##*_}"          # 01.csv  (removes up to last _)

# Remove shortest suffix: ${var%pattern}
echo "${path%/*}"               # /usr/local/bin  (dirname)
echo "${filename%.csv}"         # report_2024_01  (remove extension)
echo "${filename%_*}"           # report_2024  (removes from last _)

# Remove longest suffix: ${var%%pattern}
echo "${filename%%_*}"          # report  (removes from first _)
```

### Search and Replace

```bash
str="banana"

# Replace first occurrence: ${var/search/replace}
echo "${str/a/o}"               # bonana

# Replace all occurrences: ${var//search/replace}
echo "${str//a/o}"              # bonono

# Replace prefix: ${var/#search/replace}
echo "${str/#ban/van}"          # vanilla → vanana

# Replace suffix: ${var/%search/replace}
echo "${str/%ana/one}"          # banone

# Delete (replace with nothing)
echo "${str//a/}"               # bnn
```

### Case Modification (Bash 4+)

```bash
text="hello WORLD"

echo "${text^}"                 # Hello WORLD (capitalize first char)
echo "${text^^}"                # HELLO WORLD (uppercase all)
echo "${text,}"                 # hello WORLD (lowercase first char)
echo "${text,,}"                # hello world (lowercase all)

# With pattern (convert matching characters)
echo "${text^^[aeiou]}"         # hEllO wOrld (uppercase vowels only)
```

## Here-Strings and Here-Docs

```bash
# Here-string: pass single string as stdin
grep "error" <<< "this has an error in it"
base64 --decode <<< "SGVsbG8="

# Read from variable
read -r line <<< "single line string"

# Here-doc: pass multi-line string as stdin
cat <<EOF
Line 1
Line 2
Line $USER
EOF
# ↑ Expands variables and command substitutions

# Quoted here-doc: no expansion (single-quoted delimiter)
cat <<'EOF'
Literal $HOME
No $(expansion) here
EOF

# Indented here-doc (strip leading tabs, NOT spaces): <<-
cat <<-EOF
	Indented with tabs
	This is cleaner
	EOF
# ↑ Leading tabs are removed from body and closing delimiter

# Assign to variable
readme=$(cat <<EOF
# My Project
Version: $VERSION
Built: $(date)
EOF
)
```

## IFS — Internal Field Separator

`IFS` controls how bash splits words in unquoted expansions and `read` commands.

```bash
# Default IFS: space + tab + newline
echo "$IFS" | cat -A    # shows the invisible characters

# Read CSV line
IFS=',' read -r name age city <<< "Alice,30,New York"
echo "Name: $name, Age: $age, City: $city"

# Split string into array
IFS=':' read -ra parts <<< "$PATH"
for dir in "${parts[@]}"; do
  echo "$dir"
done

# Read a file line by line (safe with spaces in names)
while IFS= read -r line; do
  echo "Line: $line"
done < /etc/hosts

# Read tab-separated data
while IFS=$'\t' read -r col1 col2 col3; do
  echo "$col1 | $col2 | $col3"
done < data.tsv

# Temporarily change IFS
(
  IFS=','
  for word in $csv_string; do
    echo "$word"
  done
)  # IFS restored after subshell

# Change and restore manually
old_ifs="$IFS"
IFS=','
# ... processing ...
IFS="$old_ifs"
```

:::caution[IFS and set -u]
With `set -u`, using `IFS=` (empty IFS) to prevent word splitting is safe. But beware: changing `IFS` globally affects all word splitting until restored. Prefer subshells or local scope via functions.
:::

## Nameref (declare -n)

```bash
# Nameref allows a variable to act as a reference to another variable (Bash 4.3+)
declare -n ref=myvar
myvar="hello"
echo "$ref"         # hello
ref="world"
echo "$myvar"       # world (modified through nameref)

# Useful in functions to return data
function fill_array() {
  local -n arr_ref=$1   # nameref to the caller's array
  arr_ref=("one" "two" "three")
}

my_array=()
fill_array my_array
echo "${my_array[@]}"   # one two three

# Nameref with associative arrays
function configure() {
  local -n config_ref=$1
  config_ref[host]="localhost"
  config_ref[port]="5432"
}

declare -A db_config
configure db_config
echo "${db_config[host]}"   # localhost
```

## Indirect Reference

```bash
# ${!varname} — indirect reference: value of a variable whose name is in varname
color="red"
red="#FF0000"
green="#00FF00"

echo "${!color}"    # #FF0000 (value of variable named by $color)

# Useful for dynamic variable names
for env in dev staging prod; do
  varname="DB_HOST_${env^^}"    # DB_HOST_DEV, DB_HOST_STAGING, etc.
  echo "$env: ${!varname}"      # indirect reference
done

# List variables matching a prefix
echo "${!BASH*}"    # all variables starting with BASH
# BASH BASHOPTS BASHPID BASH_ALIASES BASH_ARGC ...
```

## mapfile / readarray

```bash
# Read lines of a file into an array (Bash 4.0+)
mapfile -t lines < /etc/hosts      # -t strips newlines (recommended)
readarray -t lines < /etc/hosts    # synonym for mapfile

echo "${#lines[@]}"  # number of lines
echo "${lines[0]}"   # first line

# Read from command output
mapfile -t processes < <(ps aux | tail -n +2)

# Read limited number of lines
mapfile -t -n 10 first_ten < bigfile.txt

# With callback function (called for each line)
mapfile -t -C mycallback -c 1 < file.txt  # call mycallback every 1 line

# Read starting at a specific index
mapfile -t -O 5 arr < file.txt   # start filling from index 5

# Practical: read CSV into array
mapfile -t csv_lines < data.csv
for line in "${csv_lines[@]}"; do
  IFS=',' read -ra fields <<< "$line"
  echo "Name: ${fields[0]}, Age: ${fields[1]}"
done
```

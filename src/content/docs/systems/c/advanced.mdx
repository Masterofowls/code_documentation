# Advanced C Programming

This guide covers the C features that separate beginner from expert: the preprocessor, linker behavior, signal handling, multi-file project architecture, and C11's threading and atomic primitives.

## The C Preprocessor

The preprocessor transforms source text before compilation. It knows nothing about C syntax — it operates purely on text.

### Include Directives

```c
// Angle brackets: search system/compiler include paths
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>

// Quotes: search current directory first, then system paths
#include "mylib.h"
#include "../../common/utils.h"

// You can compute include paths (rarely done, but possible)
#define HEADER "mylib.h"
#include HEADER
```

### Object-like and Function-like Macros

```c
// Object-like macro (constant replacement)
#define PI        3.14159265358979323846
#define MAX_SIZE  1024
#define TRUE      1
#define FALSE     0

// Prefer typed constants when possible (const variables or enum):
static const double kPi = 3.14159265358979323846;

// Function-like macros — arguments are text-substituted
// Wrap EVERYTHING in parentheses to avoid precedence issues!
#define MIN(a, b)    (((a) < (b)) ? (a) : (b))
#define MAX(a, b)    (((a) > (b)) ? (a) : (b))
#define ABS(x)       (((x) < 0) ? -(x) : (x))
#define CLAMP(v,lo,hi) (MIN(MAX((v),(lo)),(hi)))

// Array size (common idiom)
#define ARRAY_SIZE(arr) (sizeof(arr) / sizeof((arr)[0]))

// Stringify (#) — converts argument to string literal
#define STRINGIFY(x)  #x
#define XSTRINGIFY(x) STRINGIFY(x)  // Extra level for macro expansion

// Token concatenation (##) — joins two tokens into one
#define MAKE_VAR(prefix, num)  prefix##num
int MAKE_VAR(count, 1) = 0;  // Expands to: int count1 = 0;
int MAKE_VAR(count, 2) = 0;  // int count2 = 0;

// Variadic macros with __VA_ARGS__
#define LOG(fmt, ...)  fprintf(stderr, "[LOG] " fmt "\n", ##__VA_ARGS__)
#define ASSERT(cond)   \
    do { if (!(cond)) { \
        fprintf(stderr, "Assert failed: %s at %s:%d\n", \
                #cond, __FILE__, __LINE__); \
        abort(); \
    } } while(0)

// Multi-statement macro — ALWAYS wrap in do { } while(0)
#define SWAP(a, b, type) \
    do { type _tmp = (a); (a) = (b); (b) = _tmp; } while(0)

// Usage:
LOG("Value = %d", 42);
ASSERT(1 + 1 == 2);
int x = 5, y = 10;
SWAP(x, y, int);
```

### Conditional Compilation

```c
// Include guards — prevent double-inclusion
// In mylib.h:
#ifndef MYLIB_H
#define MYLIB_H

// ... header content ...

#endif // MYLIB_H

// pragma once (non-standard but widely supported, cleaner)
#pragma once
// ... header content ...

// Compile-time feature detection
#ifdef DEBUG
    #define DBG_PRINT(fmt, ...) printf("[DEBUG] " fmt "\n", ##__VA_ARGS__)
#else
    #define DBG_PRINT(fmt, ...) do { } while(0)  // Optimized away
#endif

// Platform detection
#if defined(_WIN32) || defined(_WIN64)
    #define PLATFORM_WINDOWS
    #include <windows.h>
#elif defined(__APPLE__)
    #define PLATFORM_MACOS
    #include <TargetConditionals.h>
#elif defined(__linux__)
    #define PLATFORM_LINUX
#endif

// Compiler detection
#if defined(__GNUC__) && !defined(__clang__)
    #define COMPILER_GCC
    #define LIKELY(x)   __builtin_expect(!!(x), 1)
    #define UNLIKELY(x) __builtin_expect(!!(x), 0)
    #define FORCE_INLINE __attribute__((always_inline)) inline
    #define NO_INLINE    __attribute__((noinline))
#elif defined(__clang__)
    #define COMPILER_CLANG
    #define LIKELY(x)   __builtin_expect(!!(x), 1)
    #define UNLIKELY(x) __builtin_expect(!!(x), 0)
    #define FORCE_INLINE __attribute__((always_inline)) inline
    #define NO_INLINE    __attribute__((noinline))
#elif defined(_MSC_VER)
    #define COMPILER_MSVC
    #define LIKELY(x)   (x)
    #define UNLIKELY(x) (x)
    #define FORCE_INLINE __forceinline
    #define NO_INLINE    __declspec(noinline)
#endif

// Feature testing (C11+)
#if defined(__STDC_VERSION__) && __STDC_VERSION__ >= 201112L
    #define HAS_C11
#endif

// #if with expressions
#define VERSION_MAJOR 2
#define VERSION_MINOR 4
#if VERSION_MAJOR > 1 && defined(HAS_C11)
    // Features for version 2+ with C11
#endif

// Emit a compile-time error
#if CHAR_BIT != 8
    #error "This code requires 8-bit bytes"
#endif

// Emit a compile-time warning (GCC/Clang)
#if defined(ENABLE_DEPRECATED_API)
    #warning "Deprecated API enabled — remove before release"
#endif

// __LINE__, __FILE__, __DATE__, __TIME__, __func__ (C99+)
#define WHERE  __FILE__ ":" XSTRINGIFY(__LINE__)
printf("Error at " WHERE "\n");
```

## Compilation and Linking

### Linkage

```c
// External linkage: accessible from other translation units
int global_var = 42;          // External linkage
void global_function(void) {}  // External linkage

// Internal linkage: visible only within this translation unit
static int private_var = 0;          // Internal linkage (file-private)
static void private_helper(void) {}  // Internal linkage

// No linkage: local variables (only visible in their block)
void foo(void) {
    int local = 5;  // No linkage
    static int persistent = 0;  // Static local: no linkage but persists
}

// extern: declare without defining (use definition from another TU)
extern int counter;   // Defined in counter.c
extern void init(void);  // Defined in init.c
```

### Static vs Dynamic Libraries

  <TabItem label="Static Library (.a)">
```bash
# Create object files
gcc -c -O2 math_utils.c -o math_utils.o
gcc -c -O2 string_utils.c -o string_utils.o

# Create static library (archive)

ar rcs libutils.a math_utils.o string_utils.o

# List library contents

ar t libutils.a

# Link against static library (code is copied into executable)

gcc main.c -L. -lutils -o myapp

# Or directly

gcc main.c libutils.a -o myapp

````

  <TabItem label="Shared Library (.so/.dll)">
```bash
# Create position-independent object files
gcc -c -fPIC -O2 math_utils.c -o math_utils.o

# Create shared library
gcc -shared -o libutils.so math_utils.o

# On macOS (.dylib)
gcc -dynamiclib -o libutils.dylib math_utils.o

# Link against shared library
gcc main.c -L. -lutils -o myapp

# Set rpath so loader finds the library
gcc main.c -L. -lutils -Wl,-rpath,. -o myapp

export LD_LIBRARY_PATH=.:$LD_LIBRARY_PATH
./myapp
````

## Signal Handling

```c
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

// volatile sig_atomic_t: only type guaranteed safe in signal handlers
static volatile sig_atomic_t g_running = 1;
static volatile sig_atomic_t g_reload  = 0;

// Signal handler: VERY restricted — only async-signal-safe functions
// Allowed: write() (not printf), signal(), abort(), _exit()
// NOT allowed: malloc, free, printf, fprintf, most library functions
void handle_sigint(int sig) {
    (void)sig;
    g_running = 0;
    // write() is async-signal-safe; printf is NOT
    const char msg[] = "\nReceived SIGINT — shutting down\n";
    write(STDOUT_FILENO, msg, sizeof(msg) - 1);
}

void handle_sighup(int sig) {
    (void)sig;
    g_reload = 1;  // Main loop will handle the actual reload
}

// sigaction is preferred over signal() — more portable and predictable
void setup_signals(void) {
    struct sigaction sa;
    memset(&sa, 0, sizeof(sa));

    // SIGINT (Ctrl+C)
    sa.sa_handler = handle_sigint;
    sigemptyset(&sa.sa_mask);  // Don't block other signals during handler
    sa.sa_flags = SA_RESTART;  // Restart interrupted system calls
    if (sigaction(SIGINT, &sa, NULL) == -1) { perror("sigaction"); exit(1); }

    // SIGHUP (terminal hangup / reload)
    sa.sa_handler = handle_sighup;
    if (sigaction(SIGHUP, &sa, NULL) == -1) { perror("sigaction"); exit(1); }

    // SIGTERM (graceful shutdown)
    sa.sa_handler = handle_sigint;  // Reuse same handler
    if (sigaction(SIGTERM, &sa, NULL) == -1) { perror("sigaction"); exit(1); }

    // Ignore SIGPIPE (common for network servers — broken pipe)
    sa.sa_handler = SIG_IGN;
    if (sigaction(SIGPIPE, &sa, NULL) == -1) { perror("sigaction"); exit(1); }
}

int main(void) {
    setup_signals();
    printf("Running (PID %d). Press Ctrl+C to stop.\n", getpid());

    while (g_running) {
        if (g_reload) {
            printf("Reloading configuration...\n");
            g_reload = 0;
        }
        sleep(1);
        printf("Tick\n");
    }

    printf("Graceful shutdown complete.\n");
    return 0;
}
```

## setjmp / longjmp

`setjmp`/`longjmp` implements non-local jumps — jumping out of nested function calls without stack unwinding. Useful for error recovery, but dangerous.

```c
#include <setjmp.h>
#include <stdio.h>
#include <stdlib.h>

// Error codes
typedef enum { ERR_NONE = 0, ERR_OOM = 1, ERR_IO = 2, ERR_LOGIC = 3 } Error;

// Error recovery context
static jmp_buf error_context;

// Functions that can "throw" errors
void allocate_resources(int n) {
    void *p = malloc((size_t)n * 1024 * 1024);
    if (!p) {
        longjmp(error_context, ERR_OOM);  // Jump back to setjmp point
    }
    printf("Allocated %d MB\n", n);
    free(p);
}

void complex_operation(void) {
    allocate_resources(10);
    // ... more work ...
    int trigger = 1;
    if (trigger) longjmp(error_context, ERR_LOGIC);  // Simulate error
    allocate_resources(100);  // Never reached
}

int main(void) {
    int err;

    // setjmp: saves execution context; returns 0 on first call
    // Returns non-zero when jumped to via longjmp
    if ((err = setjmp(error_context)) == 0) {
        // Normal execution path
        printf("Starting operation...\n");
        complex_operation();
        printf("Operation complete!\n");  // Not reached
    } else {
        // Error recovery path
        switch (err) {
            case ERR_OOM:   printf("Error: out of memory\n"); break;
            case ERR_IO:    printf("Error: I/O failure\n"); break;
            case ERR_LOGIC: printf("Error: logic error\n"); break;
            default:        printf("Error: unknown (%d)\n", err); break;
        }
    }

    return err != ERR_NONE ? 1 : 0;
}
```

:::caution[setjmp/longjmp Dangers]
`longjmp` skips destructors, does NOT free heap memory allocated since `setjmp`, and can corrupt the stack if the function containing `setjmp` has already returned. Variables that changed between `setjmp` and `longjmp` must be declared `volatile`. Use only as a last resort; prefer error return codes.
:::

## C11 Threads

```c
#include <threads.h>
#include <stdio.h>
#include <stdlib.h>

// Mutex for protecting shared state
mtx_t g_mutex;
int   g_counter = 0;

// Thread function signature: int fn(void*)
int worker_thread(void *arg) {
    int id = *(int*)arg;
    free(arg);

    for (int i = 0; i < 1000; i++) {
        mtx_lock(&g_mutex);
        g_counter++;
        mtx_unlock(&g_mutex);
    }

    printf("Worker %d finished\n", id);
    return 0;  // Return value retrievable via thrd_join
}

int main(void) {
    // Initialize mutex (mtx_plain, mtx_recursive, mtx_timed)
    if (mtx_init(&g_mutex, mtx_plain) != thrd_success) {
        fprintf(stderr, "mtx_init failed\n");
        return 1;
    }

    const int NUM_THREADS = 4;
    thrd_t threads[NUM_THREADS];

    // Create threads
    for (int i = 0; i < NUM_THREADS; i++) {
        int *id = malloc(sizeof(int));
        *id = i;
        if (thrd_create(&threads[i], worker_thread, id) != thrd_success) {
            fprintf(stderr, "thrd_create failed\n");
        }
    }

    // Wait for all threads
    for (int i = 0; i < NUM_THREADS; i++) {
        int ret;
        thrd_join(threads[i], &ret);
        printf("Thread %d returned %d\n", i, ret);
    }

    printf("Final counter: %d (expected %d)\n", g_counter, NUM_THREADS * 1000);

    mtx_destroy(&g_mutex);
    return 0;
}
```

## C11 Atomics

```c
#include <stdatomic.h>
#include <threads.h>
#include <stdio.h>

// Atomic types — operations on these are thread-safe without locks
// (for basic types; larger structures still need mutexes)
static atomic_int  g_ref_count  = ATOMIC_VAR_INIT(0);
static atomic_bool g_stop_flag  = ATOMIC_VAR_INIT(false);
static atomic_uint g_seq_number = ATOMIC_VAR_INIT(0);

void acquire(void) {
    // atomic_fetch_add: atomically adds and returns old value
    // memory_order_relaxed: no ordering guarantees beyond atomicity
    atomic_fetch_add_explicit(&g_ref_count, 1, memory_order_relaxed);
}

void release(void) {
    // Release ordering: ensures all prior writes are visible before decrement
    if (atomic_fetch_sub_explicit(&g_ref_count, 1, memory_order_release) == 1) {
        // Acquire ordering: ensure we see all writes from the releasing thread
        atomic_thread_fence(memory_order_acquire);
        printf("Last reference released — free the resource\n");
    }
}

// Lock-free stack using CAS (compare-and-swap)
typedef struct StackNode { int val; struct StackNode *next; } StackNode;
typedef struct { _Atomic(StackNode*) top; } LockFreeStack;

void stack_push(LockFreeStack *s, int val) {
    StackNode *node = malloc(sizeof(StackNode));
    node->val  = val;
    StackNode *top;
    do {
        top        = atomic_load(&s->top);
        node->next = top;
    } while (!atomic_compare_exchange_weak(&s->top, &top, node));
    // compare_exchange_weak: may fail spuriously (loop handles it)
    // compare_exchange_strong: never fails spuriously (used for if-else)
}

int main(void) {
    acquire(); acquire(); acquire();
    printf("Ref count: %d\n", atomic_load(&g_ref_count));  // 3

    release(); release(); release();  // Last release prints message

    LockFreeStack stack = { .top = NULL };
    stack_push(&stack, 1);
    stack_push(&stack, 2);
    stack_push(&stack, 3);

    StackNode *n = atomic_load(&stack.top);
    while (n) { printf("%d ", n->val); n = n->next; }
    printf("\n");  // 3 2 1

    // Cleanup (not thread-safe here — single-threaded cleanup)
    n = atomic_load(&stack.top);
    while (n) { StackNode *tmp = n->next; free(n); n = tmp; }

    return 0;
}
```

## inline Functions and restrict

```c
// inline: hint to compiler to expand function at call site (avoids call overhead)
// In C99+, inline in a header requires an external definition in exactly one .c file
static inline int max_int(int a, int b) { return a > b ? a : b; }
static inline float lerp(float a, float b, float t) { return a + (b - a) * t; }

// restrict: promise pointer arguments don't alias each other
// Enables aggressive auto-vectorization
void matrix_multiply(
    float * restrict C,
    const float * restrict A,
    const float * restrict B,
    int n)
{
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            float sum = 0.0f;
            for (int k = 0; k < n; k++) {
                sum += A[i*n + k] * B[k*n + j];
            }
            C[i*n + j] = sum;
        }
    }
}
```

## Designated Initializers and Compound Literals (C99+)

```c
#include <stdio.h>

typedef struct { int x, y, w, h; } Rect;
typedef struct { float r, g, b, a; } Color;

// Designated initializer: initialize specific fields by name
Rect r = { .x = 10, .y = 20, .w = 100, .h = 50 };

// Compound literal: creates a temporary object
// Useful for passing struct to a function without a named variable
void draw_rect(Rect rect, Color color);

void example(void) {
    // Compound literal — anonymous temporary struct
    draw_rect(
        (Rect){ .x = 0, .y = 0, .w = 640, .h = 480 },
        (Color){ .r = 1.0f, .g = 0.0f, .b = 0.0f, .a = 1.0f }
    );

    // Compound literal for arrays
    int *arr = (int[]){ 1, 2, 3, 4, 5 };
    for (int i = 0; i < 5; i++) printf("%d ", arr[i]);
    printf("\n");
}
```

:::tip[Multi-File Project Layout]
A well-organized C project:

```
project/
├── include/         ← public headers (only what external code needs)
│   └── mylib.h
├── src/             ← implementation files
│   ├── core.c
│   ├── utils.c
│   └── network.c
├── tests/           ← test programs
├── examples/        ← example programs
└── CMakeLists.txt
```

Keep implementation details out of headers. Use `static` for file-private functions. Use `extern` declarations in headers for functions and global variables that need to cross translation unit boundaries.
:::

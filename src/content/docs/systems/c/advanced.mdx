# Advanced C Programming

This guide covers the advanced facilities of C: the preprocessor, linker internals, signal handling, non-local jumps, thread primitives, and modern C11/C23 features.

---

## The C Preprocessor

The preprocessor runs before compilation and handles all `#` directives. It performs purely textual transformations.

### Include Directives

```c
#include <stdio.h>      // Search system include paths (angle brackets)
#include "myheader.h"   // Search local directory first, then system paths
#include "../utils.h"   // Relative paths are allowed

// Include guards — prevent double inclusion
#ifndef MYLIB_H
#define MYLIB_H

// ... header content ...

#endif // MYLIB_H

// #pragma once — non-standard but universally supported alternative
#pragma once
// ... header content ...
```

### Macro Definitions

```c
// Object-like macros (constants)
#define PI         3.14159265358979f
#define MAX_BUFFER 4096
#define VERSION    "1.2.3"

// Function-like macros — always parenthesize arguments and result
#define MAX(a, b)         ((a) > (b) ? (a) : (b))
#define MIN(a, b)         ((a) < (b) ? (a) : (b))
#define CLAMP(v, lo, hi)  ((v) < (lo) ? (lo) : ((v) > (hi) ? (hi) : (v)))
#define ARRAY_SIZE(arr)   (sizeof(arr) / sizeof((arr)[0]))
#define BIT(n)            (1u << (n))

// Multi-line macros use backslash continuation
#define SWAP(T, a, b) do { \
    T _tmp = (a);          \
    (a) = (b);             \
    (b) = _tmp;            \
} while (0)
// The do { ... } while(0) idiom makes the macro safe to use in if/else without {}
```

### Stringification and Token Pasting

```c
// # operator — stringify a macro argument
#define STRINGIFY(x)  #x
#define TO_STR(x)     STRINGIFY(x)

printf("%s\n", STRINGIFY(hello world));  // "hello world"
printf("Line: %s\n", TO_STR(__LINE__)); // "42" (as string)

// ## operator — token concatenation
#define CONCAT(a, b) a##b
#define MAKE_NAME(prefix, num) prefix##num

int MAKE_NAME(var, 1) = 10;   // expands to: int var1 = 10;
int MAKE_NAME(var, 2) = 20;   // expands to: int var2 = 20;

// Generating unique names with __LINE__
#define UNIQUE_NAME(base) CONCAT(base, __LINE__)
```

### Variadic Macros

```c
#include <stdio.h>

// __VA_ARGS__ captures all extra arguments
#define LOG(fmt, ...)   fprintf(stderr, "[LOG] " fmt "\n", ##__VA_ARGS__)
#define DIE(fmt, ...)   do { fprintf(stderr, "FATAL: " fmt "\n", ##__VA_ARGS__); exit(1); } while(0)

// Debug macro that includes file/line info
#define DEBUG(fmt, ...) fprintf(stderr, "[%s:%d] " fmt "\n", __FILE__, __LINE__, ##__VA_ARGS__)

// Usage
LOG("Server started on port %d", 8080);
DEBUG("Value of x = %d", x);
```

### Conditional Compilation

```c
// Platform detection
#ifdef _WIN32
    #include <windows.h>
    #define PATH_SEP '\\'
#elif defined(__APPLE__)
    #include <TargetConditionals.h>
    #define PATH_SEP '/'
#else
    #define PATH_SEP '/'
#endif

// Feature flags
#ifdef ENABLE_LOGGING
    #define LOG(msg)  printf("[LOG] %s\n", (msg))
#else
    #define LOG(msg)  ((void)0)   // compiles to nothing
#endif

// Version checks
#if __STDC_VERSION__ >= 201112L
    // C11 or later
    #define STATIC_ASSERT(cond, msg) _Static_assert(cond, msg)
#else
    // Fallback
    #define STATIC_ASSERT(cond, msg) typedef char _sa_##__LINE__[(cond) ? 1 : -1]
#endif

// defined() — test if a macro is defined (works in #if)
#if defined(DEBUG) && !defined(NDEBUG)
    // debug build
#endif
```

### Predefined Macros

```c
printf("File:     %s\n", __FILE__);     // current source filename
printf("Line:     %d\n", __LINE__);     // current line number
printf("Function: %s\n", __func__);     // current function name (C99)

printf("GCC version: %d.%d.%d\n", __GNUC__, __GNUC_MINOR__, __GNUC_PATCHLEVEL__);
#endif
```

## Compilation and Linking

### Linkage

```c
// External linkage — symbol visible across translation units
int   global_var = 42;          // external linkage (default for globals)
void  global_func(void) { }     // external linkage

// Internal linkage — visible only in this translation unit (.c file)
static int   module_var = 10;   // internal linkage
static void  helper(void) { }   // internal linkage (preferred for private functions)

// No linkage — local scope only
void foo(void) {
    int local = 5;              // no linkage
    static int count = 0;       // no linkage (static only affects storage duration here)
}

// extern declaration — declare without defining
extern int global_var;          // defined in another .c file
extern void other_func(void);
```

### Header and Source File Pattern

```c
// --- vector.h ---
#pragma once
#include <stddef.h>

typedef struct { float x, y, z; } Vec3;

// Declare public API
Vec3  vec3_add(Vec3 a, Vec3 b);
float vec3_dot(Vec3 a, Vec3 b);
Vec3  vec3_normalize(Vec3 v);

// Inline functions defined in header (C99)
static inline float vec3_length_sq(Vec3 v) {
    return v.x*v.x + v.y*v.y + v.z*v.z;
}
```

```c
// --- vector.c ---
#include "vector.h"
#include <math.h>

Vec3 vec3_add(Vec3 a, Vec3 b) {
    return (Vec3){ a.x+b.x, a.y+b.y, a.z+b.z };
}
float vec3_dot(Vec3 a, Vec3 b) {
    return a.x*b.x + a.y*b.y + a.z*b.z;
}
Vec3 vec3_normalize(Vec3 v) {
    float len = sqrtf(vec3_length_sq(v));
    if (len < 1e-6f) return (Vec3){0,0,0};
    return (Vec3){ v.x/len, v.y/len, v.z/len };
}
```

### Static and Dynamic Libraries

```bash
# Build a static library (.a)
gcc -c -O2 vector.c -o vector.o
ar rcs libvector.a vector.o
gcc main.c -L. -lvector -o program   # link against static lib

# Build a shared library (.so on Linux)
gcc -fPIC -c vector.c -o vector.o    # -fPIC: position-independent code
gcc -shared vector.o -o libvector.so
gcc main.c -L. -lvector -Wl,-rpath,. -o program

# On macOS: .dylib instead of .so
# On Windows (MinGW): .dll with import lib .a
```

## Signal Handling

```c
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Signal handler — must be async-signal-safe (very limited operations allowed)
volatile sig_atomic_t g_running = 1;    // volatile sig_atomic_t for signal-safe flag
volatile sig_atomic_t g_got_sigusr1 = 0;

// sigaction handler — preferred over signal()
static void handle_sigint(int sig) {
    (void)sig;
    g_running = 0;    // set flag — check in main loop
    // DO NOT: printf, malloc, free, or most library functions here
}

static void handle_sigusr1(int sig) {
    (void)sig;
    g_got_sigusr1 = 1;
}

int main(void) {
    // sigaction — more portable and reliable than signal()
    struct sigaction sa;
    memset(&sa, 0, sizeof(sa));
    sa.sa_handler = handle_sigint;
    sigemptyset(&sa.sa_mask);             // no signals blocked during handler
    sa.sa_flags = SA_RESTART;             // restart interrupted system calls

    if (sigaction(SIGINT, &sa, NULL) < 0) {
        perror("sigaction");
        return 1;
    }

    sa.sa_handler = handle_sigusr1;
    sigaction(SIGUSR1, &sa, NULL);

    // Ignore SIGPIPE (common for network servers)
    signal(SIGPIPE, SIG_IGN);

    printf("Running. Press Ctrl+C to stop.\n");
    while (g_running) {
        if (g_got_sigusr1) {
            printf("Got SIGUSR1\n");
            g_got_sigusr1 = 0;
        }
        // do work...
        pause();   // sleep until any signal
    }

    printf("Shutting down cleanly\n");
    return 0;
}
```

## setjmp and longjmp — Non-Local Jumps

```c
#include <setjmp.h>
#include <stdio.h>

jmp_buf error_handler;

void risky_operation(int depth) {
    if (depth > 3) {
        longjmp(error_handler, 1);   // jump back to setjmp site, return value 1
    }
    printf("Depth: %d\n", depth);
    risky_operation(depth + 1);
}

int main(void) {
    int error = setjmp(error_handler);
    // setjmp returns 0 on initial call
    // returns the value passed to longjmp on non-local return

    if (error == 0) {
        printf("Starting risky operation\n");
        risky_operation(0);
        printf("Completed normally\n");   // not reached
    } else {
        printf("Error recovered! Code: %d\n", error);
    }
    return 0;
}
```

:::caution[setjmp/longjmp Dangers]
`longjmp` bypasses destructors (none in C, but matters in mixed C/C++), does not free stack-allocated resources, and can leave program state inconsistent. Use only for top-level error recovery where you understand the full call stack.
:::

## C11 Threads

```c
#include <threads.h>   // C11 threads
#include <stdio.h>
#include <stdatomic.h>

// Shared state
atomic_int g_counter = 0;
mtx_t      g_mutex;

typedef struct { int id; int iterations; } WorkerArgs;

int worker_thread(void *arg) {
    WorkerArgs *args = arg;

    for (int i = 0; i < args->iterations; i++) {
        atomic_fetch_add(&g_counter, 1);   // atomic increment — no mutex needed
    }

    printf("Thread %d done. Counter: %d\n", args->id, atomic_load(&g_counter));
    return 0;
}

// Thread-local storage (C11)
_Thread_local int tls_id = 0;

int tls_thread(void *arg) {
    tls_id = *(int *)arg;   // each thread has its own copy
    printf("Thread-local ID: %d\n", tls_id);
    return 0;
}

int main(void) {
    mtx_init(&g_mutex, mtx_plain);

    thrd_t threads[4];
    WorkerArgs args[4];

    for (int i = 0; i < 4; i++) {
        args[i] = (WorkerArgs){ .id = i, .iterations = 100000 };
        thrd_create(&threads[i], worker_thread, &args[i]);
    }

    for (int i = 0; i < 4; i++) {
        thrd_join(threads[i], NULL);
    }

    printf("Final counter: %d (expected: 400000)\n", atomic_load(&g_counter));

    mtx_destroy(&g_mutex);
    return 0;
}
```

## C11 Atomics

```c
#include <stdatomic.h>
#include <stdio.h>

// Atomic types
_Atomic int   atomic_counter = 0;
_Atomic _Bool atomic_flag    = 0;

// Memory orders (from weakest to strongest):
// memory_order_relaxed  — no ordering constraints (only atomicity)
// memory_order_consume  — data dependency ordering
// memory_order_acquire  — see all writes before corresponding release
// memory_order_release  — writes visible before corresponding acquire
// memory_order_acq_rel  — acquire + release
// memory_order_seq_cst  — total sequential consistency (default)

void atomics_demo(void) {
    // Load/store
    int val = atomic_load_explicit(&atomic_counter, memory_order_relaxed);
    atomic_store_explicit(&atomic_counter, 42, memory_order_relaxed);

    // Arithmetic operations
    int old = atomic_fetch_add(&atomic_counter, 1);   // returns OLD value
    atomic_fetch_sub(&atomic_counter, 1);
    atomic_fetch_or(&atomic_counter, BIT(3));

    // Compare-and-swap (CAS) — foundation of lock-free algorithms
    int expected = 42;
    int desired  = 43;
    // Returns true if swap succeeded, updates expected if failed
    if (atomic_compare_exchange_strong(&atomic_counter, &expected, desired)) {
        printf("CAS succeeded\n");
    } else {
        printf("CAS failed, current value: %d\n", expected);
    }

    // Lock-free flag
    _Bool old_flag = atomic_exchange(&atomic_flag, 1);  // set and get old value
}
```

## Designated Initializers and Compound Literals (C99)

```c
#include <stdio.h>

typedef struct { int x, y, z; } Vec3i;

// Compound literal — unnamed object of specified type
Vec3i v = (Vec3i){ .x=1, .y=2, .z=3 };

// Pass directly to function without a named variable
void print_vec(Vec3i v) { printf("(%d,%d,%d)\n", v.x, v.y, v.z); }
print_vec((Vec3i){ 4, 5, 6 });

// Array compound literal
int *primes = (int[]){ 2, 3, 5, 7, 11, 13 };
// Note: compound literal lifetime is the enclosing block (like a local variable)

// Designated initializer for arrays
int sparse[100] = {
    [0]  = 1,
    [50] = 42,
    [99] = -1,
    // all others are 0
};
```

## Inline Functions (C99)

```c
// inline hint — compiler may inline the function body at call site
// In C99, a function with 'inline' and no 'extern' has no external linkage
// Must provide extern definition in exactly one translation unit if called externally

// Best practice: define in header with static inline
static inline int imax(int a, int b) {
    return a > b ? a : b;
}

static inline float fast_inv_sqrt(float x) {
    // Quake III famous fast inverse square root
    long i = *(long *)&x;
    i = 0x5f3759df - (i >> 1);
    x = *(float *)&i;
    x = x * (1.5f - 0.5f * x * x * x);
    return x;
}
```

## restrict Qualifier in Depth

```c
#include <stddef.h>

// restrict: pointer is the ONLY way to access that memory in this function
// Enables auto-vectorization and other aggressive optimizations
void memcpy_restrict(void * restrict dst, const void * restrict src, size_t n) {
    // Compiler knows dst and src don't overlap — can vectorize
    unsigned char       *d = dst;
    const unsigned char *s = src;
    for (size_t i = 0; i < n; i++) d[i] = s[i];
}

// Matrix multiplication — restrict enables many register optimizations
void matmul(float * restrict C,
            const float * restrict A,
            const float * restrict B,
            int N) {
    for (int i = 0; i < N; i++)
        for (int k = 0; k < N; k++)
            for (int j = 0; j < N; j++)
                C[i*N+j] += A[i*N+k] * B[k*N+j];
}
```

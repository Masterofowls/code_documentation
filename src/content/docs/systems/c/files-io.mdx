# File I/O in C

C provides two layers of file I/O: the standard library's buffered `FILE*` interface (portable, convenient) and the POSIX `open()`/`read()`/`write()` interface (lower-level, more control). Both are essential for systems programming.

## FILE\* Streams

The three standard streams are automatically available:

```c
#include <stdio.h>

// Pre-defined FILE* streams (auto-opened)
// stdin  — standard input (keyboard by default)
// stdout — standard output (terminal by default)
// stderr — standard error (terminal, unbuffered by default)

// Basic usage:
fprintf(stdout, "Hello from stdout\n");
fprintf(stderr, "Error: something went wrong\n");
fputs("Line to stdout\n", stdout);
```

## Opening and Closing Files

```c
#include <stdio.h>
#include <errno.h>
#include <string.h>

int main(void) {
    // fopen returns NULL on failure
    // Mode strings:
    //   "r"  — read, file must exist
    //   "w"  — write, truncates/creates
    //   "a"  — append, creates if needed
    //   "r+" — read+write, file must exist
    //   "w+" — read+write, truncates/creates
    //   "a+" — read+append, creates if needed
    //   Add "b" for binary mode: "rb", "wb", "ab", "rb+", "wb+", "ab+"

    FILE *f = fopen("data.txt", "r");
    if (f == NULL) {
        // errno contains the error code
        fprintf(stderr, "Failed to open 'data.txt': %s\n", strerror(errno));
        return 1;
    }

    // ... use f ...

    // fclose flushes buffers and releases resources
    if (fclose(f) != 0) {
        fprintf(stderr, "fclose failed: %s\n", strerror(errno));
        return 1;
    }

    // Binary mode is important on Windows (converts \r\n on text vs binary)
    FILE *bin = fopen("data.bin", "wb");
    if (!bin) { perror("fopen binary"); return 1; }
    fclose(bin);

    return 0;
}
```

### Error Handling

```c
#include <stdio.h>
#include <errno.h>
#include <string.h>

void process_file(const char *path) {
    FILE *f = fopen(path, "r");
    if (!f) {
        perror(path);  // perror prints: "path: error message\n"
        return;
    }

    char buf[256];
    while (fgets(buf, sizeof(buf), f)) {
        // fgets returns NULL on EOF or error
        // Use feof() and ferror() to distinguish
        printf("%s", buf);
    }

    if (ferror(f)) {
        fprintf(stderr, "Read error on '%s': %s\n", path, strerror(errno));
        clearerr(f);  // Clear the error/EOF indicator
    }
    // feof(f) would be non-zero here after reading to end

    fclose(f);
}

// Checking return values is mandatory
int write_data(const char *path, const int *arr, int n) {
    FILE *f = fopen(path, "w");
    if (!f) { perror(path); return -1; }

    for (int i = 0; i < n; i++) {
        if (fprintf(f, "%d\n", arr[i]) < 0) {
            perror("fprintf");
            fclose(f);
            return -1;
        }
    }

    if (fclose(f) != 0) { perror("fclose"); return -1; }
    return 0;
}
```

## Formatted I/O

```c
#include <stdio.h>
#include <string.h>

int main(void) {
    // Writing formatted output
    FILE *f = fopen("output.txt", "w");
    if (!f) { perror("fopen"); return 1; }

    fprintf(f, "Name: %s\n", "Alice");
    fprintf(f, "Age: %d\n", 30);
    fprintf(f, "Score: %.2f\n", 98.6);
    fclose(f);

    // sprintf: format to string (risk of overflow — avoid)
    char buf[64];
    sprintf(buf, "%d + %d = %d", 1, 2, 3);  // Dangerous if buf too small

    // snprintf: safe version with size limit
    char safe[64];
    int n = snprintf(safe, sizeof(safe), "Value: %d", 42);
    if (n >= (int)sizeof(safe)) printf("Output was truncated!\n");
    printf("safe: %s\n", safe);

    // sscanf: parse formatted string
    const char *input = "Alice 30 98.6";
    char name[32];
    int age;
    float score;
    int read = sscanf(input, "%31s %d %f", name, &age, &score);
    if (read == 3) {
        printf("Parsed: %s, %d, %.1f\n", name, age, score);
    }

    // fscanf: read formatted from file
    FILE *fin = fopen("output.txt", "r");
    if (fin) {
        char key[32], val[64];
        while (fscanf(fin, "%31[^:]: %63[^\n]\n", key, val) == 2) {
            printf("  key='%s' val='%s'\n", key, val);
        }
        fclose(fin);
    }

    return 0;
}
```

## Character and Line I/O

```c
#include <stdio.h>

// Character I/O
void copy_chars(FILE *in, FILE *out) {
    int c;
    // fgetc returns int (not char!) to represent EOF = -1
    while ((c = fgetc(in)) != EOF) {
        if (fputc(c, out) == EOF) { perror("fputc"); break; }
    }
}

// Unget a character (push one char back into the stream)
int peek_char(FILE *f) {
    int c = fgetc(f);
    if (c != EOF) ungetc(c, f);  // Can only unget one character
    return c;
}

// Line I/O
void read_lines(const char *path) {
    FILE *f = fopen(path, "r");
    if (!f) { perror(path); return; }

    char line[4096];
    int lineno = 0;

    // fgets: safe — reads at most (size-1) chars, keeps newline, adds '\0'
    // NEVER use gets() — it has no buffer limit and was removed in C11
    while (fgets(line, sizeof(line), f)) {
        lineno++;
        // Remove trailing newline if present
        size_t len = strlen(line);
        if (len > 0 && line[len-1] == '\n') line[len-1] = '\0';

        printf("%4d: %s\n", lineno, line);

        // Check if line was truncated (no newline found, not EOF)
        if (len == sizeof(line) - 1 && !feof(f)) {
            fprintf(stderr, "Line %d was truncated\n", lineno);
        }
    }

    fclose(f);
}

// Write line to file
void write_lines(FILE *f, const char **lines, int n) {
    for (int i = 0; i < n; i++) {
        // fputs doesn't add a newline — add it yourself
        if (fputs(lines[i], f) == EOF || fputc('\n', f) == EOF) {
            perror("write_lines");
            break;
        }
    }
}
```

## Binary I/O

```c
#include <stdio.h>
#include <stdint.h>
#include <string.h>

// Binary format for a simple data file
typedef struct {
    uint32_t magic;      // File identifier
    uint32_t version;
    uint32_t count;      // Number of records
    uint32_t reserved;
} FileHeader;

typedef struct {
    int32_t id;
    float   x, y, z;
    char    name[32];
} Record;

// Write binary file
int write_records(const char *path, const Record *records, uint32_t count) {
    FILE *f = fopen(path, "wb");
    if (!f) { perror(path); return -1; }

    FileHeader header = {
        .magic    = 0xDEADBEEF,
        .version  = 1,
        .count    = count,
        .reserved = 0
    };

    // fwrite(ptr, element_size, count, stream) — returns elements written
    if (fwrite(&header, sizeof(header), 1, f) != 1) goto err;
    if (fwrite(records, sizeof(Record), count, f) != count) goto err;

    fclose(f);
    return 0;

err:
    perror("fwrite");
    fclose(f);
    return -1;
}

// Read binary file
int read_records(const char *path, Record **out_records, uint32_t *out_count) {
    FILE *f = fopen(path, "rb");
    if (!f) { perror(path); return -1; }

    FileHeader header;
    if (fread(&header, sizeof(header), 1, f) != 1) goto err;

    // Validate
    if (header.magic != 0xDEADBEEF) {
        fprintf(stderr, "Invalid file magic: 0x%08X\n", header.magic);
        goto err;
    }
    if (header.version != 1) {
        fprintf(stderr, "Unsupported version: %u\n", header.version);
        goto err;
    }

    Record *records = malloc(header.count * sizeof(Record));
    if (!records) { perror("malloc"); goto err; }

    if (fread(records, sizeof(Record), header.count, f) != header.count) {
        free(records);
        goto err;
    }

    *out_records = records;
    *out_count   = header.count;
    fclose(f);
    return 0;

err:
    fclose(f);
    return -1;
}
```

## File Positioning

```c
#include <stdio.h>

void seek_demo(const char *path) {
    FILE *f = fopen(path, "rb");
    if (!f) { perror(path); return; }

    // fseek(stream, offset, whence)
    // SEEK_SET: from beginning
    // SEEK_CUR: from current position
    // SEEK_END: from end (negative offset reads backward)

    // Get file size
    fseek(f, 0, SEEK_END);
    long file_size = ftell(f);   // Returns current position as long
    printf("File size: %ld bytes\n", file_size);

    // Seek back to beginning
    rewind(f);    // Equivalent to fseek(f, 0, SEEK_SET) + clearerr

    // Read last 10 bytes
    if (file_size >= 10) {
        fseek(f, -10, SEEK_END);
        char tail[11] = {0};
        fread(tail, 1, 10, f);
        printf("Last 10 bytes: %s\n", tail);
    }

    // fgetpos/fsetpos: for very large files (uses fpos_t, not long)
    fpos_t pos;
    fseek(f, 100, SEEK_SET);
    fgetpos(f, &pos);            // Save position
    // ... read some data ...
    fsetpos(f, &pos);            // Restore position

    fclose(f);
}
```

## Buffering Control

```c
#include <stdio.h>
#include <stdlib.h>

void buffering_demo(void) {
    FILE *f = fopen("log.txt", "w");
    if (!f) return;

    // setbuf(stream, buf):
    //   buf = NULL  → unbuffered
    //   buf = array → fully buffered using that array (must be BUFSIZ bytes)

    // setvbuf(stream, buf, mode, size): more control
    //   _IOFBF — full buffering (writes when buffer is full)
    //   _IOLBF — line buffering (flushes on newline; stdout when connected to terminal)
    //   _IONBF — no buffering (writes immediately; stderr default)

    // Use a custom 64KB buffer
    char *buf = malloc(65536);
    setvbuf(f, buf, _IOFBF, 65536);

    for (int i = 0; i < 1000; i++) {
        fprintf(f, "Line %d: some data here\n", i);
    }

    // fflush: force-write buffered data to OS
    // Essential before: fork(), reading after writing with same FILE*, crash recovery
    fflush(f);

    // After fclose, buf is no longer used by the stream
    fclose(f);
    free(buf);

    // Make stderr line-buffered (useful for log output)
    setvbuf(stderr, NULL, _IOLBF, 0);

    // Make stdout unbuffered (every character written immediately)
    setbuf(stdout, NULL);
}
```

## POSIX File I/O

POSIX provides lower-level file operations through file descriptors (integers, not `FILE*`).

```c
#include <unistd.h>   // read, write, close, lseek
#include <fcntl.h>    // open, O_* flags
#include <sys/stat.h> // stat, mkdir
#include <dirent.h>   // opendir, readdir
#include <stdio.h>
#include <string.h>
#include <errno.h>

void posix_read_write_demo(void) {
    // open: returns file descriptor (int), or -1 on error
    // Flags: O_RDONLY, O_WRONLY, O_RDWR
    //        O_CREAT (create if not exists), O_TRUNC (truncate), O_APPEND
    //        O_NONBLOCK (non-blocking), O_SYNC (synchronous writes)
    // Mode: permissions (used with O_CREAT); e.g., 0644 = rw-r--r--
    int fd = open("data.bin", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) { perror("open"); return; }

    const char data[] = "Hello, POSIX world!\n";
    ssize_t written = write(fd, data, strlen(data));
    if (written == -1) perror("write");
    else if ((size_t)written < strlen(data)) printf("Partial write!\n");

    close(fd);  // Always close; returns 0 on success, -1 on error

    // Read back
    fd = open("data.bin", O_RDONLY);
    if (fd == -1) { perror("open"); return; }

    char buf[256];
    ssize_t nread;
    while ((nread = read(fd, buf, sizeof(buf) - 1)) > 0) {
        buf[nread] = '\0';
        printf("%s", buf);
    }
    if (nread == -1) perror("read");

    // lseek: reposition file offset
    // SEEK_SET, SEEK_CUR, SEEK_END (same as fseek)
    off_t pos = lseek(fd, 0, SEEK_END);
    printf("File size: %lld bytes\n", (long long)pos);

    close(fd);
}

void posix_directory_demo(const char *path) {
    // stat: get file information
    struct stat st;
    if (stat(path, &st) == -1) { perror(path); return; }

    printf("File: %s\n", path);
    printf("  Size:  %lld bytes\n", (long long)st.st_size);
    printf("  Inode: %llu\n", (unsigned long long)st.st_ino);
    printf("  Is dir: %s\n", S_ISDIR(st.st_mode) ? "yes" : "no");
    printf("  Perms: %o\n", (unsigned)(st.st_mode & 0777));

    // List directory contents
    DIR *dir = opendir(path);
    if (!dir) { perror("opendir"); return; }

    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        if (entry->d_name[0] == '.') continue;  // Skip . and ..
        printf("  %s\n", entry->d_name);
    }

    closedir(dir);
}

// dup/dup2: duplicate file descriptors
void redirect_stdout_demo(const char *logfile) {
    int log_fd = open(logfile, O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (log_fd == -1) { perror("open"); return; }

    // Save original stdout
    int saved_stdout = dup(STDOUT_FILENO);

    // Redirect stdout to logfile
    dup2(log_fd, STDOUT_FILENO);
    close(log_fd);

    printf("This goes to the log file!\n");
    fflush(stdout);

    // Restore stdout
    dup2(saved_stdout, STDOUT_FILENO);
    close(saved_stdout);

    printf("This goes to the terminal again\n");
}
```

## Memory-Mapped Files

```c
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

// mmap: map file contents directly into virtual address space
// Benefits: zero-copy I/O, OS manages caching/paging, huge file support
void mmap_demo(const char *path) {
    int fd = open(path, O_RDONLY);
    if (fd == -1) { perror("open"); return; }

    struct stat st;
    fstat(fd, &st);
    size_t size = (size_t)st.st_size;

    // mmap(addr, length, prot, flags, fd, offset)
    // PROT_READ, PROT_WRITE, PROT_EXEC
    // MAP_SHARED (visible to other processes), MAP_PRIVATE (copy-on-write)
    void *data = mmap(NULL, size, PROT_READ, MAP_PRIVATE, fd, 0);
    if (data == MAP_FAILED) { perror("mmap"); close(fd); return; }

    // Now we can read file data as regular memory!
    const char *text = (const char*)data;
    printf("First 100 chars: %.100s\n", text);

    // Count newlines efficiently
    size_t lines = 0;
    for (size_t i = 0; i < size; i++) {
        if (text[i] == '\n') lines++;
    }
    printf("Line count: %zu\n", lines);

    // Advise OS about access pattern
    madvise(data, size, MADV_SEQUENTIAL);  // Prefetch ahead

    // Unmap when done
    munmap(data, size);
    close(fd);
}
```

:::tip[mmap Performance]
`mmap` is ideal for read-once large files (log parsing, binary formats) and for files accessed randomly by multiple processes. For small files or files written frequently, `read()`/`write()` is often simpler and faster due to less setup overhead.
:::

## File Locking

```c
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

// Advisory locking with fcntl (POSIX)
// Multiple processes agree to check locks before accessing files
int lock_file(int fd, int lock_type) {
    // lock_type: F_RDLCK (shared read lock), F_WRLCK (exclusive write lock), F_UNLCK
    struct flock fl = {
        .l_type   = lock_type,
        .l_whence = SEEK_SET,
        .l_start  = 0,
        .l_len    = 0,   // 0 = entire file
    };
    return fcntl(fd, F_SETLKW, &fl);  // F_SETLKW: blocking wait for lock
}

int try_lock_file(int fd, int lock_type) {
    struct flock fl = {
        .l_type = lock_type, .l_whence = SEEK_SET, .l_start = 0, .l_len = 0
    };
    return fcntl(fd, F_SETLK, &fl);  // F_SETLK: non-blocking, returns -1 if locked
}

void locked_write_demo(const char *path) {
    int fd = open(path, O_WRONLY | O_CREAT | O_APPEND, 0644);
    if (fd == -1) { perror("open"); return; }

    // Acquire exclusive write lock
    if (lock_file(fd, F_WRLCK) == -1) { perror("lock"); close(fd); return; }

    const char *msg = "Atomic line\n";
    write(fd, msg, strlen(msg));

    // Release lock
    lock_file(fd, F_UNLCK);
    close(fd);
}
```

# File I/O in C

C provides two levels of file I/O: the **standard library** (`stdio.h`) with buffered streams, and the **POSIX API** with unbuffered file descriptors. Understanding both layers — and when to use each — is essential for systems programming.

---

## Standard Streams

Three streams are always available at program startup:

```c
#include <stdio.h>

// stdin  — standard input  (fd 0) — keyboard by default
// stdout — standard output (fd 1) — terminal by default (line-buffered)
// stderr — standard error  (fd 2) — terminal by default (unbuffered)

fprintf(stdout, "This is stdout\n");
fprintf(stderr, "This is stderr (not buffered)\n");

// printf is just fprintf(stdout, ...)
printf("Hello\n");

// fflush forces buffered data to be written
fflush(stdout);   // flush stdout buffer
fflush(NULL);     // flush ALL open output streams
```

## fopen and fclose

```c
#include <stdio.h>
#include <string.h>
#include <errno.h>

int main(void) {
    // fopen(filename, mode) — returns FILE* or NULL on error
    FILE *f = fopen("data.txt", "w");
    if (!f) {
        perror("fopen");   // prints: fopen: <error description>
        return 1;
    }

    fprintf(f, "Line 1\n");
    fprintf(f, "Value: %d\n", 42);

    fclose(f);   // always close! flushes buffer and releases resources

    return 0;
}
```

### Mode Strings

| Mode | Description | File must exist? | Truncate? |
|---|---|---|---|
| `"r"` | Read only | ✅ yes | — |
| `"w"` | Write only | ❌ creates | ✅ yes |
| `"a"` | Append | ❌ creates | — |
| `"r+"` | Read + write | ✅ yes | — |
| `"w+"` | Read + write | ❌ creates | ✅ yes |
| `"a+"` | Read + append | ❌ creates | — |
| `"rb"` | Binary read | ✅ yes | — |
| `"wb"` | Binary write | ❌ creates | ✅ yes |
| `"ab"` | Binary append | ❌ creates | — |

:::tip[Always Use Binary Mode for Non-Text]
On Windows, text mode translates `\n` ↔ `\r\n`. For image files, serialized data, or any non-text content, always open with the `b` flag (`"rb"`, `"wb"`) to prevent corruption.
:::

## Error Handling

```c
#include <stdio.h>
#include <errno.h>
#include <string.h>

int main(void) {
    FILE *f = fopen("nonexistent.txt", "r");
    if (!f) {
        // perror prints: "<message>: <errno string>"
        perror("Cannot open file");

        // strerror returns error string for given errno
        printf("Error: %s (errno=%d)\n", strerror(errno), errno);
        return 1;
    }

    char buf[256];
    while (fgets(buf, sizeof(buf), f) != NULL) {
        printf("%s", buf);
    }

    // Check why the loop ended
    if (ferror(f)) {
        perror("Read error");
        clearerr(f);   // clear error and EOF indicators
    }

    if (feof(f)) {
        printf("Reached end of file\n");
    }

    fclose(f);
    return 0;
}
```

## Formatted I/O

```c
#include <stdio.h>

void formatted_io_examples(void) {
    // fprintf — write formatted text to FILE*
    FILE *out = fopen("output.txt", "w");
    fprintf(out, "Name: %-20s Age: %3d GPA: %.2f\n", "Alice", 21, 3.85);
    fclose(out);

    // fscanf — read formatted values from FILE*
    FILE *in = fopen("data.txt", "r");
    char  name[64];
    int   age;
    float gpa;
    int count = fscanf(in, "%63s %d %f", name, &age, &gpa);
    if (count == 3) {
        printf("Read: %s %d %.2f\n", name, age, gpa);
    }
    fclose(in);

    // sprintf / snprintf — format to a string buffer
    char buf[128];
    int n = snprintf(buf, sizeof(buf), "Value: %d, Pi: %.4f", 42, 3.14159f);
    printf("Formatted (%d chars): %s\n", n, buf);
    // NOTE: always prefer snprintf over sprintf — sprintf can overflow

    // sscanf — parse a string
    const char *line = "Alice 21 3.85";
    char pname[64]; int page; float pgpa;
    if (sscanf(line, "%63s %d %f", pname, &page, &pgpa) == 3) {
        printf("Parsed: %s %d %.2f\n", pname, page, pgpa);
    }
}
```

:::caution[Never Use gets()]
`gets()` was removed in C11. It cannot be used safely — it has no buffer size parameter and will overflow. Always use `fgets(buf, sizeof(buf), stdin)` instead.
:::

## Character I/O

```c
#include <stdio.h>

// Character-by-character reading
void char_io_examples(void) {
    FILE *f = fopen("text.txt", "r");
    if (!f) return;

    int c;
    while ((c = fgetc(f)) != EOF) {
        // c is int, not char — EOF (-1) doesn't fit in char
        putchar(c);   // write to stdout
    }

    // Push back one character (for one-character lookahead)
    ungetc(c, f);   // puts c back into the stream buffer

    fclose(f);

    // Convenient functions for stdin/stdout
    int ch = getchar();   // read one char from stdin
    putchar(ch);           // write one char to stdout

    // fputc — write to specific stream
    fputc('!', stdout);
    fputc('\n', stdout);
}

// Line-by-line reading with fgets
void line_io_examples(void) {
    FILE *f = fopen("lines.txt", "r");
    if (!f) return;

    char line[1024];
    int linenum = 1;

    while (fgets(line, sizeof(line), f) != NULL) {
        // fgets keeps the \n at end of line (if it fits)
        // Remove trailing newline:
        size_t len = strlen(line);
        if (len > 0 && line[len-1] == '\n') line[len-1] = '\0';

        printf("%4d: %s\n", linenum++, line);
    }

    fclose(f);

    // fputs — write string to file (no formatting, no automatic newline)
    fputs("Hello, World!\n", stdout);
}
```

## Binary I/O

```c
#include <stdio.h>
#include <stdint.h>

typedef struct {
    uint32_t magic;
    uint32_t version;
    uint64_t timestamp;
    float    data[16];
} FileHeader;

// Write binary data
int write_binary(const char *path, const float *samples, int count) {
    FILE *f = fopen(path, "wb");   // MUST use "wb" on Windows
    if (!f) return -1;

    FileHeader hdr = {
        .magic     = 0xDEADBEEF,
        .version   = 1,
        .timestamp = 1234567890ULL,
    };

    // fwrite(ptr, element_size, element_count, file) — returns elements written
    if (fwrite(&hdr, sizeof(hdr), 1, f) != 1) { fclose(f); return -1; }
    if (fwrite(samples, sizeof(float), count, f) != (size_t)count) {
        fclose(f); return -1;
    }

    fclose(f);
    return 0;
}

// Read binary data
int read_binary(const char *path, float *out_samples, int max_count) {
    FILE *f = fopen(path, "rb");
    if (!f) return -1;

    FileHeader hdr;
    if (fread(&hdr, sizeof(hdr), 1, f) != 1) { fclose(f); return -1; }

    if (hdr.magic != 0xDEADBEEF) {
        printf("Invalid magic: 0x%08X\n", hdr.magic);
        fclose(f);
        return -1;
    }

    // fread(ptr, element_size, element_count, file) — returns elements read
    int count = (int)fread(out_samples, sizeof(float), max_count, f);
    fclose(f);
    return count;
}
```

## File Positioning

```c
#include <stdio.h>

void positioning_examples(void) {
    FILE *f = fopen("data.bin", "rb");
    if (!f) return;

    // ftell — current position (bytes from start)
    long pos = ftell(f);
    printf("Initial position: %ld\n", pos);   // 0

    // fseek(file, offset, origin)
    // SEEK_SET — absolute from start
    // SEEK_CUR — relative to current position
    // SEEK_END — relative to end (usually negative offset)
    fseek(f, 0, SEEK_END);
    long file_size = ftell(f);
    printf("File size: %ld bytes\n", file_size);

    fseek(f, 0, SEEK_SET);   // rewind to start
    // equivalent: rewind(f);

    fseek(f, 10, SEEK_SET);   // jump to byte 10
    fseek(f, 5, SEEK_CUR);    // advance 5 bytes from current

    // fgetpos/fsetpos — for files > 2GB (uses fpos_t, not long)
    fpos_t save_pos;
    fgetpos(f, &save_pos);    // save position
    // ... do some reading ...
    fsetpos(f, &save_pos);    // restore position

    fclose(f);
}
```

## I/O Buffering

```c
#include <stdio.h>
#include <stdlib.h>

void buffering_examples(void) {
    FILE *f = fopen("output.txt", "w");

    // setvbuf — set buffering mode BEFORE any I/O
    // _IOFBF: fully buffered (default for files)
    // _IOLBF: line buffered (default for stdout when connected to terminal)
    // _IONBF: unbuffered (default for stderr)
    char *mybuf = malloc(8192);
    setvbuf(f, mybuf, _IOFBF, 8192);   // use custom 8KB buffer

    // setbuf — simpler version (full buffering with BUFSIZ, or unbuffered)
    // setbuf(f, NULL);    // unbuffered

    fprintf(f, "Buffered output\n");   // not written to disk yet

    fflush(f);   // force write buffer to OS
    // after fflush, data is in OS page cache (not necessarily on disk)
    // to guarantee disk write, use fsync(fileno(f)) on POSIX

    fclose(f);   // implicitly flushes before closing
    free(mybuf);
}
```

## POSIX File I/O — File Descriptors

```c
#include <fcntl.h>      // open, O_RDONLY, O_WRONLY, etc.
#include <unistd.h>     // read, write, close, lseek
#include <sys/stat.h>   // stat, mkdir, S_IRUSR, etc.
#include <stdio.h>
#include <dirent.h>     // opendir, readdir, closedir

void posix_io_examples(void) {
    // open(path, flags, mode) — returns fd (int) or -1
    int fd = open("data.txt", O_RDONLY);
    if (fd < 0) { perror("open"); return; }

    // read(fd, buf, count) — returns bytes read, 0 on EOF, -1 on error
    char buf[4096];
    ssize_t n;
    while ((n = read(fd, buf, sizeof(buf))) > 0) {
        write(1, buf, n);   // write to stdout (fd 1)
    }
    close(fd);

    // Create/write file
    int wfd = open("output.txt",
                   O_WRONLY | O_CREAT | O_TRUNC,
                   S_IRUSR | S_IWUSR | S_IRGRP);  // permissions: rw-r-----
    if (wfd < 0) { perror("open for write"); return; }
    const char *msg = "Hello from POSIX\n";
    write(wfd, msg, strlen(msg));
    close(wfd);

    // lseek(fd, offset, whence) — repositioning
    int fd2 = open("data.txt", O_RDONLY);
    off_t size = lseek(fd2, 0, SEEK_END);   // get file size
    lseek(fd2, 0, SEEK_SET);               // rewind
    close(fd2);
    printf("File size: %lld\n", (long long)size);

    // stat — file metadata
    struct stat st;
    if (stat("data.txt", &st) == 0) {
        printf("size=%lld mode=%o mtime=%ld\n",
               (long long)st.st_size, st.st_mode, (long)st.st_mtime);
    }

    // Directory operations
    mkdir("newdir", 0755);
    DIR *dir = opendir(".");
    struct dirent *entry;
    while ((entry = readdir(dir)) != NULL) {
        printf("%s\n", entry->d_name);
    }
    closedir(dir);
}
```

## File Descriptor Duplication

```c
#include <unistd.h>
#include <fcntl.h>
#include <stdio.h>

void redirect_stdout(void) {
    // Redirect stdout to a file
    int log = open("app.log", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    // dup2(old_fd, new_fd) — makes new_fd refer to same file as old_fd
    int saved_stdout = dup(1);   // save stdout
    dup2(log, 1);                // stdout now goes to log file
    close(log);

    printf("This goes to app.log\n");   // redirected!
    fflush(stdout);

    dup2(saved_stdout, 1);   // restore stdout
    close(saved_stdout);

    printf("This goes to terminal\n");
}
```

## Memory-Mapped Files (mmap)

```c
#include <sys/mman.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>

void mmap_example(void) {
    int fd = open("largefile.bin", O_RDONLY);
    if (fd < 0) return;

    struct stat st;
    fstat(fd, &st);
    size_t size = st.st_size;

    // Map file into virtual address space
    void *data = mmap(
        NULL,              // let OS choose address
        size,             // map entire file
        PROT_READ,        // read-only
        MAP_PRIVATE,      // private (copy-on-write)
        fd,               // file descriptor
        0                 // offset into file
    );

    close(fd);   // fd can be closed after mmap

    if (data == MAP_FAILED) { perror("mmap"); return; }

    // Access file contents directly as memory — zero-copy!
    const uint8_t *bytes = (const uint8_t *)data;
    printf("First byte: 0x%02X\n", bytes[0]);

    // The OS handles reading pages from disk on demand (demand paging)
    // Excellent for random access of large files

    munmap(data, size);   // release mapping
}
```

## File Locking (POSIX)

```c
#include <fcntl.h>
#include <unistd.h>

void file_locking(void) {
    int fd = open("shared.dat", O_RDWR);

    // fcntl advisory locks (POSIX)
    struct flock fl = {
        .l_type   = F_WRLCK,   // exclusive write lock (F_RDLCK, F_UNLCK)
        .l_whence = SEEK_SET,
        .l_start  = 0,
        .l_len    = 0,         // 0 = lock entire file
    };

    // F_SETLKW — set lock, wait if blocked
    if (fcntl(fd, F_SETLKW, &fl) < 0) { perror("lock"); return; }

    // Critical section — only one process holds this lock
    // ... read/write shared.dat ...

    // Release lock
    fl.l_type = F_UNLCK;
    fcntl(fd, F_SETLK, &fl);

    close(fd);
}
```

# Getting Started with C

This guide covers the essential tooling and language fundamentals you need to write, build, and run real C programs from day one.

## Compiler Command Line

### GCC and Clang Essentials

```bash
# Basic compilation
gcc main.c -o program
clang main.c -o program

# Enable strict warnings (recommended for all projects)
gcc -Wall -Wextra -Wpedantic main.c -o program

# Specify C standard
gcc -std=c11 main.c -o program    # C11
gcc -std=c99 main.c -o program    # C99
gcc -std=c23 main.c -o program    # C23 (GCC 13+)

# Debug build — include debug symbols, disable optimizations
gcc -std=c11 -Wall -Wextra -g -O0 main.c -o program_debug

# Optimized release build
gcc -std=c11 -O2 main.c -o program_release
gcc -std=c11 -O3 -march=native main.c -o program_fast  # CPU-specific opts

# Compile to object file only (no linking)
gcc -std=c11 -c main.c -o main.o
gcc -std=c11 -c utils.c -o utils.o

# Link multiple object files
gcc main.o utils.o -o program

# Link a system library (e.g., math library)
gcc main.c -o program -lm           # -lm links libm (math)
gcc main.c -o program -lpthread     # POSIX threads

# Sanitizers — catch bugs at runtime
gcc -fsanitize=address,undefined -g main.c -o program_san
```

:::tip[Warning Flags Explained]
- `-Wall` — enables most important warnings (despite the name, not *all*)
- `-Wextra` — additional warnings Wall doesn't include
- `-Wpedantic` — strict ISO C conformance warnings
- `-Werror` — treat all warnings as errors (great for CI pipelines)
:::

### Useful Diagnostic Flags

```bash
# Show preprocessor output
gcc -E main.c -o main.i

# Show assembly output
gcc -S -O2 main.c -o main.s

# Show macro definitions from headers
gcc -dM -E -std=c11 - < /dev/null

# Optimization report
gcc -O2 -fopt-info-vec main.c -o program     # vectorization info

# Static analysis pass
gcc -fanalyzer main.c -o program             # GCC 10+
clang --analyze main.c                        # Clang static analyzer
```

## Makefile Basics

Make automates the build process. It tracks file modification timestamps and only rebuilds what has changed.

```makefile
# Makefile

# Variables
CC      := gcc
CFLAGS  := -std=c11 -Wall -Wextra -Wpedantic
DBGFLAGS:= -g -O0 -fsanitize=address,undefined
RELFLAGS:= -O2 -DNDEBUG
SRCS    := main.c utils.c parser.c
OBJS    := $(SRCS:.c=.o)
TARGET  := program

# Default target
all: $(TARGET)

# Link rule
$(TARGET): $(OBJS)
	$(CC) $(CFLAGS) $^ -o $@

# Pattern rule: compile each .c to .o
%.o: %.c
	$(CC) $(CFLAGS) -c $< -o $@

# Debug target
debug: CFLAGS += $(DBGFLAGS)
debug: $(TARGET)

# Release target
release: CFLAGS += $(RELFLAGS)
release: $(TARGET)

# Phony targets — not real files
.PHONY: all clean debug release

clean:
	rm -f $(OBJS) $(TARGET)
```

Key Make syntax:
- `$@` — the target name of the current rule
- `$&lt;` — first prerequisite (dependency)
- `$^` — all prerequisites
- `:=` — immediate assignment (evaluate once)
- `?=` — assign only if not already set

## CMake Basics for C

CMake generates Makefiles, Ninja build files, or Visual Studio projects from a single `CMakeLists.txt`.

```cmake
# CMakeLists.txt
cmake_minimum_required(VERSION 3.20)
project(MyProject VERSION 1.0 LANGUAGES C)

# Require C11
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)   # Disable GNU extensions for portability

# Compiler warnings
add_compile_options(-Wall -Wextra -Wpedantic)

# Sources
set(SOURCES
    src/main.c
    src/utils.c
    src/parser.c
)

# Executable target
add_executable(my_program ${SOURCES})

# Include directories
target_include_directories(my_program PRIVATE include/)

# Link libraries
target_link_libraries(my_program PRIVATE m)   # math library

# Optional: sanitizer build
option(ENABLE_SANITIZERS "Enable address and UB sanitizers" OFF)
if(ENABLE_SANITIZERS)
    target_compile_options(my_program PRIVATE -fsanitize=address,undefined)
    target_link_options(my_program PRIVATE -fsanitize=address,undefined)
endif()
```

```bash
# Build workflow
mkdir build && cd build
cmake ..                             # configure
cmake --build .                      # build
cmake --build . --target clean       # clean
cmake .. -DENABLE_SANITIZERS=ON      # with option
cmake .. -DCMAKE_BUILD_TYPE=Release  # release mode
```

## Data Types

### Fundamental Types

```c
#include <stdio.h>
#include <limits.h>   // INT_MAX, CHAR_MIN, etc.
#include <float.h>    // DBL_MAX, FLT_EPSILON, etc.

int main(void)
{
    /* Integer types — sizes are implementation-defined minimums */
    char   c  = 'A';        // at least 8 bits; may be signed or unsigned
    short  s  = 32767;      // at least 16 bits
    int    i  = 2147483647; // at least 16 bits; typically 32
    long   l  = 2147483647L;// at least 32 bits
    long long ll = 9223372036854775807LL; // at least 64 bits

    /* Unsigned variants */
    unsigned char  uc = 255U;
    unsigned short us = 65535U;
    unsigned int   ui = 4294967295U;
    unsigned long  ul = 4294967295UL;
    unsigned long long ull = 18446744073709551615ULL;

    /* Floating-point types */
    float  f  = 3.14159f;       // typically 32-bit IEEE 754
    double d  = 3.14159265358979323846; // typically 64-bit IEEE 754
    long double ld = 3.14159265358979323846L; // 80-bit extended on x86

    /* Boolean (C99+) */
    #include <stdbool.h>
    bool flag = true;   // _Bool is the keyword; bool is the macro

    /* Print sizes */
    printf("char:        %zu bytes\n", sizeof(char));
    printf("short:       %zu bytes\n", sizeof(short));
    printf("int:         %zu bytes\n", sizeof(int));
    printf("long:        %zu bytes\n", sizeof(long));
    printf("long long:   %zu bytes\n", sizeof(long long));
    printf("float:       %zu bytes\n", sizeof(float));
    printf("double:      %zu bytes\n", sizeof(double));
    printf("long double: %zu bytes\n", sizeof(long double));
    printf("pointer:     %zu bytes\n", sizeof(void*));

    /* Limits */
    printf("INT_MAX  = %d\n", INT_MAX);
    printf("INT_MIN  = %d\n", INT_MIN);
    printf("DBL_MAX  = %e\n", DBL_MAX);
    printf("FLT_EPSILON = %e\n", FLT_EPSILON); // smallest x: 1.0f + x != 1.0f

    return 0;
}
```

### Fixed-Width Types (`<stdint.h>`)

When exact sizes matter — networking, binary file formats, embedded registers — use `<stdint.h>`:

```c
#include <stdint.h>
#include <inttypes.h>  // printf macros for fixed-width types

int main(void)
{
    /* Exact-width types */
    int8_t   a = -128;
    uint8_t  b = 255;
    int16_t  c = -32768;
    uint16_t d = 65535;
    int32_t  e = -2147483648;
    uint32_t f = 4294967295U;
    int64_t  g = INT64_MIN;
    uint64_t h = UINT64_MAX;

    /* Fastest types of at least N bits */
    int_fast8_t  fast8  = 42;   // fastest type >= 8 bits
    int_fast16_t fast16 = 1000;
    int_fast32_t fast32 = 100000;

    /* Smallest types of at least N bits */
    int_least8_t  least8  = 10;
    int_least16_t least16 = 1000;

    /* Pointer-sized integer */
    intptr_t  ptr_val = (intptr_t)&a;  // can hold a pointer
    uintptr_t uptr    = (uintptr_t)&b;
    ptrdiff_t diff    = &b - &a;       // pointer difference
    size_t    sz      = sizeof(a);     // sizeof result type

    /* Print using PRId32, PRIu64, etc. macros */
    printf("int32: %"  PRId32 "\n", e);
    printf("uint64: %" PRIu64 "\n", h);
    printf("size_t: %zu\n", sz);
    printf("ptrdiff_t: %td\n", diff);

    return 0;
}
```

## Literals

```c
/* Integer literals */
int dec  = 255;        // decimal
int hex  = 0xFF;       // hexadecimal (0x prefix)
int oct  = 0377;       // octal (0 prefix) — easy to confuse!
int bin  = 0b11111111; // binary (C23 / GNU extension)

/* Integer suffixes */
unsigned int  u_val  = 100U;
long          l_val  = 100L;
unsigned long ul_val = 100UL;
long long     ll_val = 100LL;
unsigned long long ull_val = 100ULL;

/* Floating-point literals */
float  f1 = 3.14f;     // float (F or f suffix)
float  f2 = 3.14F;
double d1 = 3.14;      // double (default)
double d2 = 3.14e2;    // scientific: 314.0
double d3 = 0x1.fp4;   // hexadecimal float: 1.9375 * 2^4 = 31.0 (C99)
long double ld = 3.14L;

/* Character literals */
char  ch  = 'A';       // single character
char  esc = '\n';      // escape sequences: \n \t \r \\ \' \" \0
char  hex_ch = '\x41'; // hex escape: 'A'
char  oct_ch = '\101'; // octal escape: 'A'
wchar_t wch = L'€';    // wide character

/* String literals */
const char *str = "Hello, World!";    // null-terminated char array
const char *multiline =
    "Line one\n"
    "Line two\n";   // adjacent string literals are concatenated
```

:::caution[Octal Literals]
Any integer literal starting with `0` is **octal** in C. This is a common source of bugs: `int x = 010;` is **8**, not **10**. Avoid leading zeros in integer literals unless you intend octal.
:::

## printf / scanf Format Specifiers

```c
#include <stdio.h>
#include <stdint.h>

int main(void)
{
    int   i = 42;
    long  l = 1234567890L;
    long long ll = 9876543210LL;
    unsigned u = 0xDEAD;
    float  f = 3.14f;
    double d = 2.718281828;
    char   c = 'Z';
    char  *s = "hello";
    void  *p = &i;
    size_t sz = sizeof(int);

    /* Signed integers */
    printf("%d\n",   i);       // int (decimal)
    printf("%i\n",   i);       // int (decimal — same as %d for printf)
    printf("%ld\n",  l);       // long
    printf("%lld\n", ll);      // long long

    /* Unsigned integers */
    printf("%u\n",   u);       // unsigned int (decimal)
    printf("%o\n",   u);       // unsigned int (octal)
    printf("%x\n",   u);       // unsigned int (hex, lowercase)
    printf("%X\n",   u);       // unsigned int (hex, uppercase)
    printf("%lu\n",  (unsigned long)l);   // unsigned long
    printf("%llu\n", (unsigned long long)ll); // unsigned long long

    /* Floating-point */
    printf("%f\n",   d);       // decimal notation (default 6 decimal places)
    printf("%e\n",   d);       // scientific notation: 2.718282e+00
    printf("%g\n",   d);       // shorter of %f and %e
    printf("%.10f\n",d);       // 10 decimal places
    printf("%10.3f\n",d);      // width 10, 3 decimal places

    /* Other */
    printf("%c\n",   c);       // char
    printf("%s\n",   s);       // string (null-terminated)
    printf("%p\n",   p);       // pointer (hex address)
    printf("%zu\n",  sz);      // size_t
    printf("%%\n");            // literal percent sign

    /* scanf (read from stdin) */
    int n;
    double x;
    char buf[64];
    scanf("%d", &n);           // read int — NOTE: pass ADDRESS (&n)
    scanf("%lf", &x);          // read double — use %lf not %f for double!
    scanf("%63s", buf);        // read string, limit to 63 chars + null
    scanf(" %c", &c);          // space before %c skips whitespace

    return 0;
}
```

## Operators

### Arithmetic Operators

```c
int a = 17, b = 5;

int sum  = a + b;    // 22
int diff = a - b;    // 12
int prod = a * b;    // 85
int quot = a / b;    // 3 (integer division — truncates toward zero)
int rem  = a % b;    // 2 (remainder; sign matches dividend in C99+)

// Unary
int neg = -a;        // -17
int pos = +a;        // 17 (no-op)
int inc = ++a;       // pre-increment: a becomes 18, inc = 18
int dec = b--;       // post-decrement: dec = 5, b becomes 4
```

### Bitwise Operators

```c
unsigned int x = 0b10110011;  // 0xB3 = 179
unsigned int y = 0b01101010;  // 0x6A = 106

unsigned int and_  = x & y;   // 0b00100010 = AND each bit
unsigned int or_   = x | y;   // 0b11111011 = OR each bit
unsigned int xor_  = x ^ y;   // 0b11011001 = XOR each bit
unsigned int not_  = ~x;      // 0b01001100 = flip all bits
unsigned int shl   = x << 2;  // shift left 2  (multiply by 4)
unsigned int shr   = x >> 1;  // shift right 1 (divide by 2)

/* Common bit tricks */
// Test bit N
int bit_n_set = (x >> N) & 1;

// Set bit N
x |= (1u << N);

// Clear bit N
x &= ~(1u << N);

// Toggle bit N
x ^= (1u << N);

// Check if power of 2
bool is_pow2 = (x != 0) && ((x & (x - 1)) == 0);
```

:::caution[Shift Behavior]
Shifting a **signed** negative integer right is **implementation-defined**. Shifting by more than or equal to the bit width is **undefined behavior**. Always shift **unsigned** types for predictable results.
:::

### Logical and Comparison Operators

```c
int p = 1, q = 0;

// Logical — short-circuit evaluation
int land = p && q;    // 0 (false) — q not evaluated if p is false
int lor  = p || q;    // 1 (true)  — q not evaluated if p is true
int lnot = !p;        // 0

// Comparison — return 0 or 1 (int in C)
int eq  = (p == q);   // 0
int neq = (p != q);   // 1
int lt  = (p <  q);   // 0
int gt  = (p >  q);   // 1
int lte = (p <= q);   // 0
int gte = (p >= q);   // 1
```

### Assignment and Other Operators

```c
int x = 10;

/* Compound assignment */
x += 5;   // x = x + 5  = 15
x -= 3;   // x = x - 3  = 12
x *= 2;   // x = x * 2  = 24
x /= 4;   // x = x / 4  = 6
x %= 4;   // x = x % 4  = 2
x <<= 1;  // x = x << 1 = 4
x >>= 1;  // x = x >> 1 = 2
x &= 0xF; // x = x & 0xF= 2
x |= 0x8; // x = x | 8  = 10
x ^= 0x3; // x = x ^ 3  = 9

/* Ternary operator */
int max = (a > b) ? a : b;   // if(a>b) max=a else max=b

/* Comma operator — evaluates both, yields rightmost */
int y = (x = 5, x + 1);     // x = 5, y = 6 (rarely useful, avoid)

/* sizeof operator — compile-time, returns size_t */
size_t sz_int  = sizeof(int);
size_t sz_x    = sizeof x;    // no parens needed for variables
size_t sz_type = sizeof(double);

/* Compound literal (C99) */
int *arr = (int[]){ 1, 2, 3, 4, 5 };   // array of 5 ints
```

## Type Casting and Conversions

### Implicit Conversions (Integer Promotion)

```c
/* Integer promotions — types smaller than int are promoted to int */
char  c = 200;
short s = 1000;
int   i = c + s;    // c and s both promoted to int before addition

/* Arithmetic conversion (usual arithmetic conversions) */
// When operands have different types, they are converted to a common type:
// 1. If either is long double → both to long double
// 2. If either is double     → both to double
// 3. If either is float      → both to float
// 4. Integer promotions apply, then:
//    - If both same signedness → lesser rank converted
//    - Mixed sign: complex rules — avoid mixing signed/unsigned!

int    a = 10;
double d = a * 2.5;   // a promoted to double: 10.0 * 2.5 = 25.0

unsigned int u = 1;
int          n = -1;
// DANGER: -1 < 1 is true, but:
if (u + n > 0) { /* TRUE — n converted to unsigned! 0xFFFFFFFF */ }
```

### Explicit Casting

```c
double pi = 3.14159;
int    truncated = (int)pi;         // 3 — truncates, no round

int    a = 7, b = 2;
double ratio = (double)a / b;       // 3.5 — cast before division!
// Without cast: 7 / 2 = 3 (integer division)

char   c = (char)300;               // implementation-defined (overflow)
void  *ptr = malloc(sizeof(int));
int   *iptr = (int *)ptr;           // cast void* to concrete pointer type

// Cast to access bytes of any object (legal via unsigned char*)
int x = 0x12345678;
unsigned char *bytes = (unsigned char *)&x;
printf("%02X %02X %02X %02X\n",
       bytes[0], bytes[1], bytes[2], bytes[3]);   // endian-dependent
```

:::tip[Cast Arithmetic Before Division]
`(double)a / b` casts `a` to double before the division. `(double)(a / b)` divides as integers first, then converts — usually not what you want.
:::

## Integer Overflow

```c
#include <limits.h>

/* Unsigned overflow — well-defined: wraps modulo 2^N */
unsigned int u = UINT_MAX;   // 4294967295
u++;                          // 0 (wraps to 0 — defined behavior)

/* Signed overflow — UNDEFINED BEHAVIOR */
int s = INT_MAX;             // 2147483647
s++;                          // UB: do NOT rely on wrapping

/* Safe signed overflow check */
#include <stdint.h>
bool will_overflow_add(int a, int b)
{
    return (b > 0 && a > INT_MAX - b) ||
           (b < 0 && a < INT_MIN - b);
}

/* Or use compiler builtins (GCC/Clang) */
int result;
if (__builtin_add_overflow(a, b, &result)) {
    // overflow occurred
}
```

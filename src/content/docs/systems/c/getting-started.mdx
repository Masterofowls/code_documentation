# Getting Started with C

This guide covers everything you need to go from zero to a working C project: compiler flags, build systems, types, operators, and I/O.

## Compiler Command Line

### GCC / Clang Flags Reference

<Tabs>
  <TabItem label="GCC">
```bash
# Basic compilation
gcc hello.c -o hello

# Essential warning flags (always use these)

gcc -Wall -Wextra -Wpedantic hello.c -o hello

# With C11 standard

gcc -std=c11 -Wall -Wextra -Wpedantic hello.c -o hello

# Debug build (symbols, no optimization)

gcc -std=c11 -Wall -Wextra -g hello.c -o hello_debug

# Release build (optimize for speed)

gcc -std=c11 -O2 hello.c -o hello_release

# Maximum optimization (may increase binary size)

gcc -std=c11 -O3 -march=native hello.c -o hello_fast

# Compile to object file only (no linking)

gcc -std=c11 -c hello.c -o hello.o

# Link multiple object files

gcc main.o utils.o network.o -o myapp

# Link with a library (e.g., libm for math)

gcc main.c -o main -lm

# Sanitizers (invaluable for debugging)

gcc -std=c11 -g -fsanitize=address,undefined hello.c -o hello_asan

# Produce assembly output

gcc -std=c11 -S -O2 hello.c -o hello.s

# Show preprocessed output

gcc -E hello.c -o hello.i

# Define a macro on the command line

gcc -DDEBUG=1 -DVERSION=\"1.0\" hello.c -o hello

# Add include search path

gcc -I./include -I/usr/local/include hello.c -o hello

# Add library search path

gcc hello.c -L./lib -lmylib -o hello

````
</TabItem>
  <TabItem label="Clang">
```bash
# Clang accepts the same flags as GCC for most purposes
clang -std=c11 -Wall -Wextra -Wpedantic hello.c -o hello

# Clang has better error messages — try -fcolor-diagnostics
clang -std=c11 -Wall -fcolor-diagnostics hello.c -o hello

# Clang static analyzer
clang --analyze hello.c

# Generate LLVM IR
clang -std=c11 -S -emit-llvm hello.c -o hello.ll

# Sanitizers (same as GCC)
clang -std=c11 -g -fsanitize=address,undefined hello.c -o hello_asan

# Memory sanitizer (detects uninitialized reads — clang only)
clang -std=c11 -g -fsanitize=memory hello.c -o hello_msan
````

</TabItem>
  <TabItem label="MSVC">
```bash
# Basic compilation (from Developer Command Prompt)
cl hello.c

# With output name

cl hello.c /Fe:hello.exe

# C11 standard

cl /std:c11 hello.c /Fe:hello.exe

# Warning level 4 (highest practical level)

cl /W4 /std:c11 hello.c /Fe:hello.exe

# Debug build

cl /W4 /std:c11 /Zi /Od hello.c /Fe:hello_debug.exe

# Release build

cl /W4 /std:c11 /O2 hello.c /Fe:hello.exe

# Link with library

cl main.c /link ws2_32.lib

````
</TabItem>
</Tabs>

:::tip[Warning Flags are Mandatory]
Always compile with `-Wall -Wextra -Wpedantic`. These flags catch a huge class of bugs — uninitialized variables, implicit function declarations, sign comparison mismatches, and more. Treat warnings as errors in CI with `-Werror`.
:::

## Make

Make is the traditional build tool for C projects. A `Makefile` describes targets and their dependencies.

```makefile
# Makefile — a well-structured example

# Variables
CC      := gcc
CFLAGS  := -std=c11 -Wall -Wextra -Wpedantic
DBGFLAGS:= -g -fsanitize=address,undefined
RELFLAGS:= -O2 -DNDEBUG
LDFLAGS :=
LDLIBS  := -lm

# Directories
SRC_DIR := src
OBJ_DIR := build/obj
BIN_DIR := build/bin

# Source files (find all .c files in src/)
SRCS := $(wildcard $(SRC_DIR)/*.c)

# Object files (replace src/*.c with build/obj/*.o)
OBJS := $(SRCS:$(SRC_DIR)/%.c=$(OBJ_DIR)/%.o)

# Target name
TARGET := $(BIN_DIR)/myapp

# Default target
.PHONY: all
all: $(TARGET)

# Link step
$(TARGET): $(OBJS) | $(BIN_DIR)
	$(CC) $(LDFLAGS) $^ $(LDLIBS) -o $@

# Compile step — pattern rule
$(OBJ_DIR)/%.o: $(SRC_DIR)/%.c | $(OBJ_DIR)
	$(CC) $(CFLAGS) $(RELFLAGS) -c $< -o $@

# Create directories if they don't exist
$(OBJ_DIR) $(BIN_DIR):
	mkdir -p $@

# Debug build
.PHONY: debug
debug: CFLAGS += $(DBGFLAGS)
debug: $(TARGET)

# Clean build artifacts
.PHONY: clean
clean:
	rm -rf build/

# Run the program
.PHONY: run
run: all
	./$(TARGET)

# Auto-generate dependency files (tracks #include changes)
DEPS := $(OBJS:.o=.d)
-include $(DEPS)
$(OBJ_DIR)/%.o: $(SRC_DIR)/%.c | $(OBJ_DIR)
	$(CC) $(CFLAGS) -MMD -MP -c $< -o $@
````

Key Make concepts:

- `$@` — The target name
- `$<` — The first prerequisite
- `$^` — All prerequisites
- `.PHONY` — Targets that are not files
- `| $(OBJ_DIR)` — Order-only prerequisite (create dir but don't rebuild if dir changes)

## CMake Basics

CMake generates platform-native build files (Makefiles on Linux, .vcxproj on Windows, etc.).

```cmake
# CMakeLists.txt — minimal but correct

cmake_minimum_required(VERSION 3.20)
project(MyApp VERSION 1.0.0 LANGUAGES C)

# Set C standard globally
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)  # Disable GNU extensions for portability

# Collect source files
file(GLOB_RECURSE SOURCES "src/*.c")

# Create executable
add_executable(myapp ${SOURCES})

# Warning flags (per-target, not global)
target_compile_options(myapp PRIVATE
    $<$<C_COMPILER_ID:GNU,Clang>:-Wall -Wextra -Wpedantic>
    $<$<C_COMPILER_ID:MSVC>:/W4>
)

# Include directories
target_include_directories(myapp PRIVATE include)

# Link libraries
target_link_libraries(myapp PRIVATE m)  # libm

# Debug/Release configurations
target_compile_options(myapp PRIVATE
    $<$<CONFIG:Debug>:-g -fsanitize=address,undefined>
    $<$<CONFIG:Release>:-O2 -DNDEBUG>
)
target_link_options(myapp PRIVATE
    $<$<CONFIG:Debug>:-fsanitize=address,undefined>
)
```

```bash
# Build with CMake
mkdir build && cd build
cmake .. -DCMAKE_BUILD_TYPE=Release
cmake --build . --parallel

# Debug build
cmake .. -DCMAKE_BUILD_TYPE=Debug
cmake --build .
```

## Data Types

### Fundamental Types

```c
#include <stdio.h>
#include <limits.h>   // INT_MAX, CHAR_MIN, etc.
#include <float.h>    // FLT_MAX, DBL_EPSILON, etc.
#include <stdint.h>   // Fixed-width integer types
#include <stdbool.h>  // bool, true, false (C99+)

int main(void) {
    // Character types
    char c = 'A';               // Signed or unsigned — implementation-defined
    signed char sc = -128;      // Always signed, -128..127
    unsigned char uc = 255;     // Always unsigned, 0..255

    // Integer types (minimum sizes — actual sizes platform-dependent)
    short s = 32767;            // At least 16 bits
    int i = 2147483647;         // At least 16 bits (typically 32)
    long l = 2147483647L;       // At least 32 bits
    long long ll = 9223372036854775807LL; // At least 64 bits

    // Unsigned variants
    unsigned short us = 65535U;
    unsigned int ui = 4294967295U;
    unsigned long ul = 4294967295UL;
    unsigned long long ull = 18446744073709551615ULL;

    // Floating point
    float f = 3.14159265f;      // Single precision, ~7 decimal digits
    double d = 3.14159265358979; // Double precision, ~15 decimal digits
    long double ld = 3.14159265358979323846L; // Extended, 80-bit on x86

    // Boolean (C99+, include stdbool.h)
    bool flag = true;
    bool other = false;

    // Print sizes on this platform
    printf("char: %zu bytes\n", sizeof(char));       // Always 1
    printf("short: %zu bytes\n", sizeof(short));     // Usually 2
    printf("int: %zu bytes\n", sizeof(int));         // Usually 4
    printf("long: %zu bytes\n", sizeof(long));       // 4 (Windows) or 8 (Linux)
    printf("long long: %zu bytes\n", sizeof(long long)); // Always 8
    printf("float: %zu bytes\n", sizeof(float));     // Usually 4
    printf("double: %zu bytes\n", sizeof(double));   // Usually 8
    printf("pointer: %zu bytes\n", sizeof(void*));   // 8 on 64-bit

    // Limits
    printf("INT_MAX = %d\n", INT_MAX);       // 2147483647
    printf("INT_MIN = %d\n", INT_MIN);       // -2147483648
    printf("UINT_MAX = %u\n", UINT_MAX);     // 4294967295
    printf("LLONG_MAX = %lld\n", LLONG_MAX); // 9223372036854775807
    printf("DBL_EPSILON = %e\n", DBL_EPSILON); // ~2.22e-16

    (void)c; (void)sc; (void)uc;
    (void)s; (void)i; (void)l; (void)ll;
    (void)us; (void)ui; (void)ul; (void)ull;
    (void)f; (void)d; (void)ld;
    (void)flag; (void)other;
    return 0;
}
```

### Fixed-Width Integer Types (stdint.h)

Always use these when you need a specific bit width:

```c
#include <stdint.h>
#include <inttypes.h>  // PRId32, PRIu64, etc. for printf

// Exact-width types (may not exist on all platforms)
int8_t   i8  = -128;
uint8_t  u8  = 255;
int16_t  i16 = -32768;
uint16_t u16 = 65535;
int32_t  i32 = -2147483648;
uint32_t u32 = 4294967295U;
int64_t  i64 = INT64_MIN;
uint64_t u64 = UINT64_MAX;

// Minimum-width types (guaranteed to be AT LEAST N bits)
int_least8_t  least8  = 100;
uint_least16_t least16 = 50000;

// Fastest types at least N bits wide
int_fast8_t  fast8  = 42;
uint_fast32_t fast32 = 1000000;

// Pointer-sized integers
intptr_t  iptr = (intptr_t)NULL;
uintptr_t uptr = (uintptr_t)NULL;
ptrdiff_t diff = 0;  // Result of pointer subtraction
size_t    sz   = 0;  // sizeof result, array indices

// Printf format specifiers from inttypes.h
printf("%" PRId32 "\n", i32);   // %d equivalent for int32_t
printf("%" PRIu64 "\n", u64);   // %llu equivalent for uint64_t
printf("%" PRIx32 "\n", u32);   // Hex for uint32_t
```

### Literals and Suffixes

```c
// Integer literals
int decimal     = 255;       // Base 10
int octal       = 0377;      // Base 8 (leading zero)
int hex         = 0xFF;      // Base 16 (0x or 0X prefix)
int binary      = 0b11111111; // Base 2 (C23 — 0b or 0B prefix)

// Integer suffixes
unsigned int  u_val  = 255U;   // U or u — unsigned
long          l_val  = 255L;   // L or l — long
unsigned long ul_val = 255UL;  // UL — unsigned long
long long     ll_val = 255LL;  // LL — long long
unsigned long long ull_val = 255ULL; // ULL

// Floating point literals
float  f_val  = 3.14f;     // f or F — float
double d_val  = 3.14;      // No suffix — double (default)
double hex_fp = 0x1.8p+1;  // Hex float: 1.5 * 2^1 = 3.0 (C99+)

// Character literals
char newline = '\n';
char tab     = '\t';
char null    = '\0';
char bell    = '\a';
char bs      = '\b';
char cr      = '\r';
char backsl  = '\\';
char quote   = '\'';
char hex_c   = '\x41';  // 'A'
char oct_c   = '\101';  // 'A'

// String literals are null-terminated char arrays
const char *str = "Hello, World!";
// Adjacent string literals are concatenated by the preprocessor:
const char *long_str = "This is a very "
                        "long string "
                        "split across lines.";
```

## printf / scanf Format Specifiers

```c
#include <stdio.h>
#include <stdint.h>

int main(void) {
    int i = 42;
    unsigned u = 4294967295U;
    long l = 1234567890L;
    long long ll = 9876543210LL;
    float f = 3.14159f;
    double d = 2.718281828;
    char c = 'X';
    char *s = "hello";
    void *p = &i;
    size_t sz = sizeof(int);

    // Integer formats
    printf("%d\n",   i);    // Signed decimal int
    printf("%i\n",   i);    // Same as %d
    printf("%u\n",   u);    // Unsigned decimal
    printf("%o\n",   u);    // Unsigned octal
    printf("%x\n",   u);    // Unsigned hex (lowercase)
    printf("%X\n",   u);    // Unsigned hex (uppercase)
    printf("%ld\n",  l);    // Long decimal
    printf("%lld\n", ll);   // Long long decimal
    printf("%llu\n", (unsigned long long)ll); // Unsigned long long

    // Floating point formats
    printf("%f\n",   d);    // Decimal notation: 2.718282
    printf("%e\n",   d);    // Scientific: 2.718282e+00
    printf("%E\n",   d);    // Scientific uppercase: 2.718282E+00
    printf("%g\n",   d);    // Shorter of %f or %e
    printf("%a\n",   d);    // Hex float (C99+)
    printf("%.2f\n", d);    // 2 decimal places: 2.72
    printf("%10.3f\n", d);  // Width 10, 3 decimals:      2.718
    printf("%-10.3f|\n", d); // Left-aligned: 2.718     |

    // Character and string
    printf("%c\n",   c);    // Character: X
    printf("%s\n",   s);    // String: hello
    printf("%.3s\n", s);    // String truncated to 3: hel
    printf("%10s\n", s);    // Right-aligned in width 10:      hello

    // Pointer and special
    printf("%p\n",   p);    // Pointer address
    printf("%zu\n",  sz);   // size_t (z modifier)
    printf("%td\n", (ptrdiff_t)0); // ptrdiff_t (t modifier)
    printf("%%\n");          // Literal percent sign

    // scanf reads formatted input
    int n;
    float x;
    char buf[64];
    printf("Enter int float string: ");
    // ALWAYS check scanf return value (number of items read)
    if (scanf("%d %f %63s", &n, &x, buf) == 3) {
        printf("Read: %d, %.2f, %s\n", n, x, buf);
    }

    return 0;
}
```

## Operators

### Arithmetic Operators

```c
int a = 17, b = 5;
int sum  = a + b;   // 22
int diff = a - b;   // 12
int prod = a * b;   // 85
int quot = a / b;   // 3  (integer division truncates toward zero)
int rem  = a % b;   // 2  (remainder — same sign as dividend)
int neg  = -a;      // -17 (unary minus)

// Integer overflow: SIGNED overflow is UB; UNSIGNED wraps modulo 2^N
unsigned int x = UINT_MAX;
x++;  // x == 0 (wraps, defined)
int y = INT_MAX;
y++;  // UB! Don't do this.

// Floating point arithmetic
double fa = 1.0 / 3.0;  // 0.333...  (not exact in binary)
double fb = 0.1 + 0.2;  // Not exactly 0.3!
// Never compare floats with ==; use epsilon comparison:
#include <math.h>
if (fabs(fb - 0.3) < 1e-9) { /* equal enough */ }
```

### Bitwise Operators

```c
unsigned int a = 0b10110100;  // 180
unsigned int b = 0b11001010;  // 202

unsigned int and_result  = a & b;   // 0b10000000 = 128  (AND)
unsigned int or_result   = a | b;   // 0b11111110 = 254  (OR)
unsigned int xor_result  = a ^ b;   // 0b01111110 = 126  (XOR)
unsigned int not_result  = ~a;      // 0b01001011 = 75   (bitwise NOT)
unsigned int shl_result  = a << 2;  // 0b1011010000 = 720 (left shift)
unsigned int shr_result  = a >> 2;  // 0b00101101 = 45  (right shift)

// Common bit tricks
#define SET_BIT(n, bit)    ((n) |=  (1U << (bit)))
#define CLEAR_BIT(n, bit)  ((n) &= ~(1U << (bit)))
#define TOGGLE_BIT(n, bit) ((n) ^=  (1U << (bit)))
#define TEST_BIT(n, bit)   (((n) >> (bit)) & 1U)

unsigned int flags = 0;
SET_BIT(flags, 3);        // Set bit 3: flags = 0b00001000 = 8
TEST_BIT(flags, 3);       // 1 (true)
CLEAR_BIT(flags, 3);      // Clear bit 3: flags = 0
TOGGLE_BIT(flags, 0);     // Toggle bit 0: flags = 1

// Check if power of 2
int is_pow2(unsigned int n) { return n && !(n & (n - 1)); }

// Round up to next power of 2
unsigned next_pow2(unsigned n) {
    n--;
    n |= n >> 1; n |= n >> 2; n |= n >> 4; n |= n >> 8; n |= n >> 16;
    return ++n;
}
```

### All Operators by Category

```c
// Assignment operators
int x = 5;
x += 3;   // x = x + 3 = 8
x -= 2;   // x = x - 2 = 6
x *= 4;   // x = x * 4 = 24
x /= 6;   // x = x / 6 = 4
x %= 3;   // x = x % 3 = 1
x <<= 2;  // x = x << 2 = 4
x >>= 1;  // x = x >> 1 = 2
x &= 0xF; // x = x & 0xF = 2
x |= 0x8; // x = x | 0x8 = 10
x ^= 0x5; // x = x ^ 0x5 = 15

// Increment/decrement
int a = 5;
int pre_inc = ++a;  // a becomes 6, pre_inc = 6
int post_inc = a++; // post_inc = 6, a becomes 7
int pre_dec = --a;  // a becomes 6, pre_dec = 6
int post_dec = a--; // post_dec = 6, a becomes 5

// Comparison (return int: 0 or 1)
int eq  = (a == b);
int neq = (a != b);
int lt  = (a < b);
int gt  = (a > b);
int lte = (a <= b);
int gte = (a >= b);

// Logical (short-circuit evaluation)
int land = (a > 0) && (b > 0);  // Both must be true
int lor  = (a > 0) || (b > 0);  // At least one true
int lnot = !(a > 0);            // Negation

// Ternary (conditional expression)
int max = (a > b) ? a : b;
const char *sign = (x > 0) ? "positive" : (x < 0) ? "negative" : "zero";

// Comma operator (evaluate left, discard, return right)
int val = (printf("side effect\n"), 42);  // val = 42

// sizeof operator (evaluated at compile time for types)
size_t int_size  = sizeof(int);       // 4
size_t arr_size  = sizeof(int[10]);   // 40
size_t expr_size = sizeof(1 + 2.0);   // sizeof(double) = 8
```

### Type Casting and Promotion

```c
// Implicit promotions (the "usual arithmetic conversions")
// Short/char are promoted to int before arithmetic
char c = 100;
short s = 200;
int result = c + s;  // Both promoted to int before addition

// Integer promotions in expressions
unsigned char uc = 255;
int promoted = uc + 1;  // uc promoted to int, result is 256 (not 0!)

// When mixing signed and unsigned: signed converts to unsigned
int neg = -1;
unsigned pos = 1;
// -1 < 1 is TRUE, but:
if (neg < pos) { /* This branch is NOT taken! -1 converts to UINT_MAX */ }

// Explicit casts
double d = 3.99;
int truncated = (int)d;          // 3 (truncation, not rounding)
float f = (float)(1.0 / 3.0);   // Precision lost
unsigned u = (unsigned)(-1);     // UINT_MAX
char ch = (char)300;             // Implementation-defined (300 > CHAR_MAX)

// Safe integer conversion check
#include <limits.h>
long big = 1000000L;
if (big <= INT_MAX && big >= INT_MIN) {
    int safe = (int)big;  // Safe to cast
    (void)safe;
}
```

:::caution[Signed/Unsigned Mixing]
Mixing signed and unsigned integers in comparisons is one of the most common C bugs. Enable `-Wsign-compare` (included in `-Wextra`) to catch these. When in doubt, use explicit casts or use the same type throughout.
:::

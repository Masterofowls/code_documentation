# Memory Management in C

C gives you direct control over memory — the most powerful and most dangerous feature of the language. Understanding where your data lives, how long it lives, and how to manage it correctly is the foundation of writing correct, fast C programs.

## Memory Layout of a C Program

When a C program runs, the OS maps virtual memory into several distinct regions:

```
High addresses
┌─────────────────────────────┐
│        Stack                │  ← grows downward; automatic variables, return addresses
│         ↓                   │
│   (unmapped / guard page)   │
│         ↑                   │
│        Heap                 │  ← grows upward; malloc/calloc/realloc allocations
├─────────────────────────────┤
│    BSS Segment              │  ← zero-initialized global/static data
├─────────────────────────────┤
│    Data Segment             │  ← initialized global/static data
├─────────────────────────────┤
│    Text Segment             │  ← read-only; compiled machine code + string literals
└─────────────────────────────┘
Low addresses
```

```c
#include <stdio.h>
#include <stdlib.h>

// Text segment: the code itself
// Data segment: initialized statics
int global_init = 42;          // data segment
static float static_init = 3.14f;  // data segment

// BSS segment: zero-initialized
int global_zero;               // BSS — guaranteed to be 0
static char static_zero[100];  // BSS

int main(void)
{
    int stack_var = 10;        // stack
    int *heap_var = malloc(4); // heap

    printf("stack:  %p\n", (void*)&stack_var);
    printf("heap:   %p\n", (void*)heap_var);
    printf("global: %p\n", (void*)&global_init);
    printf("bss:    %p\n", (void*)&global_zero);
    // Typical output: stack > heap > global > bss

    free(heap_var);
    return 0;
}
```

## Storage Duration

| Duration | Where | Lifetime | Examples |
|----------|-------|----------|---------|
| **Automatic** | Stack | From declaration to end of enclosing block | Local variables, function parameters |
| **Static** | Data/BSS | Entire program duration | Global variables, `static` local variables |
| **Thread-local** | Per-thread stack | Thread's lifetime | `_Thread_local` variables (C11) |
| **Allocated** | Heap | From `malloc` until `free` | Dynamic arrays, linked lists, trees |

```c
#include <stdio.h>

void demonstrate_static_local(void)
{
    // Static local: lives in BSS/data, initialized once, persists across calls
    static int count = 0;
    count++;
    printf("called %d times\n", count);
}

int main(void)
{
    demonstrate_static_local();  // 1
    demonstrate_static_local();  // 2
    demonstrate_static_local();  // 3
    return 0;
}
```

## Stack Frames

Each function call creates a **stack frame** (activation record) on the stack:

```
Stack frame for function foo(int a, int b):
┌─────────────────────┐  ← high address (frame bottom)
│  Return address     │  where to jump back to when foo() returns
│  Saved registers    │  callee-saved registers (rbx, r12, etc.)
│  Parameter a        │  (on x86-64 first 6 params are in registers)
│  Parameter b        │
│  Local variable x   │
│  Local variable arr │  ← arr[0]
│  ...                │
└─────────────────────┘  ← low address (stack grows down)
```

```c
#include <stdio.h>

void leaf(void)
{
    int x = 999;
    printf("leaf: &x = %p\n", (void*)&x);
}

void middle(void)
{
    int y = 42;
    printf("middle: &y = %p\n", (void*)&y);
    leaf();
}

int main(void)
{
    int z = 1;
    printf("main: &z = %p\n", (void*)&z);
    middle();
    return 0;
}
// Address of z > y > x — each is deeper (lower address) in the call chain
```

## Dynamic Memory: malloc, calloc, realloc, free

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(void)
{
    /* malloc — allocate N bytes, UNINITIALIZED (may contain garbage) */
    int *arr = malloc(10 * sizeof(int));
    if (arr == NULL) {
        fprintf(stderr, "malloc failed\n");
        return 1;
    }

    /* calloc — allocate N elements of size M, ZERO-INITIALIZED */
    double *data = calloc(20, sizeof(double));
    if (data == NULL) {
        free(arr);
        fprintf(stderr, "calloc failed\n");
        return 1;
    }
    // data[0..19] are all 0.0

    /* realloc — resize an existing allocation */
    // Returns new pointer (may or may not be same address)
    // If it fails, returns NULL — the ORIGINAL pointer is still valid!
    int *bigger = realloc(arr, 20 * sizeof(int));
    if (bigger == NULL) {
        // arr is STILL VALID here — don't lose it!
        free(arr);
        free(data);
        fprintf(stderr, "realloc failed\n");
        return 1;
    }
    arr = bigger;   // now arr points to the larger buffer

    /* Using the memory */
    for (int i = 0; i < 20; i++) arr[i] = i * i;

    /* free — release heap memory */
    free(arr);    // always free what you malloc/calloc/realloc
    free(data);
    // After free: arr and data are DANGLING POINTERS
    arr  = NULL;  // neutralize
    data = NULL;

    /* free(NULL) is always safe — it's a no-op */
    free(NULL);

    return 0;
}
```

:::caution[realloc Anti-Pattern]
Never write `ptr = realloc(ptr, new_size)`. If `realloc` fails, it returns `NULL` and the original allocation is still valid — but you've just lost the only pointer to it. Always assign to a temporary variable first.
:::

### Safe realloc Pattern

```c
#include <stdlib.h>
#include <errno.h>

// Returns new pointer on success; returns NULL and preserves old_ptr on failure
void *safe_realloc(void *old_ptr, size_t new_size)
{
    void *new_ptr = realloc(old_ptr, new_size);
    if (new_ptr == NULL && new_size > 0) {
        // old_ptr is still valid
        return NULL;
    }
    return new_ptr;
}

// Dynamic array growth pattern
typedef struct {
    int    *data;
    size_t  count;
    size_t  capacity;
} IntVec;

int vec_push(IntVec *v, int value)
{
    if (v->count == v->capacity) {
        size_t new_cap = (v->capacity == 0) ? 8 : v->capacity * 2;
        int *new_data = safe_realloc(v->data, new_cap * sizeof(int));
        if (!new_data) return -1;   // out of memory
        v->data     = new_data;
        v->capacity = new_cap;
    }
    v->data[v->count++] = value;
    return 0;
}
```

## Alignment and aligned_alloc (C11)

Data types have alignment requirements — they must be stored at addresses that are multiples of their size (on most architectures):

```c
#include <stdalign.h>  // alignas, alignof
#include <stdlib.h>
#include <stdio.h>

// SIMD types often require 32-byte alignment
typedef struct {
    alignas(32) float data[8];  // aligned to 32-byte boundary
} AVXVec;

int main(void)
{
    printf("alignof(char)   = %zu\n", alignof(char));    // 1
    printf("alignof(int)    = %zu\n", alignof(int));     // 4
    printf("alignof(double) = %zu\n", alignof(double));  // 8
    printf("alignof(AVXVec) = %zu\n", alignof(AVXVec));  // 32

    // aligned_alloc(alignment, size) — C11
    // alignment must be a power of 2; size must be multiple of alignment
    float *simd_buf = aligned_alloc(32, 32 * sizeof(float));
    if (!simd_buf) { /* handle error */ }
    printf("simd_buf address: %p (should end in 0)\n", (void*)simd_buf);
    free(simd_buf);  // free works on aligned_alloc memory

    // Stack-aligned variable
    alignas(64) char cache_line[64];  // 64-byte cache line aligned
    printf("cache_line: %p\n", (void*)cache_line);

    return 0;
}
```

### Struct Padding and Memory Layout

```c
#include <stddef.h>
#include <stdio.h>

// Poorly ordered — lots of padding
struct Padded {
    char   a;    // 1 byte + 3 bytes padding
    int    b;    // 4 bytes
    char   c;    // 1 byte + 7 bytes padding
    double d;    // 8 bytes
};               // total: 24 bytes

// Well-ordered — minimal padding
struct Packed {
    double d;    // 8 bytes
    int    b;    // 4 bytes
    char   a;    // 1 byte
    char   c;    // 1 byte + 2 bytes padding
};               // total: 16 bytes

int main(void)
{
    printf("sizeof(struct Padded) = %zu\n", sizeof(struct Padded)); // 24
    printf("sizeof(struct Packed) = %zu\n", sizeof(struct Packed)); // 16

    // offsetof — byte offset of a member
    printf("offsetof Padded.b = %zu\n", offsetof(struct Padded, b)); // 4
    printf("offsetof Padded.d = %zu\n", offsetof(struct Padded, d)); // 16

    // Force no padding — NOT portable, but common for network/file formats
    #pragma pack(push, 1)
    struct NoPadding {
        char   a;
        int    b;
        double d;
    };
    #pragma pack(pop)
    printf("sizeof(NoPadding) = %zu\n", sizeof(struct NoPadding));  // 13

    return 0;
}
```

## Memory Errors and Detection Tools

### Common Memory Errors

```c
#include <stdlib.h>
#include <string.h>

void buffer_overflow(void)
{
    char buf[8];
    strcpy(buf, "This string is way too long!");  // OVERFLOW
    // Use strncpy(buf, src, sizeof(buf) - 1); buf[sizeof(buf)-1] = '\0';
}

void heap_buffer_overflow(void)
{
    int *p = malloc(5 * sizeof(int));
    p[5] = 99;   // one past end — heap corruption
    free(p);
}

void use_after_free_example(void)
{
    int *p = malloc(sizeof(int));
    *p = 42;
    free(p);
    printf("%d\n", *p);  // USE-AFTER-FREE — undefined behavior
}

void memory_leak(void)
{
    for (int i = 0; i < 1000; i++) {
        int *p = malloc(sizeof(int));
        *p = i;
        // forgot to free(p) — leak 4000 bytes per call!
    }
}

void stack_overflow_risk(int depth)
{
    char big_array[1024 * 1024];  // 1 MB on stack each call
    if (depth > 0) stack_overflow_risk(depth - 1);
}
```

### Detection with Valgrind

```bash
# Compile with debug symbols
gcc -g -O0 main.c -o program

# Full leak check
valgrind --leak-check=full \
         --show-leak-kinds=all \
         --track-origins=yes \
         --verbose \
         ./program

# Valgrind output categories:
# "definitely lost" — memory you allocated and lost the pointer to
# "indirectly lost" — memory reachable only through other lost blocks
# "possibly lost"   — pointer to middle of allocation (bad practice)
# "still reachable" — allocated but not freed at exit (often acceptable)
```

### AddressSanitizer (ASan)

```bash
# Compile with ASan
gcc -fsanitize=address,undefined -g -fno-omit-frame-pointer main.c -o program_asan

# Run normally — ASan intercepts allocations
./program_asan

# Environment variables
ASAN_OPTIONS=detect_leaks=1:abort_on_error=1 ./program_asan
```

## Flexible Array Members (C99)

```c
#include <stdlib.h>
#include <stdio.h>
#include <string.h>

// Flexible array member — must be last; struct has incomplete size
typedef struct {
    size_t length;
    char   data[];   // zero-length array (flexible)
} String;

String *create_string(const char *s)
{
    size_t len = strlen(s);
    // Allocate struct + space for the string data
    String *str = malloc(sizeof(String) + len + 1);
    if (!str) return NULL;
    str->length = len;
    memcpy(str->data, s, len + 1);  // copy including null terminator
    return str;
}

int main(void)
{
    String *s = create_string("Hello, World!");
    printf("length=%zu, data=%s\n", s->length, s->data);
    free(s);  // one free for both header and data
    return 0;
}
```

## Arena Allocator Pattern

An arena (also called a linear or bump allocator) allocates by incrementing a pointer. **Freeing is instant** — just reset the pointer. No per-object overhead.

```c
#include <stdlib.h>
#include <stdio.h>
#include <string.h>
#include <stdint.h>
#include <stdalign.h>

#define ARENA_SIZE (1024 * 1024)  // 1 MB

typedef struct {
    unsigned char *base;
    size_t         used;
    size_t         capacity;
} Arena;

static Arena arena_init(size_t size)
{
    Arena a;
    a.base     = malloc(size);
    a.used     = 0;
    a.capacity = size;
    return a;
}

// Aligned bump allocation
static void *arena_alloc(Arena *a, size_t size, size_t align)
{
    // Align the current offset
    size_t offset = (a->used + align - 1) & ~(align - 1);
    if (offset + size > a->capacity) return NULL;  // out of memory
    a->used = offset + size;
    return a->base + offset;
}

static void arena_reset(Arena *a)
{
    a->used = 0;   // "free" everything — O(1)
}

static void arena_destroy(Arena *a)
{
    free(a->base);
    a->base = NULL;
}

// Convenience macros
#define ARENA_ALLOC(arena, T) \
    ((T *)arena_alloc((arena), sizeof(T), alignof(T)))
#define ARENA_ALLOC_N(arena, T, n) \
    ((T *)arena_alloc((arena), sizeof(T) * (n), alignof(T)))

int main(void)
{
    Arena a = arena_init(ARENA_SIZE);

    // Allocate many objects — no individual frees needed
    int *nums = ARENA_ALLOC_N(&a, int, 100);
    for (int i = 0; i < 100; i++) nums[i] = i;

    double *vals = ARENA_ALLOC_N(&a, double, 50);
    for (int i = 0; i < 50; i++) vals[i] = (double)i * 0.1;

    printf("Arena used: %zu / %zu bytes\n", a.used, a.capacity);

    // "Free" all at once
    arena_reset(&a);
    printf("Arena reset: used = %zu\n", a.used);  // 0

    arena_destroy(&a);
    return 0;
}
```

:::tip[When to Use an Arena]
Arenas excel for **request-scoped** allocations (e.g., parse a JSON document, process it, free everything at once), **frame allocations** in games (allocate scratch memory per frame, reset at frame end), and **compiler symbol tables**.
:::

## Cache-Friendly Data Layouts

```c
// SLOW: Array of Structs (AoS) — poor cache utilization when only one field accessed
typedef struct {
    float x, y, z;  // position
    float vx, vy, vz; // velocity
    float mass;
} Particle_AoS;

Particle_AoS particles_aos[10000];

void update_positions_aos(float dt)
{
    for (int i = 0; i < 10000; i++) {
        // Loads 28 bytes per particle, uses 12 bytes of position
        // Cache line: 64 bytes — fits ~2 particles
        particles_aos[i].x += particles_aos[i].vx * dt;
        particles_aos[i].y += particles_aos[i].vy * dt;
        particles_aos[i].z += particles_aos[i].vz * dt;
    }
}

// FAST: Struct of Arrays (SoA) — all x values contiguous
typedef struct {
    float x[10000], y[10000], z[10000];    // positions
    float vx[10000], vy[10000], vz[10000]; // velocities
    float mass[10000];
} Particles_SoA;

Particles_SoA particles_soa;

void update_positions_soa(float dt)
{
    // SIMD-friendly: operates on 16 floats per cache line
    for (int i = 0; i < 10000; i++) {
        particles_soa.x[i] += particles_soa.vx[i] * dt;
        particles_soa.y[i] += particles_soa.vy[i] * dt;
        particles_soa.z[i] += particles_soa.vz[i] * dt;
    }
}
```

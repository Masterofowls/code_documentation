# Memory Management in C

C gives you complete control over memory. This power demands responsibility: manual allocation and deallocation, awareness of lifetime, and vigilance against the many ways memory management can go wrong.

## Memory Layout of a C Program

A running C program's virtual address space is divided into segments:

```
High addresses
┌──────────────────────────────────────┐
│              Stack                   │ ← grows downward
│   (local variables, function frames) │
├──────────────────────────────────────┤
│                ↓                     │
│             (gap)                    │
│                ↑                     │
├──────────────────────────────────────┤
│              Heap                    │ ← grows upward
│   (malloc/calloc/realloc memory)     │
├──────────────────────────────────────┤
│         BSS Segment                  │
│   (uninitialized global/static vars) │
├──────────────────────────────────────┤
│         Data Segment                 │
│   (initialized global/static vars)   │
├──────────────────────────────────────┤
│         Text Segment                 │
│   (compiled code — read-only)        │
└──────────────────────────────────────┘
Low addresses
```

```c
#include <stdio.h>
#include <stdlib.h>

// Text segment: the function code itself is here

// Data segment: initialized global/static
int global_init = 42;
static double pi = 3.14159;

// BSS segment: uninitialized (zero-initialized by OS at start)
int global_uninit;
static char big_buffer[4096];

int main(void) {
    // Stack: local variables
    int stack_var = 100;
    char stack_arr[64];

    // Heap: dynamically allocated
    int *heap_var = malloc(sizeof(int));
    *heap_var = 200;

    printf("Text (main):        %p\n", (void*)main);
    printf("Data (global_init): %p\n", (void*)&global_init);
    printf("BSS  (global_unin): %p\n", (void*)&global_uninit);
    printf("Stack (stack_var):  %p\n", (void*)&stack_var);
    printf("Heap (heap_var):    %p\n", (void*)heap_var);

    free(heap_var);
    (void)stack_arr;
    (void)big_buffer;
    (void)pi;
    return 0;
}
```

## Storage Duration

C has three storage durations:

| Duration      | Where    | Lifetime                                   | Example                      |
| ------------- | -------- | ------------------------------------------ | ---------------------------- |
| **Automatic** | Stack    | From declaration to end of enclosing block | Local variables              |
| **Static**    | Data/BSS | Entire program lifetime                    | Global vars, `static` locals |
| **Dynamic**   | Heap     | From `malloc` to `free`                    | Explicitly allocated         |

```c
#include <stdio.h>
#include <stdlib.h>

// Static storage duration (file scope)
static int call_count = 0;  // Persists forever; zero-initialized

void demonstrate_storage(void) {
    // Automatic storage duration
    int auto_var = 42;       // Created on stack entry, destroyed on exit
    int auto_arr[10];        // Uninitialized — contains garbage

    // Static local variable: persists across calls
    static int static_local = 0;  // Initialized once, persists
    static_local++;
    call_count++;

    printf("auto_var=%d, static_local=%d, call_count=%d\n",
           auto_var, static_local, call_count);

    // Dynamic storage duration
    int *dyn = malloc(sizeof(int));
    if (dyn) {
        *dyn = 99;
        printf("heap value: %d\n", *dyn);
        free(dyn);  // Must free explicitly
    }
}

int main(void) {
    demonstrate_storage();  // auto_var=42, static_local=1, call_count=1
    demonstrate_storage();  // auto_var=42, static_local=2, call_count=2
    demonstrate_storage();  // auto_var=42, static_local=3, call_count=3
    return 0;
}
```

## Dynamic Memory Allocation

### malloc — Allocate Uninitialized Memory

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(void) {
    // malloc(size): returns void* to at least 'size' bytes, or NULL on failure
    // Contents are UNINITIALIZED (may contain garbage)
    int *arr = malloc(10 * sizeof(int));

    // ALWAYS check for NULL
    if (arr == NULL) {
        fprintf(stderr, "malloc failed: out of memory\n");
        return 1;
    }

    // Initialize manually
    for (int i = 0; i < 10; i++) arr[i] = i * i;

    printf("arr[5] = %d\n", arr[5]);  // 25

    // Free when done
    free(arr);
    arr = NULL;  // Good practice: prevent use-after-free

    return 0;
}
```

### calloc — Allocate Zero-Initialized Memory

```c
// calloc(count, element_size): allocates count*element_size bytes, zero-initialized
// Safer than malloc + memset for large allocations (can detect overflow)
int *zeroed = calloc(100, sizeof(int));
if (!zeroed) { /* handle error */ }

// All elements are guaranteed to be zero
printf("zeroed[50] = %d\n", zeroed[50]);  // 0

free(zeroed);
```

### realloc — Resize Allocated Memory

```c
#include <stdlib.h>
#include <stdio.h>

// Dynamic array with realloc
typedef struct {
    int *data;
    size_t size;
    size_t capacity;
} IntVec;

void vec_push(IntVec *v, int val) {
    if (v->size == v->capacity) {
        // Double capacity
        size_t new_cap = v->capacity ? v->capacity * 2 : 4;

        // CRITICAL: don't assign realloc result directly to v->data
        // If realloc fails, it returns NULL and the original pointer is lost!
        int *new_data = realloc(v->data, new_cap * sizeof(int));
        if (!new_data) {
            fprintf(stderr, "realloc failed\n");
            return;
        }
        v->data     = new_data;
        v->capacity = new_cap;
    }
    v->data[v->size++] = val;
}

void vec_free(IntVec *v) {
    free(v->data);
    v->data     = NULL;
    v->size     = 0;
    v->capacity = 0;
}

int main(void) {
    IntVec v = {0};  // Zero-initialize all fields

    for (int i = 0; i < 20; i++) {
        vec_push(&v, i * i);
    }

    printf("size=%zu, cap=%zu\n", v.size, v.capacity);
    for (size_t i = 0; i < v.size; i++) {
        printf("%d ", v.data[i]);
    }
    printf("\n");

    vec_free(&v);
    return 0;
}
```

### Correct Size Calculations

```c
// Preferred: use sizeof(*ptr) to avoid type mismatch bugs
int *p = malloc(10 * sizeof(*p));          // sizeof(int) = 4 -> 40 bytes
double *d = malloc(100 * sizeof(*d));      // sizeof(double) = 8 -> 800 bytes

// Allocating a struct
typedef struct { int x, y; char label[32]; } Point;
Point *pt = malloc(sizeof(*pt));           // sizeof(Point)

// Allocating an array of structs
Point *points = malloc(100 * sizeof(*points));

// String duplication (standard pattern)
char *strdup_impl(const char *s) {
    size_t len = strlen(s) + 1;  // +1 for null terminator
    char *copy = malloc(len);
    if (copy) memcpy(copy, s, len);
    return copy;
}
```

:::caution[Realloc Danger]
Never write `ptr = realloc(ptr, new_size)`. If `realloc` fails, it returns NULL but the original allocation is still valid — you just leaked it by overwriting your only pointer. Always use a temporary: `tmp = realloc(ptr, size); if (tmp) ptr = tmp;`
:::

## Memory Alignment

```c
#include <stdio.h>
#include <stddef.h>   // offsetof
#include <stdalign.h> // alignas, alignof

// Struct padding demonstration
struct Unpadded {
    char  a;    // 1 byte at offset 0
    // 3 bytes padding
    int   b;    // 4 bytes at offset 4
    char  c;    // 1 byte at offset 8
    // 7 bytes padding
    double d;   // 8 bytes at offset 16
};              // Total: 24 bytes

struct Packed {
    double d;   // 8 bytes at offset 0
    int   b;    // 4 bytes at offset 8
    char  a;    // 1 byte at offset 12
    char  c;    // 1 byte at offset 13
    // 2 bytes padding
};              // Total: 16 bytes (reordered!)

int main(void) {
    printf("sizeof(Unpadded) = %zu\n", sizeof(struct Unpadded));  // 24
    printf("sizeof(Packed)   = %zu\n", sizeof(struct Packed));    // 16

    printf("offsetof(Unpadded, a) = %zu\n", offsetof(struct Unpadded, a)); // 0
    printf("offsetof(Unpadded, b) = %zu\n", offsetof(struct Unpadded, b)); // 4
    printf("offsetof(Unpadded, c) = %zu\n", offsetof(struct Unpadded, c)); // 8
    printf("offsetof(Unpadded, d) = %zu\n", offsetof(struct Unpadded, d)); // 16

    // Aligned allocation (C11+)
    // aligned_alloc(alignment, size) — alignment must be power of 2
    // size must be a multiple of alignment
    float *simd_data = aligned_alloc(32, 256 * sizeof(float));  // 32-byte aligned for AVX
    if (simd_data) {
        printf("SIMD alignment: %zu\n", (uintptr_t)simd_data % 32);  // 0
        free(simd_data);
    }

    // alignas: force specific alignment on stack variable
    alignas(64) char cache_line[64];  // One full cache line
    printf("cache_line alignment check: %zu\n", (uintptr_t)cache_line % 64);

    return 0;
}
```

## Memory Error Detection

### Valgrind

```bash
# Compile with debug info
gcc -g -O0 buggy.c -o buggy

# Check for memory leaks
valgrind --leak-check=full --show-leak-kinds=all ./buggy

# Check for invalid accesses
valgrind --tool=memcheck ./buggy

# Generate suppression file to ignore known issues
valgrind --gen-suppressions=all ./buggy 2> suppressions.txt
```

Valgrind output categories:

- **definitely lost** — memory with no remaining pointers (true leak)
- **indirectly lost** — memory lost because a container holding it was leaked
- **possibly lost** — memory with interior pointers (may be intentional)
- **still reachable** — memory with valid pointers at exit (not freed, but not lost)

### AddressSanitizer (ASan)

```bash
# Compile with ASan (GCC/Clang)
gcc -g -fsanitize=address,undefined -o buggy buggy.c
./buggy  # ASan will print detailed error reports with stack traces
```

ASan detects:

- Heap buffer overflow/underflow
- Stack buffer overflow
- Use-after-free
- Use-after-return
- Double free
- Memory leaks (with `ASAN_OPTIONS=detect_leaks=1`)

```c
// Example that ASan will catch:
#include <stdlib.h>
int main(void) {
    int *p = malloc(10 * sizeof(int));
    p[10] = 42;  // ASan: heap-buffer-overflow (off by 1)
    free(p);
    p[0] = 1;    // ASan: heap-use-after-free
    free(p);     // ASan: double-free
    return 0;
}
```

## Arena Allocator Pattern

An arena (also called a bump allocator or linear allocator) is vastly faster than `malloc/free` for short-lived allocations with the same lifetime.

```c
#include <stdio.h>
#include <stdlib.h>
#include <stddef.h>
#include <string.h>
#include <stdalign.h>

typedef struct Arena {
    unsigned char *buf;
    size_t         size;
    size_t         pos;
} Arena;

// Initialize arena with pre-allocated buffer
void arena_init(Arena *a, void *buf, size_t size) {
    a->buf  = buf;
    a->size = size;
    a->pos  = 0;
}

// Allocate from arena — O(1), no fragmentation
void *arena_alloc(Arena *a, size_t size, size_t align) {
    // Align current position
    size_t aligned_pos = (a->pos + align - 1) & ~(align - 1);
    if (aligned_pos + size > a->size) return NULL;  // Out of space
    void *ptr = a->buf + aligned_pos;
    a->pos = aligned_pos + size;
    return ptr;
}

// Reset arena — free ALL allocations at once (O(1))
void arena_reset(Arena *a) {
    a->pos = 0;
}

// Convenience macros
#define ARENA_ALLOC(arena, type)       arena_alloc((arena), sizeof(type), alignof(type))
#define ARENA_ALLOC_N(arena, type, n)  arena_alloc((arena), sizeof(type)*(n), alignof(type))

typedef struct { float x, y, z; } Vec3;

int main(void) {
    // Pre-allocate a large block
    unsigned char backing[1024 * 1024];  // 1 MB on stack
    Arena arena;
    arena_init(&arena, backing, sizeof(backing));

    // Allocate from arena — extremely fast
    Vec3 *positions = ARENA_ALLOC_N(&arena, Vec3, 1000);
    int  *indices   = ARENA_ALLOC_N(&arena, int, 3000);
    char *names     = ARENA_ALLOC_N(&arena, char, 256);

    if (!positions || !indices || !names) {
        printf("Arena exhausted\n");
        return 1;
    }

    // Use the memory
    for (int i = 0; i < 1000; i++) {
        positions[i] = (Vec3){ (float)i, 0.0f, 0.0f };
    }

    printf("Arena used: %zu / %zu bytes\n", arena.pos, arena.size);

    // Free everything in one shot — extremely fast
    arena_reset(&arena);
    printf("Arena reset: %zu bytes free\n", arena.size);

    return 0;
}
```

## Memory Pool Pattern

A pool allocator is optimal when you frequently allocate and free many objects of the same size.

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// A free-list pool allocator
typedef struct PoolNode {
    struct PoolNode *next;
} PoolNode;

typedef struct Pool {
    unsigned char *memory;
    PoolNode      *free_list;
    size_t         object_size;
    size_t         count;
} Pool;

Pool pool_create(size_t object_size, size_t count) {
    // object_size must be at least sizeof(PoolNode) for the free-list trick
    if (object_size < sizeof(PoolNode)) object_size = sizeof(PoolNode);

    Pool pool = {
        .memory      = malloc(object_size * count),
        .free_list   = NULL,
        .object_size = object_size,
        .count       = count
    };

    if (!pool.memory) return pool;

    // Build the free list
    for (size_t i = 0; i < count; i++) {
        PoolNode *node = (PoolNode*)(pool.memory + i * object_size);
        node->next     = pool.free_list;
        pool.free_list = node;
    }

    return pool;
}

void *pool_alloc(Pool *pool) {
    if (!pool->free_list) return NULL;
    PoolNode *node   = pool->free_list;
    pool->free_list  = node->next;
    memset(node, 0, pool->object_size);
    return node;
}

void pool_free(Pool *pool, void *ptr) {
    PoolNode *node  = ptr;
    node->next      = pool->free_list;
    pool->free_list = node;
}

void pool_destroy(Pool *pool) {
    free(pool->memory);
    pool->memory    = NULL;
    pool->free_list = NULL;
}

typedef struct { int id; float x, y; char name[32]; } Enemy;

int main(void) {
    Pool enemy_pool = pool_create(sizeof(Enemy), 256);

    // Allocate enemies (O(1), no heap fragmentation)
    Enemy *e1 = pool_alloc(&enemy_pool);
    Enemy *e2 = pool_alloc(&enemy_pool);

    if (e1 && e2) {
        e1->id = 1; e1->x = 10.0f; snprintf(e1->name, 32, "Goblin");
        e2->id = 2; e2->x = 20.0f; snprintf(e2->name, 32, "Orc");
        printf("e1: %s at %.1f\n", e1->name, e1->x);
        printf("e2: %s at %.1f\n", e2->name, e2->x);

        // Return to pool (O(1))
        pool_free(&enemy_pool, e1);
        pool_free(&enemy_pool, e2);
    }

    pool_destroy(&enemy_pool);
    return 0;
}
```

:::tip[Cache-Friendly Layouts]
For game and simulation code, consider **Structure of Arrays (SoA)** instead of **Array of Structures (AoS)**:

- AoS: `struct Entity { float x, y, z; int hp; } entities[1000]`
- SoA: `float xs[1000], ys[1000], zs[1000]; int hps[1000]`

When iterating only over `x` and `y`, SoA gives better cache utilization because you don't load `z` and `hp` unnecessarily.
:::

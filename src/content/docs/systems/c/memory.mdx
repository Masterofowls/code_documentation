# Memory Management in C

C gives you complete control over memory. With that power comes responsibility: every allocation must eventually be freed, every pointer must be valid, and every buffer must be within bounds. This guide covers everything from the fundamentals to advanced patterns like arena allocators.

---

## Memory Layout of a C Program

When a C program runs, the OS maps its virtual address space into several distinct regions:

```
High addresses
┌──────────────────────┐
│       Stack          │  ← Grows downward. Local variables, function call frames.
│         ↓            │
│                      │
│         ↑            │
│        Heap          │  ← Grows upward. malloc/calloc/realloc.
├──────────────────────┤
│    BSS Segment       │  ← Uninitialized global/static variables (zeroed by OS).
├──────────────────────┤
│    Data Segment      │  ← Initialized global/static variables.
├──────────────────────┤
│    Text Segment      │  ← Executable code (read-only).
└──────────────────────┘
Low addresses
```

```c
#include <stdio.h>
#include <stdlib.h>

int global_init   = 42;        // Data segment — initialized global
int global_uninit;             // BSS segment  — uninitialized (zeroed)
static int static_var = 10;    // Data segment — static duration

int main(void) {
    int  local = 5;            // Stack — automatic storage duration
    int *heap  = malloc(4);    // Heap  — dynamic storage duration

    printf("Text  (code):  %p\n", (void *)main);
    printf("Data  (global):%p\n", (void *)&global_init);
    printf("BSS   (uninit):%p\n", (void *)&global_uninit);
    printf("Stack (local): %p\n", (void *)&local);
    printf("Heap  (malloc):%p\n", (void *)heap);

    free(heap);
    return 0;
}
```

## Storage Duration

| Duration | Lifetime | Location | Example |
|---|---|---|---|
| **Automatic** | Block scope (until `}`) | Stack | `int x = 5;` inside function |
| **Static** | Entire program | Data/BSS | `static int count;` or global variables |
| **Dynamic** | `malloc` to `free` | Heap | `int *p = malloc(n);` |
| **Thread-local** (C11) | Thread lifetime | Thread-local storage | `_Thread_local int id;` |

```c
#include <stdio.h>
#include <stdlib.h>

void counter_demo(void) {
    int        automatic_var = 0;   // resets each call
    static int static_var    = 0;   // persists across calls

    automatic_var++;
    static_var++;

    printf("automatic=%d  static=%d\n", automatic_var, static_var);
}

int main(void) {
    counter_demo();   // automatic=1  static=1
    counter_demo();   // automatic=1  static=2
    counter_demo();   // automatic=1  static=3
    return 0;
}
```

## Stack Frames

Each function call pushes a **stack frame** containing:
- Return address (where to continue after return)
- Saved register values
- Local variables
- Padding for alignment

```c
void inner(int a) {
    // Stack frame for inner: [return addr | saved rbp | a | ...]
    int local = a * 2;
    printf("inner: a=%d local=%d\n", a, local);
    // Frame popped when inner returns
}

void outer(void) {
    // Stack frame for outer: [return addr | saved rbp | x | ...]
    int x = 10;
    inner(x);  // pushes new frame on top
    // outer's frame still exists during inner's execution
}
```

:::caution[Stack Overflow]
Stack size is limited (typically 1–8 MB on Linux, 1 MB on Windows). Deep recursion or large local arrays can cause stack overflow — a crash with no recovery. Never declare large arrays as local variables. Use heap allocation instead.
:::

```c
// ❌ Stack overflow risk — 1MB local array
void bad(void) {
    int huge[250000];   // ~1 MB on stack — likely crashes
}

// ✅ Heap allocation
void good(void) {
    int *heap_arr = malloc(250000 * sizeof(int));
    if (!heap_arr) return;
    // use heap_arr...
    free(heap_arr);
}
```

## Dynamic Memory Allocation

### malloc — Allocate Uninitialized Memory

```c
#include <stdlib.h>
#include <stdio.h>

int main(void) {
    // malloc(size) — returns void*, or NULL on failure
    // Memory is UNINITIALIZED — contains garbage
    int *arr = malloc(10 * sizeof(int));

    // ALWAYS check for NULL
    if (arr == NULL) {
        perror("malloc failed");
        return 1;
    }

    // Use it
    for (int i = 0; i < 10; i++) arr[i] = i * i;

    // ALWAYS free when done
    free(arr);
    arr = NULL;   // Good practice: nullify after free

    // Allocating a struct
    typedef struct { int x, y; } Point;
    Point *p = malloc(sizeof(Point));
    if (!p) return 1;
    p->x = 3; p->y = 4;
    free(p);

    return 0;
}
```

### calloc — Allocate Zero-Initialized Memory

```c
// calloc(count, size) — allocates count * size bytes, zeroed
int *arr = calloc(10, sizeof(int));    // all elements are 0
if (!arr) return 1;

// Useful for: lookup tables, bitfields, anything needing zero init
// Slightly slower than malloc due to zeroing, but safer
free(arr);
```

### realloc — Resize an Allocation

```c
#include <stdlib.h>
#include <string.h>

// Dynamic array with realloc
typedef struct {
    int   *data;
    int    size;
    int    capacity;
} IntVec;

void vec_push(IntVec *v, int value) {
    if (v->size == v->capacity) {
        // Grow by 2x — classic amortized strategy
        int new_cap = v->capacity == 0 ? 4 : v->capacity * 2;

        // CRITICAL: use a temporary pointer — realloc may return NULL
        int *new_data = realloc(v->data, new_cap * sizeof(int));
        if (!new_data) {
            // Handle error — old v->data is still valid
            perror("realloc failed");
            return;
        }
        v->data = new_data;
        v->capacity = new_cap;
    }
    v->data[v->size++] = value;
}

void vec_free(IntVec *v) {
    free(v->data);
    v->data = NULL;
    v->size = v->capacity = 0;
}
```

:::caution[realloc NULL Trap]
Never write `ptr = realloc(ptr, new_size)`. If `realloc` fails, it returns `NULL` and the original allocation is unchanged — but you've just lost the original pointer, creating a memory leak. Always use a temporary pointer.
:::

### aligned_alloc (C11)

```c
#include <stdlib.h>
#include <stdalign.h>

// aligned_alloc(alignment, size) — alignment must be power of 2, size must be multiple of alignment
float *simd_arr = aligned_alloc(32, 32 * sizeof(float));  // 32-byte aligned for AVX
if (!simd_arr) return 1;
// use with AVX intrinsics...
free(simd_arr);

// Alternative: posix_memalign (POSIX)
void *buf;
posix_memalign(&buf, 64, 512);   // 64-byte aligned, 512 bytes
free(buf);
```

## Memory Alignment

The CPU accesses memory most efficiently when data is aligned to its natural size. Misaligned access may be slower or cause faults on some architectures.

```c
#include <stddef.h>
#include <stdio.h>
#include <stdalign.h>

struct Padded {
    char   a;     // 1 byte at offset 0
                  // 3 bytes padding
    int    b;     // 4 bytes at offset 4
    char   c;     // 1 byte at offset 8
                  // 7 bytes padding  
    double d;     // 8 bytes at offset 16
};                // Total: 24 bytes

struct Compact {
    double d;     // 8 bytes at offset 0
    int    b;     // 4 bytes at offset 8
    char   a;     // 1 byte at offset 12
    char   c;     // 1 byte at offset 13
                  // 2 bytes padding
};                // Total: 16 bytes

int main(void) {
    printf("sizeof(Padded)  = %zu\n", sizeof(struct Padded));   // 24
    printf("sizeof(Compact) = %zu\n", sizeof(struct Compact));  // 16

    // offsetof — byte offset of a member within a struct
    printf("offsetof(Padded, b) = %zu\n", offsetof(struct Padded, b));   // 4
    printf("offsetof(Padded, d) = %zu\n", offsetof(struct Padded, d));   // 16

    // alignof / _Alignof — required alignment of a type
    printf("alignof(int)    = %zu\n", alignof(int));     // 4
    printf("alignof(double) = %zu\n", alignof(double));  // 8

    // Force specific alignment
    struct __attribute__((packed)) Unpadded {
        char   a;   // no padding added
        int    b;   // at offset 1 — misaligned!
    };
    printf("sizeof(Unpadded) = %zu\n", sizeof(struct Unpadded));  // 5

    return 0;
}
```

## Memory Leak Detection

### Valgrind

```bash
# Compile with debug info
gcc -g -O0 program.c -o program

# Run under Valgrind
valgrind --leak-check=full --show-leak-kinds=all --track-origins=yes ./program

# Example output on leak:
# LEAK SUMMARY:
#    definitely lost: 40 bytes in 1 blocks
#    indirectly lost: 0 bytes in 0 blocks
#      possibly lost: 0 bytes in 0 blocks
```

### AddressSanitizer (ASan)

```bash
# Compile with ASan (GCC or Clang)
gcc -g -fsanitize=address,undefined program.c -o program

# Just run it — ASan hooks malloc/free and reports errors immediately
./program

# Reports: heap-buffer-overflow, use-after-free, double-free, stack-overflow
```

```c
// Example that ASan catches
int main(void) {
    int *arr = malloc(3 * sizeof(int));
    arr[3] = 99;   // heap-buffer-overflow — ASan reports this immediately
    free(arr);
    *arr = 0;      // use-after-free — ASan reports this too
    return 0;
}
```

## Flexible Array Members (C99)

```c
#include <stdlib.h>
#include <string.h>

// Flexible array member: last member is an incomplete array type
typedef struct {
    int  length;
    char data[];   // flexible array member — no size
} Buffer;

Buffer *buffer_create(int len) {
    // Allocate struct + inline array in one malloc call
    Buffer *buf = malloc(sizeof(Buffer) + len * sizeof(char));
    if (!buf) return NULL;
    buf->length = len;
    return buf;
}

// Contrast with the old "struct hack" (before C99):
// struct OldBuffer { int length; char data[1]; };  // UB in strict C89
```

## Arena Allocator Pattern

An arena (also called a linear allocator or bump allocator) allocates from a contiguous block and frees everything at once. It's extremely fast for short-lived allocations.

```c
#include <stdlib.h>
#include <string.h>
#include <stdio.h>
#include <stdalign.h>

typedef struct {
    unsigned char *buf;     // backing memory
    size_t         cap;     // total capacity
    size_t         pos;     // current allocation offset
} Arena;

// Initialize arena with a fixed-size buffer
Arena arena_init(size_t capacity) {
    return (Arena){
        .buf = malloc(capacity),
        .cap = capacity,
        .pos = 0,
    };
}

// Allocate aligned memory from arena — O(1), no fragmentation
void *arena_alloc(Arena *a, size_t size, size_t align) {
    // Round up pos to required alignment
    size_t aligned_pos = (a->pos + align - 1) & ~(align - 1);
    if (aligned_pos + size > a->cap) return NULL;   // out of space
    void *ptr = a->buf + aligned_pos;
    a->pos = aligned_pos + size;
    return ptr;
}

// Reset arena — O(1), does NOT zero memory
void arena_reset(Arena *a) {
    a->pos = 0;
}

// Free the backing buffer
void arena_destroy(Arena *a) {
    free(a->buf);
    a->buf = NULL;
    a->cap = a->pos = 0;
}

// Usage: per-frame allocation in a game loop
typedef struct { float x, y, z; } Vec3;

int main(void) {
    Arena frame_arena = arena_init(1024 * 1024);   // 1 MB per-frame arena

    // Every frame:
    while (0) {  // placeholder loop
        arena_reset(&frame_arena);   // free all at once — blazing fast

        Vec3 *positions = arena_alloc(&frame_arena, 1000 * sizeof(Vec3), alignof(Vec3));
        int  *indices   = arena_alloc(&frame_arena, 3000 * sizeof(int),  alignof(int));
        // use positions and indices...
        // No individual frees needed!
    }

    arena_destroy(&frame_arena);
    return 0;
}
```

## Memory Pool (Slab Allocator Concept)

A memory pool pre-allocates fixed-size objects, eliminating malloc overhead for hot-path allocations:

```c
#include <stdlib.h>
#include <stdio.h>
#include <stddef.h>

// Fixed-size pool — all objects same size
typedef struct FreeNode {
    struct FreeNode *next;
} FreeNode;

typedef struct {
    void     *memory;      // backing allocation
    FreeNode *free_list;   // linked list of free slots
    size_t    obj_size;
    int       capacity;
    int       in_use;
} Pool;

Pool pool_create(size_t obj_size, int capacity) {
    // Ensure minimum size for the free-list pointer
    if (obj_size < sizeof(FreeNode)) obj_size = sizeof(FreeNode);

    Pool p = {
        .memory    = malloc(obj_size * capacity),
        .free_list = NULL,
        .obj_size  = obj_size,
        .capacity  = capacity,
        .in_use    = 0,
    };

    // Build the free list
    unsigned char *cur = p.memory;
    for (int i = 0; i < capacity; i++) {
        FreeNode *node = (FreeNode *)cur;
        node->next = p.free_list;
        p.free_list = node;
        cur += obj_size;
    }
    return p;
}

void *pool_alloc(Pool *p) {
    if (!p->free_list) return NULL;   // pool exhausted
    FreeNode *node = p->free_list;
    p->free_list = node->next;
    p->in_use++;
    return (void *)node;
}

void pool_free(Pool *p, void *ptr) {
    FreeNode *node = (FreeNode *)ptr;
    node->next = p->free_list;
    p->free_list = node;
    p->in_use--;
}

void pool_destroy(Pool *p) {
    free(p->memory);
    p->memory = p->free_list = NULL;
}
```

## Cache-Friendly Data Layouts

CPU caches work on cache lines (typically 64 bytes). Accessing memory sequentially maximizes cache hits.

```c
// ❌ Array of Structures (AoS) — bad for processing only one field
typedef struct {
    float x, y, z;    // position
    float vx, vy, vz; // velocity
    float mass;
    int   id;
} Particle_AoS;

Particle_AoS particles_aos[1000];

// Processing positions requires loading all 8 fields per particle
for (int i = 0; i < 1000; i++) {
    particles_aos[i].x += particles_aos[i].vx;
}

// ✅ Structure of Arrays (SoA) — excellent cache locality when processing single fields
typedef struct {
    float x[1000], y[1000], z[1000];    // positions
    float vx[1000], vy[1000], vz[1000]; // velocities
    float mass[1000];
    int   id[1000];
} Particles_SoA;

Particles_SoA particles_soa;

// Processing positions loads only x[] and vx[] — fully cache-friendly, SIMD-friendly
for (int i = 0; i < 1000; i++) {
    particles_soa.x[i] += particles_soa.vx[i];
}
```

# Pointers in C

Pointers are the most powerful — and most dangerous — feature of C. A pointer is a variable that holds the memory address of another variable. Mastering pointers is essential for systems programming, data structures, and interfacing with hardware.

## Pointer Basics

```c
#include <stdio.h>

int main(void) {
    int x = 42;

    // Declare a pointer to int
    // Read type declarations right-to-left: "p is a pointer to int"
    int *p;

    // Initialize: take the address of x with the & operator
    p = &x;

    // Dereference: access the value at the address with *
    printf("Address of x: %p\n", (void*)p);
    printf("Value at p:   %d\n", *p);          // 42
    printf("Value of x:   %d\n", x);           // 42

    // Modify through pointer
    *p = 100;
    printf("x is now: %d\n", x);               // 100

    // Size of a pointer (always same regardless of pointed-to type)
    printf("sizeof(int*):    %zu\n", sizeof(int*));    // 8 on 64-bit
    printf("sizeof(double*): %zu\n", sizeof(double*)); // 8 on 64-bit
    printf("sizeof(char*):   %zu\n", sizeof(char*));   // 8 on 64-bit

    return 0;
}
```

### NULL Pointer

A null pointer points to no valid memory location. Dereferencing it is undefined behavior (typically a segfault).

```c
#include <stddef.h>  // NULL

// NULL is typically defined as ((void*)0) or 0
int *p = NULL;

// Always check before dereferencing
if (p != NULL) {
    printf("%d\n", *p);
}

// Common pattern: initialize to NULL, set later
int *result = NULL;
// ... some computation ...
if (result == NULL) {
    fprintf(stderr, "Error: result was not set\n");
    return 1;
}
```

## Pointer Arithmetic

```c
#include <stdio.h>

int main(void) {
    int arr[5] = {10, 20, 30, 40, 50};
    int *p = arr;  // p points to arr[0]

    // Adding integer to pointer moves by sizeof(pointed-to type)
    printf("p[0] = %d\n",   *p);        // 10
    printf("p[1] = %d\n",   *(p + 1));  // 20 (p + 1 * sizeof(int) bytes)
    printf("p[2] = %d\n",   *(p + 2));  // 30

    // Increment/decrement pointer
    p++;
    printf("After p++: *p = %d\n", *p); // 20

    p += 2;
    printf("After p+=2: *p = %d\n", *p); // 40

    p--;
    printf("After p--: *p = %d\n", *p);  // 30

    // Pointer subtraction: result is ptrdiff_t (number of elements between)
    int *start = arr;
    int *end   = arr + 5;  // One past the end (valid but not dereferenceable)
    ptrdiff_t count = end - start;
    printf("Elements between: %td\n", count);  // 5

    // Array indexing is equivalent to pointer arithmetic
    // arr[i]  ===  *(arr + i)
    // &arr[i] ===  arr + i
    for (int i = 0; i < 5; i++) {
        printf("arr[%d]=%d  *(arr+%d)=%d  *(p_base+%d)=%d\n",
               i, arr[i], i, *(arr + i), i, *(start + i));
    }

    // Traversal with pointer (often faster in tight loops)
    int sum = 0;
    for (const int *q = arr; q < arr + 5; q++) {
        sum += *q;
    }
    printf("Sum: %d\n", sum);  // 150

    return 0;
}
```

:::caution[Pointer Arithmetic Bounds]
Only perform pointer arithmetic within an array (or one past the end). Arithmetic on a pointer that goes out of bounds is undefined behavior, even if you never dereference the out-of-bounds pointer. One past the end (`arr + N`) is valid for comparison but not for dereference.
:::

## Pointer to Pointer

```c
#include <stdio.h>
#include <stdlib.h>

void set_value(int **pp, int val) {
    // Allocate and assign through pointer-to-pointer
    *pp = malloc(sizeof(int));
    if (*pp) **pp = val;
}

int main(void) {
    int x = 42;
    int *p = &x;
    int **pp = &p;   // Pointer to pointer to int

    printf("x    = %d\n",   x);    // 42
    printf("*p   = %d\n",   *p);   // 42
    printf("**pp = %d\n",   **pp); // 42

    // Modify through double pointer
    **pp = 100;
    printf("x is now %d\n", x);    // 100

    // Common use: function that needs to set a pointer
    int *allocated = NULL;
    set_value(&allocated, 999);
    if (allocated) {
        printf("allocated = %d\n", *allocated);  // 999
        free(allocated);
    }

    // 2D dynamic array using pointer-to-pointer
    int rows = 3, cols = 4;
    int **matrix = malloc(rows * sizeof(int*));
    for (int i = 0; i < rows; i++) {
        matrix[i] = malloc(cols * sizeof(int));
        for (int j = 0; j < cols; j++) {
            matrix[i][j] = i * cols + j;
        }
    }
    printf("matrix[2][3] = %d\n", matrix[2][3]);  // 11

    // Free 2D array (must free inner arrays first)
    for (int i = 0; i < rows; i++) free(matrix[i]);
    free(matrix);

    return 0;
}
```

## Void Pointer

A `void*` is a generic pointer — it can hold the address of any type but cannot be dereferenced directly.

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Generic swap using void pointer
void swap(void *a, void *b, size_t size) {
    // Use a temporary byte buffer
    unsigned char tmp[256];  // Stack buffer; for large sizes, malloc
    if (size <= sizeof(tmp)) {
        memcpy(tmp, a, size);
        memcpy(a, b, size);
        memcpy(b, tmp, size);
    }
}

// Generic print function
void print_value(const void *data, char type) {
    switch (type) {
        case 'i': printf("%d\n", *(const int*)data); break;
        case 'f': printf("%f\n", *(const float*)data); break;
        case 's': printf("%s\n", (const char*)data); break;
    }
}

int main(void) {
    // malloc returns void* — no cast needed in C (unlike C++)
    int *arr = malloc(10 * sizeof(int));
    double *darr = malloc(10 * sizeof(double));

    // void* can be assigned to/from any pointer type without cast
    void *generic = arr;
    int *back = generic;  // Implicit conversion back

    // Use swap
    int x = 10, y = 20;
    swap(&x, &y, sizeof(int));
    printf("x=%d, y=%d\n", x, y);  // x=20, y=10

    double dx = 3.14, dy = 2.71;
    swap(&dx, &dy, sizeof(double));
    printf("dx=%.2f, dy=%.2f\n", dx, dy);  // dx=2.71, dy=3.14

    int val = 42;
    float fval = 1.5f;
    print_value(&val, 'i');
    print_value(&fval, 'f');
    print_value("hello", 's');

    free(arr);
    free(darr);
    (void)back;
    return 0;
}
```

## const Pointer Variants

```c
// Four combinations of const with pointers:

int x = 10, y = 20;

// 1. Pointer to const int — can't modify value, can redirect pointer
const int *p1 = &x;
p1 = &y;          // OK: redirecting pointer
// *p1 = 30;      // ERROR: can't modify through pointer

// 2. Const pointer to int — can modify value, can't redirect pointer
int * const p2 = &x;
*p2 = 30;         // OK: modifying value
// p2 = &y;       // ERROR: can't redirect

// 3. Const pointer to const int — can't modify value OR redirect
const int * const p3 = &x;
// *p3 = 30;      // ERROR
// p3 = &y;       // ERROR

// 4. Plain pointer to int — can do both
int *p4 = &x;
*p4 = 30;         // OK
p4 = &y;          // OK

// Function parameters: use const for read-only access
size_t my_strlen(const char *s) {
    const char *p = s;
    while (*p) p++;
    return (size_t)(p - s);
}

// const correctness propagates:
// Never cast away const (undefined behavior if the underlying data is const)
const char *str = "hello";
// char *mutable = (char*)str;  // Technically valid cast but dangerous
// mutable[0] = 'H';            // UB: modifying string literal
```

## Function Pointers

Function pointers allow storing and passing functions as values — essential for callbacks, dispatch tables, and plugins.

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Declare a function pointer type
// Syntax: return_type (*name)(param_types)
typedef int (*CompareFunc)(const void*, const void*);
typedef double (*MathFunc)(double);

// Regular functions to point to
double square(double x) { return x * x; }
double cube(double x)   { return x * x * x; }
double negate(double x) { return -x; }

// Function accepting a function pointer (callback pattern)
void apply_and_print(double *arr, int n, MathFunc fn) {
    for (int i = 0; i < n; i++) {
        printf("%.2f -> %.2f\n", arr[i], fn(arr[i]));
    }
}

// Comparison functions for qsort
int compare_int_asc(const void *a, const void *b) {
    int ia = *(const int*)a;
    int ib = *(const int*)b;
    return (ia > ib) - (ia < ib);  // Branch-free comparison
}

int compare_int_desc(const void *a, const void *b) {
    return compare_int_asc(b, a);  // Reversed
}

int main(void) {
    // Assign function pointer
    MathFunc fn = square;
    printf("square(5) = %.2f\n", fn(5.0));

    fn = cube;
    printf("cube(3) = %.2f\n", fn(3.0));

    // Array of function pointers (dispatch table)
    MathFunc math_table[] = { square, cube, negate };
    const char *names[] = { "square", "cube", "negate" };
    for (int i = 0; i < 3; i++) {
        printf("%s(4.0) = %.2f\n", names[i], math_table[i](4.0));
    }

    // Using callback
    double values[] = {1.0, 2.0, 3.0};
    printf("\nSquares:\n");
    apply_and_print(values, 3, square);

    // qsort with function pointer
    int numbers[] = {5, 2, 8, 1, 9, 3};
    int n = sizeof(numbers) / sizeof(numbers[0]);

    qsort(numbers, n, sizeof(int), compare_int_asc);
    printf("\nAscending: ");
    for (int i = 0; i < n; i++) printf("%d ", numbers[i]);
    printf("\n");

    qsort(numbers, n, sizeof(int), compare_int_desc);
    printf("Descending: ");
    for (int i = 0; i < n; i++) printf("%d ", numbers[i]);
    printf("\n");

    // Calling via pointer: both syntaxes are equivalent
    CompareFunc cmp = compare_int_asc;
    int a = 3, b = 5;
    printf("cmp result: %d\n",   cmp(&a, &b));     // Explicit dereference
    printf("cmp result: %d\n", (*cmp)(&a, &b));    // Explicit form

    return 0;
}
```

## Pointers to Structs

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char name[64];
    int age;
    double score;
} Person;

// Pass by pointer to avoid copying large struct
void print_person(const Person *p) {
    // Arrow operator (->) is shorthand for (*p).field
    printf("Name: %s, Age: %d, Score: %.1f\n", p->name, p->age, p->score);
    // Equivalent: (*p).name, (*p).age, (*p).score
}

void birthday(Person *p) {
    p->age++;
}

Person *create_person(const char *name, int age) {
    Person *p = malloc(sizeof(Person));
    if (!p) return NULL;
    strncpy(p->name, name, sizeof(p->name) - 1);
    p->name[sizeof(p->name) - 1] = '\0';
    p->age = age;
    p->score = 0.0;
    return p;
}

int main(void) {
    // Stack-allocated struct
    Person alice = {"Alice", 30, 95.5};
    print_person(&alice);

    // Heap-allocated struct
    Person *bob = create_person("Bob", 25);
    if (bob) {
        bob->score = 87.3;
        print_person(bob);
        birthday(bob);
        printf("After birthday: %d\n", bob->age);
        free(bob);
    }

    return 0;
}
```

## Pointer Aliasing and `restrict`

```c
// Without restrict: compiler must assume p and q might alias (point to same memory)
void add_arrays(int *dest, const int *a, const int *b, int n) {
    for (int i = 0; i < n; i++) {
        dest[i] = a[i] + b[i];
    }
}

// With restrict: promise to compiler that p, a, b don't alias each other
// This enables vectorization (SIMD) optimizations
void add_arrays_fast(int * restrict dest,
                     const int * restrict a,
                     const int * restrict b,
                     int n) {
    for (int i = 0; i < n; i++) {
        dest[i] = a[i] + b[i];
    }
}

// memcpy uses restrict; memmove does not (handles overlap)
// void *memcpy(void * restrict dst, const void * restrict src, size_t n);
// void *memmove(void *dst, const void *src, size_t n);
```

:::caution[Restrict is a Promise]
If you declare a pointer `restrict` and then the actual code passes aliased pointers, the behavior is undefined. Use `restrict` only when you're certain the pointers don't alias.
:::

## Common Pointer Pitfalls

```c
#include <stdio.h>
#include <stdlib.h>

// PITFALL 1: Dangling pointer (use-after-free)
void dangling_pointer_example(void) {
    int *p = malloc(sizeof(int));
    *p = 42;
    free(p);
    // p is now a dangling pointer — points to freed memory
    // *p = 10;  // UB: use-after-free

    // FIX: set to NULL after free
    p = NULL;
    if (p) *p = 10;  // Safe: NULL check prevents dereference
}

// PITFALL 2: Returning pointer to local variable
int *bad_return(void) {
    int local = 42;
    return &local;  // UB: local ceases to exist after function returns
}

int *good_return(void) {
    int *p = malloc(sizeof(int));
    if (p) *p = 42;
    return p;  // OK: heap memory persists
}

// PITFALL 3: Buffer overflow
void buffer_overflow_example(void) {
    char buf[10];
    // strcpy(buf, "This string is too long for the buffer!"); // UB: overflow
    // FIX: use strncpy or snprintf
    snprintf(buf, sizeof(buf), "%.9s", "Too long string");
}

// PITFALL 4: Uninitialized pointer (wild pointer)
void wild_pointer_example(void) {
    int *p;  // Uninitialized — contains garbage address
    // *p = 42;  // UB: writing to random memory location

    // FIX: always initialize pointers
    int *safe = NULL;
    (void)safe;
}

// PITFALL 5: Double free
void double_free_example(void) {
    int *p = malloc(sizeof(int));
    free(p);
    // free(p);  // UB: double free — heap corruption

    // FIX: set to NULL after free
    p = NULL;
    free(p);  // free(NULL) is always safe
}
```

## Common Pointer Patterns

### Swap via Pointer

```c
void swap_int(int *a, int *b) {
    int tmp = *a;
    *a = *b;
    *b = tmp;
}

// Generic swap
void swap(void *a, void *b, size_t size) {
    unsigned char *pa = a, *pb = b, tmp;
    for (size_t i = 0; i < size; i++) {
        tmp = pa[i]; pa[i] = pb[i]; pb[i] = tmp;
    }
}
```

### Linked List Node

```c
typedef struct Node {
    int data;
    struct Node *next;  // Pointer to same struct type (forward reference)
} Node;

Node *list_push(Node *head, int val) {
    Node *node = malloc(sizeof(Node));
    if (!node) return head;
    node->data = val;
    node->next = head;
    return node;
}

void list_print(const Node *head) {
    for (const Node *n = head; n; n = n->next) {
        printf("%d -> ", n->data);
    }
    printf("NULL\n");
}

void list_free(Node *head) {
    while (head) {
        Node *next = head->next;
        free(head);
        head = next;
    }
}
```

### Return Multiple Values via Pointer Parameters

```c
#include <stdbool.h>

// Return bool for success, output values via pointers
bool divide(int a, int b, int *quotient, int *remainder) {
    if (b == 0) return false;
    *quotient  = a / b;
    *remainder = a % b;
    return true;
}

// Parse a string, return success and result
bool parse_int(const char *str, int *out) {
    if (!str || !out) return false;
    char *end;
    long val = strtol(str, &end, 10);
    if (end == str || *end != '\0') return false;
    *out = (int)val;
    return true;
}

int main(void) {
    int q, r;
    if (divide(17, 5, &q, &r)) {
        printf("17 / 5 = %d remainder %d\n", q, r);
    }

    int parsed;
    if (parse_int("42", &parsed)) {
        printf("Parsed: %d\n", parsed);
    }
    if (!parse_int("abc", &parsed)) {
        printf("Parse failed\n");
    }

    return 0;
}
```

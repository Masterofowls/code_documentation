# Pointers in C

A pointer is a variable that stores the **memory address** of another variable. Pointers are the mechanism through which C provides dynamic memory, arrays, strings, callbacks, and data structure composition. Mastering pointers is mastering C.

## Pointer Fundamentals

```c
#include <stdio.h>

int main(void)
{
    int x = 42;

    // Declaration: int *p means "p is a pointer to int"
    int *p;              // uninitialized — DANGEROUS (wild pointer)
    p = &x;             // & = address-of operator: p now holds address of x

    // Dereference: *p means "the value at the address stored in p"
    printf("x      = %d\n",  x);     // 42
    printf("&x     = %p\n",  (void*)&x); // e.g., 0x7ffd1234 (stack address)
    printf("p      = %p\n",  (void*)p);  // same address as &x
    printf("*p     = %d\n",  *p);    // 42 — same as x

    // Modifying through pointer
    *p = 100;           // writes to x through the pointer
    printf("x after *p=100: %d\n", x); // 100

    // Multiple declarations — * binds to name, not type
    int *a, b, *c;  // a and c are pointers; b is a plain int
    // Prefer: int *a; int b; int *c;  (one per line)

    return 0;
}
```

## The NULL Pointer

`NULL` is a guaranteed null pointer constant (typically `(void*)0` on most platforms). Always initialize pointers to `NULL` when they don't yet point at valid memory.

```c
#include <stdio.h>
#include <stdlib.h>

int main(void)
{
    int *p = NULL;   // safe initialization

    // Check before dereferencing
    if (p != NULL) {
        printf("%d\n", *p);
    } else {
        printf("p is null — not dereferencing\n");
    }

    // Dereferencing NULL is undefined behavior (usually a segfault)
    // *p = 5;  // CRASH

    // After free(), set to NULL to avoid use-after-free
    p = malloc(sizeof(int));
    if (p == NULL) {
        fprintf(stderr, "malloc failed\n");
        return 1;
    }
    *p = 99;
    free(p);
    p = NULL;  // good practice

    return 0;
}
```

## Pointer Arithmetic

When you add an integer to a pointer, C advances the pointer by **that many elements** of the pointed-to type — not by that many bytes.

```c
#include <stdio.h>

int main(void)
{
    int arr[5] = { 10, 20, 30, 40, 50 };
    int *p = arr;    // p points to arr[0]; arrays decay to pointer to first element

    printf("*p       = %d\n", *p);         // 10
    printf("*(p+1)   = %d\n", *(p + 1));  // 20 — p+1 advances by sizeof(int) bytes
    printf("*(p+2)   = %d\n", *(p + 2));  // 30

    // Increment pointer
    p++;              // p now points to arr[1]
    printf("after ++: *p = %d\n", *p);   // 20

    // Pointer subtraction — yields ptrdiff_t (number of elements between)
    int *start = arr;
    int *end   = arr + 5;               // one-past-end (valid, but don't dereference)
    ptrdiff_t count = end - start;      // 5
    printf("elements between: %td\n", count);

    // Array indexing and pointer arithmetic are equivalent
    // arr[i]  ≡  *(arr + i)  ≡  *(p + i)  ≡  p[i]
    for (int i = 0; i < 5; i++) {
        printf("arr[%d]=%d  *(arr+%d)=%d\n", i, arr[i], i, *(arr + i));
    }

    // Pointer comparison (only within same array/object)
    int *q = arr + 3;
    printf("p < q: %d\n", p < q);       // 1 (true) — p=arr+1, q=arr+3

    return 0;
}
```

:::caution[Pointer Arithmetic Bounds]
Pointer arithmetic is only valid **within** an array (including the one-past-end position). Arithmetic on pointers into different objects produces **undefined behavior**. Never dereference the one-past-end pointer.
:::

## Pointer to Pointer

```c
#include <stdio.h>

int main(void)
{
    int   x  = 5;
    int  *p  = &x;    // pointer to int
    int **pp = &p;    // pointer to pointer to int

    printf("x   = %d\n",  x);
    printf("*p  = %d\n",  *p);    // 5
    printf("**pp= %d\n", **pp);   // 5

    **pp = 99;        // modifies x through two levels of indirection
    printf("x after **pp=99: %d\n", x); // 99

    // Common use: modify a pointer from a function
    // (pass-by-pointer-to-pointer)
    // Also used for 2D dynamic arrays:
    int rows = 3, cols = 4;
    int **matrix = malloc(rows * sizeof(int *));
    for (int i = 0; i < rows; i++) {
        matrix[i] = malloc(cols * sizeof(int));
    }
    matrix[1][2] = 42;
    printf("matrix[1][2] = %d\n", matrix[1][2]);

    for (int i = 0; i < rows; i++) free(matrix[i]);
    free(matrix);

    return 0;
}
```

## void Pointer — Generic Pointer

A `void *` can hold any pointer type without casting (in C; not in C++). It enables generic functions like `malloc`, `memcpy`, `qsort`.

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Generic swap using void*
void swap(void *a, void *b, size_t size)
{
    unsigned char tmp[256];  // max element size (use VLA or malloc for large)
    memcpy(tmp, a,   size);
    memcpy(a,   b,   size);
    memcpy(b,   tmp, size);
}

int main(void)
{
    // malloc returns void* — assign to any pointer type without cast in C
    int    *ip = malloc(sizeof(int));        // implicit cast from void*
    double *dp = malloc(sizeof(double));
    char   *cp = malloc(10 * sizeof(char));

    // void* cannot be dereferenced directly
    void *vp = ip;
    // *vp = 5;     // ERROR: cannot dereference void*
    *(int *)vp = 5; // must cast first

    // Generic swap
    int x = 10, y = 20;
    swap(&x, &y, sizeof(int));
    printf("x=%d y=%d\n", x, y);  // x=20 y=10

    double a = 1.5, b = 2.5;
    swap(&a, &b, sizeof(double));
    printf("a=%.1f b=%.1f\n", a, b); // a=2.5 b=1.5

    free(ip); free(dp); free(cp);
    return 0;
}
```

## const Pointer Variants

There are four meaningful combinations of `const` with pointers:

```c
int value = 10;
int other = 20;

// 1. Pointer to const int — cannot modify the value pointed to
//    (read-only data; pointer itself can change)
const int *p1 = &value;
// *p1 = 99;  // ERROR: read-only location
p1 = &other; // OK: pointer can change

// 2. const pointer to int — pointer cannot be reassigned
//    (the value can still be modified through the pointer)
int *const p2 = &value;
*p2 = 99;    // OK: value can be modified
// p2 = &other; // ERROR: pointer is const

// 3. const pointer to const int — neither pointer nor value can change
const int *const p3 = &value;
// *p3 = 99;    // ERROR
// p3 = &other; // ERROR

// 4. Plain pointer to int — no restrictions
int *p4 = &value;
*p4 = 50;    // OK
p4 = &other; // OK

// Function parameters — use const to express intent
void print_array(const int *arr, size_t n)
{
    for (size_t i = 0; i < n; i++) {
        printf("%d ", arr[i]);
        // arr[i] = 0;  // ERROR: arr is pointer to const int
    }
    printf("\n");
}
```

:::tip[Reading const Declarations]
Read pointer declarations **right-to-left**: `const int *p` → "p is a pointer to const int". `int *const p` → "p is a const pointer to int". `const int *const p` → "p is a const pointer to const int".
:::

## Function Pointers

Function pointers store the address of a function. They enable callbacks, dispatch tables, and plugin architectures.

```c
#include <stdio.h>
#include <stdlib.h>

// Function pointer type: points to a function taking two ints, returning int
typedef int (*BinaryOp)(int, int);

int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }

// Apply function pointer
int apply(BinaryOp op, int x, int y)
{
    return op(x, y);   // call through pointer — same as (*op)(x, y)
}

// Array of function pointers (dispatch table)
int dispatch(int op_code, int a, int b)
{
    BinaryOp ops[] = { add, sub, mul };
    if (op_code < 0 || op_code > 2) return 0;
    return ops[op_code](a, b);
}

// qsort comparison callback
int compare_int(const void *a, const void *b)
{
    int ia = *(const int *)a;
    int ib = *(const int *)b;
    return (ia > ib) - (ia < ib);   // branchless compare
}

int main(void)
{
    // Declare and assign
    int (*fp)(int, int) = add;    // without typedef
    BinaryOp fp2 = mul;           // with typedef (cleaner)

    printf("add(3,4) = %d\n",  apply(add, 3, 4));  // 7
    printf("mul(3,4) = %d\n",  apply(mul, 3, 4));  // 12
    printf("dispatch(1,10,3) = %d\n", dispatch(1, 10, 3)); // 7 (sub)

    // qsort with callback
    int arr[] = { 5, 2, 8, 1, 9, 3 };
    qsort(arr, 6, sizeof(int), compare_int);
    for (int i = 0; i < 6; i++) printf("%d ", arr[i]);
    printf("\n");  // 1 2 3 5 8 9

    // Function pointer to NULL
    BinaryOp maybe_op = NULL;
    if (maybe_op) {
        maybe_op(1, 2);
    }

    return 0;
}
```

## Pointers to Structs and the Arrow Operator

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char name[64];
    int  age;
    float score;
} Student;

void print_student(const Student *s)
{
    // Arrow operator -> dereferences AND accesses member
    // s->name  is equivalent to  (*s).name
    printf("Name: %s, Age: %d, Score: %.1f\n",
           s->name, s->age, s->score);
}

// Linked list node
typedef struct Node {
    int data;
    struct Node *next;   // self-referential (must use struct tag, not typedef)
} Node;

Node *create_node(int data)
{
    Node *n = malloc(sizeof(Node));
    if (n) {
        n->data = data;
        n->next = NULL;
    }
    return n;
}

int main(void)
{
    // Stack-allocated struct accessed via pointer
    Student alice = { "Alice", 20, 95.5f };
    Student *sp = &alice;
    print_student(sp);

    // Dynamically allocated struct
    Student *bob = malloc(sizeof(Student));
    strncpy(bob->name, "Bob", sizeof(bob->name) - 1);
    bob->age   = 22;
    bob->score = 88.0f;
    print_student(bob);
    free(bob);

    // Build a simple linked list
    Node *head = create_node(1);
    head->next = create_node(2);
    head->next->next = create_node(3);

    // Traverse
    for (Node *cur = head; cur != NULL; cur = cur->next) {
        printf("%d -> ", cur->data);
    }
    printf("NULL\n");

    // Free
    Node *cur = head;
    while (cur) {
        Node *next = cur->next;
        free(cur);
        cur = next;
    }

    return 0;
}
```

## Pointer Aliasing and restrict

The `restrict` keyword (C99) tells the compiler that a pointer is the **only** way to access the object it points to — no other pointer aliases it. This enables optimizations the compiler couldn't safely make otherwise.

```c
// Without restrict — compiler must assume a and b might alias;
// c may point to same memory as a or b
void add_arrays(float *a, const float *b, const float *c, size_t n)
{
    for (size_t i = 0; i < n; i++)
        a[i] = b[i] + c[i];
    // Compiler must reload b[i] each iteration in case a aliases b
}

// With restrict — compiler knows no aliasing, can vectorize freely
void add_arrays_restricted(
    float *restrict a,
    const float *restrict b,
    const float *restrict c,
    size_t n)
{
    for (size_t i = 0; i < n; i++)
        a[i] = b[i] + c[i];
    // Compiler can use SIMD — no aliasing possible
}

// memcpy uses restrict (from <string.h>):
// void *memcpy(void *restrict dst, const void *restrict src, size_t n);
// vs memmove (no restrict — handles overlapping regions):
// void *memmove(void *dst, const void *src, size_t n);
```

## Pointer Pitfalls

<CardGrid>
  <Card title="Dangling Pointer" icon="warning">
    A pointer that points to memory that has been freed or has gone out of scope.
  </Card>
  <Card title="Wild Pointer" icon="warning">
    An uninitialized pointer containing a garbage address.
  </Card>
  <Card title="Buffer Overflow" icon="warning">
    Writing past the end of an array via pointer arithmetic.
  </Card>
  <Card title="Double Free" icon="warning">
    Calling `free()` on the same pointer twice — corrupts the heap.
  </Card>
</CardGrid>

```c
#include <stdlib.h>

// 1. DANGLING POINTER
int *dangling_example(void)
{
    int local = 42;
    return &local;  // WRONG: local lives on the stack; stack frame gone on return
}

void use_after_free(void)
{
    int *p = malloc(sizeof(int));
    *p = 5;
    free(p);
    *p = 10;  // USE-AFTER-FREE: undefined behavior
    // Fix: p = NULL; after free
}

// 2. WILD POINTER
void wild_pointer(void)
{
    int *p;           // uninitialized — contains garbage address
    *p = 42;          // UNDEFINED BEHAVIOR — writing to unknown location
    // Fix: int *p = NULL; or int *p = malloc(...);
}

// 3. BUFFER OVERFLOW
void buffer_overflow(void)
{
    int arr[5];
    for (int i = 0; i <= 5; i++) {  // BUG: should be i < 5
        arr[i] = i;                  // arr[5] is out of bounds!
    }
    // Overwrites adjacent memory — could corrupt stack frame
}

// 4. DOUBLE FREE
void double_free(void)
{
    int *p = malloc(sizeof(int));
    free(p);
    free(p);  // DOUBLE FREE: heap corruption, possible crash or security issue
    // Fix: free(p); p = NULL;  — freeing NULL is always safe (no-op)
}
```

:::caution[AddressSanitizer]
Use `-fsanitize=address` during development. AddressSanitizer (ASan) catches use-after-free, buffer overflows, use-after-scope, and double-free at runtime with minimal overhead. Enable it always for debug and test builds.
:::

## Common Pointer Patterns

```c
#include <stdio.h>
#include <stdlib.h>

// Pattern 1: Swap two values via pointers
void swap_int(int *a, int *b)
{
    int tmp = *a;
    *a = *b;
    *b = tmp;
}

// Pattern 2: Return multiple values via pointer parameters
void min_max(const int *arr, size_t n, int *out_min, int *out_max)
{
    *out_min = *out_max = arr[0];
    for (size_t i = 1; i < n; i++) {
        if (arr[i] < *out_min) *out_min = arr[i];
        if (arr[i] > *out_max) *out_max = arr[i];
    }
}

// Pattern 3: Iterator pattern — return pointer to next element (or NULL)
const int *find_first(const int *arr, size_t n, int value)
{
    for (size_t i = 0; i < n; i++) {
        if (arr[i] == value) return &arr[i];
    }
    return NULL;
}

int main(void)
{
    // Swap
    int x = 3, y = 7;
    swap_int(&x, &y);
    printf("x=%d y=%d\n", x, y);  // x=7 y=3

    // Multiple return values
    int data[] = { 3, 1, 4, 1, 5, 9, 2, 6 };
    int mn, mx;
    min_max(data, 8, &mn, &mx);
    printf("min=%d max=%d\n", mn, mx);  // min=1 max=9

    // Find
    const int *found = find_first(data, 8, 5);
    if (found) {
        printf("Found 5 at index %td\n", found - data);  // index 4
    }

    return 0;
}
```

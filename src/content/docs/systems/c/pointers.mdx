# Pointers in C

A pointer is a variable that stores a memory address. Mastering pointers is the single most important skill in C programming — they enable dynamic memory, efficient data structures, callbacks, and direct hardware access.

---

## Declaration and Initialization

```c
#include <stdio.h>
#include <stddef.h>   // NULL

int main(void) {
    int  value = 42;

    // Declare a pointer: type* name (the * is part of the declarator, not the type)
    int *ptr;          // uninitialized — dangerous (wild pointer)
    int *p = &value;   // p holds the address of value
    int *q = NULL;     // safe uninitialized pointer — always initialize to NULL

    // Address-of operator &: produces the address of a variable
    printf("Address of value: %p\n", (void *)&value);

    // Dereference operator *: access the value at the stored address
    printf("Value via pointer: %d\n", *p);    // 42

    // Modifying through a pointer
    *p = 100;
    printf("value is now: %d\n", value);      // 100

    // Multiple pointer declarations (the * binds to the name, not the type)
    int *a, b;     // a is int*, b is int  ← common mistake!
    int *c, *d;   // both are int* — declare separately to avoid confusion

    return 0;
}
```

:::tip[Pointer Initialization]
**Always** initialize pointers, either to `NULL` or to a valid address. Reading or writing through an uninitialized pointer is undefined behavior and can corrupt memory silently.
:::

## The NULL Pointer

```c
#include <stddef.h>   // NULL (= (void*)0 in C)
#include <stdio.h>

int *find_value(int *arr, int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) {
            return &arr[i];   // return pointer to found element
        }
    }
    return NULL;   // sentinel: not found
}

int main(void) {
    int data[] = {10, 20, 30, 40, 50};
    int *result = find_value(data, 5, 30);

    // ALWAYS check for NULL before dereferencing
    if (result != NULL) {
        printf("Found: %d\n", *result);
    } else {
        printf("Not found\n");
    }

    // In C, NULL is typically defined as (void*)0 or just 0
    // In C11, you can use the null pointer constant directly
    int *np = 0;     // valid, but prefer NULL for clarity
    int *np2 = NULL; // idiomatic

    return 0;
}
```

## Pointer Arithmetic

Pointer arithmetic operates in units of the pointed-to type — adding 1 to `int *` advances by `sizeof(int)` bytes.

```c
#include <stdio.h>

int main(void) {
    int arr[] = {10, 20, 30, 40, 50};
    int *p = arr;    // p points to arr[0]

    // Adding an integer to a pointer
    int *p1 = p + 1;   // points to arr[1] — advances by sizeof(int) bytes
    int *p2 = p + 4;   // points to arr[4]

    printf("arr[0] = %d\n", *p);       // 10
    printf("arr[1] = %d\n", *p1);      // 20
    printf("arr[4] = %d\n", *p2);      // 50

    // Pointer increment/decrement
    p++;             // p now points to arr[1]
    printf("After p++: %d\n", *p);   // 20

    // Pointer subtraction — result is ptrdiff_t (number of elements between)
    int *start = arr;
    int *end   = arr + 5;
    ptrdiff_t count = end - start;   // 5
    printf("Elements: %td\n", count);

    // Array subscript is equivalent to pointer arithmetic
    printf("arr[2] = %d\n", arr[2]);       // 30
    printf("*(arr+2) = %d\n", *(arr + 2)); // 30 — identical

    // Walking an array with a pointer
    for (int *cur = arr; cur < arr + 5; cur++) {
        printf("%d ", *cur);
    }
    printf("\n");

    // IMPORTANT: arithmetic is only valid within the same array (or one past end)
    // Going out of bounds: UB

    return 0;
}
```

## Pointer to Pointer

```c
#include <stdio.h>
#include <stdlib.h>

// Classic use: modifying a pointer passed to a function
void allocate_buffer(char **out_ptr, int size) {
    *out_ptr = malloc(size);  // modify the caller's pointer
}

int main(void) {
    int  value = 42;
    int *p     = &value;
    int **pp   = &p;     // pointer to pointer

    printf("value  = %d\n",  value);  // 42
    printf("*p     = %d\n",  *p);     // 42
    printf("**pp   = %d\n",  **pp);   // 42

    // Modify through double pointer
    **pp = 100;
    printf("value is now %d\n", value);  // 100

    // 2D dynamic array (array of pointers to rows)
    int rows = 3, cols = 4;
    int **matrix = malloc(rows * sizeof(int *));
    for (int i = 0; i < rows; i++) {
        matrix[i] = malloc(cols * sizeof(int));
        for (int j = 0; j < cols; j++) {
            matrix[i][j] = i * cols + j;
        }
    }

    printf("matrix[1][2] = %d\n", matrix[1][2]);  // 6

    // Cleanup
    for (int i = 0; i < rows; i++) free(matrix[i]);
    free(matrix);

    // allocate via double pointer
    char *buf = NULL;
    allocate_buffer(&buf, 256);
    if (buf) {
        buf[0] = 'H';
        free(buf);
    }

    return 0;
}
```

## void Pointer — Generic Pointer

`void *` is a generic pointer type. Any object pointer can be implicitly converted to and from `void *` in C. This is how `malloc` works.

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Generic swap using void*
void swap(void *a, void *b, size_t size) {
    unsigned char temp[size];  // VLA (C99) or use a fixed max
    memcpy(temp, a, size);
    memcpy(a, b, size);
    memcpy(b, temp, size);
}

int main(void) {
    // malloc returns void* — implicit conversion to any pointer type
    int *arr = malloc(10 * sizeof(int));  // no cast needed in C (unlike C++)
    if (!arr) return 1;

    // void* cannot be dereferenced directly — must cast first
    void *vp = arr;
    // *vp = 5;           // ❌ compile error: cannot dereference void*
    *(int *)vp = 42;    // ✅ cast to int* first, then dereference

    // Generic swap
    int x = 10, y = 20;
    printf("Before: x=%d y=%d\n", x, y);
    swap(&x, &y, sizeof(int));
    printf("After:  x=%d y=%d\n", x, y);  // x=20, y=10

    double a = 1.5, b = 2.5;
    swap(&a, &b, sizeof(double));
    printf("Doubles: a=%.1f b=%.1f\n", a, b);  // a=2.5, b=1.5

    free(arr);
    return 0;
}
```

## const Pointer Variants

There are three distinct `const` combinations for pointers:

```c
int value = 42;
int other = 99;

// 1. Pointer to const int — cannot modify the value through this pointer
const int *p1 = &value;
// *p1 = 100;        // ❌ error: read-only location
p1 = &other;         // ✅ can reassign the pointer itself

// 2. const pointer to int — cannot change where the pointer points
int * const p2 = &value;
*p2 = 100;           // ✅ can modify the value
// p2 = &other;      // ❌ error: assignment to const pointer

// 3. const pointer to const int — neither can be changed
const int * const p3 = &value;
// *p3 = 100;        // ❌ error
// p3 = &other;      // ❌ error

// Reading tip: read right-to-left
// const int *p    → p is a pointer to int that is const
// int * const p   → p is a const pointer to int
// const int * const p → p is a const pointer to const int

// Function parameters: use const to promise not to modify
void print_array(const int *arr, int n) {
    for (int i = 0; i < n; i++) printf("%d ", arr[i]);
}
```

## Function Pointers

Function pointers store the address of a function, enabling callbacks, dispatch tables, and runtime polymorphism.

```c
#include <stdio.h>
#include <stdlib.h>

// Function pointer type: return_type (*name)(param_types)
int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }

// Typedef for readability
typedef int (*BinaryOp)(int, int);

// Function accepting a callback
void apply_and_print(int a, int b, BinaryOp op) {
    printf("Result: %d\n", op(a, b));
}

// Array of function pointers (dispatch table)
int main(void) {
    // Declare and assign function pointer
    int (*fp)(int, int);
    fp = add;
    printf("add(3,4) = %d\n", fp(3, 4));   // 7

    fp = sub;
    printf("sub(10,3) = %d\n", fp(10, 3)); // 7

    // Using typedef
    BinaryOp ops[] = { add, sub, mul };
    const char *names[] = { "add", "sub", "mul" };

    for (int i = 0; i < 3; i++) {
        printf("%s(6, 3) = %d\n", names[i], ops[i](6, 3));
    }

    // Passing as callback
    apply_and_print(10, 5, add);
    apply_and_print(10, 5, sub);

    // qsort example — stdlib sort with comparison callback
    int arr[] = { 5, 2, 8, 1, 9, 3 };
    int compare_ints(const void *a, const void *b) {
        return *(const int *)a - *(const int *)b;
    }
    qsort(arr, 6, sizeof(int), compare_ints);
    for (int i = 0; i < 6; i++) printf("%d ", arr[i]);  // 1 2 3 5 8 9
    printf("\n");

    return 0;
}
```

## Pointers to Structs

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char name[64];
    int  age;
    double gpa;
} Student;

// Pass large struct by pointer (avoids copy)
void print_student(const Student *s) {
    printf("Name: %s, Age: %d, GPA: %.2f\n", s->name, s->age, s->gpa);
    //                                           ^^ arrow operator: (*s).name
}

void birthday(Student *s) {
    s->age++;  // modify through pointer
}

// Linked list node
typedef struct Node {
    int          data;
    struct Node *next;  // pointer to same type (struct name required here)
} Node;

Node *create_node(int data) {
    Node *n = malloc(sizeof(Node));
    if (!n) return NULL;
    n->data = data;
    n->next = NULL;
    return n;
}

int main(void) {
    Student alice = { .name = "Alice", .age = 20, .gpa = 3.8 };
    Student *sp = &alice;

    // Arrow operator -> is equivalent to (*sp).name
    printf("Name: %s\n", sp->name);      // Alice
    printf("Name: %s\n", (*sp).name);    // same thing

    print_student(sp);
    birthday(sp);
    printf("After birthday: %d\n", sp->age);  // 21

    // Dynamic allocation of struct
    Student *bob = malloc(sizeof(Student));
    strncpy(bob->name, "Bob", sizeof(bob->name) - 1);
    bob->age = 22;
    bob->gpa = 3.5;
    print_student(bob);
    free(bob);

    return 0;
}
```

## restrict Qualifier

`restrict` is a hint to the compiler that no other pointer aliases the pointed-to memory within the current scope, enabling more aggressive optimization:

```c
// Without restrict: compiler must assume possible aliasing
void add_arrays(float *dst, const float *a, const float *b, int n) {
    for (int i = 0; i < n; i++) dst[i] = a[i] + b[i];
}

// With restrict: compiler knows dst, a, b don't overlap — can vectorize
void add_arrays_fast(float * restrict dst,
                     const float * restrict a,
                     const float * restrict b,
                     int n) {
    for (int i = 0; i < n; i++) dst[i] = a[i] + b[i];
}

// memcpy is declared with restrict; memmove is not (handles overlap)
// void *memcpy(void * restrict dst, const void * restrict src, size_t n);
```

## Common Pointer Pitfalls

:::caution[The Four Deadly Pointer Sins]
Most C security vulnerabilities trace back to one of these four pointer errors.
:::

```c
#include <stdlib.h>
#include <stdio.h>

// 1. DANGLING POINTER — pointer to memory that has been freed or gone out of scope
int *dangling_local(void) {
    int local = 42;
    return &local;   // ❌ local is destroyed when function returns
}

int *dangling_heap(void) {
    int *p = malloc(sizeof(int));
    *p = 99;
    free(p);
    return p;        // ❌ pointer to freed memory
}

// 2. BUFFER OVERFLOW — writing past the end of an allocation
void overflow_example(void) {
    int arr[5];
    arr[5] = 99;     // ❌ UB — writes one past end
    // May corrupt adjacent memory, overwrite return address, etc.
}

// 3. DOUBLE FREE — freeing the same memory twice
void double_free_example(void) {
    int *p = malloc(sizeof(int));
    free(p);
    free(p);         // ❌ UB — corrupts malloc's internal state
    // Fix: set p = NULL after free
}

// 4. WILD POINTER — using uninitialized pointer
void wild_pointer_example(void) {
    int *p;          // uninitialized — points to random address
    *p = 42;         // ❌ UB — writing to unknown memory
}

// BEST PRACTICES
void safe_patterns(void) {
    int *p = malloc(sizeof(int));
    if (!p) return;   // always check malloc
    *p = 42;
    free(p);
    p = NULL;        // ✅ nullify after free — catches double-free
}
```

## Common Pointer Patterns

```c
#include <stdio.h>

// Pattern 1: Swap two values via pointers
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

// Pattern 2: Return multiple values via pointer parameters
void min_max(const int *arr, int n, int *out_min, int *out_max) {
    *out_min = *out_max = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] < *out_min) *out_min = arr[i];
        if (arr[i] > *out_max) *out_max = arr[i];
    }
}

// Pattern 3: String traversal
int my_strlen(const char *s) {
    const char *p = s;
    while (*p != '\0') p++;       // walk until null terminator
    return (int)(p - s);           // pointer subtraction = length
}

// Pattern 4: Pointer to end of array (sentinel)
int sum_array(const int *begin, const int *end) {
    int total = 0;
    for (const int *p = begin; p != end; p++) {
        total += *p;
    }
    return total;
}

int main(void) {
    int x = 10, y = 20;
    swap(&x, &y);
    printf("x=%d y=%d\n", x, y);   // x=20, y=10

    int arr[] = {3, 1, 4, 1, 5, 9, 2, 6};
    int mn, mx;
    min_max(arr, 8, &mn, &mx);
    printf("min=%d max=%d\n", mn, mx);  // min=1, max=9

    printf("strlen=\"hello\" = %d\n", my_strlen("hello"));  // 5

    printf("sum = %d\n", sum_array(arr, arr + 8));  // 31

    return 0;
}
```

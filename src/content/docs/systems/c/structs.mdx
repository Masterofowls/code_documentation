# Structs, Unions, and Enums in C

Structs are the primary tool for grouping related data in C. Unions provide overlapping memory layouts. Enums define named integer constants. Together they form the foundation for building complex data structures.

---

## Struct Declaration and Definition

```c
#include <stdio.h>
#include <string.h>

// Struct declaration — introduces a new aggregate type
struct Point {
    int x;
    int y;
};

// typedef creates an alias — allows using Point instead of struct Point
typedef struct Point Point;

// Combined struct + typedef (most common style)
typedef struct {
    char   first[32];
    char   last[32];
    int    age;
    double gpa;
} Student;

// Forward declaration (incomplete type — useful for linked lists, circular refs)
typedef struct Node Node;
struct Node {
    int   data;
    Node *next;   // pointer to incomplete type is allowed
};

int main(void) {
    // Stack-allocated struct
    struct Point p1;
    p1.x = 3;
    p1.y = 4;

    // Compound literal initialization (C99+)
    struct Point p2 = { 10, 20 };          // positional
    struct Point p3 = { .x = 10, .y = 20 }; // designated initializer (C99)

    // Zero-initialization
    Student s = {0};     // all fields set to zero
    strncpy(s.first, "Alice", sizeof(s.first) - 1);
    s.age = 21;
    s.gpa = 3.9;

    printf("Name: %s, Age: %d\n", s.first, s.age);

    // Struct copy (member-wise)
    Student s2 = s;     // copies all fields
    s2.age = 22;        // does not affect s

    return 0;
}
```

## Struct Initialization — Designated Initializers (C99)

Designated initializers let you initialize specific fields by name, in any order. Unspecified fields are zero-initialized.

```c
typedef struct {
    int   id;
    char  name[64];
    float score;
    int   active;
} Record;

// C89 style — positional (fragile, breaks on field reorder)
Record r1 = { 1, "Alice", 95.5f, 1 };

// C99 designated initializers — explicit and order-independent
Record r2 = {
    .id     = 2,
    .name   = "Bob",     // string literal assignment works in initializer
    .score  = 87.3f,
    .active = 1,
};

// Partial initialization — unspecified fields are 0
Record r3 = { .id = 3, .name = "Carol" };
// r3.score == 0.0f, r3.active == 0

// Nested struct with designated initializers
typedef struct {
    struct { int x, y; } position;
    struct { int w, h; } size;
} Rect;

Rect window = {
    .position = { .x = 100, .y = 200 },
    .size     = { .w = 800, .h = 600 },
};
```

## Struct Padding and Alignment

The compiler inserts padding bytes between struct members to satisfy alignment requirements:

```c
#include <stddef.h>
#include <stdio.h>

// Poorly ordered — wastes 7 bytes to padding
struct Wasteful {
    char   c;     // 1 byte  @ offset 0
                  // 7 bytes padding
    double d;     // 8 bytes @ offset 8
    char   c2;    // 1 byte  @ offset 16
                  // 7 bytes padding
};                // Total: 24 bytes

// Ordered largest to smallest — no waste
struct Efficient {
    double d;     // 8 bytes @ offset 0
    char   c;     // 1 byte  @ offset 8
    char   c2;    // 1 byte  @ offset 9
                  // 6 bytes padding (for next struct alignment)
};                // Total: 16 bytes

int main(void) {
    printf("Wasteful:  %zu bytes\n", sizeof(struct Wasteful));   // 24
    printf("Efficient: %zu bytes\n", sizeof(struct Efficient));  // 16

    printf("offsetof(Wasteful.d)  = %zu\n", offsetof(struct Wasteful, d));   // 8
    printf("offsetof(Wasteful.c2) = %zu\n", offsetof(struct Wasteful, c2));  // 16
    return 0;
}
```

### Packing Pragmas

```c
// Disable padding — USE WITH CAUTION (misaligned access may be slow or crash)

// GCC / Clang attribute (preferred)
typedef struct __attribute__((packed)) {
    char   a;   // offset 0
    int    b;   // offset 1 — no padding!
    short  c;   // offset 5
} PackedStruct;   // 7 bytes total

// MSVC pragma (also works in GCC/Clang)
#pragma pack(push, 1)
typedef struct {
    char  a;
    int   b;
    short c;
} PackedMSVC;
#pragma pack(pop)

// Use case: network protocols, binary file formats (exact byte layouts)
```

## Nested Structs and Anonymous Structs (C11)

```c
// Nested structs — regular
typedef struct {
    struct {
        float x, y;
    } position;
    struct {
        float w, h;
    } size;
} Widget;

Widget w;
w.position.x = 100.0f;
w.size.w     = 200.0f;

// Anonymous struct/union members (C11) — access fields directly
typedef struct {
    union {
        struct {
            float x, y, z;    // anonymous struct — no member name
        };
        float v[3];            // same memory
    };
    float w;
} Vec4;

Vec4 q;
q.x = 1.0f;    // access anonymous struct member directly
q.y = 0.0f;
q.z = 0.0f;
q.w = 1.0f;
printf("q.v[0] = %.1f\n", q.v[0]);  // 1.0 — same as q.x
```

## Passing Structs to Functions

```c
#include <stdio.h>

typedef struct { float x, y; } Vec2;

// Pass by value — copies entire struct (safe, but expensive for large structs)
float vec2_length_val(Vec2 v) {
    return sqrtf(v.x * v.x + v.y * v.y);
}

// Pass by const pointer — efficient, caller retains ownership
float vec2_length_ptr(const Vec2 *v) {
    return sqrtf(v->x * v->x + v->y * v->y);
}

// Return struct by value — compiler often uses NRVO to avoid copy
Vec2 vec2_add(Vec2 a, Vec2 b) {
    return (Vec2){ a.x + b.x, a.y + b.y };  // compound literal
}

// Modify in place via pointer
void vec2_normalize(Vec2 *v) {
    float len = vec2_length_ptr(v);
    if (len > 0.0001f) {
        v->x /= len;
        v->y /= len;
    }
}

int main(void) {
    Vec2 a = { 3.0f, 4.0f };
    Vec2 b = { 1.0f, 2.0f };

    printf("length = %.2f\n", vec2_length_val(a));    // 5.00
    printf("length = %.2f\n", vec2_length_ptr(&a));   // 5.00

    Vec2 sum = vec2_add(a, b);
    printf("sum = (%.1f, %.1f)\n", sum.x, sum.y);    // (4.0, 6.0)

    vec2_normalize(&a);
    printf("unit = (%.2f, %.2f)\n", a.x, a.y);       // (0.60, 0.80)

    return 0;
}
```

## Unions — Shared Memory Layout

A union stores all members at the same memory address. Only one member can hold a valid value at a time.

```c
#include <stdio.h>
#include <stdint.h>
#include <string.h>

// Union for type punning — reading float bits as uint32 (defined in C via union)
union FloatBits {
    float    f;
    uint32_t u;
};

float float_abs(float x) {
    union FloatBits fb;
    fb.f = x;
    fb.u &= 0x7FFFFFFF;   // clear sign bit
    return fb.f;
}

// Tagged union — discriminated union (variant type)
typedef enum { TYPE_INT, TYPE_FLOAT, TYPE_STRING } ValueType;

typedef struct {
    ValueType type;
    union {
        int    i;
        float  f;
        char   s[32];
    } data;
} Value;

void print_value(const Value *v) {
    switch (v->type) {
    case TYPE_INT:    printf("int: %d\n",    v->data.i); break;
    case TYPE_FLOAT:  printf("float: %.2f\n",v->data.f); break;
    case TYPE_STRING: printf("str: %s\n",    v->data.s); break;
    }
}

// Endianness detection via union
int is_little_endian(void) {
    union { uint32_t u; uint8_t b[4]; } x = { .u = 0x01020304 };
    return x.b[0] == 4;   // little-endian: LSB at lowest address
}

int main(void) {
    printf("abs(-3.14) = %.2f\n", float_abs(-3.14f));  // 3.14

    Value v1 = { .type = TYPE_INT,    .data.i = 42 };
    Value v2 = { .type = TYPE_FLOAT,  .data.f = 3.14f };
    Value v3 = { .type = TYPE_STRING };
    strncpy(v3.data.s, "hello", sizeof(v3.data.s) - 1);
    v3.type = TYPE_STRING;

    print_value(&v1);
    print_value(&v2);
    print_value(&v3);

    printf("Little endian: %s\n", is_little_endian() ? "yes" : "no");

    return 0;
}
```

## Bit Fields

Bit fields allow packing multiple integer values into a single word:

```c
#include <stdio.h>
#include <stdint.h>

// CPU status register example
typedef struct {
    unsigned int carry     : 1;   // 1-bit field
    unsigned int zero      : 1;
    unsigned int overflow  : 1;
    unsigned int sign      : 1;
    unsigned int parity    : 1;
    unsigned int reserved  : 3;  // unused bits
    unsigned int mode      : 2;  // 0-3 four modes
    unsigned int           : 6;  // unnamed padding
} StatusReg;

// Network packet header (big-endian fields — PORTABILITY WARNING)
typedef struct {
    uint8_t  version  : 4;   // IP version
    uint8_t  ihl      : 4;   // header length
    uint8_t  dscp     : 6;
    uint8_t  ecn      : 2;
} IPHeader;

int main(void) {
    StatusReg sr = {0};
    sr.carry    = 1;
    sr.zero     = 0;
    sr.mode     = 2;   // 0b10

    printf("carry=%u zero=%u mode=%u\n", sr.carry, sr.zero, sr.mode);
    printf("sizeof(StatusReg) = %zu\n", sizeof(StatusReg));

    // ❌ You cannot take the address of a bit field
    // &sr.carry;  // compile error

    return 0;
}
```

:::caution[Bit Field Portability]
Bit field layout (order, padding, straddling word boundaries) is implementation-defined. For portable binary formats, use bit masking and shifting on `uint8_t`/`uint16_t`/`uint32_t` instead.
:::

## Enums

```c
#include <stdio.h>

// Basic enum — values start at 0, increment by 1
typedef enum {
    DIRECTION_NORTH,   // 0
    DIRECTION_SOUTH,   // 1
    DIRECTION_EAST,    // 2
    DIRECTION_WEST,    // 3
    DIRECTION_COUNT,   // 4 — use for array sizing
} Direction;

// Enum with explicit values
typedef enum {
    LOG_DEBUG   = 0,
    LOG_INFO    = 1,
    LOG_WARNING = 2,
    LOG_ERROR   = 3,
    LOG_FATAL   = 4,
} LogLevel;

// Bitmask enum (powers of 2 for flag combinations)
typedef enum {
    PERM_NONE    = 0,
    PERM_READ    = 1 << 0,   // 1
    PERM_WRITE   = 1 << 1,   // 2
    PERM_EXECUTE = 1 << 2,   // 4
    PERM_ALL     = PERM_READ | PERM_WRITE | PERM_EXECUTE,
} Permission;

const char *direction_name(Direction d) {
    static const char *names[] = {
        [DIRECTION_NORTH] = "North",
        [DIRECTION_SOUTH] = "South",
        [DIRECTION_EAST]  = "East",
        [DIRECTION_WEST]  = "West",
    };
    if (d < DIRECTION_COUNT) return names[d];
    return "Unknown";
}

int main(void) {
    Direction dir = DIRECTION_NORTH;
    printf("Going: %s\n", direction_name(dir));

    Permission perms = PERM_READ | PERM_WRITE;
    if (perms & PERM_READ)  printf("Can read\n");
    if (perms & PERM_WRITE) printf("Can write\n");
    if (!(perms & PERM_EXECUTE)) printf("Cannot execute\n");

    // In C, enum values are int — no type safety!
    int n = DIRECTION_NORTH;   // ✅ compiles (unlike C++ scoped enums)
    Direction d2 = 42;         // ✅ compiles — no range check

    return 0;
}
```

## Struct of Arrays vs Array of Structs

```c
// Array of Structs (AoS) — intuitive, bad cache for partial field access
typedef struct {
    float x, y, z;
    float health;
    int   id;
    int   flags;
} Entity_AoS;
Entity_AoS entities[10000];

// Processing health: loads x,y,z,flags etc even though we only need health
for (int i = 0; i < 10000; i++) {
    entities[i].health -= 1.0f;  // cache misses on every access
}

// Struct of Arrays (SoA) — best performance for SIMD and partial access
typedef struct {
    float x[10000], y[10000], z[10000];  // positions together
    float health[10000];                  // health contiguous
    int   id[10000];
    int   flags[10000];
} Entities_SoA;
Entities_SoA entities_soa;

// Processing health: only health[] loaded — 100% cache utilization
for (int i = 0; i < 10000; i++) {
    entities_soa.health[i] -= 1.0f;   // sequential memory — perfect for prefetcher
}
```

## Opaque Pointer — Handle Pattern

The opaque pointer pattern hides struct implementation details, providing encapsulation in C:

```c
// --- queue.h (public header — users only see this) ---
typedef struct Queue Queue;   // incomplete type — size and fields hidden

Queue *queue_create(int capacity);
void   queue_destroy(Queue *q);
int    queue_push(Queue *q, int value);
int    queue_pop(Queue *q, int *out_value);
int    queue_size(const Queue *q);

// --- queue.c (implementation — users never include this) ---
#include "queue.h"
#include <stdlib.h>

struct Queue {          // full definition hidden in .c file
    int *data;
    int  head, tail;
    int  size, capacity;
};

Queue *queue_create(int capacity) {
    Queue *q = malloc(sizeof(Queue));
    if (!q) return NULL;
    q->data     = malloc(capacity * sizeof(int));
    q->head     = 0;
    q->tail     = 0;
    q->size     = 0;
    q->capacity = capacity;
    return q;
}

void queue_destroy(Queue *q) {
    if (q) { free(q->data); free(q); }
}
```

:::tip[Opaque Handles]
The opaque pointer pattern is C's equivalent of a class with private fields. Users get a handle (`Queue *`) they can pass around but cannot inspect or modify directly. This enables ABI stability — you can change the struct layout without recompiling users.
:::

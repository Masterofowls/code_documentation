# Structs, Unions & Enums in C

Structs are the primary mechanism for grouping related data in C. Combined with unions, bit fields, and enumerations, they enable rich, efficient data structures.

## Struct Basics

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// Declaration: introduces the type
struct Point {
    float x;
    float y;
};

// typedef: creates an alias — avoids writing 'struct' everywhere
typedef struct {
    float x;
    float y;
} Vec2;

// Both of these are equivalent after typedef:
struct Point p1;
Vec2 p2;

// Initialization — C99+ designated initializers (order doesn't matter)
struct Point origin = { .x = 0.0f, .y = 0.0f };
Vec2 unit_x = { .x = 1.0f, .y = 0.0f };

// Zero-initialization shorthand
struct Point zeroed = {0};   // All fields set to 0

// Nested struct initialization
typedef struct {
    Vec2 position;
    Vec2 velocity;
    int  id;
    char name[32];
} Entity;

Entity player = {
    .position = { .x = 100.0f, .y = 200.0f },
    .velocity = { .x = 0.0f,   .y = 0.0f   },
    .id       = 1,
    .name     = "Player",
};

// Accessing members
printf("Player pos: (%.1f, %.1f)\n", player.position.x, player.position.y);
player.velocity.x = 5.0f;
player.id = 2;
```

## Struct Padding and Alignment

The compiler inserts padding bytes between fields to ensure proper alignment. Misunderstanding this can waste memory and cause subtle bugs.

```c
#include <stdio.h>
#include <stddef.h>

// BAD layout — lots of padding
struct BadLayout {
    char   a;    // 1 byte  @ offset 0
    // 7 bytes padding (align double to 8-byte boundary)
    double b;    // 8 bytes @ offset 8
    char   c;    // 1 byte  @ offset 16
    // 3 bytes padding (align int to 4-byte boundary)
    int    d;    // 4 bytes @ offset 20
    // Total: 24 bytes
};

// GOOD layout — fields ordered largest to smallest
struct GoodLayout {
    double b;    // 8 bytes @ offset 0
    int    d;    // 4 bytes @ offset 8
    char   a;    // 1 byte  @ offset 12
    char   c;    // 1 byte  @ offset 13
    // 2 bytes padding (struct alignment = 8, size must be multiple of 8)
    // Total: 16 bytes
};

void print_layout(void) {
    printf("BadLayout size:  %zu bytes\n", sizeof(struct BadLayout));   // 24
    printf("GoodLayout size: %zu bytes\n", sizeof(struct GoodLayout));  // 16

    printf("BadLayout offsets: a=%zu b=%zu c=%zu d=%zu\n",
        offsetof(struct BadLayout, a),
        offsetof(struct BadLayout, b),
        offsetof(struct BadLayout, c),
        offsetof(struct BadLayout, d));

    printf("GoodLayout offsets: b=%zu d=%zu a=%zu c=%zu\n",
        offsetof(struct GoodLayout, b),
        offsetof(struct GoodLayout, d),
        offsetof(struct GoodLayout, a),
        offsetof(struct GoodLayout, c));
}

// Force packing (removes padding — may cause performance penalty on some CPUs)
#pragma pack(push, 1)
struct Packed {
    char   a;    // 1 byte @ 0
    double b;    // 8 bytes @ 1 (unaligned!)
    char   c;    // 1 byte @ 9
    int    d;    // 4 bytes @ 10 (unaligned!)
};               // Total: 14 bytes — no padding
#pragma pack(pop)

// GCC alternative:
struct __attribute__((packed)) GccPacked {
    char a; double b; char c; int d;
};
```

:::caution[Packed Structs and Performance]
Packed structs may cause **misaligned memory access**, which is slow on x86 and causes a fault on some ARM and RISC-V configurations. Use packing only for serialization formats, network packets, or hardware register maps where you need exact byte layout.
:::

## Nested Structs and Pointers

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct Node {
    int data;
    struct Node *left;   // Self-referential: must use 'struct Node' here
    struct Node *right;
} Node;

// Forward declaration (incomplete type)
typedef struct HashTable HashTable;  // Type declared but not yet defined
// Functions can accept/return HashTable* because pointer size is known
void hash_table_insert(HashTable *ht, const char *key, int value);

// Now define it (in the same or another translation unit)
struct HashTable {
    Node **buckets;
    size_t bucket_count;
    size_t item_count;
};

// Passing by value vs pointer
typedef struct { float x, y, z; } Vec3;

// Pass by value: works for small structs (copied on the stack)
float vec3_dot(Vec3 a, Vec3 b) {
    return a.x*b.x + a.y*b.y + a.z*b.z;
}

// Pass by pointer: preferred for large structs
void vec3_normalize(Vec3 *v) {
    float len = sqrtf(v->x*v->x + v->y*v->y + v->z*v->z);
    if (len > 0.0f) { v->x /= len; v->y /= len; v->z /= len; }
}

// Return struct by value (modern compilers use NRVO — no copy)
Vec3 vec3_add(Vec3 a, Vec3 b) {
    return (Vec3){ a.x + b.x, a.y + b.y, a.z + b.z };
}
```

## Unions

A union allocates all members at the same memory location. Only one member is "active" at any time. The size of a union is the size of its largest member.

```c
#include <stdio.h>
#include <stdint.h>

// Union: shared memory
union Number {
    int    i;
    float  f;
    double d;
};

// Common use: type-punning (read memory as a different type)
// C11+ makes this defined behavior (was technically UB in C99)
union FloatBits {
    float    f;
    uint32_t bits;
};

float fast_inverse_sqrt(float x) {
    union FloatBits fb;
    fb.f   = x;
    uint32_t i = 0x5f3759df - (fb.bits >> 1);
    fb.bits = i;
    float y = fb.f;
    y = y * (1.5f - 0.5f * x * y * y);  // Newton's method iteration
    return y;
}

// Endian detection with union
union EndianTest {
    uint32_t word;
    uint8_t  bytes[4];
};

int is_little_endian(void) {
    union EndianTest et;
    et.word = 0x01020304;
    return et.bytes[0] == 0x04;  // Little-endian: LSB at lowest address
}

// Tagged union: union with a type tag (manual variant type)
typedef enum { TAG_INT, TAG_FLOAT, TAG_STRING } Tag;

typedef struct {
    Tag tag;
    union {
        int    i;
        float  f;
        char  *s;
    };
} Value;

void print_value(const Value *v) {
    switch (v->tag) {
        case TAG_INT:    printf("int: %d\n", v->i); break;
        case TAG_FLOAT:  printf("float: %.2f\n", v->f); break;
        case TAG_STRING: printf("string: %s\n", v->s); break;
    }
}

int main(void) {
    printf("Is little-endian: %s\n", is_little_endian() ? "yes" : "no");

    Value vals[] = {
        { .tag = TAG_INT,    .i = 42 },
        { .tag = TAG_FLOAT,  .f = 3.14f },
        { .tag = TAG_STRING, .s = "hello" },
    };

    for (int i = 0; i < 3; i++) print_value(&vals[i]);

    float x = 4.0f;
    printf("1/sqrt(%.1f) ≈ %.6f\n", x, fast_inverse_sqrt(x));

    return 0;
}
```

## Anonymous Structs and Unions (C11)

```c
#include <stdio.h>

// Anonymous union inside struct: members accessible directly
typedef struct {
    int type;
    union {             // Anonymous: no member name
        int    i_val;   // Access as event.i_val
        float  f_val;   // Access as event.f_val
        struct {        // Anonymous struct inside anonymous union
            int x, y;   // Access as event.x, event.y
        };
    };
} Event;

// Anonymous struct inside union (common for vector math)
typedef struct {
    union {
        struct { float x, y, z, w; };  // Access as v.x, v.y, v.z, v.w
        float e[4];                      // Access as v.e[0], v.e[1]...
    };
} Vec4;

int main(void) {
    Event e = { .type = 1, .i_val = 42 };
    printf("type=%d, i_val=%d\n", e.type, e.i_val);

    e.type  = 2;
    e.f_val = 3.14f;
    printf("type=%d, f_val=%.2f\n", e.type, e.f_val);

    Vec4 v = { .x = 1.0f, .y = 2.0f, .z = 3.0f, .w = 1.0f };
    printf("v.x=%.1f, v.e[0]=%.1f\n", v.x, v.e[0]);  // Same memory

    return 0;
}
```

## Bit Fields

Bit fields allow packing multiple values into a single integer, useful for flags, hardware registers, and compact data.

```c
#include <stdio.h>

// CPU flags register simulation
typedef struct {
    unsigned int carry    : 1;   // 1 bit wide
    unsigned int zero     : 1;
    unsigned int overflow : 1;
    unsigned int negative : 1;
    unsigned int : 4;            // 4 unnamed bits (padding)
    unsigned int mode     : 2;   // 2-bit value (0-3)
    unsigned int : 6;            // 6 unnamed bits
    unsigned int reserved : 16;  // Reserved
} StatusFlags;

// Network packet header (packed layout with bit fields)
typedef struct {
    unsigned int version  : 4;   // 4 bits: IP version
    unsigned int ihl      : 4;   // 4 bits: header length
    unsigned int dscp     : 6;   // 6 bits: differentiated services
    unsigned int ecn      : 2;   // 2 bits: explicit congestion
    unsigned int length   : 16;  // 16 bits: total length
} IPHeader;

int main(void) {
    StatusFlags sf = {0};
    sf.zero  = 1;
    sf.mode  = 2;

    printf("sizeof(StatusFlags) = %zu\n", sizeof(StatusFlags));  // 4
    printf("zero=%u, mode=%u\n", sf.zero, sf.mode);

    // Bit fields are NOT portable: layout depends on compiler and endianness
    // Don't use for serialization; use explicit bitwise operations instead

    // Explicit bit manipulation (portable alternative):
    uint32_t flags = 0;
    flags |= (1U << 1);  // Set "zero" flag
    flags |= (2U << 8);  // Set "mode" = 2
    printf("flags via bits: 0x%08X\n", flags);

    return 0;
}
```

## Enumerations

```c
#include <stdio.h>

// Basic enum (underlying type is int in C)
enum Direction { NORTH, SOUTH, EAST, WEST };  // 0, 1, 2, 3

// Enum with explicit values
enum Color {
    RED   = 0xFF0000,
    GREEN = 0x00FF00,
    BLUE  = 0x0000FF,
};

// Enum as flags (powers of 2)
typedef enum {
    FLAG_NONE     = 0,
    FLAG_READABLE = 1 << 0,   // 1
    FLAG_WRITABLE = 1 << 1,   // 2
    FLAG_EXEC     = 1 << 2,   // 4
    FLAG_HIDDEN   = 1 << 3,   // 8
    FLAG_ALL      = FLAG_READABLE | FLAG_WRITABLE | FLAG_EXEC
} FileFlags;

const char *direction_name(enum Direction d) {
    switch (d) {
        case NORTH: return "North";
        case SOUTH: return "South";
        case EAST:  return "East";
        case WEST:  return "West";
        default:    return "Unknown";
    }
}

int main(void) {
    enum Direction dir = NORTH;
    printf("Direction: %s (%d)\n", direction_name(dir), dir);

    FileFlags perms = FLAG_READABLE | FLAG_WRITABLE;
    if (perms & FLAG_READABLE) printf("Readable\n");
    if (perms & FLAG_WRITABLE) printf("Writable\n");
    if (!(perms & FLAG_EXEC))  printf("Not executable\n");

    // Note: C enums are just ints — no type safety
    enum Direction x = 99;  // Compiles! No range check.
    printf("Invalid direction: %d\n", x);

    return 0;
}
```

## SoA vs AoS Performance

Structure of Arrays (SoA) vs Array of Structures (AoS) is a crucial optimization for data-intensive code.

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define N 1000000

// AoS: Array of Structures — common intuitive layout
typedef struct {
    float x, y, z;   // Position
    float vx, vy, vz; // Velocity
    float mass;
    int   alive;
} ParticleAoS;

// SoA: Structure of Arrays — cache-friendly for partial field access
typedef struct {
    float *x,  *y,  *z;    // All X positions, all Y positions...
    float *vx, *vy, *vz;
    float *mass;
    int   *alive;
} ParticleSoA;

// AoS update: loads all fields even though we only need x,y,z,vx,vy,vz
void update_aos(ParticleAoS *particles, int n, float dt) {
    for (int i = 0; i < n; i++) {
        if (particles[i].alive) {
            particles[i].x += particles[i].vx * dt;
            particles[i].y += particles[i].vy * dt;
            particles[i].z += particles[i].vz * dt;
        }
    }
}

// SoA update: accesses only the needed arrays — better cache behavior
// Compiler can also vectorize (SIMD) this more easily
void update_soa(ParticleSoA *p, int n, float dt) {
    for (int i = 0; i < n; i++) {
        if (p->alive[i]) {
            p->x[i] += p->vx[i] * dt;
            p->y[i] += p->vy[i] * dt;
            p->z[i] += p->vz[i] * dt;
        }
    }
}
```

:::tip[When to Use SoA]
Use SoA when:

- You iterate through all entities processing only a few fields (e.g., game physics updating position)
- SIMD vectorization is desired (packed arrays of the same type vectorize naturally)
- The number of objects is large (thousands+)

Use AoS when:

- You always access all fields of an entity together
- The number of objects is small
- Readability/ergonomics matter more than throughput
  :::

## Opaque Pointer Pattern

The opaque pointer (handle) pattern hides implementation details — a form of encapsulation in C.

```c
// queue.h — public interface
#ifndef QUEUE_H
#define QUEUE_H

// Incomplete type: callers see only the pointer, not the internals
typedef struct Queue Queue;

Queue *queue_create(size_t capacity);
void   queue_destroy(Queue *q);
int    queue_push(Queue *q, int value);
int    queue_pop(Queue *q, int *out_value);
int    queue_empty(const Queue *q);
size_t queue_size(const Queue *q);

#endif

// queue.c — private implementation (callers never see this)
#include "queue.h"
#include <stdlib.h>
#include <string.h>

struct Queue {         // Full definition hidden from callers
    int   *data;
    size_t head, tail, size, capacity;
};

Queue *queue_create(size_t capacity) {
    Queue *q = malloc(sizeof(Queue));
    if (!q) return NULL;
    q->data = malloc(capacity * sizeof(int));
    if (!q->data) { free(q); return NULL; }
    q->head = q->tail = q->size = 0;
    q->capacity = capacity;
    return q;
}

void queue_destroy(Queue *q) {
    if (q) { free(q->data); free(q); }
}

int queue_push(Queue *q, int value) {
    if (q->size == q->capacity) return 0;  // Full
    q->data[q->tail] = value;
    q->tail = (q->tail + 1) % q->capacity;
    q->size++;
    return 1;
}

int queue_pop(Queue *q, int *out) {
    if (q->size == 0) return 0;  // Empty
    *out = q->data[q->head];
    q->head = (q->head + 1) % q->capacity;
    q->size--;
    return 1;
}

int    queue_empty(const Queue *q) { return q->size == 0; }
size_t queue_size(const Queue *q)  { return q->size; }
```

## Flexible Array Members

A flexible array member allows a struct to have a variable-length array at the end (C99+):

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    size_t length;
    int    data[];  // Flexible array member — must be last; no size specified
} IntArray;

IntArray *intarray_create(size_t n) {
    // Allocate struct + n integers in one allocation
    IntArray *arr = malloc(sizeof(IntArray) + n * sizeof(int));
    if (!arr) return NULL;
    arr->length = n;
    memset(arr->data, 0, n * sizeof(int));
    return arr;
}

int main(void) {
    IntArray *a = intarray_create(5);
    if (!a) return 1;

    for (size_t i = 0; i < a->length; i++) a->data[i] = (int)(i * i);
    for (size_t i = 0; i < a->length; i++) printf("%d ", a->data[i]);
    printf("\n");  // 0 1 4 9 16

    free(a);  // Single free for the whole thing
    return 0;
}
```

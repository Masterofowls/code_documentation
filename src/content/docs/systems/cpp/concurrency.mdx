# Concurrency in C++

C++11 introduced a portable threading model directly in the standard library. C++20 extended it with coroutines, `jthread`, semaphores, latches, and barriers.

## std::thread

```cpp
#include <thread>
#include <iostream>
#include <vector>
#include <numeric>

void worker(int id, int start, int end, std::vector<int>& results) {
    int sum = 0;
    for (int i = start; i < end; i++) sum += i;
    results[id] = sum;
}

int main() {
    // Spawn a thread
    std::thread t([]() {
        std::cout << "Hello from thread " << std::this_thread::get_id() << '\n';
    });

    // Must either join (wait) or detach (let run independently)
    t.join();   // blocks until thread finishes

    // Hardware parallelism hint
    unsigned n = std::thread::hardware_concurrency();  // logical CPU count

    // Parallel sum example
    std::vector<int> results(4);
    std::vector<std::thread> threads;
    int chunk = 1000 / 4;
    for (int i = 0; i < 4; i++) {
        threads.emplace_back(worker, i, i * chunk, (i + 1) * chunk, std::ref(results));
    }
    for (auto& th : threads) th.join();

    int total = std::accumulate(results.begin(), results.end(), 0);
    std::cout << "Sum: " << total << '\n';

    // thread_local — each thread gets its own copy
    thread_local int thread_id_counter = 0;
    (void)thread_id_counter;

    return 0;
}
```

## Mutex and Lock Guards

```cpp
#include <mutex>
#include <shared_mutex>  // C++17

std::mutex             g_mtx;
std::recursive_mutex   g_rec_mtx;  // same thread can lock multiple times
std::shared_mutex      g_rw_mtx;   // C++17: multiple readers OR one writer

// std::lock_guard — simplest; non-movable
{
    std::lock_guard<std::mutex> lock(g_mtx);
    // critical section
}  // unlocked automatically

// std::unique_lock — more flexible: deferred, timed, movable
{
    std::unique_lock<std::mutex> lock(g_mtx);
    lock.unlock();  // can unlock early
    // ... non-critical work ...
    lock.lock();    // re-lock
}

// std::scoped_lock (C++17) — lock multiple mutexes atomically (deadlock-free)
std::mutex m1, m2;
{
    std::scoped_lock lock(m1, m2);  // locks both in deadlock-free order
    // critical section
}

// std::shared_lock — reader lock (C++17)
{
    std::shared_lock<std::shared_mutex> read_lock(g_rw_mtx);
    // Multiple threads can hold shared_lock simultaneously
}
{
    std::unique_lock<std::shared_mutex> write_lock(g_rw_mtx);
    // Only one thread can hold unique_lock
}
```

## Deadlock Prevention

```cpp
#include <mutex>
#include <thread>

struct Account {
    std::mutex mtx;
    double balance;
    int    id;
};

// WRONG — transfer(A, B) and transfer(B, A) can deadlock
void transfer_bad(Account& from, Account& to, double amount) {
    std::lock_guard lk1(from.mtx);  // thread 1 locks A, thread 2 locks B
    std::lock_guard lk2(to.mtx);    // thread 1 waits for B, thread 2 waits for A → deadlock
    from.balance -= amount;
    to.balance   += amount;
}

// CORRECT — std::lock acquires both atomically
void transfer_good(Account& from, Account& to, double amount) {
    // Lock both simultaneously — no deadlock
    std::scoped_lock lk(from.mtx, to.mtx);  // C++17
    from.balance -= amount;
    to.balance   += amount;
}

// Consistent ordering — another deadlock prevention approach
void transfer_ordered(Account& from, Account& to, double amount) {
    Account* first  = (from.id < to.id) ? &from : &to;
    Account* second = (from.id < to.id) ? &to   : &from;
    std::scoped_lock lk(first->mtx, second->mtx);
    from.balance -= amount;
    to.balance   += amount;
}
```

## Condition Variables

```cpp
#include <mutex>
#include <condition_variable>
#include <queue>

// Thread-safe queue using condition variable
template<typename T>
class BoundedQueue {
public:
    explicit BoundedQueue(size_t cap) : capacity_(cap) {}

    void push(T val) {
        std::unique_lock lock(mtx_);
        not_full_.wait(lock, [this] { return queue_.size() < capacity_; });
        queue_.push(std::move(val));
        lock.unlock();
        not_empty_.notify_one();
    }

    T pop() {
        std::unique_lock lock(mtx_);
        // Loop guards against spurious wakeups
        not_empty_.wait(lock, [this] { return !queue_.empty(); });
        T val = std::move(queue_.front());
        queue_.pop();
        lock.unlock();
        not_full_.notify_one();
        return val;
    }

    // Timed wait
    std::optional<T> try_pop(std::chrono::milliseconds timeout) {
        std::unique_lock lock(mtx_);
        if (!not_empty_.wait_for(lock, timeout, [this] { return !queue_.empty(); })) {
            return std::nullopt;
        }
        T val = std::move(queue_.front());
        queue_.pop();
        return val;
    }

private:
    std::queue<T>           queue_;
    size_t                  capacity_;
    std::mutex              mtx_;
    std::condition_variable not_empty_;
    std::condition_variable not_full_;
};
```

## std::async and std::future

```cpp
#include <future>
#include <chrono>

int compute(int x) {
    std::this_thread::sleep_for(std::chrono::milliseconds(100));
    return x * x;
}

int main() {
    // std::async — run function asynchronously
    auto fut = std::async(std::launch::async, compute, 10);

    // Do other work while compute() runs in background
    std::cout << "Computing...\n";

    // get() blocks until result available
    int result = fut.get();  // 100
    std::cout << "Result: " << result << '\n';

    // launch::deferred — run lazily (only when .get() called)
    auto lazy = std::async(std::launch::deferred, compute, 5);
    // compute(5) hasn't run yet
    int lazy_result = lazy.get();  // runs now

    // std::promise — manually fulfil a future
    std::promise<int> prom;
    std::future<int>  fut2 = prom.get_future();

    std::thread([p = std::move(prom)]() mutable {
        std::this_thread::sleep_for(std::chrono::milliseconds(50));
        p.set_value(42);  // fulfil the promise
    }).detach();

    std::cout << fut2.get() << '\n';  // blocks until promise fulfilled

    // std::packaged_task — wraps callable, provides future
    std::packaged_task<int(int)> task(compute);
    std::future<int> task_fut = task.get_future();
    std::thread(std::move(task), 7).join();
    std::cout << task_fut.get() << '\n';  // 49

    return 0;
}
```

## C++20 jthread — Cooperative Cancellation

```cpp
#include <thread>
#include <stop_token>
#include <chrono>

int main() {
    // jthread joins automatically on destruction — no need to call join()
    // Accepts stop_token as first parameter (optional)
    std::jthread worker([](std::stop_token st) {
        int count = 0;
        while (!st.stop_requested()) {  // check cancellation
            std::cout << "Working... " << count++ << '\n';
            std::this_thread::sleep_for(std::chrono::milliseconds(200));
        }
        std::cout << "Stopped gracefully\n";
    });

    std::this_thread::sleep_for(std::chrono::milliseconds(1000));
    worker.request_stop();  // signal cancellation
    // worker destructor calls join() automatically

    // stop_callback — called when stop is requested
    std::stop_source source;
    std::stop_callback cb(source.get_token(), []() {
        std::cout << "Stop requested!\n";
    });
    source.request_stop();  // fires the callback

    return 0;
}
```

## C++20 Semaphores, Latches, and Barriers

```cpp
#include <semaphore>
#include <latch>
#include <barrier>
#include <thread>
#include <vector>

void semaphore_example() {
    // Binary semaphore (0 or 1) — like mutex but can release from different thread
    std::binary_semaphore ready{0};

    std::thread producer([&] {
        std::cout << "Producing...\n";
        ready.release();  // signal consumer
    });
    ready.acquire();       // wait for signal
    std::cout << "Consumer got signal\n";
    producer.join();

    // Counting semaphore — limit concurrent access
    std::counting_semaphore<4> pool{4};  // max 4 threads at once
    std::vector<std::thread> workers;
    for (int i = 0; i < 8; i++) {
        workers.emplace_back([&, i] {
            pool.acquire();             // enter critical section
            std::cout << i << " working\n";
            std::this_thread::sleep_for(std::chrono::milliseconds(100));
            pool.release();             // leave critical section
        });
    }
    for (auto& t : workers) t.join();
}

void latch_example() {
    // std::latch — count down once; threads wait until count reaches 0
    std::latch done{4};  // wait for 4 threads
    std::vector<std::thread> threads;
    for (int i = 0; i < 4; i++) {
        threads.emplace_back([&, i] {
            std::cout << "Thread " << i << " ready\n";
            done.count_down();   // decrement
        });
    }
    done.wait();  // block until count == 0
    std::cout << "All threads ready\n";
    for (auto& t : threads) t.join();
}

void barrier_example() {
    // std::barrier — reusable synchronization point (unlike latch)
    const int N = 4;
    std::barrier sync(N, []() noexcept {
        std::cout << "All threads reached barrier\n";
    });

    std::vector<std::thread> threads;
    for (int t = 0; t < N; t++) {
        threads.emplace_back([&, t] {
            for (int phase = 0; phase < 3; phase++) {
                std::cout << "Thread " << t << " phase " << phase << '\n';
                sync.arrive_and_wait();  // wait for all threads
            }
        });
    }
    for (auto& t : threads) t.join();
}
```

## Race Conditions and False Sharing

```cpp
#include <atomic>
#include <thread>

// Race condition — two threads modify shared data without synchronization
int g_counter_bad = 0;  // data race if accessed concurrently!

// Fix 1: atomic
std::atomic<int> g_counter_atomic{0};

// Fix 2: mutex-protected
struct ThreadSafeCounter {
    std::mutex mtx;
    int value = 0;
    void increment() {
        std::lock_guard lock(mtx);
        ++value;
    }
};

// False sharing — different variables in same cache line cause performance degradation
struct FalseSharing {
    int a;  // threads 0 touches this
    int b;  // thread 1 touches this — BUT they share a 64-byte cache line!
};

struct NoFalseSharing {
    alignas(64) int a;  // each on its own 64-byte cache line
    alignas(64) int b;
};
```

:::caution[Thread Safety Rules]
- **Data race = UB**: Any concurrent access where at least one is a write (without synchronization) is undefined behavior in C++.
- **Atomics are NOT a silver bullet**: `std::atomic` prevents data races but doesn't prevent logical races (check-then-act issues).
- **Lock-free ≠ wait-free**: Lock-free code has no deadlock but can still have starvation.
- **Prefer higher-level abstractions**: Use `std::async`, `std::packaged_task`, or thread pools over raw `std::thread`.
:::

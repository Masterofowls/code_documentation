# Getting Started with C++

This guide covers the tooling and language fundamentals needed to write, build, and manage modern C++ projects.

## Compiler Flags

```bash
# Basic compilation
g++ main.cpp -o program
clang++ main.cpp -o program

# Modern C++20 with full warnings
g++ -std=c++20 -Wall -Wextra -Wpedantic -Werror main.cpp -o program

# Debug build (with sanitizers — use during development)
g++ -std=c++20 -g -O0 -fsanitize=address,undefined main.cpp -o program_debug

# Release build
g++ -std=c++20 -O2 -DNDEBUG main.cpp -o program_release

# Profile-guided optimization
g++ -std=c++20 -O3 -march=native -flto main.cpp -o program_fast

# Type-checking additions
g++ -std=c++20 -Wall -Wextra                 \
    -Wshadow -Wnon-virtual-dtor              \
    -Woverloaded-virtual -Wold-style-cast    \
    -Wcast-align -Wconversion               \
    -Wsign-conversion -Wmisleading-indentation \
    main.cpp -o program
```

### Useful Clang-Specific Flags

```bash
# Use Clang's include-what-you-use analysis
clang++ -std=c++20 -Weverything -Wno-c++98-compat main.cpp

# Generate compilation database (for clangd language server)
cmake -DCMAKE_EXPORT_COMPILE_COMMANDS=ON ..
```

## CMake for C++

CMake is the de-facto build system for C++ projects.

```cmake
# CMakeLists.txt — Modern CMake (3.20+)
cmake_minimum_required(VERSION 3.20)
project(MyApp VERSION 1.0 LANGUAGES CXX)

# Require C++20
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)  # disable GNU extensions

# Export compile commands for clangd
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# Library target
add_library(mylib STATIC
    src/utils.cpp
    src/math.cpp
)
target_include_directories(mylib PUBLIC include/)
target_compile_features(mylib PUBLIC cxx_std_20)

# Executable target
add_executable(myapp src/main.cpp)
target_link_libraries(myapp PRIVATE mylib)

# Compiler warnings
target_compile_options(myapp PRIVATE
    $<$<CXX_COMPILER_ID:GNU,Clang>:
        -Wall -Wextra -Wpedantic -Wshadow -Wnon-virtual-dtor>
    $<$<CXX_COMPILER_ID:MSVC>:
        /W4 /permissive->
)

# Find external packages
find_package(OpenSSL REQUIRED)
target_link_libraries(myapp PRIVATE OpenSSL::SSL)

# Enable sanitizers in Debug mode
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    target_compile_options(myapp PRIVATE -fsanitize=address,undefined)
    target_link_options(myapp PRIVATE -fsanitize=address,undefined)
endif()

# Tests
enable_testing()
add_executable(tests tests/test_utils.cpp)
target_link_libraries(tests PRIVATE mylib)
add_test(NAME unit_tests COMMAND tests)
```

```bash
# Configure and build
mkdir build && cd build
cmake .. -DCMAKE_BUILD_TYPE=Release
cmake --build . --parallel $(nproc)
ctest --output-on-failure

# Or with Ninja (faster than Make)
cmake .. -G Ninja -DCMAKE_BUILD_TYPE=Release
ninja
```

## Namespaces

Namespaces prevent name collisions in large projects and libraries.

```cpp
#include <iostream>
#include <string>

// Declare a namespace
namespace graphics {
    struct Color { float r, g, b, a; };

    Color make_color(float r, float g, float b) {
        return { r, g, b, 1.0f };
    }

    namespace detail {  // nested namespace
        void validate_color(const Color& c) { /* ... */ }
    }

    // C++17 nested namespace shorthand
    // namespace graphics::detail { ... }
}

namespace physics {
    struct Vector3 { float x, y, z; };
}

int main() {
    // Fully-qualified name
    graphics::Color red = graphics::make_color(1.0f, 0.0f, 0.0f);

    // using declaration: brings specific name into scope
    using graphics::Color;
    Color blue = { 0.0f, 0.0f, 1.0f, 1.0f };

    // using directive: brings entire namespace into scope (avoid in headers!)
    using namespace graphics;
    Color green = make_color(0.0f, 1.0f, 0.0f);

    return 0;
}

// Anonymous namespace: internal linkage (replaces C's static for functions/globals)
namespace {
    constexpr int kMaxRetries = 3;

    void internal_helper() {
        // Only visible in this translation unit
    }
}

// Inline namespace (C++11): versioning
namespace api {
    inline namespace v2 {    // v2 is the "current" version
        void process() {}    // api::process() works
    }
    namespace v1 {
        void process() {}    // api::v1::process() for old version
    }
}
```

## auto Type Deduction

`auto` lets the compiler deduce the type from the initializer.

```cpp
#include <vector>
#include <map>
#include <string>

int main() {
    // Basic auto
    auto i   = 42;            // int
    auto f   = 3.14f;         // float
    auto d   = 3.14;          // double
    auto s   = std::string{"hello"};  // std::string
    auto *p  = &i;            // int*

    // auto with containers
    std::vector<int> v = {1, 2, 3};
    auto it = v.begin();      // std::vector<int>::iterator

    // Range-based for with auto
    for (auto x : v) { /* x is int */ }
    for (auto& x : v) { x *= 2; /* x is int&, can modify */ }
    for (const auto& x : v) { /* x is const int&, read-only */ }

    // auto with structured bindings (C++17)
    std::map<std::string, int> scores = {{"Alice", 95}, {"Bob", 87}};
    for (const auto& [name, score] : scores) {
        std::cout << name << ": " << score << '\n';
    }

    // decltype: query type without value
    decltype(i) j = 0;        // j is int
    decltype(auto) k = i;     // k is int (not reference)
    decltype(auto) ref = (i); // ref is int& (parentheses make it reference)

    return 0;
}
```

:::tip[When to Use auto]
Use `auto` when the type is:
- Long and obvious from context (iterators, complex template types)
- A lambda type (you can't spell it anyway)
- Returned by a function where the exact type doesn't matter

Avoid `auto` when type clarity matters for readability (`int count`, `float ratio`).
:::

## Range-Based for Loop

```cpp
#include <vector>
#include <array>
#include <string>
#include <map>

void range_for_examples() {
    // Works on: arrays, containers, anything with begin()/end()
    int arr[] = {1, 2, 3, 4, 5};
    for (int x : arr) { std::cout << x << ' '; }

    std::vector<std::string> words = {"hello", "world"};
    for (const auto& word : words) { std::cout << word << ' '; }

    // Modify elements
    std::vector<double> vals = {1.0, 2.0, 3.0};
    for (auto& v : vals) { v *= 2.0; }

    // With index (C++20 init-statement in range-for)
    for (int i = 0; const auto& word : words) {
        std::cout << i++ << ": " << word << '\n';
    }
}
```

## std::string

```cpp
#include <string>
#include <string_view>
#include <sstream>

void string_examples() {
    // Construction
    std::string s1 = "hello";
    std::string s2(10, 'x');      // "xxxxxxxxxx"
    std::string s3 = s1 + " world"; // concatenation (new string)

    // Size and access
    s1.size();          // 5
    s1.length();        // 5 (same)
    s1.empty();         // false
    s1[0];              // 'h' (no bounds check)
    s1.at(0);           // 'h' (throws std::out_of_range)
    s1.front();         // 'h'
    s1.back();          // 'o'
    s1.data();          // const char* (C++11: always null-terminated)
    s1.c_str();         // const char* (null-terminated)

    // Modification
    s1.append(" there");
    s1 += "!";
    s1.insert(5, " dear");
    s1.erase(5, 5);
    s1.replace(0, 5, "bye");
    s1.clear();

    // Search
    size_t pos = s2.find("xx");      // find substring
    s2.find_first_of("aeiou");       // find first vowel
    s2.rfind("xx");                  // find last occurrence

    // Substring
    std::string sub = s2.substr(2, 5);  // 5 chars starting at index 2

    // Conversion
    int n = std::stoi("42");
    double d = std::stod("3.14");
    std::string ns = std::to_string(42);

    // string_view: non-owning, zero-copy view
    // Prefer over const std::string& for read-only parameters
    std::string_view sv = s1;
    // sv does not own the string data — don't outlive the original string
}
```

## References

```cpp
void reference_examples() {
    int x = 42;

    // Lvalue reference: alias for an existing object
    int& ref = x;       // ref IS x — same memory
    ref = 100;          // modifies x
    std::cout << x;     // 100

    // References cannot be reseated (unlike pointers)
    int y = 50;
    ref = y;            // copies y's value into x — does NOT rebind ref

    // Const reference: bind to rvalue or const
    const int& cref = 42;    // OK: extends lifetime of temporary
    const int& cref2 = x;   // OK: read-only alias

    // Reference vs pointer comparison:
    // - References cannot be null (always valid if initialized)
    // - References cannot be reseated
    // - References are syntactically transparent (no * to dereference)
    // - Pointers can be null, reseated, used in arrays, used in algorithms

    // Pass by reference: avoids copy, allows modification
    auto increment = [](int& n) { ++n; };
    increment(x);  // x is now 101

    // Pass by const reference: avoids copy, read-only
    auto print = [](const std::string& s) { std::cout << s; };

    // Return by reference (careful: don't return reference to local!)
    // OK if returning member reference or reference to something with longer lifetime
}
```

## nullptr

C++11 introduced `nullptr` — a strongly-typed null pointer constant of type `std::nullptr_t`.

```cpp
#include <cstddef>  // nullptr_t

// Always use nullptr, never NULL or 0
int *p = nullptr;

// Why nullptr is better than NULL:
void foo(int n) { /* ... */ }
void foo(int* p) { /* ... */ }

foo(NULL);     // Ambiguous! NULL might be 0 (int) — calls foo(int)
foo(nullptr);  // Unambiguous: calls foo(int*)

// nullptr_t in templates
template <typename T>
void process(T *ptr) {
    if (ptr == nullptr) return;  // works for any T*
}

// nullptr converts to any pointer type
int    *ip  = nullptr;
double *dp  = nullptr;
void   *vp  = nullptr;
```

## override / final

```cpp
class Shape {
public:
    virtual double area() const = 0;
    virtual void draw() const;
    virtual ~Shape() = default;
};

class Circle : public Shape {
public:
    explicit Circle(double r) : radius_(r) {}

    double area() const override;   // override: compiler checks it really overrides
    void draw() const override;

    // final: this override cannot be further overridden
    // void move() override final { ... }

private:
    double radius_;
};

// final class: no further inheritance allowed
class Singleton final {
    // ...
};
// class SubSingleton : public Singleton {};  // error: Singleton is final
```

## default / delete / explicit

```cpp
class Resource {
public:
    // Explicitly request compiler-generated implementations
    Resource()  = default;
    ~Resource() = default;

    // Prevent copying (non-copyable resource)
    Resource(const Resource&)            = delete;
    Resource& operator=(const Resource&) = delete;

    // Allow moving
    Resource(Resource&&) noexcept            = default;
    Resource& operator=(Resource&&) noexcept = default;
};

class Wrapper {
public:
    // explicit: prevents implicit conversion
    explicit Wrapper(int value) : value_(value) {}
    explicit operator bool() const { return value_ != 0; }

private:
    int value_;
};

void process(Wrapper w) {}
// process(42);        // ERROR: implicit conversion from int prevented
process(Wrapper(42));  // OK: explicit construction

Wrapper w(1);
// if (w == true) {}    // ERROR: implicit conversion to bool prevented
if (static_cast<bool>(w)) {}  // OK: explicit cast
```

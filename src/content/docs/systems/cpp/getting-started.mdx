# Getting Started with C++

## Compiler Flags

<Tabs>
  <TabItem label="g++">
```bash
# Minimum recommended flags for any C++20 project
g++ -std=c++20 -Wall -Wextra -Wpedantic -o myapp main.cpp

# Debug build

g++ -std=c++20 -Wall -Wextra -g -O0 -o myapp_debug main.cpp

# Release build

g++ -std=c++20 -O2 -DNDEBUG -o myapp main.cpp

# With sanitizers (development only)

g++ -std=c++20 -g -fsanitize=address,undefined,leak -o myapp main.cpp

# Profile-guided optimization (two-pass)

g++ -std=c++20 -O2 -fprofile-generate main.cpp -o myapp
./myapp # Run to collect profile data
g++ -std=c++20 -O2 -fprofile-use main.cpp -o myapp_pgo

# Link-time optimization

g++ -std=c++20 -O2 -flto -o myapp main.cpp utils.cpp

# All warnings as errors (use in CI)

g++ -std=c++20 -Wall -Wextra -Werror -o myapp main.cpp

# Additional useful warnings

g++ -std=c++20 -Wall -Wextra \
 -Wshadow \
 -Wnon-virtual-dtor \
 -Wold-style-cast \
 -Wcast-align \
 -Wunused \
 -Woverloaded-virtual \
 -Wconversion \
 -Wsign-conversion \
 -Wdouble-promotion \
 -Wformat=2 \
 -o myapp main.cpp

````
</TabItem>
  <TabItem label="clang++">
```bash
# clang++ with libc++
clang++ -std=c++20 -stdlib=libc++ -Wall -Wextra -o myapp main.cpp

# Or with libstdc++ (GCC's standard library)
clang++ -std=c++20 -stdlib=libstdc++ -Wall -Wextra -o myapp main.cpp

# Clang-specific: enable all useful warnings
clang++ -std=c++20 -Weverything \
    -Wno-c++98-compat \
    -Wno-padded \
    -o myapp main.cpp

# Static analysis
clang-tidy main.cpp -- -std=c++20
scan-build g++ -std=c++20 main.cpp -o myapp
````

</TabItem>

## CMake for C++

```cmake
# CMakeLists.txt — Production-quality C++20 project

cmake_minimum_required(VERSION 3.25)

project(MyApp
    VERSION 1.0.0
    DESCRIPTION "A C++20 application"
    LANGUAGES CXX
)

# Require C++20 globally
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)   # No compiler-specific extensions

# Improve IDE integration
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

# Source files
file(GLOB_RECURSE APP_SOURCES "src/*.cpp")
file(GLOB_RECURSE APP_HEADERS "include/*.hpp" "include/*.h")

# Main executable
add_executable(myapp ${APP_SOURCES} ${APP_HEADERS})

# Compile features (alternative to setting CMAKE_CXX_STANDARD)
target_compile_features(myapp PUBLIC cxx_std_20)

# Include directories
target_include_directories(myapp
    PUBLIC  include/          # Part of public API
    PRIVATE src/internal/     # Implementation only
)

# Compiler-specific warning flags
target_compile_options(myapp PRIVATE
    $<$<CXX_COMPILER_ID:GNU>:   -Wall -Wextra -Wpedantic -Wshadow -Wnon-virtual-dtor>
    $<$<CXX_COMPILER_ID:Clang>: -Wall -Wextra -Wpedantic -Wshadow -Wnon-virtual-dtor>
    $<$<CXX_COMPILER_ID:MSVC>:  /W4 /permissive->
)

# Build type-specific flags
target_compile_options(myapp PRIVATE
    $<$<CONFIG:Debug>:   $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-O0 -g>>
    $<$<CONFIG:Release>: $<$<NOT:$<CXX_COMPILER_ID:MSVC>>:-O2 -DNDEBUG>>
)

# Link libraries
target_link_libraries(myapp PRIVATE)

# Find and use packages
find_package(fmt QUIET)
if(fmt_FOUND)
    target_link_libraries(myapp PRIVATE fmt::fmt)
    target_compile_definitions(myapp PRIVATE HAS_FMT)
endif()

# Tests
enable_testing()
add_subdirectory(tests)
````

```bash
# Configure and build
cmake -B build -DCMAKE_BUILD_TYPE=Release
cmake --build build --parallel $(nproc)

# Run with sanitizers
cmake -B build_san -DCMAKE_BUILD_TYPE=Debug \
      -DCMAKE_CXX_FLAGS="-fsanitize=address,undefined"
cmake --build build_san
```

## Namespaces

```cpp
#include <iostream>
#include <string>

// Define a namespace
namespace mylib {

    // Everything inside is in the mylib:: scope
    class Timer {
    public:
        void start();
        void stop();
        double elapsed_ms() const;
    };

    void log(const std::string& msg);

    // Nested namespace (C++17: nested namespace shorthand)
    namespace detail {
        void internal_helper();
    }

    // C++17 nested namespace shorthand
    namespace io::net {
        void connect(const std::string& host, int port);
    }

} // namespace mylib

// Anonymous namespace: internal linkage within translation unit
// Better than `static` for limiting visibility of types and functions
namespace {
    const int kMaxRetries = 3;

    bool internal_check(int value) {
        return value >= 0 && value < 100;
    }
}

// Inline namespace (C++11): symbols accessible from enclosing namespace
// Useful for versioning ABI
namespace mylib {
    inline namespace v2 {
        class Widget { /* v2 implementation */ };
    }
    namespace v1 {
        class Widget { /* old implementation */ };
    }
}

// Using declarations and directives
void using_demo() {
    // Using declaration: bring specific names into scope
    using mylib::Timer;
    Timer t;           // OK without mylib::

    // Using directive: bring entire namespace into scope
    // NEVER put "using namespace std;" in a header file
    using namespace mylib;
    log("Hello");      // OK

    // Namespace alias (useful for long namespace names)
    namespace ml = mylib;
    ml::Timer t2;
}
```

## auto Type Deduction

```cpp
#include <vector>
#include <map>
#include <string>
#include <memory>

void auto_demo() {
    // auto deduces from initializer
    auto x = 42;          // int
    auto y = 3.14;        // double
    auto z = 3.14f;       // float
    auto s = std::string("hello");  // std::string
    auto b = true;        // bool

    // auto with complex types — saves typing
    auto it = std::vector<int>().begin();  // std::vector<int>::iterator

    std::map<std::string, std::vector<int>> data;
    auto& ref = data["key"];  // std::vector<int>&

    // auto preserves const/reference with modifiers
    const auto& cref = x;    // const int&
    auto* ptr = &x;          // int*

    // auto in range-for
    std::vector<std::pair<int, std::string>> pairs = {{1, "a"}, {2, "b"}};
    for (const auto& [id, name] : pairs) {   // Structured binding (C++17)
        // id is const int&, name is const std::string&
    }

    // Return type deduction (C++14+)
    auto add = [](auto a, auto b) { return a + b; };  // Generic lambda
    auto result = add(1, 2);     // int
    auto result2 = add(1.5, 2);  // double

    // decltype: get type of expression without evaluating
    decltype(x) copy = x;     // int
    decltype(auto) ref2 = x;  // int (exact type, including refs)
}
```

## References

```cpp
#include <iostream>
#include <string>

// Lvalue reference — alias to existing variable
void lvalue_refs() {
    int x = 42;
    int& ref = x;   // ref IS x — same object

    ref = 100;
    std::cout << x << '\n';   // 100 — x was modified

    // References must be initialized and cannot be rebound
    // int& bad;          // Error: uninitialized reference
    // ref = y;           // This assigns y's value to x, doesn't rebind ref

    // Const reference: read-only alias, can bind to temporaries
    const int& cref = x;       // Alias to x
    const int& tmp_ref = 42;   // Binds to temporary — extends lifetime
    // cref = 200;  // Error: const
}

// Passing by const reference — no copy, read-only
void print_string(const std::string& s) {
    std::cout << s << '\n';
    // s += '!';  // Error: can't modify through const ref
}

// Passing by non-const reference — modify the original
void to_upper_inplace(std::string& s) {
    for (char& c : s) {
        c = std::toupper(c);
    }
}

// References vs pointers comparison
void ref_vs_ptr() {
    int x = 10;

    // Reference: always valid, no null, no arithmetic, must init
    int& r = x;
    r = 20;           // Modify x

    // Pointer: can be null, arithmetic possible, must dereference
    int* p = &x;
    if (p) *p = 30;   // Null check, then dereference
    p++;              // Pointer arithmetic (dangerous here)
}
```

## First Class: A Complete Example

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <format>
#include <stdexcept>
#include <algorithm>

class Person {
private:
    std::string m_name;
    int m_age;

public:
    // Constructor with member initializer list (preferred over assignment in body)
    Person(std::string name, int age)
        : m_name(std::move(name))  // Move instead of copy
        , m_age(age)
    {
        if (m_age < 0 || m_age > 150) {
            throw std::invalid_argument(
                std::format("Invalid age: {}", m_age)
            );
        }
    }

    // Const member functions: promise not to modify *this
    const std::string& name() const { return m_name; }
    int age() const { return m_age; }

    // Non-const: modifies the object
    void birthday() { ++m_age; }
    void rename(std::string new_name) { m_name = std::move(new_name); }

    // Comparison operator (C++20 spaceship <=> auto-generates all six)
    auto operator<=>(const Person& other) const {
        if (auto cmp = m_name <=> other.m_name; cmp != 0) return cmp;
        return m_age <=> other.m_age;
    }
    bool operator==(const Person& other) const = default;

    // Stream output
    friend std::ostream& operator<<(std::ostream& os, const Person& p) {
        return os << std::format("Person({}, {})", p.m_name, p.m_age);
    }
};

int main() {
    std::vector<Person> people;
    people.emplace_back("Alice", 30);
    people.emplace_back("Bob", 25);
    people.emplace_back("Charlie", 35);

    // Range-based for with structured output
    for (const auto& p : people) {
        std::cout << p << '\n';
    }

    // Sort using <=>
    std::ranges::sort(people, [](const Person& a, const Person& b) {
        return a.age() < b.age();
    });

    std::cout << "\nSorted by age:\n";
    for (const auto& p : people) {
        std::cout << std::format("  {:15s} age {}\n", p.name(), p.age());
    }

    // Exception handling
    try {
        Person invalid("Bad", -1);
    } catch (const std::invalid_argument& e) {
        std::cerr << "Caught: " << e.what() << '\n';
    }

    return 0;
}
```

:::tip[nullptr vs NULL vs 0]
Always use `nullptr` in C++11+. `NULL` is typically `0` (an integer constant), which causes ambiguity in overload resolution. `nullptr` is of type `std::nullptr_t` and only implicitly converts to pointer types.

```cpp
void foo(int x);
void foo(int* p);

foo(NULL);     // Ambiguous or calls foo(int)!
foo(nullptr);  // Always calls foo(int*)
```

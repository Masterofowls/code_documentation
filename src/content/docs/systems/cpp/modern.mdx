# Modern C++ Features

## C++17 Key Features

### Structured Bindings

```cpp
#include <map>
#include <tuple>
#include <array>

int main() {
    // Decompose pairs
    std::pair<int, std::string> p = {42, "hello"};
    auto& [num, str] = p;
    std::cout << num << ' ' << str << '\n';

    // Decompose tuples
    auto t = std::make_tuple(1, 2.0, "three");
    auto [a, b, c] = t;

    // Decompose arrays
    int arr[] = {1, 2, 3};
    auto [x, y, z] = arr;

    // Decompose structs (aggregate types)
    struct Point { int x, y; };
    Point pt{10, 20};
    auto [px, py] = pt;

    // Especially useful with maps
    std::map<std::string, int> scores = {{"Alice", 95}, {"Bob", 87}};
    for (const auto& [name, score] : scores) {
        std::cout << name << ": " << score << '\n';
    }

    // insert returns pair<iterator, bool>
    auto [iter, inserted] = scores.insert({"Carol", 92});
    if (inserted) std::cout << "Inserted " << iter->first << '\n';
}
```

```cpp
// Limit scope of variable to the if statement

    std::cout << "Found: " << it->second << '\n';

// switch with initializer
```

### std::optional, std::variant, std::any

```cpp
#include <optional>
#include <variant>
#include <any>

// std::optional — nullable value without heap allocation
std::optional<int> safe_divide(int a, int b) {
    if (b == 0) return std::nullopt;
    return a / b;
}

auto result = safe_divide(10, 2);
if (result) std::cout << *result << '\n';          // 5
std::cout << result.value_or(-1) << '\n';          // 5
std::cout << safe_divide(10, 0).value_or(-1) << '\n';  // -1

// std::variant — type-safe union
using Token = std::variant<int, double, std::string>;

Token tok = 42;
tok = 3.14;
tok = std::string("identifier");

// std::visit with overloaded visitor (C++17 pattern)
struct Visitor {
    void operator()(int i)               { std::cout << "int: "    << i << '\n'; }
    void operator()(double d)            { std::cout << "double: " << d << '\n'; }
    void operator()(const std::string& s){ std::cout << "string: " << s << '\n'; }
};
std::visit(Visitor{}, tok);

// Generic visitor with if constexpr
std::visit([](auto&& val) {
    using T = std::decay_t<decltype(val)>;
    if constexpr (std::is_same_v<T, int>)         std::cout << "int\n";
    else if constexpr (std::is_same_v<T, double>) std::cout << "double\n";
    else                                           std::cout << "other\n";
}, tok);

// std::any — type-erased container (like void*, but safe)
std::any a = 42;
a = std::string("hello");
a = 3.14;
std::cout << std::any_cast<double>(a) << '\n';  // 3.14
// std::any_cast<int>(a); // throws std::bad_any_cast
```

### std::filesystem

```cpp
#include <filesystem>
namespace fs = std::filesystem;

void filesystem_examples() {
    // Path manipulation
    fs::path p = "/home/user/docs/file.txt";
    std::cout << p.parent_path()   << '\n';  // /home/user/docs
    std::cout << p.filename()      << '\n';  // file.txt
    std::cout << p.stem()          << '\n';  // file
    std::cout << p.extension()     << '\n';  // .txt

    fs::path combined = fs::path("/home") / "user" / "file.txt";

    // File operations
    fs::create_directories("path/to/deep/dir");
    fs::copy("source.txt", "dest.txt");
    fs::rename("old.txt", "new.txt");
    fs::remove("file.txt");
    fs::remove_all("directory");

    // File info
    if (fs::exists("myfile.txt")) {
        std::cout << "Size: " << fs::file_size("myfile.txt") << '\n';
        auto time = fs::last_write_time("myfile.txt");
    }

    // Directory iteration
    for (const auto& entry : fs::directory_iterator(".")) {
        if (entry.is_regular_file()) {
            std::cout << entry.path().filename() << '\n';
        }
    }

    // Recursive iteration
    for (const auto& entry : fs::recursive_directory_iterator("src")) {
        if (entry.path().extension() == ".cpp") {
            std::cout << entry.path() << '\n';
        }
    }
}
```

## C++20 Key Features

### std::format

```cpp
#include <format>
#include <string>

int main() {
    // Basic formatting
    auto s1 = std::format("Hello, {}!", "World");
    auto s2 = std::format("{} + {} = {}", 1, 2, 3);

    // Positional arguments
    auto s3 = std::format("{0} {1} {0}", "hello", "world");  // hello world hello

    // Format specifiers
    auto s4 = std::format("{:>10}", "right");     // right-aligned, width 10
    auto s5 = std::format("{:<10}", "left");      // left-aligned
    auto s6 = std::format("{:^10}", "center");    // centered
    auto s7 = std::format("{:0>5}", 42);          // zero-padded: 00042
    auto s8 = std::format("{:.3f}", 3.14159);     // 3.142
    auto s9 = std::format("{:#010x}", 255);       // 0x000000ff
    auto s10 = std::format("{:+d}", 42);          // +42

    // Print directly (C++23 std::print, C++20 use format+cout)
    std::cout << std::format("Pi = {:.5f}\n", 3.14159265);

    // Custom formatter for user type
    struct Color { uint8_t r, g, b; };
}

// Custom formatter
template<>
struct std::formatter<Color> {
    constexpr auto parse(std::format_parse_context& ctx) { return ctx.begin(); }
    auto format(const Color& c, std::format_context& ctx) const {
        return std::format_to(ctx.out(), "#{:02X}{:02X}{:02X}", c.r, c.g, c.b);
    }
};
```

### C++20 Ranges — Advanced Usage

```cpp
#include <ranges>
#include <vector>
#include <string>

int main() {
    // Lazy pipeline — nothing executed until iteration
    auto pipeline = std::views::iota(1)         // infinite range: 1, 2, 3, ...
        | std::views::filter([](int n) { return n % 2 == 0; })  // even
        | std::views::transform([](int n) { return n * n; })    // square
        | std::views::take(5);                  // first 5

    for (int x : pipeline) std::cout << x << ' ';  // 4 16 36 64 100
    std::cout << '\n';

    // views::split — split string on delimiter
    std::string csv = "alice,bob,carol,dave";
    for (auto word : std::views::split(csv, ',')) {
        std::cout << std::string_view(word) << '\n';
    }

    // views::chunk (C++23) — split into chunks of N
    std::vector<int> v{1,2,3,4,5,6};
    for (auto chunk : v | std::views::chunk(2)) {
        for (int x : chunk) std::cout << x << ' ';
        std::cout << '|';
    }  // 1 2 |3 4 |5 6 |
}
```

### C++20 Modules (Preview)

```cpp
export namespace math {
    constexpr double PI = 3.14159265358979;

    double circle_area(double r) { return PI * r * r; }
    double square(double x)      { return x * x; }
}

// Internal (not exported)
namespace detail {
    double internal_helper() { return 42.0; }
}

// main.cpp — consuming the module
import math;

int main() {
    std::cout << math::circle_area(5.0) << '\n';
    std::cout << math::PI << '\n';
    return 0;
}
```

### Three-Way Comparison (Spaceship Operator)

```cpp
#include <compare>

struct Version {
    int major, minor, patch;

    // Generates all 6 comparison operators from spaceship
    auto operator<=>(const Version&) const = default;

    // Or customize ordering:
    std::strong_ordering operator<=>(const Version& other) const {
        if (auto cmp = major <=> other.major; cmp != 0) return cmp;
        if (auto cmp = minor <=> other.minor; cmp != 0) return cmp;
        return patch <=> other.patch;
    }
    bool operator==(const Version&) const = default;
};

int main() {
    Version v1{1, 2, 3};
    Version v2{1, 3, 0};
    std::cout << (v1 < v2) << '\n';   // 1 (true)
    std::cout << (v1 > v2) << '\n';   // 0 (false)
    std::cout << (v1 != v2) << '\n';  // 1 (true)
}
```

## C++23 Highlights

```cpp
// std::expected — error without exceptions (like Rust's Result<T, E>)
#include <expected>
#include <system_error>

std::expected<int, std::string> safe_parse(std::string_view s) {
    try {
        return std::stoi(std::string(s));
    } catch (...) {
        return std::unexpected("Not a valid integer: " + std::string(s));
    }
}

int main() {
    auto result = safe_parse("42");
    if (result) {
        std::cout << "Value: " << *result << '\n';
    } else {
        std::cerr << "Error: " << result.error() << '\n';
    }

    // Monadic operations (C++23)
    auto doubled = safe_parse("21")
        .transform([](int n) { return n * 2; })  // map if success
        .or_else([](auto& e) -> std::expected<int, std::string> {
            return 0;  // default on error
        });
    std::cout << *doubled << '\n';  // 42

    // std::print / std::println (C++23)
    // std::print("Hello, {}!\n", "world");
    // std::println("Value: {}", 42);

    // std::stacktrace (C++23)
    // auto trace = std::stacktrace::current();
    // std::cout << trace << '\n';
}
```

:::tip[Adoption Strategy]
When migrating existing C++ code to modern standards:
1. Enable C++17 first: structured bindings, `if constexpr`, and `optional`/`variant` have high ROI with minimal risk.
2. Enable C++20 for new files: concepts and ranges significantly improve template error messages.
3. Add `[[nodiscard]]` to functions returning error codes or owning pointers.
4. Replace raw `new`/`delete` with `make_unique`/`make_shared`.
5. Replace `NULL` with `nullptr` and `0` literal returns.
:::

# Object-Oriented Programming in C++

C++ OOP goes far deeper than Java or Python — with manual control over copying, moving, destruction, and memory layout. Understanding these mechanisms is essential for writing correct, efficient C++.

## Classes In Depth

```cpp
#include <iostream>
#include <string>

class BankAccount {
public:
    // Access specifiers: public, protected, private
    static int total_accounts;    // Shared across all instances

    // Constructor
    BankAccount(std::string owner, double balance = 0.0)
        : m_owner(std::move(owner))
        , m_balance(balance)
        , m_id(++total_accounts)
    {}

    // Const member function — does not modify *this
    double balance() const { return m_balance; }
    const std::string& owner() const { return m_owner; }
    int id() const { return m_id; }

    // Non-const member function
    bool deposit(double amount) {
        if (amount <= 0) return false;
        m_balance += amount;
        return true;
    }

    bool withdraw(double amount) {
        if (amount <= 0 || amount > m_balance) return false;
        m_balance -= amount;
        return true;
    }

    // mutable: can be modified even in const member functions
    // Useful for caches, lazy initialization, mutexes
    mutable int m_access_count = 0;

    double balance_cached() const {
        ++m_access_count;   // OK: mutable field
        return m_balance;
    }

    // Friend: grants access to private/protected members
    friend std::ostream& operator<<(std::ostream& os, const BankAccount& a);

    // this pointer: pointer to current object
    BankAccount& transfer_to(BankAccount& other, double amount) {
        if (this == &other) return *this;  // Self-transfer check
        if (withdraw(amount)) other.deposit(amount);
        return *this;  // Return *this for chaining
    }

private:
    std::string m_name_prefix = "ACC";  // In-class default
    std::string m_owner;
    double      m_balance;
    int         m_id;
};

int BankAccount::total_accounts = 0;  // Static member definition

std::ostream& operator<<(std::ostream& os, const BankAccount& a) {
    return os << "Account[" << a.m_id << "] " << a.m_owner
              << " $" << a.m_balance;
}
```

## Constructors

```cpp
#include <string>
#include <iostream>
#include <initializer_list>

class Point {
public:
    // Default constructor (compiler generates if no user-provided constructor)
    Point() : x(0.0), y(0.0) {}

    // Parameterized constructor
    Point(double x, double y) : x(x), y(y) {}

    // Explicit prevents implicit conversion
    explicit Point(double scalar) : x(scalar), y(scalar) {}

    // Delegating constructor (C++11): calls another constructor
    Point(double val) : Point(val, val) {}  // Wait — this conflicts with explicit above

    // Copy constructor
    Point(const Point& other) : x(other.x), y(other.y) {}

    // Move constructor (C++11)
    Point(Point&& other) noexcept : x(other.x), y(other.y) {
        other.x = other.y = 0.0;  // Leave moved-from in valid state
    }

    // default/delete
    Point& operator=(const Point&) = default;  // Compiler generates
    // ~Point() = default;

    double x, y;
};

// Aggregate initialization (C++20 designated initializers work with aggregates)
struct Config {
    int width  = 800;
    int height = 600;
    bool fullscreen = false;
    std::string title = "Window";
};

Config cfg{
    .width      = 1920,
    .height     = 1080,
    .fullscreen = true,
    // title uses default: "Window"
};

// std::initializer_list constructor
class NumberList {
public:
    NumberList(std::initializer_list<int> vals)
        : data(vals.begin(), vals.end()) {}

    std::vector<int> data;
};

NumberList nl = {1, 2, 3, 4, 5};  // Uses initializer_list constructor

// explicit prevents: NumberList nl2 = {1}; using copy-init
// Without explicit: NumberList nl3(42); and NumberList nl3 = 42; would work
```

## Destructor and RAII

RAII (Resource Acquisition Is Initialization) ties resource lifetime to object lifetime.

```cpp
#include <cstdio>
#include <stdexcept>
#include <memory>

// RAII file wrapper: file is closed automatically
class FileGuard {
public:
    explicit FileGuard(const char* path, const char* mode) {
        m_file = fopen(path, mode);
        if (!m_file) {
            throw std::runtime_error(std::string("Cannot open: ") + path);
        }
    }

    // Destructor: runs automatically when object goes out of scope
    // Even if an exception is thrown — guaranteed cleanup
    ~FileGuard() {
        if (m_file) {
            fclose(m_file);
            m_file = nullptr;
        }
    }

    // Delete copy (resource should not be shared)
    FileGuard(const FileGuard&) = delete;
    FileGuard& operator=(const FileGuard&) = delete;

    // Allow move
    FileGuard(FileGuard&& other) noexcept : m_file(other.m_file) {
        other.m_file = nullptr;
    }

    FILE* get() const { return m_file; }

private:
    FILE* m_file = nullptr;
};

void process_file(const char* path) {
    FileGuard f(path, "r");  // Opens file
    // ... use f.get() ...
    // File automatically closed when f goes out of scope — even on exception
}
```

## Rule of 0/3/5

```cpp
#include <string>
#include <cstring>
#include <utility>

// RULE OF 0: If your class doesn't manage resources directly,
// let the compiler generate all special members.
class GoodClass {
    std::string name;
    std::vector<int> data;
    // No user-defined destructor, copy, or move needed
    // std::string and std::vector handle their own resources
};

// RULE OF 3 (pre-C++11): If you define any of:
//   destructor, copy constructor, copy assignment
// you should define ALL THREE.
class OldResource {
public:
    OldResource(size_t n) : data(new int[n]), size(n) {}
    ~OldResource() { delete[] data; }

    // Copy constructor
    OldResource(const OldResource& other)
        : data(new int[other.size]), size(other.size)
    {
        std::memcpy(data, other.data, size * sizeof(int));
    }

    // Copy assignment
    OldResource& operator=(const OldResource& other) {
        if (this == &other) return *this;  // Self-assignment check
        delete[] data;
        size = other.size;
        data = new int[size];
        std::memcpy(data, other.data, size * sizeof(int));
        return *this;
    }

private:
    int* data;
    size_t size;
};

// RULE OF 5 (C++11): Add move constructor and move assignment
class Resource {
public:
    Resource(size_t n) : m_data(new int[n]), m_size(n) {}
    ~Resource() { delete[] m_data; }

    // Copy
    Resource(const Resource& o) : m_data(new int[o.m_size]), m_size(o.m_size) {
        std::memcpy(m_data, o.m_data, m_size * sizeof(int));
    }

    Resource& operator=(const Resource& o) {
        Resource tmp(o);  // Copy-and-swap idiom — exception safe
        swap(tmp);
        return *this;
    }

    // Move — transfer ownership without copying
    Resource(Resource&& o) noexcept
        : m_data(o.m_data), m_size(o.m_size)
    {
        o.m_data = nullptr;  // Leave moved-from in valid state
        o.m_size = 0;
    }

    Resource& operator=(Resource&& o) noexcept {
        if (this != &o) {
            delete[] m_data;
            m_data = o.m_data;
            m_size = o.m_size;
            o.m_data = nullptr;
            o.m_size = 0;
        }
        return *this;
    }

    void swap(Resource& o) noexcept {
        std::swap(m_data, o.m_data);
        std::swap(m_size, o.m_size);
    }

private:
    int*   m_data = nullptr;
    size_t m_size = 0;
};
```

## Move Semantics

```cpp
#include <string>
#include <vector>
#include <utility>

// std::move doesn't move — it casts to rvalue reference
// enabling the move constructor/assignment to be selected
void move_demo() {
    std::string s1 = "Hello, World!";
    std::string s2 = std::move(s1);  // s1 now in valid but unspecified state
    // s1 may be empty; s2 has "Hello, World!"

    std::vector<int> v1 = {1, 2, 3, 4, 5};
    std::vector<int> v2 = std::move(v1);  // O(1) — just pointer swap
    // v1 is now empty (implementation-defined but typically cleared)

    // std::forward: perfect forwarding — preserve lvalue/rvalue-ness
    // Used in generic code (templates)
    auto wrapper = [](auto&& arg) {
        process(std::forward<decltype(arg)>(arg));
    };
}

// Return value optimization (RVO/NRVO): compiler eliminates copies
std::vector<int> make_vector(int n) {
    std::vector<int> result;
    result.reserve(n);
    for (int i = 0; i < n; i++) result.push_back(i);
    return result;  // NRVO: no copy or move — constructed in-place at call site
}
```

## Operator Overloading

```cpp
#include <ostream>
#include <compare>
#include <cmath>

struct Vec2 {
    double x, y;

    Vec2(double x = 0, double y = 0) : x(x), y(y) {}

    // Arithmetic (member functions)
    Vec2 operator+(const Vec2& o) const { return {x+o.x, y+o.y}; }
    Vec2 operator-(const Vec2& o) const { return {x-o.x, y-o.y}; }
    Vec2 operator*(double scalar) const { return {x*scalar, y*scalar}; }
    Vec2 operator/(double scalar) const { return {x/scalar, y/scalar}; }
    Vec2 operator-() const { return {-x, -y}; }  // Unary minus

    Vec2& operator+=(const Vec2& o) { x+=o.x; y+=o.y; return *this; }
    Vec2& operator-=(const Vec2& o) { x-=o.x; y-=o.y; return *this; }
    Vec2& operator*=(double s) { x*=s; y*=s; return *this; }

    // Subscript operator
    double& operator[](int i) { return i == 0 ? x : y; }
    double  operator[](int i) const { return i == 0 ? x : y; }

    // Comparison (C++20 spaceship operator — generates ==, !=, <, <=, >, >=)
    auto operator<=>(const Vec2&) const = default;

    // Conversion operator
    explicit operator bool() const { return x != 0.0 || y != 0.0; }

    double length() const { return std::sqrt(x*x + y*y); }

    // Non-member friend: scalar * vec2 (free function, accesses privates via friend)
    friend Vec2 operator*(double scalar, const Vec2& v) { return v * scalar; }

    // Stream output
    friend std::ostream& operator<<(std::ostream& os, const Vec2& v) {
        return os << '(' << v.x << ", " << v.y << ')';
    }
};
```

## Inheritance and Virtual Dispatch

```cpp
#include <iostream>
#include <memory>
#include <vector>
#include <string>

// Abstract base class
class Shape {
public:
    // Virtual destructor: REQUIRED for polymorphic base classes
    virtual ~Shape() = default;

    // Pure virtual: must be overridden in derived class
    virtual double area()      const = 0;
    virtual double perimeter() const = 0;

    // Virtual with default implementation
    virtual std::string describe() const {
        return "Shape with area " + std::to_string(area());
    }

    // Non-virtual (not intended to be overridden)
    void print() const {
        std::cout << describe() << '\n';
    }
};

class Circle final : public Shape {  // final: cannot be derived from
public:
    explicit Circle(double r) : m_radius(r) {}

    // override: explicit that we're overriding a virtual function
    // The compiler errors if the base class doesn't have this virtual function
    double area()      const override { return 3.14159265 * m_radius * m_radius; }
    double perimeter() const override { return 2 * 3.14159265 * m_radius; }
    std::string describe() const override {
        return "Circle(r=" + std::to_string(m_radius) + ")";
    }

private:
    double m_radius;
};

class Rectangle : public Shape {
public:
    Rectangle(double w, double h) : m_width(w), m_height(h) {}
    double area()      const override { return m_width * m_height; }
    double perimeter() const override { return 2*(m_width + m_height); }

private:
    double m_width, m_height;
};

// Polymorphism through base class pointers
void print_shapes(const std::vector<std::unique_ptr<Shape>>& shapes) {
    for (const auto& s : shapes) {
        s->print();  // Virtual dispatch: calls the correct override
        std::cout << "  Area: " << s->area() << '\n';
    }
}

// CRTP: Curiously Recurring Template Pattern
// Enables static polymorphism (no vtable overhead)
template<typename Derived>
class Printable {
public:
    void print() const {
        // static_cast to derived — safe because Derived inherits from Printable<Derived>
        static_cast<const Derived*>(this)->print_impl();
    }
};

class Foo : public Printable<Foo> {
public:
    void print_impl() const { std::cout << "Foo!\n"; }
};
```

:::caution[Virtual Destructor]
**Always declare the destructor `virtual` in a polymorphic base class.** Without it, deleting a derived object through a base pointer only calls the base destructor, leaking derived resources.

```cpp
Base* b = new Derived();
delete b;  // UB if ~Base() is not virtual — ~Derived() never called!
```

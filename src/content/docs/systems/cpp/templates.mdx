# Templates in C++

Templates are C++'s mechanism for generic programming — writing code that works with any type. They enable zero-overhead abstractions: the compiler generates type-specific code at compile time.

## Function Templates

```cpp
#include <algorithm>
#include <iostream>
#include <vector>
#include <concepts>

// Basic function template
template<typename T>
T max_val(T a, T b) {
    return a > b ? a : b;
}

// Multiple type parameters
template<typename T, typename U>
auto add(T a, U b) -> decltype(a + b) {  // Trailing return type
    return a + b;
}

// Non-type template parameter: value known at compile time
template<typename T, std::size_t N>
T sum_array(const T (&arr)[N]) {
    T result{};
    for (const auto& x : arr) result += x;
    return result;
}

// Template with default parameter
template<typename T, typename Allocator = std::allocator<T>>
class MyContainer { /* ... */ };

void function_template_demo() {
    // Explicit instantiation
    std::cout << max_val<int>(3, 7) << '\n';

    // Implicit instantiation (compiler deduces T from arguments)
    std::cout << max_val(3.14, 2.71) << '\n';   // T = double
    std::cout << max_val('a', 'z') << '\n';      // T = char

    int arr[] = {1, 2, 3, 4, 5};
    std::cout << sum_array(arr) << '\n';  // N = 5 deduced from array type

    auto r = add(1, 2.5);  // T=int, U=double; result is double
    std::cout << r << '\n';
}
```

## Class Templates

```cpp
#include <cstddef>
#include <stdexcept>
#include <iostream>
#include <array>

// Fixed-size stack using class template
template<typename T, std::size_t Capacity = 16>
class Stack {
public:
    void push(const T& value) {
        if (m_size == Capacity) throw std::overflow_error("Stack full");
        m_data[m_size++] = value;
    }

    void push(T&& value) {
        if (m_size == Capacity) throw std::overflow_error("Stack full");
        m_data[m_size++] = std::move(value);
    }

    T pop() {
        if (empty()) throw std::underflow_error("Stack empty");
        return std::move(m_data[--m_size]);
    }

    const T& top() const {
        if (empty()) throw std::underflow_error("Stack empty");
        return m_data[m_size - 1];
    }

    bool empty() const noexcept { return m_size == 0; }
    std::size_t size() const noexcept { return m_size; }
    std::size_t capacity() const noexcept { return Capacity; }

private:
    std::array<T, Capacity> m_data{};
    std::size_t m_size = 0;
};

// Member function defined outside class template
template<typename T, std::size_t Cap>
void Stack<T, Cap>::push(const T& value) {
    // (defined inline above — shown for syntax reference)
}

// Usage
void class_template_demo() {
    Stack<int, 8> int_stack;
    int_stack.push(1);
    int_stack.push(2);
    int_stack.push(3);
    std::cout << int_stack.pop() << '\n';  // 3

    Stack<std::string> str_stack;  // Uses default capacity 16
    str_stack.push("hello");
    std::cout << str_stack.top() << '\n';  // hello
}
```

## Template Specialization

```cpp
#include <iostream>
#include <cstring>
#include <type_traits>

// Primary template
template<typename T>
class TypeInfo {
public:
    static const char* name() { return "unknown"; }
    static bool is_numeric() { return false; }
};

// Full specialization for int
template<>
class TypeInfo<int> {
public:
    static const char* name() { return "int"; }
    static bool is_numeric() { return true; }
};

// Full specialization for bool
template<>
class TypeInfo<bool> {
public:
    static const char* name() { return "bool"; }
    static bool is_numeric() { return false; }
};

// Partial specialization: for any pointer type
template<typename T>
class TypeInfo<T*> {
public:
    static const char* name() { return "pointer"; }
    static bool is_numeric() { return false; }
};

// Function template full specialization
template<typename T>
void print_typed(const T& val) {
    std::cout << TypeInfo<T>::name() << ": " << val << '\n';
}

template<>
void print_typed<bool>(const bool& val) {
    std::cout << "bool: " << (val ? "true" : "false") << '\n';
}

// Max for strings (specialization by behavior)
template<>
const char* max_val<const char*>(const char* a, const char* b) {
    return std::strcmp(a, b) > 0 ? a : b;
}
```

## Class Template Argument Deduction (CTAD) — C++17

```cpp
#include <vector>
#include <pair>
#include <tuple>

// Before C++17: had to write template arguments explicitly
std::vector<int> v1 = {1, 2, 3};  // OK (special case)
std::pair<int, double> p1 = {1, 2.5};  // Had to specify both

// C++17 CTAD: compiler deduces template arguments from constructor
std::pair p2 = {1, 2.5};       // Deduces pair<int, double>
std::tuple t = {1, 2.5, "hi"}; // Deduces tuple<int, double, const char*>

// Custom class with deduction guide
template<typename T>
class Box {
public:
    Box(T value) : m_value(std::move(value)) {}
    T m_value;
};

// Explicit deduction guide: Box("hello") deduces Box<std::string>, not Box<const char*>
Box(const char*) -> Box<std::string>;

void ctad_demo() {
    Box b1(42);        // Box<int>
    Box b2(3.14);      // Box<double>
    Box b3("hello");   // Box<std::string> (via deduction guide)
}
```

## Variadic Templates

```cpp
#include <iostream>
#include <string>
#include <tuple>
#include <utility>

// Variadic function template — accepts any number of arguments
template<typename... Args>
void print_all(Args&&... args) {
    // Fold expression (C++17): expands pack with operator
    ((std::cout << std::forward<Args>(args) << ' '), ...);
    std::cout << '\n';
}

// Count arguments at compile time
template<typename... Args>
constexpr std::size_t count_args(Args&&...) {
    return sizeof...(Args);  // sizeof... for parameter packs
}

// Recursive variadic template (pre-C++17 style)
void print_recursive() { std::cout << '\n'; }  // Base case

template<typename First, typename... Rest>
void print_recursive(const First& first, const Rest&... rest) {
    std::cout << first << ' ';
    print_recursive(rest...);  // Peel off first, recurse with rest
}

// C++17 Fold expressions
template<typename... T>
auto sum(T... vals) { return (vals + ... + 0); }  // Left fold with init

template<typename... T>
auto product(T... vals) { return (vals * ...); }   // Right fold, no init

template<typename... T>
bool all_true(T... vals) { return (vals && ...); }  // Logical AND fold

// std::tuple operations
template<typename... Types>
void print_tuple_impl(const std::tuple<Types...>& t, std::index_sequence<>) {}

template<std::size_t I, std::size_t... Is, typename... Types>
void print_tuple_impl(const std::tuple<Types...>& t, std::index_sequence<I, Is...>) {
    std::cout << std::get<I>(t);
    if constexpr (sizeof...(Is) > 0) std::cout << ", ";
    print_tuple_impl(t, std::index_sequence<Is...>{});
}

template<typename... Types>
void print_tuple(const std::tuple<Types...>& t) {
    std::cout << '(';
    print_tuple_impl(t, std::index_sequence_for<Types...>{});
    std::cout << ")\n";
}

void variadic_demo() {
    print_all(1, 2.5, "hello", true);        // 1 2.5 hello 1
    std::cout << count_args(1, 2, 3) << '\n'; // 3
    std::cout << sum(1, 2, 3, 4, 5) << '\n'; // 15

    auto t = std::make_tuple(1, 3.14, std::string("hi"));
    print_tuple(t);  // (1, 3.14, hi)
}
```

## Concepts (C++20)

Concepts constrain template parameters — replacing SFINAE with readable, compiler-checked requirements.

```cpp
#include <concepts>
#include <numeric>
#include <iterator>
#include <ranges>

// Requires clause (simplest form)
template<typename T>
    requires std::integral<T>
T next(T val) { return val + 1; }

// Abbreviated function template (C++20 — terse syntax)
auto next_auto(std::integral auto val) { return val + 1; }

// Requires expression: defines a concept inline
template<typename T>
concept Addable = requires(T a, T b) {
    { a + b } -> std::convertible_to<T>;  // Must return T-compatible
    { a += b } -> std::same_as<T&>;       // Must return T&
};

// Compound concept using &&, ||
template<typename T>
concept Numeric = std::integral<T> || std::floating_point<T>;

// Custom concept using type traits
template<typename T>
concept Sortable = requires(T container) {
    { container.begin() } -> std::input_iterator;
    { container.end()   } -> std::input_iterator;
    { container.size()  } -> std::convertible_to<std::size_t>;
    requires std::totally_ordered<typename T::value_type>;
};

// Using standard concepts from <concepts>:
// std::same_as<T, U>
// std::convertible_to<From, To>
// std::integral<T>
// std::floating_point<T>
// std::signed_integral<T>
// std::unsigned_integral<T>
// std::invocable<F, Args...>
// std::predicate<F, Args...>
// std::copyable<T>
// std::movable<T>
// std::equality_comparable<T>
// std::totally_ordered<T>

// Constrained algorithm
template<Numeric T>
T clamp(T val, T lo, T hi) {
    return val < lo ? lo : (val > hi ? hi : val);
}

void concepts_demo() {
    std::cout << next(41) << '\n';    // int: 42
    std::cout << next('A') << '\n';   // char: 'B'
    // next(3.14);  // Compile error: double is not integral

    std::cout << clamp(15, 0, 10) << '\n';    // 10
    std::cout << clamp(5.0, 0.0, 10.0) << '\n'; // 5.0
}
```

## constexpr and consteval

```cpp
#include <array>
#include <numeric>

// constexpr function: CAN be evaluated at compile time
constexpr int factorial(int n) {
    return n <= 1 ? 1 : n * factorial(n - 1);
}

// consteval (C++20): MUST be evaluated at compile time
consteval int compile_time_only(int n) {
    return n * n;
}

// constinit (C++20): variable initialized at compile time, but not const
constinit int global_counter = factorial(5);  // Initialized to 120 at compile time

// Compile-time computations
constexpr auto factorials = []() {
    std::array<int, 10> arr{};
    arr[0] = 1;
    for (int i = 1; i < 10; i++) arr[i] = arr[i-1] * i;
    return arr;
}();  // IIFE (immediately invoked function expression)

static_assert(factorials[5] == 120, "5! should be 120");

// if constexpr (C++17): compile-time branching in templates
template<typename T>
std::string to_string_nice(T val) {
    if constexpr (std::is_same_v<T, bool>) {
        return val ? "true" : "false";
    } else if constexpr (std::is_floating_point_v<T>) {
        return std::to_string(val).substr(0, 6);  // Trim long decimals
    } else {
        return std::to_string(val);
    }
    // No need for SFINAE or overloads — one function handles all cases
}
```

## SFINAE and Type Traits

```cpp
#include <type_traits>

// type_traits: compile-time type queries
static_assert(std::is_integral_v<int>);
static_assert(!std::is_integral_v<double>);
static_assert(std::is_same_v<int, int>);
static_assert(!std::is_same_v<int, long>);

// Type transformations
std::remove_reference_t<int&>  // int
std::remove_const_t<const int> // int
std::decay_t<int[5]>           // int*
std::decay_t<const int&>       // int
std::add_const_t<int>          // const int

// std::enable_if: enable/disable overloads based on type properties
template<typename T>
std::enable_if_t<std::is_integral_v<T>, bool>
is_even(T n) { return n % 2 == 0; }

// std::conditional: choose type at compile time
template<typename T>
using BigEnough = std::conditional_t<sizeof(T) >= 8, T, std::int64_t>;

// std::void_t: detect if expression is valid (C++17)
template<typename T, typename = void>
struct has_serialize : std::false_type {};

template<typename T>
struct has_serialize<T, std::void_t<decltype(std::declval<T>().serialize())>>
    : std::true_type {};

// Helper variable template
template<typename T>
inline constexpr bool has_serialize_v = has_serialize<T>::value;

struct Serializable { std::string serialize() const { return "{}"; } };
struct NotSerializable {};

static_assert(has_serialize_v<Serializable>);
static_assert(!has_serialize_v<NotSerializable>);
```

:::tip[Concepts over SFINAE]
In C++20, always prefer **Concepts** over SFINAE (`std::enable_if`). Concepts produce dramatically better error messages, are easier to read, and can be combined with `&&` and `||`. SFINAE remains relevant when targeting C++17 or when working with pre-C++20 code.
:::

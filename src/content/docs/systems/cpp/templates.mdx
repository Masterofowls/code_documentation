# Templates in C++

Templates are C++'s mechanism for generic programming. They allow writing code that works with any type satisfying a set of requirements — with zero runtime overhead (all resolved at compile time).

## Function Templates

```cpp
#include <algorithm>
#include <string>
#include <type_traits>

// Basic function template
template <typename T>
T maximum(T a, T b) {
    return a > b ? a : b;
}

// Multiple type parameters
template <typename T, typename U>
auto add(T a, U b) -> decltype(a + b) {
    return a + b;
}

// Non-type template parameter
template <int N>
void print_n_times(const char *msg) {
    for (int i = 0; i < N; i++) std::puts(msg);
}

// Template with default argument
template <typename T = double>
T pi() { return static_cast<T>(3.14159265358979); }

// Abbreviated function template (C++20 — auto parameter)
auto add_modern(auto a, auto b) { return a + b; }  // same as template

// Usage
int    m1 = maximum(3, 7);           // T = int
double m2 = maximum(3.14, 2.71);     // T = double
// maximum(3, 2.71);                 // Error: T can't be both int and double
double m3 = maximum<double>(3, 2.71); // Explicit: T = double
print_n_times<3>("hello");            // N = 3, compile-time loop
float pi_f = pi<float>();
```

## Class Templates

```cpp
#include <memory>
#include <stdexcept>

// Stack implemented as class template
template <typename T, size_t Capacity = 256>
class StaticStack {
public:
    void push(const T& value) {
        if (size_ >= Capacity) throw std::overflow_error("Stack full");
        data_[size_++] = value;
    }

    void push(T&& value) {
        if (size_ >= Capacity) throw std::overflow_error("Stack full");
        data_[size_++] = std::move(value);
    }

    T pop() {
        if (empty()) throw std::underflow_error("Stack empty");
        return std::move(data_[--size_]);
    }

    const T& top() const {
        if (empty()) throw std::underflow_error("Stack empty");
        return data_[size_ - 1];
    }

    bool   empty() const { return size_ == 0; }
    size_t size()  const { return size_; }
    static constexpr size_t capacity() { return Capacity; }

private:
    T      data_[Capacity];
    size_t size_ = 0;
};

// Usage
StaticStack<int>          int_stack;          // Capacity = 256
StaticStack<std::string, 16> str_stack;       // Capacity = 16
int_stack.push(42);
int_stack.push(100);
std::cout << int_stack.pop() << '\n';  // 100

// Class template with type parameter and value parameter
template <typename T, size_t Rows, size_t Cols>
class Matrix {
public:
    T& at(size_t r, size_t c) { return data_[r * Cols + c]; }
    const T& at(size_t r, size_t c) const { return data_[r * Cols + c]; }

    constexpr size_t rows() const { return Rows; }
    constexpr size_t cols() const { return Cols; }

private:
    T data_[Rows * Cols] = {};
};

Matrix<float, 4, 4> transform;  // 4x4 float matrix, stack-allocated
```

## Template Specialization

```cpp
#include <cstring>

// Primary template
template <typename T>
bool is_zero(T value) {
    return value == T{};  // default-constructed T is "zero" for numeric types
}

// Full specialization for float (handle floating-point imprecision)
template <>
bool is_zero<float>(float value) {
    return std::abs(value) < 1e-6f;
}

// Full specialization for const char*
template <>
bool is_zero<const char*>(const char *s) {
    return s == nullptr || s[0] == '\0';
}

// Partial specialization (only for class templates, not function templates)
template <typename T>
class Container { /* general case */ };

template <typename T>
class Container<T*> { /* specialization for pointer types */ };

template <>
class Container<bool> { /* full specialization for bool (bitset optimization) */ };

// Partial specialization for pairs
template <typename T1, typename T2>
struct Pair { T1 first; T2 second; };

template <typename T>
struct Pair<T, T> {
    T first, second;
    T sum() const { return first + second; }  // extra feature for same-type pairs
};
```

## Class Template Argument Deduction (CTAD — C++17)

```cpp
#include <vector>
#include <string>
#include <utility>

// Before C++17: had to specify template arguments explicitly
std::vector<int>   v1 = {1, 2, 3};
std::pair<int, double> p1 = {1, 3.14};

// C++17 CTAD: deduce from constructor arguments
std::vector  v2 = {1, 2, 3};        // deduced: vector<int>
std::pair    p2 = {1, 3.14};        // deduced: pair<int, double>
std::string  s  = "hello";          // always worked

// Custom deduction guides
template <typename T>
class MyWrapper {
public:
    MyWrapper(T value) : value_(value) {}
private:
    T value_;
};
// Deduction guide: MyWrapper(T) -> MyWrapper<T>   (often implicit)

// For our StaticStack:
// StaticStack stack = {1, 2, 3};  -- would need a deduction guide
```

## Variadic Templates

```cpp
#include <iostream>
#include <tuple>

// Base case
template <typename T>
void print(const T& value) {
    std::cout << value;
}

// Variadic template: pack expansion
template <typename T, typename... Ts>
void print(const T& first, const Ts&... rest) {
    std::cout << first;
    if constexpr (sizeof...(rest) > 0) {
        std::cout << ", ";
        print(rest...);  // recursive: peel off first argument
    }
}

// sizeof...: number of arguments in pack
template <typename... Ts>
constexpr size_t count() { return sizeof...(Ts); }

// C++17 fold expressions — eliminate recursion
template <typename... Args>
auto sum(Args... args) {
    return (args + ...);   // unary right fold: (a + (b + (c + 0)))
}

template <typename... Args>
auto product(Args... args) {
    return (1 * ... * args);  // unary left fold: (((1 * a) * b) * c)
}

template <typename... Args>
void print_all(Args&&... args) {
    ((std::cout << std::forward<Args>(args) << ' '), ...);  // comma fold
    std::cout << '\n';
}

// Perfect forwarding with variadic templates
template <typename T, typename... Args>
std::unique_ptr<T> make(Args&&... args) {
    return std::unique_ptr<T>(new T(std::forward<Args>(args)...));
}

// std::tuple: uses variadic templates internally
std::tuple<int, std::string, double> t = {1, "hello", 3.14};
auto [i, s, d] = t;  // structured binding (C++17)
```

## Concepts (C++20)

Concepts are named constraints on template parameters. They replace SFINAE with readable, descriptive requirements.

```cpp
#include <concepts>
#include <string>
#include <iterator>

// Using predefined concepts
template <std::integral T>
T gcd(T a, T b) {
    while (b) { a %= b; std::swap(a, b); }
    return a;
}

template <std::floating_point T>
T circle_area(T r) {
    return static_cast<T>(3.14159265) * r * r;
}

// requires clause: constrain with a concept expression
template <typename T>
requires std::same_as<T, int> || std::same_as<T, long>
T safe_div(T a, T b) { return a / b; }

// requires expression: ad-hoc constraints
template <typename T>
requires requires(T a, T b) {
    { a + b } -> std::convertible_to<T>;  // must support + returning T
    { a < b } -> std::convertible_to<bool>;
}
T min_val(T a, T b) { return a < b ? a : b; }

// Define a custom concept
template <typename T>
concept Printable = requires(const T& t) {
    { std::cout << t };  // must support stream output
};

template <typename T>
concept Numeric = std::integral<T> || std::floating_point<T>;

template <typename C>
concept Container = requires(C c) {
    typename C::value_type;
    typename C::iterator;
    { c.begin() } -> std::same_as<typename C::iterator>;
    { c.end() }   -> std::same_as<typename C::iterator>;
    { c.size() }  -> std::convertible_to<std::size_t>;
};

// Concept used in multiple forms:
// 1. Trailing requires
template <typename T> T add1(T a, T b) requires Numeric<T> { return a + b; }
// 2. Concept as parameter kind
template <Numeric T>  T add2(T a, T b) { return a + b; }
// 3. Abbreviated function template
Numeric auto add3(Numeric auto a, Numeric auto b) { return a + b; }
```

## constexpr Functions and Variables

```cpp
#include <array>

// constexpr function: evaluated at compile time when inputs are constant
constexpr int factorial(int n) {
    return n <= 1 ? 1 : n * factorial(n - 1);
}

constexpr int fib(int n) {
    if (n <= 1) return n;
    return fib(n-1) + fib(n-2);
}

// Computed at compile time
constexpr int fact10 = factorial(10);  // 3628800 — no runtime cost
constexpr int fib20  = fib(20);        // 6765

// constexpr variable
constexpr double PI = 3.14159265358979;
constexpr size_t BUF_SIZE = 1024;

// constexpr with arrays and complex logic (C++14+)
constexpr std::array<int, 5> make_squares() {
    std::array<int, 5> arr{};
    for (int i = 0; i < 5; i++) arr[i] = i * i;
    return arr;
}
constexpr auto squares = make_squares();  // {0, 1, 4, 9, 16} — compile time

// consteval (C++20): MUST be evaluated at compile time
consteval int must_be_compile_time(int n) {
    return n * 2;
}
constexpr int val = must_be_compile_time(21);  // OK
// int runtime_n = get_n();
// int bad = must_be_compile_time(runtime_n);  // ERROR: not a constant expression

// constinit (C++20): variable must be initialized at compile time
// (but can be modified at runtime — unlike constexpr)
constinit int g_server_port = 8080;
```

## if constexpr (C++17)

```cpp
#include <type_traits>

// Compile-time branching: eliminates the branch not taken
template <typename T>
std::string to_string_nice(const T& val) {
    if constexpr (std::is_same_v<T, bool>) {
        return val ? "true" : "false";
    } else if constexpr (std::is_floating_point_v<T>) {
        return std::to_string(val);
    } else if constexpr (std::is_integral_v<T>) {
        return std::to_string(val);
    } else {
        // Assumes T has operator<<
        std::ostringstream oss;
        oss << val;
        return oss.str();
    }
    // Only the matching branch is compiled — others don't need to compile!
}
```

## Template Metaprogramming (TMP) and Type Traits

```cpp
#include <type_traits>

// std::type_traits — compile-time type inspection
static_assert(std::is_integral_v<int>);              // true
static_assert(std::is_floating_point_v<double>);     // true
static_assert(std::is_same_v<int, int>);             // true
static_assert(!std::is_same_v<int, long>);           // true (on most platforms)
static_assert(std::is_pointer_v<int*>);              // true
static_assert(std::is_reference_v<int&>);            // true
static_assert(std::is_const_v<const int>);           // true
static_assert(std::is_trivially_copyable_v<int>);    // true

// Type transformations
using NoRef  = std::remove_reference_t<int&>;       // int
using NoCV   = std::remove_cv_t<const volatile int>; // int
using AddPtr = std::add_pointer_t<int>;              // int*
using Decay  = std::decay_t<int[5]>;                // int*

// std::conditional: if-then-else for types
template <bool Condition, typename T, typename F>
using Select = std::conditional_t<Condition, T, F>;

using BigInt = Select<sizeof(void*) == 8, int64_t, int32_t>;  // int64_t on 64-bit

// std::enable_if: SFINAE-based constraint (prefer concepts in C++20)
template <typename T>
typename std::enable_if_t<std::is_integral_v<T>, T>
safe_abs(T value) {
    return value < 0 ? -value : value;
}

// std::void_t: detect whether an expression is well-formed (SFINAE)
template <typename T, typename = void>
struct has_size : std::false_type {};

template <typename T>
struct has_size<T, std::void_t<decltype(std::declval<T>().size())>>
    : std::true_type {};

static_assert(has_size<std::vector<int>>::value);   // true
static_assert(!has_size<int>::value);               // true
```

:::tip[Concepts over SFINAE]
In C++20, prefer concepts over SFINAE and `std::enable_if`. Concepts produce dramatically better error messages and are much easier to read and write.
```cpp
// SFINAE (C++17): hard to read, cryptic errors
template <typename T>
std::enable_if_t<std::is_integral_v<T>, T> abs_val(T v);

// Concept (C++20): clear intent, good errors
template <std::integral T>
T abs_val(T v);
```
:::

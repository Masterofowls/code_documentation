Mocking is the practice of replacing real dependencies with controlled substitutes so you can test a unit in isolation. Jest provides a complete mocking toolkit built into the framework — no external libraries needed.

## jest.fn() — Mock Functions

`jest.fn()` creates a mock function that records every call made to it. You can control what it returns and inspect how it was used.

```typescript
const mockAdd = jest.fn();

// Call the mock
mockAdd(2, 3);
mockAdd(5, 10);

// Inspect calls
expect(mockAdd).toHaveBeenCalledTimes(2);
expect(mockAdd).toHaveBeenCalledWith(2, 3);
expect(mockAdd).toHaveBeenLastCalledWith(5, 10);
expect(mockAdd).toHaveBeenNthCalledWith(1, 2, 3);

// Access call data directly
console.log(mockAdd.mock.calls);          // [[2, 3], [5, 10]]
console.log(mockAdd.mock.results);        // [{type: 'return', value: undefined}, ...]
console.log(mockAdd.mock.instances);      // [undefined, undefined] (no `this`)
```

### Controlling Return Values

```typescript
// Return a fixed value on every call
const mockGetUser = jest.fn().mockReturnValue({ id: 1, name: 'Alice' });

expect(mockGetUser()).toEqual({ id: 1, name: 'Alice' });
expect(mockGetUser()).toEqual({ id: 1, name: 'Alice' }); // same every time

// Return different values on successive calls
const mockRandom = jest.fn()
  .mockReturnValueOnce(0.1)
  .mockReturnValueOnce(0.5)
  .mockReturnValueOnce(0.9)
  .mockReturnValue(0);          // default for all subsequent calls

expect(mockRandom()).toBe(0.1);
expect(mockRandom()).toBe(0.5);
expect(mockRandom()).toBe(0.9);
expect(mockRandom()).toBe(0);   // falls back to default
expect(mockRandom()).toBe(0);
```

### Controlling Async Return Values

```typescript
// Resolve a promise
const mockFetch = jest.fn().mockResolvedValue({ status: 200, data: [] });
const result = await mockFetch('/api/users');
expect(result).toEqual({ status: 200, data: [] });

// Reject a promise
const mockFailingFetch = jest.fn().mockRejectedValue(new Error('Network error'));
await expect(mockFailingFetch()).rejects.toThrow('Network error');

// Alternate success and failure
const mockUnreliable = jest.fn()
  .mockResolvedValueOnce({ data: [1, 2, 3] })
  .mockRejectedValueOnce(new Error('Timeout'))
  .mockResolvedValue({ data: [] });
```

### Custom Implementations with `mockImplementation`

```typescript
// Full custom implementation
const mockDivide = jest.fn().mockImplementation((a: number, b: number) => {
  if (b === 0) throw new Error('Division by zero');
  return a / b;
});

expect(mockDivide(10, 2)).toBe(5);
expect(() => mockDivide(10, 0)).toThrow('Division by zero');

// One-time custom implementation
const mockSave = jest.fn()
  .mockImplementationOnce(() => Promise.resolve({ id: 1 }))
  .mockImplementation(() => Promise.reject(new Error('DB connection lost')));
```

## jest.spyOn() — Spy on Existing Methods

`jest.spyOn()` wraps an existing method with a mock, letting you observe calls without necessarily replacing the implementation.

```typescript
// Spy without replacing the implementation
const consoleSpy = jest.spyOn(console, 'log');

console.log('hello');

expect(consoleSpy).toHaveBeenCalledWith('hello');
expect(consoleSpy).toHaveBeenCalledTimes(1);

// Restore the original implementation
consoleSpy.mockRestore();
```

### Replacing the Implementation with spyOn

```typescript
import * as mathUtils from './mathUtils';

describe('computeScore', () => {
  it('uses the cached result when available', () => {
    // Replace the real implementation temporarily
    const spy = jest.spyOn(mathUtils, 'expensiveCalculation')
      .mockReturnValue(42);

    const result = computeScore(userId);

    expect(spy).toHaveBeenCalledTimes(1);
    expect(result).toBe(42);

    spy.mockRestore(); // Restore original after test
  });
});
```

:::tip[spyOn vs jest.mock()]
Use `jest.spyOn()` when you want to:
- Spy on a method without fully replacing it
- Temporarily replace a method for one test, then restore it
- Mock a method on an already-imported object or class instance

Use `jest.mock()` when you want to replace an entire module wholesale for all tests in a file.
:::

```typescript
// Spying on class methods
class EmailService {
  async send(to: string, subject: string, body: string) {
    // Real implementation would call an SMTP server
  }
}

const emailService = new EmailService();
const spy = jest.spyOn(emailService, 'send').mockResolvedValue(undefined);

await userService.register({ email: 'alice@example.com' });

expect(spy).toHaveBeenCalledWith(
  'alice@example.com',
  'Welcome to our platform',
  expect.stringContaining('Alice'),
);
```

## jest.mock() — Module Mocking

`jest.mock()` replaces an entire module at the module registry level. It is hoisted to the top of the test file by Babel/ts-jest, meaning it runs before any imports.

### Auto-Mocking

```typescript
// jest.mock() with no factory = auto-mock
// All exports become jest.fn(), return undefined by default
jest.mock('./emailService');

import { EmailService } from './emailService';

it('sends a welcome email on registration', async () => {
  const mockSend = EmailService.prototype.send as jest.Mock;
  mockSend.mockResolvedValue({ messageId: 'abc123' });

  await registerUser({ email: 'alice@example.com' });

  expect(mockSend).toHaveBeenCalledTimes(1);
});
```

### Manual Mock with Factory Function

```typescript
// Provide a factory function to control exactly what the mock exports
jest.mock('./database', () => ({
  db: {
    query: jest.fn(),
    transaction: jest.fn(),
    close: jest.fn(),
  },
}));

import { db } from './database';

it('queries the database with correct parameters', async () => {
  (db.query as jest.Mock).mockResolvedValue([{ id: 1, name: 'Alice' }]);

  const users = await userRepository.findAll();

  expect(db.query).toHaveBeenCalledWith(
    'SELECT * FROM users WHERE active = $1',
    [true],
  );
  expect(users).toHaveLength(1);
});
```

### Partial Mock with `jest.requireActual`

Keep real implementations for most exports but replace specific ones:

```typescript
jest.mock('./config', () => ({
  // Import and spread real module
  ...jest.requireActual('./config'),
  // Override only what you need
  API_URL: 'http://localhost:3000',
  FEATURE_FLAGS: { newDashboard: true, betaSignup: false },
}));
```

## The `__mocks__` Directory

Manual mocks live in a `__mocks__` directory adjacent to the module they replace. Jest automatically uses them when `jest.mock('moduleName')` is called (or always for `node_modules` mocks).

```
src/
├── services/
│   ├── __mocks__/
│   │   └── emailService.ts    ← manual mock for emailService.ts
│   └── emailService.ts
└── __mocks__/
    └── axios.ts               ← manual mock for node_modules/axios
```

```typescript
// src/services/__mocks__/emailService.ts
export class EmailService {
  send = jest.fn().mockResolvedValue({ messageId: 'mock-message-id' });
  verify = jest.fn().mockResolvedValue(true);
}
```

```typescript
// src/__mocks__/axios.ts
const axios = {
  get: jest.fn(),
  post: jest.fn(),
  put: jest.fn(),
  delete: jest.fn(),
  create: jest.fn(() => axios),
  defaults: { headers: { common: {} } },
};
export default axios;
```

## Mocking ES Modules

ES Modules require special handling because they are statically bound. The recommended approach with ts-jest or Babel:

```typescript
// Named export mock
jest.mock('./utils', () => ({
  formatDate: jest.fn(() => '2024-01-01'),
  parseDate: jest.fn(),
}));

// Default export mock
jest.mock('./logger', () => ({
  __esModule: true,
  default: {
    info: jest.fn(),
    error: jest.fn(),
    warn: jest.fn(),
  },
}));
```

### Mocking Default Exports

```typescript
// Module: src/analytics.ts
// export default class Analytics { ... }

jest.mock('./analytics', () => {
  return {
    __esModule: true,
    default: jest.fn().mockImplementation(() => ({
      track: jest.fn(),
      identify: jest.fn(),
    })),
  };
});

import Analytics from './analytics';

it('tracks page view on mount', () => {
  const mockTrack = jest.fn();
  (Analytics as jest.Mock).mockImplementation(() => ({ track: mockTrack }));

  renderComponent();

  expect(mockTrack).toHaveBeenCalledWith('page_view', expect.any(Object));
});
```

### Mocking Named Exports

```typescript
import { fetchUser, updateUser } from './userApi';

jest.mock('./userApi');

const mockFetchUser = fetchUser as jest.MockedFunction<typeof fetchUser>;
const mockUpdateUser = updateUser as jest.MockedFunction<typeof updateUser>;

beforeEach(() => {
  mockFetchUser.mockResolvedValue({ id: 1, name: 'Alice' });
  mockUpdateUser.mockResolvedValue({ id: 1, name: 'Alice Updated' });
});
```

## Mocking Timers

Jest can replace native timer functions with synchronous fakes, enabling you to test time-dependent code without actually waiting.

```typescript
describe('debounce', () => {
  beforeEach(() => {
    // Replace setTimeout, setInterval, clearTimeout, clearInterval, Date
    jest.useFakeTimers();
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  it('delays execution by the specified amount', () => {
    const callback = jest.fn();
    const debounced = debounce(callback, 500);

    debounced();
    expect(callback).not.toHaveBeenCalled(); // Not called yet

    jest.advanceTimersByTime(499);
    expect(callback).not.toHaveBeenCalled(); // Still not called

    jest.advanceTimersByTime(1);             // Total: 500ms
    expect(callback).toHaveBeenCalledTimes(1);
  });

  it('cancels previous calls when called again', () => {
    const callback = jest.fn();
    const debounced = debounce(callback, 500);

    debounced();
    jest.advanceTimersByTime(300);
    debounced();                             // Reset the timer
    jest.advanceTimersByTime(300);
    expect(callback).not.toHaveBeenCalled(); // 300ms since last call
    jest.advanceTimersByTime(200);
    expect(callback).toHaveBeenCalledTimes(1);
  });
});
```

### Timer Control Methods

| Method | What it does |
|---|---|
| `jest.useFakeTimers()` | Replaces all timer functions with fakes |
| `jest.useRealTimers()` | Restores real timer functions |
| `jest.runAllTimers()` | Exhausts all pending timers (including recursive) |
| `jest.runAllTimersAsync()` | Same as above but for async timers |
| `jest.runOnlyPendingTimers()` | Only runs currently-pending timers (not newly created) |
| `jest.advanceTimersByTime(ms)` | Advances fake clock by N milliseconds |
| `jest.advanceTimersByTimeAsync(ms)` | Async version of above |
| `jest.clearAllTimers()` | Removes all pending timers |
| `jest.getTimerCount()` | Returns number of pending timers |

### Mocking `Date`

```typescript
beforeEach(() => {
  jest.useFakeTimers();
  jest.setSystemTime(new Date('2024-06-15T10:00:00Z'));
});

it('stamps events with the current time', () => {
  const event = createEvent('user_login');
  expect(event.timestamp).toEqual(new Date('2024-06-15T10:00:00Z'));
});

it('expires tokens after 24 hours', () => {
  const token = generateToken();
  jest.advanceTimersByTime(24 * 60 * 60 * 1000 + 1); // 24h + 1ms

  expect(isTokenExpired(token)).toBe(true);
});
```

## Clearing, Resetting, and Restoring Mocks

| Method | What it clears | Usage |
|---|---|---|
| `mockFn.mockClear()` | Call history only (`.mock.calls`, `.mock.results`) | Between tests when you keep the implementation |
| `mockFn.mockReset()` | Call history + return values/implementations | When you want a clean mock with no behaviour |
| `mockFn.mockRestore()` | Everything + restores original (only for spies) | After `jest.spyOn()` to undo the spy |
| `jest.clearAllMocks()` | Calls `mockClear()` on every mock in the file | `afterEach` cleanup |
| `jest.resetAllMocks()` | Calls `mockReset()` on every mock | Stricter reset between tests |
| `jest.restoreAllMocks()` | Calls `mockRestore()` on every spy | Clean up all spies |

```typescript
afterEach(() => {
  jest.clearAllMocks();
});

// Or configure globally in jest.config.ts:
// clearMocks: true      — auto-calls clearAllMocks after each test
// resetMocks: true      — auto-calls resetAllMocks after each test
// restoreMocks: true    — auto-calls restoreAllMocks after each test
```

:::tip[Configure Mock Cleanup Globally]
Set `clearMocks: true` in `jest.config.ts` to automatically clear all mock call histories after each test. This eliminates the most common source of test interference (a mock called in test A affecting assertions in test B) without you needing to write `afterEach` cleanup code.
:::

:::caution[resetMocks Removes Implementations]
`resetMocks: true` resets both call history AND return value implementations. If you define `mockReturnValue` once at the describe level, it will be wiped before each test. Prefer `clearMocks` unless you specifically want this behaviour.
:::

## Practical Example: Mocking a Full Service

```typescript
// src/services/notificationService.ts
export class NotificationService {
  constructor(private readonly emailClient: EmailClient, private readonly smsClient: SmsClient) {}

  async notify(userId: number, event: string): Promise<void> {
    const user = await fetchUser(userId);
    if (user.preferences.email) await this.emailClient.send(user.email, event);
    if (user.preferences.sms) await this.smsClient.send(user.phone, event);
  }
}

// __tests__/notificationService.test.ts
jest.mock('../utils/fetchUser');

import { NotificationService } from '../services/notificationService';
import { fetchUser } from '../utils/fetchUser';

const mockFetchUser = fetchUser as jest.MockedFunction<typeof fetchUser>;

describe('NotificationService', () => {
  const mockEmailClient = { send: jest.fn().mockResolvedValue(undefined) };
  const mockSmsClient = { send: jest.fn().mockResolvedValue(undefined) };
  let service: NotificationService;

  beforeEach(() => {
    service = new NotificationService(mockEmailClient as any, mockSmsClient as any);
    mockFetchUser.mockResolvedValue({
      id: 1,
      email: 'alice@example.com',
      phone: '+1234567890',
      preferences: { email: true, sms: false },
    });
  });

  it('sends email when user has email notifications enabled', async () => {
    await service.notify(1, 'order_shipped');
    expect(mockEmailClient.send).toHaveBeenCalledWith('alice@example.com', 'order_shipped');
  });

  it('does not send SMS when user has SMS notifications disabled', async () => {
    await service.notify(1, 'order_shipped');
    expect(mockSmsClient.send).not.toHaveBeenCalled();
  });

  it('sends SMS when user has SMS notifications enabled', async () => {
    mockFetchUser.mockResolvedValueOnce({
      id: 1,
      email: 'alice@example.com',
      phone: '+1234567890',
      preferences: { email: false, sms: true },
    });

    await service.notify(1, 'order_shipped');

    expect(mockSmsClient.send).toHaveBeenCalledWith('+1234567890', 'order_shipped');
    expect(mockEmailClient.send).not.toHaveBeenCalled();
  });
});
```

This guide walks you through setting up Playwright from scratch, understanding the generated project structure, and running your first tests.

## Installation

The fastest way to set up Playwright is the interactive init command:

```bash
npm init playwright@latest
```

This launches an interactive setup wizard:

```
Getting started with writing end-to-end tests with Playwright:
Initializing project in '.'

✔ Do you want to use TypeScript or JavaScript? · TypeScript
✔ Where to put your end-to-end tests? · tests
✔ Add a GitHub Actions workflow? · true
✔ Install Playwright browsers (can be done manually via 'npx playwright install')? · true
```

<Steps>
  1. **Choose TypeScript** — strongly recommended for type safety and
  autocompletion. 2. **Tests folder** — defaults to `tests/`. Adjust based on
  your project structure (e.g., `e2e/`). 3. **GitHub Actions** — generates a
  ready-to-use CI workflow file. 4. **Install browsers** — downloads Chromium,
  Firefox, and WebKit (~300MB total).
</Steps>

### Manual Installation

```bash
# Install the package
npm install --save-dev @playwright/test

# Install browser binaries
npx playwright install

# Install only specific browsers
npx playwright install chromium firefox

# Install browsers with OS dependencies (required in CI/Docker)
npx playwright install --with-deps
```

## Project Structure

After initialization, your project looks like this:

```
my-project/
├── tests/
│   └── example.spec.ts          ← generated example test
├── tests-examples/
│   └── demo-todo-app.spec.ts    ← more complete examples
├── playwright.config.ts          ← main configuration
├── .github/
│   └── workflows/
│       └── playwright.yml        ← CI workflow (if selected)
└── package.json
```

:::tip[Organizing Tests]
For larger projects, organize tests by feature:

```
tests/
├── auth/
│   ├── login.spec.ts
│   └── registration.spec.ts
├── checkout/
│   ├── cart.spec.ts
│   └── payment.spec.ts
└── shared/
    ├── fixtures.ts
    └── page-objects/
```

:::

## playwright.config.ts

The central configuration file. Key options explained:

```typescript
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  // Directory containing test files
  testDir: './tests',

  // Match test files
  testMatch: '**/*.spec.ts',

  // Maximum time per test (in milliseconds)
  timeout: 30 * 1000,

  // Maximum time for expect() assertions to succeed
  expect: { timeout: 5000 },

  // Run tests in files in parallel
  fullyParallel: true,

  // Fail the build on CI if you accidentally left test.only in source
  forbidOnly: !!process.env.CI,

  // Retry failed tests on CI
  retries: process.env.CI ? 2 : 0,

  // Parallelize tests across CPU cores on CI
  workers: process.env.CI ? '50%' : undefined,

  // Reporter configuration
  reporter: [
    ['html', { open: 'never' }], // HTML report
    ['list'], // terminal list
    process.env.CI ? ['github'] : ['dot'], // CI annotations
  ],

  // Shared settings for all projects
  use: {
    // Base URL for page.goto('/path') calls
    baseURL: process.env.BASE_URL || 'http://localhost:3000',

    // Record traces on first retry of failed tests
    trace: 'on-first-retry',

    // Record video on first retry of failed tests
    video: 'on-first-retry',

    // Take screenshot on failure
    screenshot: 'only-on-failure',

    // Run tests headless (no UI window)
    headless: true,

    // Browser viewport size
    viewport: { width: 1280, height: 720 },

    // Ignore HTTPS certificate errors
    ignoreHTTPSErrors: false,

    // Navigation timeout
    navigationTimeout: 30 * 1000,

    // Action timeout (clicks, fills, etc.)
    actionTimeout: 15 * 1000,
  },

  // Browser/device projects
  projects: [
    {
      name: 'chromium',
      use: { ...devices['Desktop Chrome'] },
    },
    {
      name: 'firefox',
      use: { ...devices['Desktop Firefox'] },
    },
    {
      name: 'webkit',
      use: { ...devices['Desktop Safari'] },
    },
    // Mobile viewports
    {
      name: 'Mobile Chrome',
      use: { ...devices['Pixel 5'] },
    },
    {
      name: 'Mobile Safari',
      use: { ...devices['iPhone 13'] },
    },
  ],

  // Start your local dev server before running tests
  webServer: {
    command: 'npm run start',
    url: 'http://localhost:3000',
    reuseExistingServer: !process.env.CI,
    timeout: 120 * 1000,
  },
});
```

## Installing Browsers

```bash
# Install all browsers
npx playwright install

# Install specific browsers
npx playwright install chromium
npx playwright install chromium firefox

# Install browsers + OS-level dependencies (Linux CI)
npx playwright install --with-deps

# Install a specific browser version
npx playwright install chromium@1091

# Show installed browsers
npx playwright --version
```

:::caution[Browser Binaries in CI]
Browser binaries are large (~300MB). In CI, always install them as part of the setup step and cache them using `npx playwright install --with-deps`. See the [CI guide](/testing/playwright/ci) for a complete caching strategy.
:::

## Running Tests

```bash
# Run all tests (headless, all configured projects)
npx playwright test

# Run a specific test file
npx playwright test tests/auth/login.spec.ts

# Run tests matching a name pattern
npx playwright test --grep "login"
npx playwright test --grep-invert "slow"

# Run tests in a specific project
npx playwright test --project=chromium
npx playwright test --project=chromium --project=firefox

# Run tests in parallel workers
npx playwright test --workers=4

# Run tests sequentially (useful for debugging)
npx playwright test --workers=1
```

### Headed Mode

Open a real browser window during test execution:

```bash
# Run with browser window visible
npx playwright test --headed

# Slow down actions (good for visual debugging)
npx playwright test --headed --slow-mo=500
```

### UI Mode

UI mode provides an interactive visual runner — see all tests, filter them, re-run individual tests, and watch them execute in a live browser:

```bash
npx playwright test --ui
```

UI mode features:

- Live browser preview
- Time-travel debugging with screenshots at each step
- Watch mode — automatically re-runs when files change
- Filter by test status (passed, failed, skipped)
- Expandable action trace per test

## Trace Viewer

Playwright traces are ZIP files containing a complete recording of your test:

```bash
# Enable tracing for all tests
npx playwright test --trace on

# View a specific trace file
npx playwright show-trace test-results/auth-login-chromium/trace.zip

# View the most recent trace
npx playwright show-trace
```

Configure trace recording in `playwright.config.ts`:

```typescript
use: {
  trace: 'on-first-retry',    // record on first retry of failed test (recommended for CI)
  trace: 'on',                // always record
  trace: 'off',               // never record
  trace: 'retain-on-failure', // save only for failed tests
}
```

## HTML Report

After running tests, Playwright generates a rich HTML report:

```bash
# Open the HTML report in your browser
npx playwright show-report

# Run tests and automatically open report
npx playwright test --reporter=html
```

The HTML report shows:

- Overall pass/fail/skip counts
- Per-test execution time
- Retry history
- Embedded screenshots and videos on failure
- Attached traces (clickable to open trace viewer)

## Your First Test

```typescript
// tests/home.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Home page', () => {
  test('has a title', async ({ page }) => {
    await page.goto('/');
    await expect(page).toHaveTitle(/My App/);
  });

  test('navigation links work', async ({ page }) => {
    await page.goto('/');

    await page.getByRole('link', { name: 'About' }).click();
    await expect(page).toHaveURL('/about');
    await expect(page.getByRole('heading', { name: 'About Us' })).toBeVisible();
  });

  test('search returns results', async ({ page }) => {
    await page.goto('/');

    await page.getByPlaceholder('Search...').fill('playwright');
    await page.keyboard.press('Enter');

    await expect(page).toHaveURL(/search\?q=playwright/);
    await expect(
      page.getByRole('list', { name: 'Search results' }),
    ).toBeVisible();
  });
});
```

Run just this test:

```bash
npx playwright test tests/home.spec.ts
npx playwright test tests/home.spec.ts --headed --project=chromium
```

## TypeScript Support

Playwright has first-class TypeScript support — no additional configuration needed. All Playwright APIs are fully typed.

```typescript
import { test, expect, type Page, type Locator } from '@playwright/test';

// Type-safe page object
class LoginPage {
  readonly page: Page;
  readonly emailInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;

  constructor(page: Page) {
    this.page = page;
    this.emailInput = page.getByLabel('Email');
    this.passwordInput = page.getByLabel('Password');
    this.submitButton = page.getByRole('button', { name: 'Sign in' });
  }

  async login(email: string, password: string): Promise<void> {
    await this.page.goto('/login');
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
    await this.page.waitForURL('/dashboard');
  }
}
```

Configure TypeScript for Playwright in `tsconfig.json`:

```json
{
  "compilerOptions": {
    "target": "ES2020",
    "module": "commonjs",
    "strict": true,
    "esModuleInterop": true,
    "types": ["node"]
  },
  "include": ["tests/**/*", "playwright.config.ts"]
}
```

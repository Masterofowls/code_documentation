Playwright provides a powerful network interception API that allows tests to mock API responses, modify requests in flight, block resources, and monitor all network traffic. This enables full E2E isolation without requiring a running backend.

## page.route() — Intercepting Requests

`page.route()` intercepts requests matching a URL pattern before they reach the network:

```typescript
// Match a specific URL
await page.route('/api/users', handler);

// Match with wildcards
await page.route('/api/users/**', handler);
await page.route('**/api/**', handler);

// Match with regex
await page.route(/\/api\/users\/\d+/, handler);

// Match by resource type
await page.route('**', (route) => {
  if (route.request().resourceType() === 'image') {
    route.abort();
  } else {
    route.continue();
  }
});
```

## route.fulfill — Mock Responses

Return a mock response instead of making a real network request:

```typescript
test('displays users from API', async ({ page }) => {
  // Mock the GET /api/users endpoint
  await page.route('/api/users', async (route) => {
    await route.fulfill({
      status: 200,
      contentType: 'application/json',
      body: JSON.stringify({
        users: [
          { id: 1, name: 'Alice Johnson', email: 'alice@example.com', role: 'admin' },
          { id: 2, name: 'Bob Smith', email: 'bob@example.com', role: 'user' },
        ],
        total: 2,
        page: 1,
      }),
    });
  });

  await page.goto('/users');

  await expect(page.getByRole('row')).toHaveCount(3); // header + 2 data rows
  await expect(page.getByText('Alice Johnson')).toBeVisible();
  await expect(page.getByText('Bob Smith')).toBeVisible();
});
```

### Fulfilling with Different Status Codes

```typescript
// Simulate a 404 Not Found
await page.route('/api/users/999', async (route) => {
  await route.fulfill({
    status: 404,
    contentType: 'application/json',
    body: JSON.stringify({ error: 'User not found', code: 'NOT_FOUND' }),
  });
});

// Simulate a 500 Internal Server Error
await page.route('/api/orders', async (route) => {
  await route.fulfill({
    status: 500,
    contentType: 'application/json',
    body: JSON.stringify({ error: 'Internal server error' }),
  });
});

test('shows error state on API failure', async ({ page }) => {
  await page.route('/api/users', (route) =>
    route.fulfill({ status: 500, body: 'Internal Server Error' }),
  );

  await page.goto('/users');

  await expect(page.getByRole('alert')).toBeVisible();
  await expect(page.getByRole('alert')).toContainText('Failed to load users');
  await expect(page.getByRole('button', { name: 'Retry' })).toBeVisible();
});

// Simulate network latency
await page.route('/api/slow-endpoint', async (route) => {
  await new Promise((resolve) => setTimeout(resolve, 2000)); // 2 second delay
  await route.fulfill({ status: 200, body: JSON.stringify({ data: 'delayed' }) });
});
```

## route.abort — Blocking Requests

```typescript
// Block all image requests to speed up tests
await page.route('**/*.{png,jpg,jpeg,gif,webp,svg}', (route) => route.abort());

// Block ads and tracking
await page.route('**/*google-analytics*', (route) => route.abort());
await page.route('**/*doubleclick.net*', (route) => route.abort());
await page.route('**/*hotjar.com*', (route) => route.abort());

// Block specific request types
await page.route('**', (route) => {
  const resourceType = route.request().resourceType();
  if (['font', 'media'].includes(resourceType)) {
    route.abort();
  } else {
    route.continue();
  }
});

test('page loads without images (accessibility test)', async ({ page }) => {
  await page.route('**/*.{png,jpg,jpeg}', (route) => route.abort());

  await page.goto('/products');

  // Page should still be fully usable without images
  await expect(page.getByRole('heading', { name: 'Products' })).toBeVisible();
  await expect(page.getByRole('button', { name: 'Add to cart' }).first()).toBeEnabled();
});
```

## route.continue — Modifying Requests

Pass the request through to the network, optionally modifying it:

```typescript
// Add authentication header to all API requests
await page.route('/api/**', async (route) => {
  await route.continue({
    headers: {
      ...route.request().headers(),
      'Authorization': `Bearer ${testToken}`,
      'X-Test-Environment': 'e2e',
    },
  });
});

// Change the request URL (redirect)
await page.route('/api/v1/**', async (route) => {
  const url = route.request().url().replace('/api/v1/', '/api/v2/');
  await route.continue({ url });
});

// Modify the request body
await page.route('/api/users', async (route) => {
  if (route.request().method() === 'POST') {
    const body = JSON.parse(route.request().postData()!);
    body.metadata = { source: 'e2e-test', timestamp: Date.now() };
    await route.continue({ postData: JSON.stringify(body) });
  } else {
    await route.continue();
  }
});
```

## waitForRequest / waitForResponse

Wait for specific network activity:

```typescript
test('submitting form sends correct API request', async ({ page }) => {
  await page.goto('/contact');

  // Set up the waiter before triggering the action
  const [request] = await Promise.all([
    page.waitForRequest('/api/contact'),
    page.getByRole('button', { name: 'Send message' }).click(),
  ]);

  const body = JSON.parse(request.postData()!);
  expect(body.name).toBe('Alice');
  expect(body.email).toBe('alice@example.com');
  expect(body.message).toContain('Hello');
});

test('shows updated data after save', async ({ page }) => {
  await page.goto('/profile');
  await page.getByLabel('Display Name').fill('Alice Johnson');

  const [response] = await Promise.all([
    page.waitForResponse('/api/users/me'),
    page.getByRole('button', { name: 'Save changes' }).click(),
  ]);

  expect(response.status()).toBe(200);
  const data = await response.json();
  expect(data.name).toBe('Alice Johnson');

  await expect(page.getByRole('alert')).toHaveText('Profile saved');
});

// Wait for response matching a predicate
const saveResponse = await page.waitForResponse(
  (response) =>
    response.url().includes('/api/users') &&
    response.request().method() === 'PATCH' &&
    response.status() === 200,
);
```

## API Testing with the request Fixture

Playwright includes a `request` fixture for testing APIs directly (without a browser):

```typescript
import { test, expect } from '@playwright/test';

test.describe('Users API', () => {
  test('GET /api/users returns a list of users', async ({ request }) => {
    const response = await request.get('/api/users', {
      headers: { Authorization: `Bearer ${process.env.TEST_API_TOKEN}` },
    });

    await expect(response).toBeOK();
    const body = await response.json();
    expect(body.users).toBeInstanceOf(Array);
    expect(body.users.length).toBeGreaterThan(0);
  });

  test('POST /api/users creates a new user', async ({ request }) => {
    const response = await request.post('/api/users', {
      data: {
        name: 'Test User',
        email: `test-${Date.now()}@example.com`,
        role: 'user',
      },
      headers: { Authorization: `Bearer ${process.env.TEST_API_TOKEN}` },
    });

    expect(response.status()).toBe(201);
    const user = await response.json();
    expect(user.id).toBeDefined();
    expect(user.name).toBe('Test User');
  });

  test('DELETE /api/users/:id removes the user', async ({ request }) => {
    // Create a user first
    const createResponse = await request.post('/api/users', {
      data: { name: 'To Delete', email: 'delete@example.com' },
    });
    const { id } = await createResponse.json();

    // Delete the user
    const deleteResponse = await request.delete(`/api/users/${id}`);
    expect(deleteResponse.status()).toBe(204);

    // Verify deletion
    const getResponse = await request.get(`/api/users/${id}`);
    expect(getResponse.status()).toBe(404);
  });
});
```

### Using APIRequestContext

```typescript
import { request, expect } from '@playwright/test';

// Create a reusable API client
const apiContext = await request.newContext({
  baseURL: 'https://api.example.com',
  extraHTTPHeaders: {
    Accept: 'application/json',
    Authorization: `Bearer ${process.env.API_TOKEN}`,
  },
});

const response = await apiContext.get('/users');
await expect(response).toBeOK();

await apiContext.dispose();
```

## HAR Recording and Playback

HAR (HTTP Archive) files record all network requests for later playback:

```typescript
// Record network activity to a HAR file
await page.routeFromHAR('./fixtures/api.har', {
  url: '/api/**',
  update: true, // update the HAR file with current responses
});

await page.goto('/dashboard');
// All API calls are recorded to api.har
```

```typescript
// Replay from HAR file (offline/deterministic testing)
await page.routeFromHAR('./fixtures/api.har', {
  url: '/api/**',
  update: false, // use recorded responses, don't make real requests
  // Fall through to real network if request not in HAR
  notFound: 'fallback',
});

await page.goto('/dashboard');
// All /api/** requests served from HAR — no real network needed
```

:::tip[HAR Files in CI]
Commit HAR files to your repository to enable fully offline E2E tests in CI. Record them once against a real environment, then replay them deterministically. Update the HAR file whenever the API changes.
:::

## Testing Offline Mode

```typescript
test('shows offline message when network is unavailable', async ({ page, context }) => {
  await page.goto('/');

  // Go offline
  await context.setOffline(true);

  // Try to perform an action that requires network
  await page.getByRole('button', { name: 'Refresh data' }).click();

  // Should show offline indicator
  await expect(page.getByRole('alert')).toHaveText(/offline|no connection/i);
  await expect(page.getByRole('status')).toHaveText(/offline/i);

  // Go back online
  await context.setOffline(false);

  // Should recover
  await expect(page.getByRole('alert')).toBeHidden();
});
```

## Monitoring All Network Requests

```typescript
test('does not make unauthorized API calls', async ({ page }) => {
  const requests: string[] = [];

  // Listen to all requests
  page.on('request', (request) => {
    if (request.url().includes('/api/')) {
      requests.push(`${request.method()} ${request.url()}`);
    }
  });

  await page.goto('/dashboard');
  await page.waitForLoadState('networkidle');

  // Verify all API calls are authenticated
  const unauthenticatedCalls = requests.filter(
    (req) => !req.includes('Authorization'),
  );
  expect(unauthenticatedCalls).toHaveLength(0);
});

test('measures API response times', async ({ page }) => {
  const apiTimings: { url: string; duration: number }[] = [];

  page.on('requestfinished', async (request) => {
    if (request.url().includes('/api/')) {
      const timing = request.timing();
      apiTimings.push({
        url: request.url(),
        duration: timing.responseEnd - timing.requestStart,
      });
    }
  });

  await page.goto('/dashboard');
  await page.waitForLoadState('networkidle');

  // All API calls should complete within 2 seconds
  for (const timing of apiTimings) {
    expect(timing.duration, `${timing.url} was too slow`).toBeLessThan(2000);
  }
});
```

## Complete Mock Setup for E2E Isolation

```typescript
// fixtures/mockApi.ts
import { type Page } from '@playwright/test';

export async function setupApiMocks(page: Page) {
  // Mock auth endpoints
  await page.route('/api/auth/**', async (route) => {
    const url = route.request().url();

    if (url.endsWith('/login')) {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({
          token: 'test-token-abc123',
          user: { id: 1, name: 'Test User', role: 'admin' },
        }),
      });
    } else if (url.endsWith('/logout')) {
      await route.fulfill({ status: 200, body: '{}' });
    } else {
      await route.continue();
    }
  });

  // Mock data endpoints
  await page.route('/api/users', async (route) => {
    if (route.request().method() === 'GET') {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify(require('./fixtures/users.json')),
      });
    } else {
      await route.continue();
    }
  });

  // Block analytics and tracking
  await page.route('**/*analytics*', (route) => route.abort());
  await page.route('**/*hotjar*', (route) => route.abort());

  // Block images to speed up tests
  await page.route(/\.(png|jpg|jpeg|gif|webp)$/, (route) => route.abort());
}
```

```typescript
// tests/dashboard.spec.ts
import { test, expect } from '@playwright/test';
import { setupApiMocks } from '../fixtures/mockApi';

test.beforeEach(async ({ page }) => {
  await setupApiMocks(page);
});

test('dashboard loads with mocked data', async ({ page }) => {
  await page.goto('/dashboard');

  await expect(page.getByRole('heading', { name: 'Dashboard' })).toBeVisible();
  await expect(page.getByRole('list', { name: 'Users' })).toBeVisible();
});
```

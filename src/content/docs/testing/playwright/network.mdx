Playwright provides powerful tools for controlling network traffic during tests — from intercepting and mocking API responses to recording HAR files and testing offline scenarios. This enables complete isolation of your frontend from backend services.

## page.route() — Request Interception

`page.route()` intercepts requests matching a URL pattern, giving you complete control over the response.

```typescript
// Intercept a specific endpoint
await page.route('**/api/users', route => {
  route.fulfill({
    status: 200,
    contentType: 'application/json',
    body: JSON.stringify([
      { id: 1, name: 'Alice', email: 'alice@example.com' },
      { id: 2, name: 'Bob', email: 'bob@example.com' },
    ]),
  });
});

await page.goto('/users');
await expect(page.getByRole('listitem')).toHaveCount(2);
await expect(page.getByText('Alice')).toBeVisible();
```

URL patterns support:

```typescript
// Exact URL
await page.route('https://api.example.com/users', handler);

// Glob pattern
await page.route('**/api/**', handler);

// Regex
await page.route(/\/api\/users\/\d+/, handler);

// Intercept everything
await page.route('**/*', handler);
```

## route.fulfill() — Mocking Responses

`route.fulfill()` sends a mock response instead of making the real network request:

```typescript
// Full response mock
await page.route('**/api/products', route => {
  route.fulfill({
    status: 200,
    contentType: 'application/json',
    headers: {
      'X-Total-Count': '42',
      'Cache-Control': 'no-cache',
    },
    body: JSON.stringify({
      items: [
        { id: 'prod-1', name: 'Widget', price: 9.99 },
        { id: 'prod-2', name: 'Gadget', price: 24.99 },
      ],
      total: 42,
    }),
  });
});

// Fulfill with a static file
await page.route('**/api/config', async route => {
  await route.fulfill({
    path: './tests/fixtures/config-response.json',
  });
});

// Simulate various error responses
await page.route('**/api/protected', route => {
  route.fulfill({ status: 401, body: 'Unauthorized' });
});

await page.route('**/api/missing', route => {
  route.fulfill({
    status: 404,
    contentType: 'application/json',
    body: JSON.stringify({ error: 'Not found', code: 'RESOURCE_NOT_FOUND' }),
  });
});

await page.route('**/api/broken', route => {
  route.fulfill({
    status: 500,
    contentType: 'application/json',
    body: JSON.stringify({ error: 'Internal server error' }),
  });
});
```

### Testing Error States

```typescript
test('shows an error banner when the API returns 500', async ({ page }) => {
  await page.route('**/api/dashboard/stats', route => {
    route.fulfill({
      status: 500,
      contentType: 'application/json',
      body: JSON.stringify({ message: 'Internal server error' }),
    });
  });

  await page.goto('/dashboard');

  await expect(page.getByRole('alert')).toBeVisible();
  await expect(page.getByRole('alert')).toContainText('Something went wrong');
});

test('shows empty state when no data is returned', async ({ page }) => {
  await page.route('**/api/users', route => {
    route.fulfill({
      status: 200,
      contentType: 'application/json',
      body: JSON.stringify({ users: [], total: 0 }),
    });
  });

  await page.goto('/users');

  await expect(page.getByText('No users found')).toBeVisible();
  await expect(page.getByRole('listitem')).toHaveCount(0);
});
```

## route.abort() — Blocking Requests

`route.abort()` cancels a request, simulating network failure:

```typescript
// Block all image requests
await page.route('**/*.{png,jpg,jpeg,gif,webp}', route => route.abort());

// Simulate a network failure for a specific endpoint
await page.route('**/api/analytics', route => route.abort('failed'));

test('handles network failure gracefully', async ({ page }) => {
  await page.route('**/api/data', route => route.abort('failed'));

  await page.goto('/data-page');

  await expect(page.getByText('Unable to load data')).toBeVisible();
  await expect(page.getByRole('button', { name: 'Retry' })).toBeEnabled();
});
```

Abort error codes: `'aborted'`, `'accessdenied'`, `'addressunreachable'`, `'blockedbyclient'`, `'connectionaborted'`, `'connectionclosed'`, `'connectionfailed'`, `'connectionrefused'`, `'connectionreset'`, `'failed'`, `'internetdisconnected'`, `'namenotresolved'`, `'timedout'`, `'failed'`.

## route.continue() — Modifying Requests

`route.continue()` passes the request through to the server, optionally with modifications:

```typescript
// Add an authentication header to all API requests
await page.route('**/api/**', async route => {
  await route.continue({
    headers: {
      ...route.request().headers(),
      'Authorization': `Bearer ${process.env.TEST_API_TOKEN}`,
      'X-Test-Mode': 'true',
    },
  });
});

// Modify the request body
await page.route('**/api/orders', async route => {
  const body = JSON.parse(route.request().postData() ?? '{}');
  body.testFlag = true; // inject a test flag
  await route.continue({
    postData: JSON.stringify(body),
  });
});

// Override the URL (redirect)
await page.route('**/cdn.external.com/**', async route => {
  const newUrl = route.request().url().replace('cdn.external.com', 'localhost:3001');
  await route.continue({ url: newUrl });
});
```

## waitForRequest and waitForResponse

Use `waitForRequest`/`waitForResponse` to wait for specific network events before asserting:

```typescript
// Wait for a specific request to be made
const requestPromise = page.waitForRequest('**/api/submit');
await page.getByRole('button', { name: 'Submit' }).click();
const request = await requestPromise;
const body = JSON.parse(request.postData() ?? '{}');
expect(body.name).toBe('Alice');
expect(body.email).toBe('alice@example.com');

// Wait for a response with status check
const responsePromise = page.waitForResponse(
  response =>
    response.url().includes('/api/users') &&
    response.status() === 200
);
await page.goto('/users');
const response = await responsePromise;
const data = await response.json();
expect(data.users).toHaveLength(10);

// With a timeout
const response2 = await page.waitForResponse('**/api/slow-endpoint', {
  timeout: 30000,
});

// Wait for request matching a predicate
const searchRequest = await page.waitForRequest(
  req => req.url().includes('/api/search') && req.method() === 'GET'
);
```

## API Testing with the request Fixture

Playwright's `request` fixture makes direct HTTP requests without a browser — perfect for testing your API in isolation or setting up state before a UI test:

```typescript
import { test, expect } from '@playwright/test';

test('GET /api/users returns a list of users', async ({ request }) => {
  const response = await request.get('/api/users');

  expect(response.status()).toBe(200);
  await expect(response).toBeOK();

  const body = await response.json();
  expect(body.users).toBeInstanceOf(Array);
  expect(body.users.length).toBeGreaterThan(0);
  expect(body.users[0]).toMatchObject({
    id: expect.any(Number),
    name: expect.any(String),
    email: expect.stringMatching(/@/),
  });
});

test('POST /api/users creates a new user', async ({ request }) => {
  const response = await request.post('/api/users', {
    data: { name: 'Carol', email: 'carol@example.com', role: 'user' },
    headers: { Authorization: `Bearer ${process.env.TEST_TOKEN}` },
  });

  expect(response.status()).toBe(201);
  const created = await response.json();
  expect(created.id).toBeDefined();
  expect(created.name).toBe('Carol');
});

test('PUT /api/users/:id updates a user', async ({ request }) => {
  const response = await request.put('/api/users/1', {
    data: { name: 'Alice Johnson Updated' },
  });

  await expect(response).toBeOK();
  const updated = await response.json();
  expect(updated.name).toBe('Alice Johnson Updated');
});

test('DELETE /api/users/:id deletes a user', async ({ request }) => {
  const response = await request.delete('/api/users/99');
  expect(response.status()).toBe(204);
});

test('handles 422 validation errors correctly', async ({ request }) => {
  const response = await request.post('/api/users', {
    data: { name: '', email: 'not-an-email' },
  });

  expect(response.status()).toBe(422);
  const body = await response.json();
  expect(body.errors).toContainEqual(
    expect.objectContaining({ field: 'email', message: expect.any(String) })
  );
});
```

## HAR Recording and Playback

HAR (HTTP Archive) files record all network interactions. You can record once against a real server and replay in tests:

```typescript
// Record a HAR file
const context = await browser.newContext();
await context.routeFromHAR('./tests/fixtures/api.har', { update: true });
const page = await context.newPage();
await page.goto('/dashboard');
// Interactions are recorded to api.har
await context.close();

// Replay from HAR — no real network requests made
await page.routeFromHAR('./tests/fixtures/api.har', {
  notFound: 'fallback', // 'abort' | 'fallback' (default is 'abort')
  url: '**/api/**',     // only mock these URLs from the HAR
});
await page.goto('/dashboard');

// Or per-context in playwright.config.ts
use: {
  serviceWorkers: 'block',
},
```

## Offline Mode

Test how your application behaves when the network is unavailable:

```typescript
test('shows offline message when network is unavailable', async ({ page, context }) => {
  await page.goto('/dashboard');

  // Take the browser context offline
  await context.setOffline(true);

  // Try to navigate or trigger a network-dependent action
  await page.getByRole('button', { name: 'Refresh data' }).click();

  // Assert offline state
  await expect(page.getByRole('alert')).toContainText(/no internet connection/i);
  await expect(page.getByTestId('offline-indicator')).toBeVisible();

  // Restore connectivity
  await context.setOffline(false);

  // Assert recovery
  await page.getByRole('button', { name: 'Retry' }).click();
  await expect(page.getByRole('alert')).not.toBeVisible();
  await expect(page.getByTestId('offline-indicator')).not.toBeVisible();
});
```

## Monitoring All Network Requests

Track every request during a test for debugging or verification:

```typescript
test('does not make unexpected API calls on the home page', async ({ page }) => {
  const requests: string[] = [];

  // Listen to all requests
  page.on('request', request => {
    if (request.url().includes('/api/')) {
      requests.push(`${request.method()} ${request.url()}`);
    }
  });

  await page.goto('/');
  await page.waitForLoadState('networkidle');

  // Assert only expected API calls were made
  expect(requests).toEqual([
    expect.stringContaining('GET /api/config'),
    expect.stringContaining('GET /api/user/me'),
  ]);
  expect(requests).toHaveLength(2);
});

test('logs all failed requests', async ({ page }) => {
  const failedRequests: string[] = [];

  page.on('requestfailed', request => {
    failedRequests.push(`${request.method()} ${request.url()} — ${request.failure()?.errorText}`);
  });

  await page.goto('/dashboard');
  await page.waitForLoadState('networkidle');

  if (failedRequests.length > 0) {
    console.error('Failed requests:', failedRequests);
  }
  expect(failedRequests).toHaveLength(0);
});
```

:::tip[Combine Route Mocking with the request Fixture]
Use `page.route()` to isolate UI tests from backend services, and the `request` fixture for pure API contract testing. Together, they give you complete control over your application's interactions with backend services.
:::

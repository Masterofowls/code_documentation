Playwright's network interception layer lets you control every HTTP request made by the browser — mock API responses for isolation, abort unwanted requests for speed, or observe all traffic during debugging.

## page.route() — Request Interception

`page.route()` registers a handler for requests matching a URL pattern. Handlers can fulfill (mock response), abort, or continue (pass through with optional modifications) the request.

```typescript
// Intercept all requests to a specific endpoint
await page.route('**/api/users', async (route) => {
  await route.fulfill({
    status: 200,
    contentType: 'application/json',
    body: JSON.stringify([
      { id: 1, name: 'Alice', role: 'admin' },
      { id: 2, name: 'Bob', role: 'user' },
    ]),
  });
});

await page.goto('/users');
await expect(page.getByRole('listitem')).toHaveCount(2);
```

### URL Matching

```typescript
// Exact URL
await page.route('https://api.example.com/users', handler);

// Glob pattern
await page.route('**/api/**', handler);
await page.route('https://api.example.com/**', handler);

// Regex
await page.route(/\/api\/users\/\d+/, handler);

// Exclude specific routes from a broad pattern
await page.route('**/*.{png,jpg,svg,gif,webp}', (route) => route.abort()); // block images
```

## route.fulfill — Mocking Responses

`route.fulfill()` responds to the request with a mock response — the real server is never contacted:

```typescript
await page.route('**/api/products', async (route) => {
  await route.fulfill({
    status: 200,
    contentType: 'application/json',
    headers: {
      'x-total-count': '42',
      'cache-control': 'no-store',
    },
    body: JSON.stringify({
      products: [
        { id: 1, name: 'Widget', price: 9.99, inStock: true },
        { id: 2, name: 'Gadget', price: 24.99, inStock: false },
      ],
      total: 42,
      page: 1,
    }),
  });
});
```

### Fulfilling with Fixture Files

Store mock responses as JSON files for maintainability:

```typescript
// tests/fixtures/users-response.json
// { "users": [...], "total": 3 }

await page.route('**/api/users', async (route) => {
  await route.fulfill({ path: 'tests/fixtures/users-response.json' });
});
```

### Simulating Error Responses

```typescript
// 404 Not Found
await page.route('**/api/users/999', async (route) => {
  await route.fulfill({
    status: 404,
    contentType: 'application/json',
    body: JSON.stringify({ error: 'User not found', code: 'USER_NOT_FOUND' }),
  });
});

// 500 Internal Server Error
await page.route('**/api/process', async (route) => {
  await route.fulfill({
    status: 500,
    contentType: 'application/json',
    body: JSON.stringify({ error: 'Internal server error' }),
  });
});

// 401 Unauthorized
await page.route('**/api/**', async (route) => {
  await route.fulfill({
    status: 401,
    contentType: 'application/json',
    body: JSON.stringify({ error: 'Unauthorized' }),
  });
});
```

### Simulating Slow Responses

```typescript
await page.route('**/api/reports/export', async (route) => {
  await new Promise((resolve) => setTimeout(resolve, 3000)); // 3 second delay
  await route.fulfill({
    status: 200,
    contentType: 'application/json',
    body: JSON.stringify({ reportUrl: '/reports/export-123.pdf' }),
  });
});

// Test that loading state is shown during slow request
await page.getByRole('button', { name: 'Export report' }).click();
await expect(page.getByRole('progressbar')).toBeVisible();
await expect(page.getByRole('link', { name: 'Download report' })).toBeVisible();
```

## route.abort — Blocking Requests

Abort requests entirely — useful for blocking analytics, ads, or slow non-critical resources:

```typescript
// Block all image requests (speeds up tests)
await page.route('**/*.{png,jpg,jpeg,gif,webp,svg,ico}', (route) =>
  route.abort(),
);

// Block analytics and tracking
await page.route('**/(analytics|tracking|gtag|segment)/**', (route) =>
  route.abort(),
);
await page.route('**/www.google-analytics.com/**', (route) => route.abort());

// Block a specific endpoint to test error handling
await page.route('**/api/recommendations', (route) => route.abort('failed'));

// Test the UI handles network failure gracefully
await page.goto('/products');
await expect(page.getByRole('alert')).toContainText(
  /recommendations unavailable/i,
);
```

Abort reasons: `'aborted'`, `'accessdenied'`, `'addressunreachable'`, `'blockedbyclient'`, `'failed'`, `'timedout'`, `'internetdisconnected'`

## route.continue — Modifying Requests

Pass the request through to the real server, optionally modifying it:

```typescript
// Add auth headers to all API requests
await page.route('**/api/**', async (route) => {
  await route.continue({
    headers: {
      ...route.request().headers(),
      authorization: `Bearer ${testToken}`,
    },
  });
});

// Rewrite the request URL
await page.route('**/api/v1/**', async (route) => {
  const url = route.request().url().replace('/api/v1/', '/api/v2/');
  await route.continue({ url });
});

// Change the request method
await page.route('**/api/submit', async (route) => {
  await route.continue({
    method: 'POST',
    postData: JSON.stringify({
      ...JSON.parse(route.request().postData() ?? '{}'),
      source: 'test',
    }),
  });
});
```

## Waiting for Requests and Responses

### page.waitForRequest

Wait for a specific request to be made:

```typescript
// Click saves and wait for the PUT request
const [request] = await Promise.all([
  page.waitForRequest('**/api/profile'),
  page.getByRole('button', { name: 'Save profile' }).click(),
]);

expect(request.method()).toBe('PUT');
const body = JSON.parse(request.postData() ?? '{}');
expect(body.name).toBe('Alice');
```

### page.waitForResponse

Wait for a specific response to arrive:

```typescript
const [response] = await Promise.all([
  page.waitForResponse('**/api/users'),
  page.getByRole('button', { name: 'Load users' }).click(),
]);

expect(response.status()).toBe(200);
const data = await response.json();
expect(data.users).toHaveLength(3);

// With predicate
const [errorResponse] = await Promise.all([
  page.waitForResponse(
    (resp) => resp.url().includes('/api/validate') && resp.status() === 422,
  ),
  page.getByRole('button', { name: 'Submit' }).click(),
]);
expect(errorResponse.status()).toBe(422);
```

## API Testing with the request Fixture

Playwright includes a `request` fixture for making HTTP requests directly — no browser required. Use it for setup/teardown, backend assertions, or pure API testing:

```typescript
import { test, expect } from '@playwright/test';

test('GET /api/users returns paginated list', async ({ request }) => {
  const response = await request.get('/api/users', {
    params: { page: 1, limit: 10 },
  });

  await expect(response).toBeOK();
  expect(response.status()).toBe(200);

  const body = await response.json();
  expect(body.users).toHaveLength(10);
  expect(body.total).toBeGreaterThan(0);
  expect(body.page).toBe(1);
});

test('POST /api/users creates a new user', async ({ request }) => {
  const response = await request.post('/api/users', {
    data: {
      name: 'Carol',
      email: 'carol@example.com',
      role: 'user',
    },
    headers: {
      authorization: `Bearer ${process.env.TEST_API_TOKEN}`,
    },
  });

  expect(response.status()).toBe(201);
  const user = await response.json();
  expect(user).toMatchObject({ name: 'Carol', email: 'carol@example.com' });
  expect(user.id).toBeDefined();
});

test('DELETE /api/users/:id removes the user', async ({ request }) => {
  // Create first
  const createResp = await request.post('/api/users', {
    data: { name: 'TempUser', email: 'temp@example.com' },
  });
  const { id } = await createResp.json();

  // Delete
  const deleteResp = await request.delete(`/api/users/${id}`);
  expect(deleteResp.status()).toBe(204);

  // Verify gone
  const getResp = await request.get(`/api/users/${id}`);
  expect(getResp.status()).toBe(404);
});
```

### Authenticated API Context

```typescript
// Create an authenticated request context outside a test
const apiContext = await request.newContext({
  baseURL: 'https://api.example.com',
  extraHTTPHeaders: {
    authorization: `Bearer ${process.env.API_TOKEN}`,
    'content-type': 'application/json',
  },
});

const response = await apiContext.get('/users');
await apiContext.dispose();
```

## HAR Recording and Playback

HAR (HTTP Archive) files record all network traffic and allow you to replay it in tests — perfect for testing against real API responses without a live backend:

```typescript
// Record a HAR file during a test run
test.use({
  recordHar: {
    path: 'tests/har/checkout-flow.har',
    mode: 'minimal', // record only status, headers, body
    urlFilter: '**/api/**', // only record API calls
  },
});

// Replay from HAR (serve recorded responses)
await page.routeFromHAR('tests/har/checkout-flow.har', {
  notFound: 'fallback', // fall through to real network if not in HAR
  update: false, // don't re-record
  urlFilter: '**/api/**',
});
```

## Testing Offline Mode

```typescript
test('shows offline message when network is unavailable', async ({
  page,
  context,
}) => {
  await page.goto('/dashboard');
  await expect(page.getByRole('main')).toBeVisible();

  // Go offline
  await context.setOffline(true);

  // Try to perform an action that requires network
  await page.getByRole('button', { name: 'Refresh data' }).click();

  // Assert offline message appears
  await expect(page.getByRole('alert')).toContainText(/offline/i);
  await expect(page.getByRole('alert')).toBeVisible();

  // Come back online
  await context.setOffline(false);

  await page.getByRole('button', { name: 'Retry' }).click();
  await expect(page.getByRole('alert')).not.toBeVisible();
});
```

## Monitoring All Network Requests

Log or assert on all network traffic without intercepting:

```typescript
test('checkout makes the expected API calls', async ({ page }) => {
  const apiCalls: string[] = [];

  // Listen to all requests
  page.on('request', (request) => {
    if (request.url().includes('/api/')) {
      apiCalls.push(`${request.method()} ${new URL(request.url()).pathname}`);
    }
  });

  await page.goto('/checkout');
  await page.getByRole('button', { name: 'Place order' }).click();
  await expect(page).toHaveURL('/order-confirmation');

  // Verify the expected API calls were made
  expect(apiCalls).toContain('POST /api/orders');
  expect(apiCalls).toContain('POST /api/payments');
  expect(apiCalls).toContain('POST /api/notifications/email');
});

// Monitor responses for errors
page.on('response', async (response) => {
  if (response.status() >= 500) {
    console.error(`Server error: ${response.status()} ${response.url()}`);
  }
});
```

:::tip[Combining Mock and Real Requests]
A common pattern is to mock slow or flaky third-party APIs while letting your own backend requests go through:

```typescript
// Mock slow payment provider
await page.route('**/stripe.com/**', async (route) => {
  await route.fulfill({
    status: 200,
    body: JSON.stringify({ status: 'succeeded', id: 'pi_test_123' }),
  });
});

// All other requests go through to the real backend
```

:::

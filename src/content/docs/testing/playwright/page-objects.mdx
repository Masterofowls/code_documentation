The Page Object Model (POM) is a design pattern that encapsulates page interactions in reusable classes. It separates test logic from page-specific selectors and actions, making tests easier to read and maintain.

## Why Page Objects?

Without page objects, tests become brittle and duplicated:

```typescript
// ✗ Without page objects — duplicated, brittle
test('admin can delete a user', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill('admin@example.com');
  await page.getByLabel('Password').fill('AdminPass123!');
  await page.getByRole('button', { name: 'Sign In' }).click();
  await expect(page).toHaveURL('/dashboard');
  // ... now navigate to user management
  await page.getByRole('link', { name: 'Users' }).click();
  // ... find the user
  await page.getByRole('row', { name: /alice/i })
    .getByRole('button', { name: 'Delete' }).click();
  // ...
});
```

With page objects, tests read like business requirements:

```typescript
// ✓ With page objects — readable, maintainable
test('admin can delete a user', async ({ loginPage, usersPage }) => {
  await loginPage.loginAs('admin@example.com', 'AdminPass123!');
  await usersPage.goto();
  await usersPage.deleteUser('Alice');
  await expect(usersPage.userRow('Alice')).not.toBeVisible();
});
```

## Base Page Class

Create a base class with common functionality shared by all page objects:

```typescript
// tests/pages/BasePage.ts
import { type Page, type Locator, expect } from '@playwright/test';

export abstract class BasePage {
  readonly page: Page;

  constructor(page: Page) {
    this.page = page;
  }

  async goto(path: string = '/'): Promise<void> {
    await this.page.goto(path);
  }

  async waitForPageLoad(): Promise<void> {
    await this.page.waitForLoadState('networkidle');
  }

  async getPageTitle(): Promise<string> {
    return this.page.title();
  }

  async dismissCookieBanner(): Promise<void> {
    const banner = this.page.getByRole('dialog', { name: /cookie/i });
    if (await banner.isVisible()) {
      await banner.getByRole('button', { name: 'Accept' }).click();
    }
  }

  async scrollToTop(): Promise<void> {
    await this.page.keyboard.press('Control+Home');
  }

  async takeScreenshot(name: string): Promise<void> {
    await this.page.screenshot({ path: `screenshots/${name}.png` });
  }
}
```

## Concrete Page Classes

### LoginPage

```typescript
// tests/pages/LoginPage.ts
import { type Page, expect } from '@playwright/test';
import { BasePage } from './BasePage';

export class LoginPage extends BasePage {
  // Locators defined as properties — lazy, reusable
  readonly emailInput = this.page.getByLabel('Email');
  readonly passwordInput = this.page.getByLabel('Password');
  readonly submitButton = this.page.getByRole('button', { name: 'Sign In' });
  readonly errorMessage = this.page.getByRole('alert');
  readonly forgotPasswordLink = this.page.getByRole('link', { name: 'Forgot password?' });

  constructor(page: Page) {
    super(page);
  }

  async goto(): Promise<void> {
    await this.page.goto('/login');
    await expect(this.submitButton).toBeVisible();
  }

  async loginAs(email: string, password: string): Promise<void> {
    await this.goto();
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
    await expect(this.page).toHaveURL('/dashboard');
  }

  async loginAsAdmin(): Promise<void> {
    await this.loginAs(
      process.env.ADMIN_EMAIL ?? 'admin@example.com',
      process.env.ADMIN_PASSWORD ?? 'AdminPass123!'
    );
  }

  async submitInvalidCredentials(): Promise<void> {
    await this.emailInput.fill('wrong@example.com');
    await this.passwordInput.fill('wrongpassword');
    await this.submitButton.click();
  }

  async getErrorMessage(): Promise<string | null> {
    return this.errorMessage.textContent();
  }
}
```

### DashboardPage

```typescript
// tests/pages/DashboardPage.ts
import { type Page, type Locator, expect } from '@playwright/test';
import { BasePage } from './BasePage';

export class DashboardPage extends BasePage {
  readonly welcomeHeading = this.page.getByRole('heading', { level: 1 });
  readonly navigation = this.page.getByRole('navigation', { name: 'Main' });
  readonly userMenu = this.page.getByRole('button', { name: /account/i });
  readonly notificationBadge = this.page.getByTestId('notification-count');

  async goto(): Promise<void> {
    await this.page.goto('/dashboard');
    await expect(this.welcomeHeading).toBeVisible();
  }

  async getWelcomeMessage(): Promise<string | null> {
    return this.welcomeHeading.textContent();
  }

  async navigateTo(section: string): Promise<void> {
    await this.navigation.getByRole('link', { name: section }).click();
    await this.waitForPageLoad();
  }

  async logout(): Promise<void> {
    await this.userMenu.click();
    await this.page.getByRole('menuitem', { name: 'Sign out' }).click();
    await expect(this.page).toHaveURL('/login');
  }

  async getUnreadNotificationCount(): Promise<number> {
    const text = await this.notificationBadge.textContent();
    return parseInt(text ?? '0', 10);
  }
}
```

## Component Objects

For reusable UI components (modals, data tables, navigation), create component objects:

```typescript
// tests/components/DataTable.ts
import { type Locator, expect } from '@playwright/test';

export class DataTable {
  private readonly table: Locator;

  constructor(table: Locator) {
    this.table = table;
  }

  row(nameOrIndex: string | number): Locator {
    if (typeof nameOrIndex === 'number') {
      return this.table.getByRole('row').nth(nameOrIndex + 1); // skip header row
    }
    return this.table.getByRole('row', { name: nameOrIndex });
  }

  async getColumnValues(header: string): Promise<string[]> {
    const headerCells = await this.table.getByRole('columnheader').allTextContents();
    const colIndex = headerCells.indexOf(header);
    const cells = await this.table.getByRole('row').all();
    const values: string[] = [];
    for (const row of cells.slice(1)) { // skip header row
      const text = await row.getByRole('cell').nth(colIndex).textContent();
      values.push(text ?? '');
    }
    return values;
  }

  async rowCount(): Promise<number> {
    const rows = await this.table.getByRole('row').all();
    return rows.length - 1; // subtract header row
  }

  async sortBy(column: string): Promise<void> {
    await this.table.getByRole('columnheader', { name: column }).click();
  }
}
```

Usage in a page object:

```typescript
// tests/pages/UsersPage.ts
import { type Page, expect } from '@playwright/test';
import { BasePage } from './BasePage';
import { DataTable } from '../components/DataTable';

export class UsersPage extends BasePage {
  readonly usersTable: DataTable;

  constructor(page: Page) {
    super(page);
    this.usersTable = new DataTable(this.page.getByRole('table', { name: 'Users' }));
  }

  async goto(): Promise<void> {
    await this.page.goto('/admin/users');
    await expect(this.page.getByRole('heading', { name: 'Users' })).toBeVisible();
  }

  async deleteUser(name: string): Promise<void> {
    const row = this.usersTable.row(name);
    await row.getByRole('button', { name: 'Delete' }).click();

    const dialog = this.page.getByRole('dialog', { name: 'Confirm deletion' });
    await dialog.getByRole('button', { name: 'Delete' }).click();
    await expect(dialog).not.toBeVisible();
  }
}
```

## Fixtures — Dependency Injection

Playwright's fixture system is the right way to inject page objects and shared state into tests:

```typescript
// tests/fixtures.ts
import { test as base } from '@playwright/test';
import { LoginPage } from './pages/LoginPage';
import { DashboardPage } from './pages/DashboardPage';
import { UsersPage } from './pages/UsersPage';

// Declare fixture types
type Fixtures = {
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
  usersPage: UsersPage;
  authenticatedPage: DashboardPage; // auto-authenticated dashboard
};

export const test = base.extend<Fixtures>({
  loginPage: async ({ page }, use) => {
    await use(new LoginPage(page));
  },

  dashboardPage: async ({ page }, use) => {
    await use(new DashboardPage(page));
  },

  usersPage: async ({ page }, use) => {
    await use(new UsersPage(page));
  },

  // Auto-login fixture — uses saved auth state
  authenticatedPage: async ({ page }, use) => {
    const dashboard = new DashboardPage(page);
    await dashboard.goto();
    await use(dashboard);
  },
});

export { expect } from '@playwright/test';
```

Use in tests:

```typescript
// tests/users.spec.ts
import { test, expect } from '../fixtures';

test('admin sees all users in the table', async ({ usersPage }) => {
  await usersPage.goto();
  const count = await usersPage.usersTable.rowCount();
  expect(count).toBeGreaterThan(0);
});

test('admin can delete a user', async ({ usersPage }) => {
  await usersPage.goto();
  const initialCount = await usersPage.usersTable.rowCount();
  await usersPage.deleteUser('Test User');
  await expect(usersPage.usersTable.row('Test User')).not.toBeAttached();
  expect(await usersPage.usersTable.rowCount()).toBe(initialCount - 1);
});
```

## Storing Auth State

Avoid logging in before every test by saving and reusing authentication state:

<Steps>
1. Create a global setup file:

   ```typescript
   // tests/global-setup.ts
   import { chromium, type FullConfig } from '@playwright/test';

   async function globalSetup(config: FullConfig) {
     const { baseURL } = config.projects[0]!.use;
     const browser = await chromium.launch();
     const context = await browser.newContext();
     const page = await context.newPage();

     // Perform login
     await page.goto(`${baseURL}/login`);
     await page.getByLabel('Email').fill(process.env.TEST_USER_EMAIL!);
     await page.getByLabel('Password').fill(process.env.TEST_USER_PASSWORD!);
     await page.getByRole('button', { name: 'Sign In' }).click();
     await page.waitForURL('/dashboard');

     // Save auth state (cookies + localStorage)
     await context.storageState({ path: 'playwright/.auth/user.json' });
     await browser.close();
   }

   export default globalSetup;
   ```

2. Configure Playwright to use the saved auth state:

   ```typescript
   // playwright.config.ts
   export default defineConfig({
     globalSetup: './tests/global-setup.ts',

     use: {
       storageState: 'playwright/.auth/user.json',
     },
   });
   ```

3. Add the auth file to `.gitignore`:

   ```bash
   echo "playwright/.auth/" >> .gitignore
   ```
</Steps>

:::tip[Per-Role Auth States]
For apps with multiple user roles (admin, regular user, read-only), create separate auth files and separate projects in `playwright.config.ts` — one per role. This lets you run admin tests with admin auth and user tests with user auth simultaneously.
:::

## TypeScript Types for Fixtures

Define strict types for custom fixtures to get full IntelliSense support:

```typescript
// tests/fixtures.ts
import { test as base, type Page } from '@playwright/test';

// Worker-scoped fixtures — shared across all tests in the same worker
type WorkerFixtures = {
  adminUser: { email: string; password: string; token: string };
};

// Test-scoped fixtures — fresh for each test
type TestFixtures = {
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
};

export const test = base.extend<TestFixtures, WorkerFixtures>({
  // Worker-scoped: created once per worker, shared between tests
  adminUser: [async ({}, use) => {
    const token = await createTestUser({ role: 'admin' });
    await use({ email: 'admin@test.com', password: 'pass', token });
    await deleteTestUser(token);
  }, { scope: 'worker' }],

  // Test-scoped: fresh instance for each test
  loginPage: async ({ page }, use) => {
    await use(new LoginPage(page));
  },

  dashboardPage: async ({ page }, use) => {
    await use(new DashboardPage(page));
  },
});
```

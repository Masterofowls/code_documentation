The Page Object Model (POM) is a design pattern that encapsulates page-specific locators and interactions into dedicated classes. This keeps tests readable, eliminates duplication, and makes tests resilient to UI changes — update the page object once instead of every test.

## Why Page Objects?

Without page objects, tests contain repeated locator definitions scattered across many files:

```typescript
// ❌ Without page objects — duplicated locator logic
test('login then view profile', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill('alice@example.com');
  await page.getByLabel('Password').fill('secret');
  await page.getByRole('button', { name: 'Sign in' }).click();
  await page.waitForURL('/dashboard');
  await page.getByRole('link', { name: 'Profile' }).click();
  // ...
});

test('login then place an order', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill('alice@example.com'); // duplicated!
  await page.getByLabel('Password').fill('secret'); // duplicated!
  await page.getByRole('button', { name: 'Sign in' }).click(); // duplicated!
  // ...
});
```

With page objects, locators and interactions are defined once:

```typescript
// ✅ With page objects — centralized, reusable
test('login then view profile', async ({ page }) => {
  const loginPage = new LoginPage(page);
  await loginPage.goto();
  await loginPage.login('alice@example.com', 'secret');
  // ...
});
```

## Base Page Class

Define a base class with shared utilities:

```typescript
// tests/page-objects/BasePage.ts
import type { Page, Locator } from '@playwright/test';

export abstract class BasePage {
  constructor(protected readonly page: Page) {}

  async goto(path: string = '/'): Promise<void> {
    await this.page.goto(path);
  }

  async waitForPageLoad(): Promise<void> {
    await this.page.waitForLoadState('networkidle');
  }

  async getTitle(): Promise<string> {
    return this.page.title();
  }

  async takeScreenshot(name: string): Promise<void> {
    await this.page.screenshot({ path: `screenshots/${name}.png` });
  }

  // Helper to get current URL
  get url(): string {
    return this.page.url();
  }
}
```

## LoginPage

```typescript
// tests/page-objects/LoginPage.ts
import { type Page, expect } from '@playwright/test';
import { BasePage } from './BasePage';

export class LoginPage extends BasePage {
  // Declare locators as readonly class properties
  private readonly emailInput = this.page.getByLabel('Email');
  private readonly passwordInput = this.page.getByLabel('Password');
  private readonly submitButton = this.page.getByRole('button', {
    name: 'Sign in',
  });
  private readonly errorMessage = this.page.getByRole('alert');
  private readonly rememberMeCheckbox = this.page.getByLabel('Remember me');

  constructor(page: Page) {
    super(page);
  }

  async goto(): Promise<void> {
    await this.page.goto('/login');
    await expect(this.submitButton).toBeVisible();
  }

  async login(email: string, password: string): Promise<void> {
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
    await this.page.waitForURL('/dashboard');
  }

  async loginExpectingError(email: string, password: string): Promise<void> {
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
  }

  async getErrorMessage(): Promise<string | null> {
    return this.errorMessage.textContent();
  }

  async checkRememberMe(): Promise<void> {
    await this.rememberMeCheckbox.check();
  }

  async isErrorVisible(): Promise<boolean> {
    return this.errorMessage.isVisible();
  }
}
```

## DashboardPage

```typescript
// tests/page-objects/DashboardPage.ts
import { type Page, expect } from '@playwright/test';
import { BasePage } from './BasePage';

export class DashboardPage extends BasePage {
  private readonly welcomeHeading = this.page.getByRole('heading', {
    level: 1,
  });
  private readonly userMenu = this.page.getByRole('button', {
    name: /account menu/i,
  });
  private readonly navLinks = this.page
    .getByRole('navigation')
    .getByRole('link');
  private readonly notificationBadge =
    this.page.getByTestId('notification-count');

  async expectLoaded(): Promise<void> {
    await expect(this.page).toHaveURL('/dashboard');
    await expect(this.welcomeHeading).toBeVisible();
  }

  async getWelcomeMessage(): Promise<string | null> {
    return this.welcomeHeading.textContent();
  }

  async openUserMenu(): Promise<void> {
    await this.userMenu.click();
  }

  async navigateTo(linkName: string): Promise<void> {
    await this.navLinks.filter({ hasText: linkName }).click();
  }

  async getNotificationCount(): Promise<number> {
    const text = await this.notificationBadge.textContent();
    return parseInt(text ?? '0', 10);
  }

  async logout(): Promise<void> {
    await this.openUserMenu();
    await this.page.getByRole('menuitem', { name: 'Sign out' }).click();
    await this.page.waitForURL('/login');
  }
}
```

## Component Objects — Reusable UI Parts

For UI components that appear in multiple pages (data tables, navigation bars, modals), create component objects:

```typescript
// tests/page-objects/components/DataTable.ts
import type { Page, Locator } from '@playwright/test';
import { expect } from '@playwright/test';

export class DataTable {
  private readonly table: Locator;
  private readonly rows: Locator;
  private readonly searchInput: Locator;

  constructor(page: Page, testId: string) {
    this.table = page.getByTestId(testId);
    this.rows = this.table
      .getByRole('row')
      .filter({ hasNot: page.getByRole('columnheader') });
    this.searchInput = this.table.getByRole('textbox', { name: /search/i });
  }

  async search(query: string): Promise<void> {
    await this.searchInput.fill(query);
    await expect(this.searchInput).toHaveValue(query);
  }

  async getRowCount(): Promise<number> {
    return this.rows.count();
  }

  async getRowByText(text: string): Promise<Locator> {
    return this.rows.filter({ hasText: text });
  }

  async clickRowAction(rowText: string, actionName: string): Promise<void> {
    const row = this.rows.filter({ hasText: rowText });
    await row.getByRole('button', { name: actionName }).click();
  }

  async sortBy(columnName: string): Promise<void> {
    await this.table.getByRole('columnheader', { name: columnName }).click();
  }
}

// Usage in a test:
const usersTable = new DataTable(page, 'users-table');
await usersTable.search('Alice');
expect(await usersTable.getRowCount()).toBe(1);
await usersTable.clickRowAction('Alice', 'Edit');
```

## Fixtures — Dependency Injection

Playwright's fixture system is the recommended way to share page objects, authenticated sessions, and test utilities across test files.

```typescript
// tests/fixtures.ts
import { test as base, expect } from '@playwright/test';
import { LoginPage } from './page-objects/LoginPage';
import { DashboardPage } from './page-objects/DashboardPage';

// Define the shape of your custom fixtures
type AppFixtures = {
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
  authenticatedPage: DashboardPage; // pre-logged-in fixture
};

export const test = base.extend<AppFixtures>({
  // Create a fresh LoginPage for each test
  loginPage: async ({ page }, use) => {
    const loginPage = new LoginPage(page);
    await loginPage.goto();
    await use(loginPage);
  },

  // Create a fresh DashboardPage for each test
  dashboardPage: async ({ page }, use) => {
    const dashboardPage = new DashboardPage(page);
    await use(dashboardPage);
  },

  // Pre-authenticated DashboardPage — runs login before yielding
  authenticatedPage: async ({ page }, use) => {
    const loginPage = new LoginPage(page);
    await loginPage.goto();
    await loginPage.login('alice@example.com', 'testpassword');
    const dashboardPage = new DashboardPage(page);
    await dashboardPage.expectLoaded();
    await use(dashboardPage);
  },
});

export { expect };
```

Use in tests:

```typescript
// tests/dashboard.spec.ts
import { test, expect } from '../fixtures';

test('shows welcome message after login', async ({ authenticatedPage }) => {
  const message = await authenticatedPage.getWelcomeMessage();
  expect(message).toContain('Welcome');
});

test('login page shows error on wrong password', async ({ loginPage }) => {
  await loginPage.loginExpectingError('alice@example.com', 'wrongpassword');
  expect(await loginPage.isErrorVisible()).toBe(true);
});
```

## Fixture Scope — Sharing Between Tests

Fixtures can be scoped to `test` (default — new instance per test) or `worker` (shared across all tests in the same worker):

```typescript
export const test = base.extend<AppFixtures, WorkerFixtures>(
  {
    // Worker-scoped: created once per worker, shared across tests in that worker
    // Declare in second type parameter
  },
  {
    workerApiClient: [
      async ({}, use) => {
        const client = new ApiClient(process.env.API_URL!);
        await client.authenticate();
        await use(client);
        await client.cleanup();
      },
      { scope: 'worker' },
    ],
  },
);
```

## Storing Auth State — storageState

The most efficient way to handle authentication in tests is to log in once and reuse the browser storage state (cookies + localStorage) across tests:

```typescript
// global-setup.ts — runs once before all tests
import { chromium, type FullConfig } from '@playwright/test';

async function globalSetup(config: FullConfig): Promise<void> {
  const { baseURL } = config.projects[0]!.use;
  const browser = await chromium.launch();
  const page = await browser.newPage();

  // Log in to the application
  await page.goto(`${baseURL}/login`);
  await page.getByLabel('Email').fill(process.env.TEST_USER_EMAIL!);
  await page.getByLabel('Password').fill(process.env.TEST_USER_PASSWORD!);
  await page.getByRole('button', { name: 'Sign in' }).click();
  await page.waitForURL('**/dashboard');

  // Save authenticated state to disk
  await page.context().storageState({ path: 'playwright/.auth/user.json' });

  await browser.close();
}

export default globalSetup;
```

Configure in `playwright.config.ts`:

```typescript
export default defineConfig({
  globalSetup: require.resolve('./global-setup'),

  projects: [
    // Run all auth-dependent tests with the saved storage state
    {
      name: 'authenticated',
      use: {
        storageState: 'playwright/.auth/user.json',
      },
      testMatch: 'tests/authenticated/**/*.spec.ts',
    },
    // Run auth tests without storage state
    {
      name: 'auth',
      testMatch: 'tests/auth/**/*.spec.ts',
    },
  ],
});
```

Add the auth directory to `.gitignore`:

```
playwright/.auth/
```

## TypeScript Types for Fixtures

```typescript
// tests/types.ts
import type { Page, Locator, BrowserContext } from '@playwright/test';

export interface UserCredentials {
  email: string;
  password: string;
  role: 'admin' | 'user' | 'viewer';
}

export interface TestFixtures {
  loginPage: import('./page-objects/LoginPage').LoginPage;
  dashboardPage: import('./page-objects/DashboardPage').DashboardPage;
  adminUser: UserCredentials;
  regularUser: UserCredentials;
}

// fixtures.ts
import { test as base } from '@playwright/test';
import type { TestFixtures } from './types';

export const test = base.extend<TestFixtures>({
  adminUser: async ({}, use) => {
    await use({
      email: process.env.ADMIN_EMAIL ?? 'admin@example.com',
      password: process.env.ADMIN_PASSWORD ?? 'adminpass',
      role: 'admin',
    });
  },

  regularUser: async ({}, use) => {
    await use({
      email: process.env.USER_EMAIL ?? 'user@example.com',
      password: process.env.USER_PASSWORD ?? 'userpass',
      role: 'user',
    });
  },
});
```

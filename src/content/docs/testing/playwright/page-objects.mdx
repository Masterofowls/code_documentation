The Page Object Model (POM) is a design pattern that encapsulates page interaction logic into reusable classes. Instead of duplicating locators and actions across test files, each "page" or "component" of your application gets its own class with typed methods — making tests readable, maintainable, and DRY.

## Why Page Objects?

Without POM, test files become brittle maintenance burdens:

```typescript
// ❌ Without POM — locators duplicated everywhere
test('user can log in', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill('alice@example.com');
  await page.getByLabel('Password').fill('secret');
  await page.getByRole('button', { name: 'Sign In' }).click();
  await expect(page).toHaveURL('/dashboard');
});

test('user sees error on bad credentials', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill('alice@example.com');
  await page.getByLabel('Password').fill('wrong'); // duplicated locators!
  await page.getByRole('button', { name: 'Sign In' }).click();
  await expect(page.getByRole('alert')).toContainText('Invalid credentials');
});
```

With POM:

```typescript
// ✅ With POM — locators defined once, tests are readable
test('user can log in', async ({ page }) => {
  const loginPage = new LoginPage(page);
  await loginPage.goto();
  await loginPage.login('alice@example.com', 'secret');
  await expect(page).toHaveURL('/dashboard');
});

test('user sees error on bad credentials', async ({ page }) => {
  const loginPage = new LoginPage(page);
  await loginPage.goto();
  await loginPage.login('alice@example.com', 'wrong');
  await expect(loginPage.errorAlert).toContainText('Invalid credentials');
});
```

## Base Page Class

A base class provides common functionality shared by all page objects:

```typescript
// tests/pages/BasePage.ts
import { type Page, type Locator } from '@playwright/test';

export abstract class BasePage {
  readonly page: Page;
  readonly url: string;

  constructor(page: Page, url: string) {
    this.page = page;
    this.url = url;
  }

  async goto(params?: Record<string, string>): Promise<void> {
    const searchParams = params ? '?' + new URLSearchParams(params).toString() : '';
    await this.page.goto(this.url + searchParams);
    await this.waitForPageLoad();
  }

  async waitForPageLoad(): Promise<void> {
    await this.page.waitForLoadState('domcontentloaded');
  }

  async getTitle(): Promise<string> {
    return this.page.title();
  }

  async takeScreenshot(name: string): Promise<Buffer> {
    return this.page.screenshot({ path: `screenshots/${name}.png` });
  }

  async scrollToBottom(): Promise<void> {
    await this.page.evaluate(() => window.scrollTo(0, document.body.scrollHeight));
  }
}
```

## Concrete Page Classes

### LoginPage

```typescript
// tests/pages/LoginPage.ts
import { type Page, type Locator, expect } from '@playwright/test';
import { BasePage } from './BasePage';
import { DashboardPage } from './DashboardPage';

export class LoginPage extends BasePage {
  // Locators as readonly properties — defined once, used everywhere
  readonly emailInput: Locator;
  readonly passwordInput: Locator;
  readonly signInButton: Locator;
  readonly errorAlert: Locator;
  readonly forgotPasswordLink: Locator;
  readonly rememberMeCheckbox: Locator;

  constructor(page: Page) {
    super(page, '/login');
    this.emailInput = page.getByLabel('Email');
    this.passwordInput = page.getByLabel('Password');
    this.signInButton = page.getByRole('button', { name: 'Sign In' });
    this.errorAlert = page.getByRole('alert');
    this.forgotPasswordLink = page.getByRole('link', { name: 'Forgot password?' });
    this.rememberMeCheckbox = page.getByRole('checkbox', { name: 'Remember me' });
  }

  async login(email: string, password: string): Promise<DashboardPage> {
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.signInButton.click();
    return new DashboardPage(this.page);
  }

  async loginAndWait(email: string, password: string): Promise<DashboardPage> {
    const dashboard = await this.login(email, password);
    await expect(this.page).toHaveURL(/\/dashboard/);
    return dashboard;
  }

  async expectErrorMessage(message: string | RegExp): Promise<void> {
    await expect(this.errorAlert).toBeVisible();
    await expect(this.errorAlert).toContainText(message);
  }

  async isLoggedIn(): Promise<boolean> {
    return this.page.url().includes('/dashboard');
  }
}
```

### DashboardPage

```typescript
// tests/pages/DashboardPage.ts
import { type Page, type Locator, expect } from '@playwright/test';
import { BasePage } from './BasePage';

export class DashboardPage extends BasePage {
  readonly welcomeHeading: Locator;
  readonly navigationMenu: Locator;
  readonly userAvatarButton: Locator;
  readonly notificationBell: Locator;
  readonly statsCards: Locator;

  constructor(page: Page) {
    super(page, '/dashboard');
    this.welcomeHeading = page.getByRole('heading', { level: 1 });
    this.navigationMenu = page.getByRole('navigation', { name: 'Main navigation' });
    this.userAvatarButton = page.getByRole('button', { name: /user menu/i });
    this.notificationBell = page.getByRole('button', { name: 'Notifications' });
    this.statsCards = page.locator('[data-component="StatsCard"]');
  }

  async navigateTo(section: string): Promise<void> {
    await this.navigationMenu.getByRole('link', { name: section }).click();
  }

  async getWelcomeText(): Promise<string> {
    return (await this.welcomeHeading.textContent()) ?? '';
  }

  async getStatsCount(): Promise<number> {
    return this.statsCards.count();
  }

  async openNotifications(): Promise<void> {
    await this.notificationBell.click();
    await expect(
      this.page.getByRole('dialog', { name: 'Notifications' })
    ).toBeVisible();
  }

  async logout(): Promise<void> {
    await this.userAvatarButton.click();
    await this.page.getByRole('menuitem', { name: 'Sign Out' }).click();
    await expect(this.page).toHaveURL('/login');
  }
}
```

## Component Objects

Reusable UI components (navigation bars, modals, data tables) get their own object classes:

```typescript
// tests/components/DataTable.ts
import { type Page, type Locator, expect } from '@playwright/test';

export class DataTable {
  readonly table: Locator;
  readonly rows: Locator;
  readonly headers: Locator;
  readonly searchInput: Locator;
  readonly paginationNext: Locator;
  readonly paginationPrev: Locator;

  constructor(page: Page, tableAriaLabel: string) {
    this.table = page.getByRole('table', { name: tableAriaLabel });
    this.rows = this.table.getByRole('row').filter({ hasNot: page.getByRole('columnheader') });
    this.headers = this.table.getByRole('columnheader');
    this.searchInput = page.getByRole('searchbox', { name: `Search ${tableAriaLabel}` });
    this.paginationNext = page.getByRole('button', { name: 'Next page' });
    this.paginationPrev = page.getByRole('button', { name: 'Previous page' });
  }

  async getRowCount(): Promise<number> {
    return this.rows.count();
  }

  async getCellValue(row: number, column: string): Promise<string> {
    const header = await this.headers.allTextContents();
    const colIndex = header.indexOf(column);
    if (colIndex === -1) throw new Error(`Column "${column}" not found`);

    const cells = this.rows.nth(row).getByRole('cell');
    return (await cells.nth(colIndex).textContent()) ?? '';
  }

  async search(query: string): Promise<void> {
    await this.searchInput.fill(query);
    await expect(this.table).toBeVisible();
  }

  async sortBy(columnName: string): Promise<void> {
    await this.table.getByRole('columnheader', { name: columnName }).click();
  }

  async goToNextPage(): Promise<void> {
    await this.paginationNext.click();
  }
}
```

## Fixtures — Dependency Injection

Playwright fixtures replace `beforeEach`/`afterEach` boilerplate with typed, injectable dependencies:

```typescript
// tests/fixtures.ts
import { test as base, type Page } from '@playwright/test';
import { LoginPage } from './pages/LoginPage';
import { DashboardPage } from './pages/DashboardPage';

// Define fixture types
type MyFixtures = {
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
  authenticatedPage: Page;
};

// Extend the base test with custom fixtures
export const test = base.extend<MyFixtures>({
  // Per-test fixture — creates a new LoginPage for each test
  loginPage: async ({ page }, use) => {
    const loginPage = new LoginPage(page);
    await use(loginPage);
    // teardown runs after the test
  },

  dashboardPage: async ({ page }, use) => {
    const dashboardPage = new DashboardPage(page);
    await use(dashboardPage);
  },

  // Pre-authenticated page — logs in before yielding
  authenticatedPage: async ({ page }, use) => {
    const loginPage = new LoginPage(page);
    await loginPage.goto();
    await loginPage.loginAndWait(
      process.env.TEST_USER_EMAIL ?? 'test@example.com',
      process.env.TEST_USER_PASSWORD ?? 'password'
    );
    await use(page);
    // page is automatically closed after the test
  },
});

// Re-export expect so tests only import from fixtures.ts
export { expect } from '@playwright/test';
```

Use fixtures in tests:

```typescript
// tests/login.spec.ts
import { test, expect } from './fixtures'; // import from our custom fixtures

test('user can log in successfully', async ({ loginPage }) => {
  await loginPage.goto();
  await loginPage.login('alice@example.com', 'secret');
  await expect(loginPage.page).toHaveURL('/dashboard');
});

test('authenticated user can view settings', async ({ authenticatedPage }) => {
  await authenticatedPage.goto('/settings');
  await expect(authenticatedPage.getByRole('heading', { name: 'Settings' })).toBeVisible();
});
```

## Fixture Scope — Worker vs Test

```typescript
export const test = base.extend<{}, { sharedApiClient: ApiClient }>({
  // Worker-scoped fixture — created once per worker, shared by all tests in the worker
  sharedApiClient: [async ({}, use) => {
    const client = await ApiClient.create({ baseURL: process.env.API_URL });
    await use(client);
    await client.cleanup();
  }, { scope: 'worker' }],
});
```

| Scope | Created | Destroyed | Use For |
|---|---|---|---|
| `test` (default) | Before each test | After each test | Isolated state (pages, form helpers) |
| `worker` | When worker starts | When worker finishes | Shared resources (DB connections, API clients) |

## Storing Auth State (storageState)

Instead of logging in before every test, Playwright can save and restore browser storage state (cookies, localStorage, sessionStorage):

```typescript
// tests/global-setup.ts
import { chromium, type FullConfig } from '@playwright/test';

async function globalSetup(config: FullConfig) {
  const { baseURL } = config.projects[0]!.use;
  const browser = await chromium.launch();
  const page = await browser.newPage();

  // Log in once and save the auth state
  await page.goto(`${baseURL}/login`);
  await page.getByLabel('Email').fill(process.env.TEST_USER_EMAIL!);
  await page.getByLabel('Password').fill(process.env.TEST_USER_PASSWORD!);
  await page.getByRole('button', { name: 'Sign In' }).click();
  await page.waitForURL('**/dashboard');

  // Save authentication state
  await page.context().storageState({ path: 'tests/.auth/user.json' });

  await browser.close();
}

export default globalSetup;
```

```typescript
// playwright.config.ts
export default defineConfig({
  globalSetup: './tests/global-setup.ts',

  projects: [
    // Setup project — logs in and saves state
    {
      name: 'setup',
      testMatch: /global\.setup\.ts/,
    },
    // Tests that depend on authentication
    {
      name: 'authenticated',
      use: {
        storageState: 'tests/.auth/user.json', // pre-loaded auth
      },
      dependencies: ['setup'],
    },
    // Tests that don't need authentication
    {
      name: 'unauthenticated',
      testMatch: '**/auth/**/*.spec.ts',
    },
  ],
});
```

## Multiple Auth Roles

```typescript
// global-setup.ts — create auth state for different roles
async function globalSetup(config: FullConfig) {
  const browser = await chromium.launch();

  // Admin user
  const adminPage = await browser.newPage();
  await loginAs(adminPage, 'admin@example.com', 'admin-pass');
  await adminPage.context().storageState({ path: 'tests/.auth/admin.json' });

  // Regular user
  const userPage = await browser.newPage();
  await loginAs(userPage, 'user@example.com', 'user-pass');
  await userPage.context().storageState({ path: 'tests/.auth/user.json' });

  await browser.close();
}
```

```typescript
// Use in a test file
test.use({ storageState: 'tests/.auth/admin.json' });

test('admin can delete users', async ({ page }) => {
  await page.goto('/admin/users');
  // already authenticated as admin
});
```

## TypeScript Types for Fixtures

```typescript
// tests/fixtures.ts
import { test as base } from '@playwright/test';
import type { LoginPage } from './pages/LoginPage';
import type { DashboardPage } from './pages/DashboardPage';
import type { DataTable } from './components/DataTable';

export type AppFixtures = {
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
  usersTable: DataTable;
};

export type WorkerFixtures = {
  dbConnection: DatabaseConnection;
};

export const test = base.extend<AppFixtures, WorkerFixtures>({
  loginPage: async ({ page }, use) => {
    const { LoginPage } = await import('./pages/LoginPage');
    await use(new LoginPage(page));
  },

  dashboardPage: async ({ page }, use) => {
    const { DashboardPage } = await import('./pages/DashboardPage');
    await use(new DashboardPage(page));
  },

  usersTable: async ({ page }, use) => {
    const { DataTable } = await import('./components/DataTable');
    await use(new DataTable(page, 'Users'));
  },

  dbConnection: [async ({}, use) => {
    const db = await DatabaseConnection.open(process.env.TEST_DB_URL!);
    await use(db);
    await db.close();
  }, { scope: 'worker' }],
});

export { expect } from '@playwright/test';
```

The Page Object Model (POM) is a design pattern that encapsulates page-specific interactions behind a clean API. Instead of repeating locators and actions across tests, you create reusable Page classes that centralize all interactions with a particular page or component.

## Why Use Page Objects?

Without Page Objects:

```typescript
// ❌ Repeated locators across multiple test files
test('admin can delete a user', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill('admin@example.com');
  await page.getByLabel('Password').fill('adminPass123');
  await page.getByRole('button', { name: 'Sign in' }).click();
  await expect(page).toHaveURL('/dashboard');
  // ... more test steps
});

test('admin can create a user', async ({ page }) => {
  // ❌ Login steps repeated identically
  await page.goto('/login');
  await page.getByLabel('Email').fill('admin@example.com');
  await page.getByLabel('Password').fill('adminPass123');
  await page.getByRole('button', { name: 'Sign in' }).click();
  await expect(page).toHaveURL('/dashboard');
  // ...
});
```

With Page Objects:

```typescript
// ✅ Login encapsulated in LoginPage
test('admin can delete a user', async ({ page }) => {
  const loginPage = new LoginPage(page);
  await loginPage.loginAs('admin@example.com', 'adminPass123');
  // ...
});
```

## Base Page Class

Create a base class with common utilities shared across all page objects:

```typescript
// pages/BasePage.ts
import { type Page, type Locator } from '@playwright/test';

export abstract class BasePage {
  protected readonly page: Page;

  constructor(page: Page) {
    this.page = page;
  }

  async navigate(path: string): Promise<void> {
    await this.page.goto(path);
  }

  async waitForPageLoad(): Promise<void> {
    await this.page.waitForLoadState('networkidle');
  }

  async getPageTitle(): Promise<string> {
    return this.page.title();
  }

  async getFlashMessage(): Promise<string | null> {
    const flash = this.page.getByRole('alert');
    if (await flash.isVisible()) {
      return flash.textContent();
    }
    return null;
  }

  async dismissModal(): Promise<void> {
    await this.page.keyboard.press('Escape');
  }

  async scrollToBottom(): Promise<void> {
    await this.page.evaluate(() => window.scrollTo(0, document.body.scrollHeight));
  }
}
```

## Concrete Page Classes

```typescript
// pages/LoginPage.ts
import { type Page, expect } from '@playwright/test';
import { BasePage } from './BasePage';

export class LoginPage extends BasePage {
  private readonly emailInput: Locator;
  private readonly passwordInput: Locator;
  private readonly submitButton: Locator;
  private readonly errorMessage: Locator;
  private readonly forgotPasswordLink: Locator;

  constructor(page: Page) {
    super(page);
    this.emailInput = page.getByLabel('Email');
    this.passwordInput = page.getByLabel('Password');
    this.submitButton = page.getByRole('button', { name: 'Sign in' });
    this.errorMessage = page.getByRole('alert');
    this.forgotPasswordLink = page.getByRole('link', { name: 'Forgot password?' });
  }

  async goto(): Promise<void> {
    await this.navigate('/login');
  }

  async login(email: string, password: string): Promise<void> {
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
  }

  async loginAs(email: string, password: string): Promise<void> {
    await this.goto();
    await this.login(email, password);
    await expect(this.page).not.toHaveURL('/login');
  }

  async getErrorMessage(): Promise<string | null> {
    if (await this.errorMessage.isVisible()) {
      return this.errorMessage.textContent();
    }
    return null;
  }

  async clickForgotPassword(): Promise<void> {
    await this.forgotPasswordLink.click();
  }
}
```

```typescript
// pages/DashboardPage.ts
import { type Page, expect } from '@playwright/test';
import { BasePage } from './BasePage';

export class DashboardPage extends BasePage {
  private readonly userMenuButton: Locator;
  private readonly logoutMenuItem: Locator;
  private readonly notificationBell: Locator;
  private readonly searchInput: Locator;

  constructor(page: Page) {
    super(page);
    this.userMenuButton = page.getByRole('button', { name: /account menu/i });
    this.logoutMenuItem = page.getByRole('menuitem', { name: 'Sign out' });
    this.notificationBell = page.getByRole('button', { name: 'Notifications' });
    this.searchInput = page.getByRole('searchbox');
  }

  async goto(): Promise<void> {
    await this.navigate('/dashboard');
  }

  async getWelcomeText(): Promise<string | null> {
    return this.page.getByRole('heading', { level: 1 }).textContent();
  }

  async logout(): Promise<void> {
    await this.userMenuButton.click();
    await this.logoutMenuItem.click();
    await expect(this.page).toHaveURL('/login');
  }

  async search(query: string): Promise<void> {
    await this.searchInput.fill(query);
    await this.searchInput.press('Enter');
  }

  async getNotificationCount(): Promise<number> {
    const badge = this.page.getByTestId('notification-count');
    if (await badge.isVisible()) {
      const text = await badge.textContent();
      return parseInt(text ?? '0', 10);
    }
    return 0;
  }

  async navigateTo(section: string): Promise<void> {
    await this.page.getByRole('navigation').getByRole('link', { name: section }).click();
  }
}
```

## Component Objects

For reusable UI components that appear across multiple pages:

```typescript
// pages/components/DataTable.ts
import { type Locator, type Page, expect } from '@playwright/test';

export class DataTable {
  private readonly root: Locator;

  constructor(page: Page, testId: string) {
    this.root = page.getByTestId(testId);
  }

  async getRowCount(): Promise<number> {
    return this.root
      .getByRole('row')
      .filter({ hasNot: this.root.getByRole('columnheader') })
      .count();
  }

  async getCell(row: number, column: number): Promise<string | null> {
    const rows = this.root.getByRole('row').filter({ hasNot: this.root.getByRole('columnheader') });
    const cells = rows.nth(row).getByRole('cell');
    return cells.nth(column).textContent();
  }

  async clickRowAction(rowText: string, action: string): Promise<void> {
    const row = this.root
      .getByRole('row')
      .filter({ hasText: rowText });
    await row.getByRole('button', { name: action }).click();
  }

  async sortBy(columnName: string): Promise<void> {
    await this.root.getByRole('columnheader', { name: columnName }).click();
  }

  async filterBy(column: string, value: string): Promise<void> {
    await this.root.getByRole('columnheader', { name: column })
      .getByRole('textbox')
      .fill(value);
  }

  async expectRowCount(count: number): Promise<void> {
    const rows = this.root
      .getByRole('row')
      .filter({ hasNot: this.root.getByRole('columnheader') });
    await expect(rows).toHaveCount(count);
  }
}
```

## Fixtures for Dependency Injection

Playwright fixtures extend the base `test` object to inject pre-built page objects and shared state:

```typescript
// fixtures/index.ts
import { test as base, expect } from '@playwright/test';
import { LoginPage } from '../pages/LoginPage';
import { DashboardPage } from '../pages/DashboardPage';
import { UsersPage } from '../pages/UsersPage';

// Define the fixture types
interface AppFixtures {
  loginPage: LoginPage;
  dashboardPage: DashboardPage;
  usersPage: UsersPage;
  authenticatedPage: Page;
}

export const test = base.extend<AppFixtures>({
  // Page object fixtures (test scope — new instance per test)
  loginPage: async ({ page }, use) => {
    await use(new LoginPage(page));
  },

  dashboardPage: async ({ page }, use) => {
    await use(new DashboardPage(page));
  },

  usersPage: async ({ page }, use) => {
    await use(new UsersPage(page));
  },

  // Pre-authenticated page (loads stored auth state)
  authenticatedPage: async ({ browser }, use) => {
    const context = await browser.newContext({
      storageState: '.auth/user.json',
    });
    const page = await context.newPage();
    await use(page);
    await context.close();
  },
});

export { expect } from '@playwright/test';
```

Use fixtures in tests:

```typescript
// tests/users.spec.ts
import { test, expect } from '../fixtures';

test('admin can view users list', async ({ authenticatedPage, usersPage }) => {
  usersPage.attachPage(authenticatedPage);
  await usersPage.goto();

  await expect(usersPage.userCount).toBeGreaterThan(0);
});

test('shows login error for invalid credentials', async ({ loginPage }) => {
  await loginPage.goto();
  await loginPage.login('bad@example.com', 'wrongpassword');

  expect(await loginPage.getErrorMessage()).toContain('Invalid credentials');
});
```

## Fixture Scope

Control how fixtures are initialized:

```typescript
export const test = base.extend<{}, WorkerFixtures>({
  // Worker scope — shared across all tests in the same worker
  // Useful for expensive setup (database seeding, server startup)
  dbConnection: [async ({}, use) => {
    const db = await DatabaseClient.connect(process.env.TEST_DB_URL!);
    await db.seed();
    await use(db);
    await db.cleanup();
    await db.disconnect();
  }, { scope: 'worker' }],
});
```

## Storing Auth State — global-setup.ts

Log in once and reuse the session across all tests to avoid logging in on every test:

```typescript
// tests/global-setup.ts
import { chromium, type FullConfig } from '@playwright/test';

export default async function globalSetup(config: FullConfig) {
  const { baseURL } = config.projects[0]!.use;
  const browser = await chromium.launch();
  const context = await browser.newContext();
  const page = await context.newPage();

  // Log in
  await page.goto(`${baseURL}/login`);
  await page.getByLabel('Email').fill(process.env.TEST_USER_EMAIL!);
  await page.getByLabel('Password').fill(process.env.TEST_USER_PASSWORD!);
  await page.getByRole('button', { name: 'Sign in' }).click();
  await page.waitForURL('/dashboard');

  // Save auth state to a file
  await context.storageState({ path: '.auth/user.json' });

  await browser.close();
}
```

```typescript
// playwright.config.ts
export default defineConfig({
  globalSetup: require.resolve('./tests/global-setup'),
  projects: [
    {
      name: 'chromium',
      use: {
        ...devices['Desktop Chrome'],
        // Use stored auth state for all tests in this project
        storageState: '.auth/user.json',
      },
    },
  ],
});
```

## Multiple Auth States

Support different user roles:

```typescript
// tests/global-setup.ts
async function saveAuthState(
  page: Page,
  email: string,
  password: string,
  path: string,
) {
  await page.goto('/login');
  await page.getByLabel('Email').fill(email);
  await page.getByLabel('Password').fill(password);
  await page.getByRole('button', { name: 'Sign in' }).click();
  await page.waitForURL('/dashboard');
  await page.context().storageState({ path });
}

export default async function globalSetup() {
  const browser = await chromium.launch();

  const adminContext = await browser.newContext();
  await saveAuthState(
    await adminContext.newPage(),
    process.env.ADMIN_EMAIL!,
    process.env.ADMIN_PASSWORD!,
    '.auth/admin.json',
  );

  const userContext = await browser.newContext();
  await saveAuthState(
    await userContext.newPage(),
    process.env.USER_EMAIL!,
    process.env.USER_PASSWORD!,
    '.auth/user.json',
  );

  await browser.close();
}
```

## test.use — Per-Test Options

Override options for a specific test or describe block:

```typescript
// Override options for all tests in this file
test.use({ viewport: { width: 375, height: 812 } }); // iPhone SE

describe('Mobile navigation', () => {
  test.use({ viewport: { width: 390, height: 844 } }); // iPhone 14

  test('shows hamburger menu on mobile', async ({ page }) => {
    await page.goto('/');
    await expect(page.getByRole('button', { name: 'Menu' })).toBeVisible();
    await expect(page.getByRole('navigation')).toBeHidden();
  });
});

// Use a specific timezone
test.use({ timezoneId: 'America/New_York' });

// Use a specific locale
test.use({ locale: 'fr-FR' });

// Use a specific color scheme
test.use({ colorScheme: 'dark' });
```

:::tip[Composing Page Objects and Fixtures]
The best pattern is to define page objects as plain classes (no Playwright-specific base), then inject them through fixtures. This makes page objects unit-testable and decoupled from the test runner.
:::

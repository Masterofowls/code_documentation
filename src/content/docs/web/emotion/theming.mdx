Emotion's `ThemeProvider` enables a design token system that flows through your entire component tree. Combined with TypeScript module augmentation, you get full type safety on every theme access.

## ThemeProvider Basics

```tsx
import { ThemeProvider } from '@emotion/react';

const theme = {
  colors: {
    primary: '#7c3aed',
    primaryHover: '#6d28d9',
    background: '#ffffff',
    surface: '#f9fafb',
    text: '#111827',
    textMuted: '#6b7280',
    border: '#e5e7eb',
  },
  spacing: {
    1: '0.25rem',
    2: '0.5rem',
    3: '0.75rem',
    4: '1rem',
    6: '1.5rem',
    8: '2rem',
    12: '3rem',
    16: '4rem',
  },
  borderRadius: {
    sm: '0.25rem',
    md: '0.5rem',
    lg: '0.75rem',
    xl: '1rem',
    full: '9999px',
  },
  typography: {
    fontSans: "'Inter', system-ui, sans-serif",
    fontMono: "'JetBrains Mono', monospace",
    sizeXs: '0.75rem',
    sizeSm: '0.875rem',
    sizeBase: '1rem',

    sizeXl: '1.25rem',
    size2xl: '1.5rem',

  return (
    <ThemeProvider theme={theme}>
      <Main />
    </ThemeProvider>
  );
}
```

## TypeScript — Declaring the Theme Type

Emotion uses module augmentation to let TypeScript know the shape of your theme:

```ts
// emotion.d.ts — create this file in your src/ directory
import '@emotion/react';
import type { AppTheme } from './theme';

declare module '@emotion/react' {
  // Augment the Theme interface with your actual theme shape
  export interface Theme extends AppTheme {}
}
```

This single declaration gives you autocomplete and type checking everywhere `theme` is used:

```tsx
import styled from '@emotion/styled';

// Theme is fully typed — TypeScript autocompletes all theme properties
const Button = styled.button`
  background-color: ${({ theme }) => theme.colors.primary};
  color: white;
  padding: ${({ theme }) => theme.spacing[2]} ${({ theme }) => theme.spacing[4]};
  border-radius: ${({ theme }) => theme.borderRadius.md};
  font-family: ${({ theme }) => theme.typography.fontSans};
  box-shadow: ${({ theme }) => theme.shadows.sm};

  &:hover {
    background-color: ${({ theme }) => theme.colors.primaryHover};
  }
`;
```

## The `useTheme` Hook

Access the theme inside any functional component:

```tsx
import { useTheme } from '@emotion/react';
import { css } from '@emotion/react';

function Card({ children }: { children: React.ReactNode }) {
  const theme = useTheme();

  return (
    <div
      css={{
        backgroundColor: theme.colors.surface,
        border: `1px solid ${theme.colors.border}`,
        borderRadius: theme.borderRadius.xl,
        padding: theme.spacing[6],
        boxShadow: theme.shadows.md,
      }}
    >
      {children}
    </div>
  );
}

function Avatar({ name, imageUrl }: { name: string; imageUrl?: string }) {
  const theme = useTheme();

  return (
    <div
      css={{
        width: 40,
        height: 40,
        borderRadius: theme.borderRadius.full,
        backgroundColor: theme.colors.primary,
        color: 'white',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        fontSize: theme.typography.sizeSm,
        fontWeight: 600,
        fontFamily: theme.typography.fontSans,
        overflow: 'hidden',
      }}
    >
      {imageUrl ? (
        <img
          src={imageUrl}
          alt={name}
          css={{ width: '100%', height: '100%', objectFit: 'cover' }}
        />
      ) : (
        name.slice(0, 2).toUpperCase()
      )}
    </div>
  );
}
```

2. **Build the ThemeProvider with toggle**

   ```tsx
   // ThemeContext.tsx
   import { createContext, useContext, useState, useCallback } from 'react';
   import { ThemeProvider as EmotionThemeProvider } from '@emotion/react';
   import { lightTheme, darkTheme, type AppTheme } from './theme';

   type ThemeMode = 'light' | 'dark';

   interface ThemeContextValue {
     mode: ThemeMode;
     toggle: () => void;
     setMode: (mode: ThemeMode) => void;
   }

   const ThemeContext = createContext<ThemeContextValue | null>(null);

   export function ThemeProvider({ children }: { children: React.ReactNode }) {
     const [mode, setMode] = useState<ThemeMode>(() => {
       if (typeof window === 'undefined') return 'light';
       const saved = localStorage.getItem('theme-mode') as ThemeMode;
       if (saved) return saved;
       return window.matchMedia('(prefers-color-scheme: dark)').matches
         ? 'dark'
         : 'light';
     });

     const toggle = useCallback(() => {
       setMode((prev) => {
         const next = prev === 'light' ? 'dark' : 'light';
         localStorage.setItem('theme-mode', next);
         return next;
       });
     }, []);

     const theme = mode === 'dark' ? darkTheme : lightTheme;

     return (
       <ThemeContext.Provider value={{ mode, toggle, setMode }}>
         <EmotionThemeProvider theme={theme}>{children}</EmotionThemeProvider>
       </ThemeContext.Provider>
     );
   }

   export function useThemeMode() {
     const ctx = useContext(ThemeContext);
     if (!ctx)
       throw new Error('useThemeMode must be used within ThemeProvider');
     return ctx;
   }
   ```

3. **Use in components**

## Nested ThemeProviders

You can nest `ThemeProvider` to apply a different theme to a sub-tree:

```tsx
import { ThemeProvider, useTheme } from '@emotion/react';

const darkPanelTheme = {
  background: '#1e293b',
  surface: '#334155',
  text: '#f1f5f9',
  muted: '#94a3b8',
  accent: '#a78bfa',
};

function DarkPanel({ children }: { children: React.ReactNode }) {
  return (
    <ThemeProvider
      theme={(outerTheme) => ({ ...outerTheme, ...darkPanelTheme })}
    >
      <div
        css={{
          backgroundColor: darkPanelTheme.background,
          color: darkPanelTheme.text,
          padding: '1.5rem',
          borderRadius: '1rem',
        }}
      >
        {children}
      </div>
    </ThemeProvider>
  );
}
```

The functional form of `ThemeProvider`'s `theme` prop receives the outer theme and lets you merge or extend it.

## Accessing Theme in the `css` Prop

When the `css` prop receives a function, it receives the theme as its first argument:

```tsx
/** @jsxImportSource @emotion/react */

function ThemedInput() {
  return (
    <input
      css={(theme) => ({
        width: '100%',
        padding: `${theme.spacing[2]} ${theme.spacing[3]}`,
        backgroundColor: theme.colors.surface,
        color: theme.colors.text,
        border: `1px solid ${theme.colors.border}`,
        borderRadius: theme.borderRadius.md,
        fontSize: theme.typography.sizeSm,
        fontFamily: theme.typography.fontSans,
        outline: 'none',
        transition: 'border-color 0.15s, box-shadow 0.15s',
        '&:focus': {
          borderColor: theme.colors.primary,
          boxShadow: `0 0 0 3px ${theme.colors.primaryMuted}`,
        },
      })}
    />
  );
}
```

## Global Styles with Theme

```tsx
import { Global } from '@emotion/react';
import { useTheme } from '@emotion/react';

function GlobalStyles() {
  const theme = useTheme();

  return (
    <Global
      styles={{
        '*, *::before, *::after': {
          boxSizing: 'border-box',
          margin: 0,
          padding: 0,
        },
        ':root': {
          fontFamily: theme.typography.fontSans,
          fontSize: '16px',
          lineHeight: 1.5,
        },
        body: {
          backgroundColor: theme.colors.background,
          color: theme.colors.text,
          minHeight: '100dvh',
          WebkitFontSmoothing: 'antialiased',
          MozOsxFontSmoothing: 'grayscale',
          transition: 'background-color 0.2s ease, color 0.2s ease',
        },
        'h1, h2, h3, h4, h5, h6': {
          lineHeight: 1.2,
          fontWeight: 700,
          color: theme.colors.text,
        },
        a: {
          color: theme.colors.primary,
          textDecoration: 'none',
          '&:hover': { textDecoration: 'underline' },
        },
        '::selection': {
          backgroundColor: theme.colors.primaryMuted,
          color: theme.colors.text,
        },
        ':focus-visible': {
          outline: `2px solid ${theme.colors.primary}`,
          outlineOffset: '2px',
        },
      }}
    />
  );
}
```

## Theme-Aware Animations

Keyframes can reference theme values via closure:

```tsx
import { keyframes, css } from '@emotion/react';
import { useTheme } from '@emotion/react';

function usePulseAnimation() {
  const theme = useTheme();

  const pulse = keyframes`
    0%, 100% {
      box-shadow: 0 0 0 0 ${theme.colors.primary}66;
    }
    50% {
      box-shadow: 0 0 0 8px ${theme.colors.primary}00;
    }
  `;

  return css`
    animation: ${pulse} 2s ease-in-out infinite;
  `;
}

function LiveIndicator() {
  const pulseStyle = usePulseAnimation();
  const theme = useTheme();

  return (
    <div css={{ display: 'flex', alignItems: 'center', gap: '0.5rem' }}>
      <div
        css={[
          {
            width: 8,
            height: 8,
            borderRadius: '50%',
            backgroundColor: theme.colors.success,
          },
          pulseStyle,
        ]}
      />
      <span
        css={{ fontSize: theme.typography.sizeSm, color: theme.colors.success }}
      >
        Live
      </span>
    </div>
  );
}
```

:::tip[Memoize theme-derived animations]
Keyframes that depend on theme values are regenerated when the theme changes. Wrap them in `useMemo` if the theme switches frequently to avoid unnecessary style recalculations:

```tsx
const pulseAnimation = useMemo(
  () => keyframes`
  0%, 100% { box-shadow: 0 0 0 0 ${theme.colors.primary}66; }
  50%       { box-shadow: 0 0 0 8px transparent; }
`,
  [theme.colors.primary],
);
```

:::

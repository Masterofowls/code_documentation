Modern browsers expose a rich set of APIs that go far beyond rendering HTML. These APIs let you store data locally, run code off the main thread, enable offline experiences, and interact with hardware.

## Web Storage API

Web Storage provides two simple key-value stores that persist data client-side:

| Feature         | `localStorage`           | `sessionStorage`        |
| --------------- | ------------------------ | ----------------------- |
| Persistence     | Until explicitly cleared | Until tab/window closes |
| Scope           | All tabs on same origin  | Current tab only        |
| Storage limit   | ~5–10MB                  | ~5MB                    |
| Accessible from | Any page on same origin  | Same tab only           |

```js
// localStorage — persists across sessions
localStorage.setItem('theme', 'dark');
localStorage.setItem('user', JSON.stringify({ id: 1, name: 'Jane' }));

const theme = localStorage.getItem('theme'); // 'dark'
const user = JSON.parse(localStorage.getItem('user')); // { id: 1, name: 'Jane' }

localStorage.removeItem('theme');
localStorage.clear(); // Remove everything

// sessionStorage — same API, cleared when tab closes
sessionStorage.setItem('cart-items', JSON.stringify([]));

// Listen for changes in other tabs
window.addEventListener('storage', (e) => {
  console.log('Key changed:', e.key);
  console.log('Old value:', e.oldValue);
  console.log('New value:', e.newValue);
  console.log('Origin:', e.url);
});
```

:::caution[Never Store Sensitive Data in Web Storage]
Web Storage is accessible by any JavaScript on the page. Never store authentication tokens, passwords, or personal data. Use HTTP-only cookies for authentication tokens.
:::

## Web Workers

Web Workers run JavaScript on a **background thread**, keeping the main thread free for UI updates:

```html
<!-- index.html -->
<button type="button" id="calculate">Calculate Fibonacci(45)</button>
<p id="result"></p>

<script>
  const btn = document.getElementById('calculate');
  const result = document.getElementById('result');

  // Create a new worker from a separate JS file
  const worker = new Worker('/workers/fibonacci.js');

  btn.addEventListener('click', () => {
    btn.disabled = true;
    result.textContent = 'Calculating…';

    // Send data to the worker
    worker.postMessage({ n: 45 });
  });

  // Receive results from the worker
  worker.addEventListener('message', (e) => {
    result.textContent = `Fibonacci(45) = ${e.data.result}`;
    btn.disabled = false;
  });

  worker.addEventListener('error', (e) => {
    console.error('Worker error:', e.message);
  });
</script>
```

```js
// workers/fibonacci.js
function fibonacci(n) {
  if (n <= 1) return n;
  return fibonacci(n - 1) + fibonacci(n - 2);
}

// Listen for messages from the main thread
self.addEventListener('message', (e) => {
  const { n } = e.data;
  const result = fibonacci(n);

  // Send result back to main thread
  self.postMessage({ result });
});
```

:::tip[Use Transferable Objects for Large Data]
Transferring large `ArrayBuffer` data between threads copies the data by default. Use `postMessage(data, [data])` to _transfer_ (not copy) ownership — this is O(1) regardless of data size.

```js
const buffer = new ArrayBuffer(1024 * 1024 * 100); // 100MB
worker.postMessage({ buffer }, [buffer]); // Transfer, not copy
```

:::

## Service Workers

Service Workers act as a **programmable network proxy** — intercepting requests to enable offline functionality, caching, and push notifications.

<Steps>
1. **Register** the service worker from your main thread:

```js
// In your main app code
if ('serviceWorker' in navigator) {
  window.addEventListener('load', async () => {
    try {
      const registration = await navigator.serviceWorker.register('/sw.js', {
        scope: '/',
      });
      console.log('Service Worker registered:', registration.scope);
    } catch (error) {
      console.error('Service Worker registration failed:', error);
    }
  });
}
```

2. **Handle the install event** — cache static assets:

```js
// sw.js
const CACHE_NAME = 'my-app-v1';
const STATIC_ASSETS = [
  '/',
  '/styles.css',
  '/app.js',
  '/offline.html',
  '/logo.svg',
];

self.addEventListener('install', (event) => {
  // Pre-cache static assets
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => cache.addAll(STATIC_ASSETS)),
  );
  // Activate immediately without waiting for old SW to go away
  self.skipWaiting();
});
```

3. **Handle the activate event** — clean up old caches:

```js
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches
      .keys()
      .then((cacheNames) =>
        Promise.all(
          cacheNames
            .filter((name) => name !== CACHE_NAME)
            .map((name) => caches.delete(name)),
        ),
      ),
  );
  // Take control of all open clients immediately
  self.clients.claim();
});
```

4. **Handle fetch events** — serve from cache, fall back to network:

```js
self.addEventListener('fetch', (event) => {
  // Skip non-GET requests
  if (event.request.method !== 'GET') return;

  event.respondWith(
    caches.match(event.request).then((cachedResponse) => {
      // Return cached version if available
      if (cachedResponse) return cachedResponse;

      // Otherwise fetch from network and cache the response
      return fetch(event.request)
        .then((response) => {
          if (
            !response ||
            response.status !== 200 ||
            response.type !== 'basic'
          ) {
            return response;
          }

          const responseToCache = response.clone();
          caches.open(CACHE_NAME).then((cache) => {
            cache.put(event.request, responseToCache);
          });

          return response;
        })
        .catch(() => {
          // Network failed — show offline page
          return caches.match('/offline.html');
        });
    }),
  );
});
```

</Steps>

## Notifications API

```js
async function requestNotificationPermission() {
  if (!('Notification' in window)) {
    console.warn('This browser does not support notifications');
    return false;
  }

  if (Notification.permission === 'granted') return true;
  if (Notification.permission === 'denied') return false;

  const permission = await Notification.requestPermission();
  return permission === 'granted';
}

async function showNotification(title, options = {}) {
  const granted = await requestNotificationPermission();
  if (!granted) return;

  const notification = new Notification(title, {
    body: options.body ?? '',
    icon: options.icon ?? '/icon-192.png',
    badge: '/badge.png',
    tag: options.tag ?? 'default', // Replace existing notification with same tag
    renotify: options.renotify ?? false, // Vibrate even if replacing existing
    silent: options.silent ?? false,
    data: options.data ?? {},
  });

  notification.addEventListener('click', () => {
    window.focus();
    notification.close();
  });
}

// Usage
showNotification('New message', {
  body: 'Jane Smith sent you a message',
  icon: '/avatars/jane.jpg',
  tag: 'chat-message',
  data: { chatId: 'abc123' },
});
```

## Geolocation API

```js
function getCurrentPosition() {
  return new Promise((resolve, reject) => {
    if (!navigator.geolocation) {
      reject(new Error('Geolocation not supported'));
      return;
    }

    navigator.geolocation.getCurrentPosition(resolve, reject, {
      enableHighAccuracy: true, // Use GPS if available (slower, more battery)
      timeout: 5000, // Fail after 5 seconds
      maximumAge: 60000, // Accept cached position up to 1 minute old
    });
  });
}

async function showUserLocation() {
  try {
    const position = await getCurrentPosition();
    const { latitude, longitude, accuracy } = position.coords;
    console.log(`Lat: ${latitude}, Lng: ${longitude}, Accuracy: ±${accuracy}m`);
  } catch (error) {
    if (error.code === error.PERMISSION_DENIED) {
      console.error('User denied geolocation');
    } else if (error.code === error.POSITION_UNAVAILABLE) {
      console.error('Location unavailable');
    } else if (error.code === error.TIMEOUT) {
      console.error('Geolocation timed out');
    }
  }
}

// Watch for ongoing position updates
const watchId = navigator.geolocation.watchPosition(
  (position) => console.log('Updated:', position.coords),
  (error) => console.error(error),
  { enableHighAccuracy: true },
);

// Stop watching
navigator.geolocation.clearWatch(watchId);
```

## Clipboard API

```js
// Write text to clipboard
async function copyToClipboard(text) {
  try {
    await navigator.clipboard.writeText(text);
    console.log('Copied to clipboard!');
  } catch (err) {
    console.error('Clipboard write failed:', err);
  }
}

// Read text from clipboard
async function readFromClipboard() {
  try {
    const text = await navigator.clipboard.readText();
    return text;
  } catch (err) {
    console.error('Clipboard read failed:', err);
    return null;
  }
}

// Copy image to clipboard
async function copyImageToClipboard(imageUrl) {
  const response = await fetch(imageUrl);
  const blob = await response.blob();
  await navigator.clipboard.write([new ClipboardItem({ [blob.type]: blob })]);
}

// Usage with UI feedback
document.querySelectorAll('[data-copy]').forEach((btn) => {
  btn.addEventListener('click', async () => {
    await copyToClipboard(btn.dataset.copy);
    btn.textContent = 'Copied!';
    setTimeout(() => (btn.textContent = 'Copy'), 2000);
  });
});
```

## Drag and Drop API

```html
<!-- Draggable items -->
<ul id="task-list">
  <li draggable="true" data-id="1">Task One</li>
  <li draggable="true" data-id="2">Task Two</li>
  <li draggable="true" data-id="3">Task Three</li>
</ul>

<!-- Drop zones -->
<div class="drop-zone" id="zone-todo" aria-label="To Do">To Do</div>
<div class="drop-zone" id="zone-doing" aria-label="In Progress">
  In Progress
</div>
<div class="drop-zone" id="zone-done" aria-label="Done">Done</div>
```

```js
let draggedItem = null;

document.querySelectorAll('[draggable]').forEach((item) => {
  item.addEventListener('dragstart', (e) => {
    draggedItem = item;
    e.dataTransfer.setData('text/plain', item.dataset.id);
    e.dataTransfer.effectAllowed = 'move';
    item.classList.add('dragging');
  });

  item.addEventListener('dragend', () => {
    item.classList.remove('dragging');
    draggedItem = null;
  });
});

document.querySelectorAll('.drop-zone').forEach((zone) => {
  zone.addEventListener('dragover', (e) => {
    e.preventDefault(); // Allow drop
    e.dataTransfer.dropEffect = 'move';
    zone.classList.add('drag-over');
  });

  zone.addEventListener('dragleave', () => {
    zone.classList.remove('drag-over');
  });

  zone.addEventListener('drop', (e) => {
    e.preventDefault();
    zone.classList.remove('drag-over');

    if (draggedItem) {
      zone.appendChild(draggedItem);
    }
  });
});
```

## History API

```js
// pushState: add entry to history without page reload
history.pushState(
  { page: 'products', filter: 'shoes' }, // State object
  '', // Title (mostly ignored)
  '/products?category=shoes', // URL
);

// replaceState: update current entry (no new history entry)
history.replaceState(
  { page: 'products', filter: 'boots' },
  '',
  '/products?category=boots',
);

// Listen for back/forward navigation
window.addEventListener('popstate', (e) => {
  if (e.state) {
    console.log('Navigated to:', e.state.page, 'with filter:', e.state.filter);
    // Re-render your UI based on state
    renderPage(e.state);
  }
});

// Navigate programmatically
history.back(); // Same as clicking Back
history.forward(); // Same as clicking Forward
history.go(-2); // Go back 2 entries
```

## Broadcast Channel API

Communicate between browser tabs / windows on the same origin:

```js
// In every tab that needs to communicate
const channel = new BroadcastChannel('app-sync');

// Send a message to all other tabs
channel.postMessage({ type: 'THEME_CHANGED', theme: 'dark' });

// Listen for messages from other tabs
channel.addEventListener('message', (e) => {
  const { type, ...data } = e.data;

  if (type === 'THEME_CHANGED') {
    applyTheme(data.theme);
  }

  if (type === 'USER_LOGGED_OUT') {
    redirectToLogin();
  }
});

// Clean up
channel.close();
```

## Page Visibility API

Detect when users switch tabs or minimise the window:

```js
document.addEventListener('visibilitychange', () => {
  if (document.hidden) {
    // Tab is hidden — pause non-essential work
    pauseVideoPlayback();
    stopPolling();
    document.title = '⏸ Paused | My App';
  } else {
    // Tab is visible again — resume
    resumeVideoPlayback();
    startPolling();
    document.title = 'My App';
  }
});

// Check current state
console.log(document.visibilityState); // 'visible' | 'hidden'
```

## Fullscreen API

```js
async function enterFullscreen(element = document.documentElement) {
  try {
    if (element.requestFullscreen) {
      await element.requestFullscreen();
    } else if (element.webkitRequestFullscreen) {
      await element.webkitRequestFullscreen(); // Safari
    }
  } catch (err) {
    console.error('Fullscreen failed:', err);
  }
}

async function exitFullscreen() {
  if (document.exitFullscreen) {
    await document.exitFullscreen();
  }
}

function toggleFullscreen(element) {
  if (!document.fullscreenElement) {
    enterFullscreen(element);
  } else {
    exitFullscreen();
  }
}

// Listen for fullscreen changes
document.addEventListener('fullscreenchange', () => {
  const isFullscreen = !!document.fullscreenElement;
  btn.setAttribute(
    'aria-label',
    isFullscreen ? 'Exit fullscreen' : 'Enter fullscreen',
  );
  btn.textContent = isFullscreen ? '⛶ Exit' : '⛶ Fullscreen';
});
```

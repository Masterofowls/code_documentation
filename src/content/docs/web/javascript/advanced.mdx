## The Event Loop in Depth

JavaScript runs on a single thread. The event loop is the mechanism that enables non-blocking I/O and asynchronous behavior.

```
┌─────────────────────────────────────────────────────────┐
│                      CALL STACK                          │
│   [currently executing function frames]                  │
└─────────────────────────────────────────────────────────┘
           ↑ event loop picks next task when empty
┌─────────────────────────────────────────────────────────┐
│                    MICROTASK QUEUE                        │
│   Promise .then/.catch/.finally callbacks                │
│   queueMicrotask() callbacks                             │
│   MutationObserver callbacks                             │
│   Runs to COMPLETION before next task                    │
└─────────────────────────────────────────────────────────┘
┌─────────────────────────────────────────────────────────┐
│                     TASK QUEUE (Macrotasks)               │
│   setTimeout / setInterval callbacks                     │
│   I/O callbacks (Node.js: fs, net)                       │
│   setImmediate (Node.js)                                 │
│   UI rendering events                                    │
└─────────────────────────────────────────────────────────┘
```

### Macrotasks vs Microtasks

```js
console.log('1 — synchronous start');

// Macrotask — goes to task queue
setTimeout(() => console.log('5 — setTimeout (macrotask)'), 0);

// Microtask — goes to microtask queue
Promise.resolve().then(() => console.log('3 — Promise microtask'));

queueMicrotask(() => console.log('4 — queueMicrotask'));

console.log('2 — synchronous end');

// output: 1 — synchronous start
// output: 2 — synchronous end
// output: 3 — Promise microtask        ← microtask queue runs first
// output: 4 — queueMicrotask           ← still microtask queue
// output: 5 — setTimeout (macrotask)  ← task queue runs after

// KEY RULE: after each macrotask, the entire microtask queue drains before
// the next macrotask runs (or before the browser renders a frame)
```

### Microtask Queue Starvation

```js
// Danger: infinite microtasks prevent ANY rendering or I/O
function recursiveMicrotask() {
  Promise.resolve().then(() => {
    // This creates a new microtask which creates another, forever
    recursiveMicrotask(); // Don't do this!
  });
}

// Compare with safe recursion via setTimeout
function recursiveMacrotask() {
  setTimeout(() => {
    doWork();
    recursiveMacrotask(); // OK — yields control between iterations
  }, 0);
}
```

### `queueMicrotask`

```js
// Explicit microtask scheduling — runs before the next task
// but after the current synchronous code

function processData(data) {
  let result = null;

  queueMicrotask(() => {
    // This runs after processData() returns but before any I/O
    if (result) notifyWatchers(result);
  });

  result = transformData(data);
  return result;
}

// Useful for deferred-but-not-async work
class EventEmitter {
  #listeners = new Map();
  #pendingEmits = [];

  emit(event, data) {
    // Batch all emits, fire after current sync code
    this.#pendingEmits.push({ event, data });
    queueMicrotask(() => this.#flush());
  }

  #flush() {
    const batch = [...this.#pendingEmits];
    this.#pendingEmits = [];
    for (const { event, data } of batch) {
      this.#listeners.get(event)?.forEach(fn => fn(data));
    }
  }
}
```

### `requestAnimationFrame`

```js
// Runs before the next browser paint — ~16ms (60fps) cadence
// Perfect for smooth animations

function animate(element, from, to, duration) {
  const start = performance.now();

  function step(timestamp) {
    const elapsed = timestamp - start;
    const progress = Math.min(elapsed / duration, 1);

    // Easing function (ease-in-out)
    const eased = progress < 0.5
      ? 2 * progress * progress
      : -1 + (4 - 2 * progress) * progress;

    const current = from + (to - from) * eased;
    element.style.transform = `translateX(${current}px)`;

    if (progress < 1) {
      requestAnimationFrame(step); // schedule next frame
    }
  }

  requestAnimationFrame(step);
}

animate(document.querySelector('.ball'), 0, 500, 1000);

// Cancel animation
const frameId = requestAnimationFrame(step);
cancelAnimationFrame(frameId); // stop it
```

### `setTimeout(fn, 0)` Explained

```js
// setTimeout(fn, 0) does NOT mean "immediately"
// It means "after the current call stack AND microtask queue are empty,
// put this callback in the task queue for the next available iteration"

console.log('a');
setTimeout(() => console.log('c — macrotask'), 0);
Promise.resolve().then(() => console.log('b — microtask'));
// output: a, b, c

// Practical use: defer DOM updates until after current synchronous work
function showLoadingAndThenWork() {
  element.textContent = 'Loading...'; // DOM mutation queued
  // Without setTimeout, 'Loading...' may never render — work blocks paint
  setTimeout(() => {
    heavyComputation(); // runs in next task — browser paints first
  }, 0);
}

// Historical trick: break up long synchronous tasks into chunks
function processChunks(items, chunkSize = 100) {
  function processChunk(index) {
    const end = Math.min(index + chunkSize, items.length);
    for (let i = index; i < end; i++) {
      processItem(items[i]);
    }
    if (end < items.length) {
      setTimeout(() => processChunk(end), 0); // yield to event loop
    }
  }
  processChunk(0);
}

// Modern alternative: scheduler API (where supported)
// scheduler.postTask(() => heavyWork(), { priority: 'background' });
```

## WeakRef and FinalizationRegistry

```js
// WeakRef — hold a reference without preventing garbage collection
class ImageCache {
  #cache = new Map(); // Map<url, WeakRef<ImageBitmap>>
  #registry = new FinalizationRegistry((url) => {
    console.log(`Cached image for ${url} was garbage collected`);
    this.#cache.delete(url); // clean up the dead ref
  });

  async get(url) {
    const ref = this.#cache.get(url);
    const cached = ref?.deref(); // may be undefined if GC'd

    if (cached) {
      console.log(`Cache hit: ${url}`);
      return cached;
    }

    const bitmap = await loadImageBitmap(url);
    const weakRef = new WeakRef(bitmap);
    this.#cache.set(url, weakRef);
    this.#registry.register(bitmap, url); // notify when GC'd
    return bitmap;
  }

  get size() {
    // Count still-live refs
    return [...this.#cache.values()]
      .filter(ref => ref.deref() !== undefined)
      .length;
  }
}

// FinalizationRegistry — callback when object is GC'd
const registry = new FinalizationRegistry((heldValue) => {
  console.log('Object was collected, cleanup for:', heldValue);
  releaseExternalResource(heldValue.resourceId);
});

let obj = { data: 'important' };
const resourceId = acquireExternalResource();
registry.register(obj, { resourceId }, obj); // third arg = unregister token

// Later, if we want to cancel the cleanup callback:
// registry.unregister(obj);

obj = null; // object can now be GC'd; callback will eventually fire
```

:::caution[Timing is Unpredictable]
`FinalizationRegistry` callbacks run at the garbage collector's discretion — not at a predictable time. Never rely on them for critical cleanup. Use them only for **additional** cleanup of external resources when explicit cleanup was missed.
:::

## Atomics and SharedArrayBuffer

For true shared memory between the main thread and workers:

```js
// main.js
const sharedBuffer = new SharedArrayBuffer(4 * Int32Array.BYTES_PER_ELEMENT);
const sharedArray = new Int32Array(sharedBuffer);

const worker = new Worker('worker.js');
worker.postMessage({ buffer: sharedBuffer });

// Atomic operations are thread-safe
Atomics.store(sharedArray, 0, 100); // atomically write 100 at index 0
const val = Atomics.load(sharedArray, 0); // atomically read index 0
console.log(val); // output: 100

// Atomic add — returns old value
const old = Atomics.add(sharedArray, 0, 5); // sharedArray[0] += 5
console.log(old); // output: 100
console.log(Atomics.load(sharedArray, 0)); // output: 105

// Atomic compare-and-swap — only swap if current value matches expected
const swapped = Atomics.compareExchange(sharedArray, 0, 105, 200);
// If sharedArray[0] === 105, set it to 200 and return 105; else return current
console.log(swapped); // output: 105 (swap succeeded)

// Thread synchronization with wait/notify
// worker.js
self.onmessage = ({ data: { buffer } }) => {
  const arr = new Int32Array(buffer);

  // Wait until index 0 is changed from 0 (blocks the worker thread)
  Atomics.wait(arr, 0, 0);
  console.log('Worker woke up! Value:', arr[0]);
};

// main.js — wake the worker
Atomics.store(sharedArray, 0, 42);
Atomics.notify(sharedArray, 0, 1); // wake 1 waiter
```

:::caution[Security Requirements]
`SharedArrayBuffer` requires the page to be cross-origin isolated. Set these headers:
```
Cross-Origin-Opener-Policy: same-origin
Cross-Origin-Embedder-Policy: require-corp
```
:::

## Regular Expressions

```js
// Basic flags
const re = /pattern/gimsuy;
//          g - global (find all matches)
//          i - case insensitive
//          m - multiline (^ and $ match line starts/ends)
//          s - dotAll (. matches newlines too)
//          u - unicode mode (correct surrogate pair handling)
//          y - sticky (match only at lastIndex position)

// Named capture groups (ES2018)
const dateRegex = /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/;
const match = '2024-06-15'.match(dateRegex);
console.log(match.groups.year);   // output: 2024
console.log(match.groups.month);  // output: 06
console.log(match.groups.day);    // output: 15

// Destructure groups directly
const { groups: { year, month, day } } = '2024-06-15'.match(dateRegex);
console.log(`${day}/${month}/${year}`); // output: 15/06/2024

// Replace with named groups
const reformatDate = (s) => s.replace(
  /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/,
  '$<day>/$<month>/$<year>'
);
console.log(reformatDate('2024-06-15')); // output: 15/06/2024
```

```js
// Lookahead and lookbehind assertions
// Lookahead: (?=...) match if followed by, (?!...) match if NOT followed by
const passwords = ['abc123', 'strongpass', 'P@ssw0rd!'];
const strongPassword = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@#$!%]).{8,}$/;
passwords.filter(p => strongPassword.test(p));
// output: ['P@ssw0rd!']

// Lookbehind: (?<=...) match if preceded by, (?<!...) match if NOT preceded by
'$100 and €200'.replace(/(?<=\$)\d+/, '***');
// output: '$***  and €200'  (only replaces number after $)

'$100 and €200'.match(/(?<=€)\d+/)[0]; // output: '200'

// matchAll — get all matches with groups
const text = 'Call 555-1234 or 555-5678 for info';
const phoneRe = /(?<area>\d{3})-(?<number>\d{4})/g;
const phones = [...text.matchAll(phoneRe)].map(m => m.groups);
console.log(phones);
// output: [{ area: '555', number: '1234' }, { area: '555', number: '5678' }]

// Replace with a function
const result = 'hello world'.replace(/\b\w/g, (match) => match.toUpperCase());
console.log(result); // output: Hello World

// Complex replace with groups
const camelToSnake = (s) =>
  s.replace(/([A-Z])/g, (_, char) => `_${char.toLowerCase()}`);
console.log(camelToSnake('camelCaseVariable')); // output: camel_case_variable

// Unicode property escapes (with /u flag)
const emoji = '😀 Hello 🌍';
const emojiRegex = /\p{Emoji}/gu;
console.log(emoji.match(emojiRegex)); // output: ['😀', '🌍']

const letterCount = 'Hello123!'.match(/\p{Letter}/gu)?.length;
console.log(letterCount); // output: 5
```

## Performance Measurement

```js
// performance.now() — high-resolution timestamp (microseconds)
const start = performance.now();
expensiveOperation();
const end = performance.now();
console.log(`Took ${(end - start).toFixed(3)}ms`); // output: Took 1.234ms

// Avoid Date.now() for performance measurement — lower resolution
// performance.now() is relative to page load, monotonically increasing
```

```js
// performance.mark and performance.measure — named measurements
performance.mark('parse-start');
parseData(rawData);
performance.mark('parse-end');

performance.mark('render-start');
renderComponents(parsedData);
performance.mark('render-end');

// Create measures between marks
performance.measure('parse-time', 'parse-start', 'parse-end');
performance.measure('render-time', 'render-start', 'render-end');
performance.measure('total-time', 'parse-start', 'render-end');

// Read measurements
const entries = performance.getEntriesByType('measure');
entries.forEach(entry => {
  console.log(`${entry.name}: ${entry.duration.toFixed(2)}ms`);
});
// output: parse-time: 12.34ms
// output: render-time: 8.56ms
// output: total-time: 20.90ms

// Get all marks
performance.getEntriesByType('mark').forEach(m => {
  console.log(`${m.name}: ${m.startTime.toFixed(2)}ms after page load`);
});

// Clean up
performance.clearMarks();
performance.clearMeasures();

// PerformanceObserver — observe performance entries as they occur
const observer = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    if (entry.duration > 50) {
      console.warn(`Slow ${entry.entryType}: ${entry.name} took ${entry.duration.toFixed(0)}ms`);
    }
  }
});

observer.observe({ entryTypes: ['measure', 'longtask', 'navigation', 'resource'] });
```

## Structured Concurrency Patterns

```js
// Task cancellation with AbortSignal
async function fetchWithCancellation(urls, signal) {
  const results = await Promise.allSettled(
    urls.map(url => fetch(url, { signal }).then(r => r.json()))
  );

  // If aborted, the AbortError propagates to allSettled
  return results.map(r => r.status === 'fulfilled' ? r.value : null);
}

// Racing with timeout
async function withTimeout(promise, ms, message = `Timed out after ${ms}ms`) {
  const timeout = new Promise((_, reject) =>
    setTimeout(() => reject(new Error(message)), ms)
  );
  return Promise.race([promise, timeout]);
}

// Using AbortSignal.timeout (modern API)
async function fetchWithSignalTimeout(url) {
  const signal = AbortSignal.timeout(5000); // built-in timeout signal
  const response = await fetch(url, { signal });
  return response.json();
}

// Throttle and debounce
function throttle(fn, limitMs) {
  let lastRun = 0;
  return function(...args) {
    const now = Date.now();
    if (now - lastRun >= limitMs) {
      lastRun = now;
      return fn.apply(this, args);
    }
  };
}

function debounce(fn, delayMs) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delayMs);
  };
}

const handleResize = throttle(() => {
  console.log('Resize handled at', performance.now().toFixed(0));
}, 100);

const handleSearch = debounce(async (query) => {
  const results = await searchAPI(query);
  displayResults(results);
}, 300);

window.addEventListener('resize', handleResize);
searchInput.addEventListener('input', e => handleSearch(e.target.value));
```

:::tip[Performance Best Practices]
1. Use `performance.now()` for timing — more precise than `Date.now()`.
2. Profile before optimizing — measure actual bottlenecks, don't guess.
3. Prefer `requestAnimationFrame` for visual updates — syncs with display refresh.
4. Use `scheduler.postTask()` (where available) for priority-based work scheduling.
5. Break up long tasks with `setTimeout(0)` or `scheduler.yield()` to keep the UI responsive.
:::

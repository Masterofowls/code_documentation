The Document Object Model (DOM) is a tree-based API that represents an HTML document as objects you can read and modify with JavaScript.

## The DOM Tree

```
document
└── html
    ├── head
    │   ├── title
    │   └── meta
    └── body
        ├── header
        │   └── h1 "Page Title"
        ├── main
        │   ├── section#hero
        │   │   └── p.intro "Welcome..."
        │   └── ul.list
        │       ├── li "Item 1"
        │       └── li "Item 2"
        └── footer
```

Every node in the tree is either an **Element**, **Text**, **Comment**, or **Document** node. JavaScript can traverse, modify, create, and delete any node.

## Querying Elements

```js
// querySelector — first matching element (CSS selector syntax)
const hero = document.querySelector('#hero');
const intro = document.querySelector('.intro');
const firstLi = document.querySelector('ul.list > li');
const submitBtn = document.querySelector('button[type="submit"]');
const checked = document.querySelector('input:checked');

// querySelectorAll — NodeList of ALL matching elements
const allLinks = document.querySelectorAll('a[href^="https"]');
const listItems = document.querySelectorAll('.list li');

// NodeList is array-like but not an Array — convert with spread or Array.from
const items = [...document.querySelectorAll('li')];
items.forEach((item) => console.log(item.textContent));

// Scoped query — search within a specific element only
const nav = document.querySelector('nav');
const navLinks = nav.querySelectorAll('a'); // only <a> inside nav

// getElementById — fastest lookup by ID
const modal = document.getElementById('modal');

// getElementsByClassName — live HTMLCollection
const cards = document.getElementsByClassName('card');

// getElementsByTagName — live HTMLCollection
const headings = document.getElementsByTagName('h2');
```

:::tip[querySelector vs getElementById]
`getElementById` is slightly faster and returns `null` for missing elements (same as `querySelector`). Use `querySelector` for complex selectors and when working with class-based or attribute selectors.
:::

## Traversing the DOM

```js
const list = document.querySelector('ul');

// Parent / children
console.log(list.parentElement); // the parent element
console.log(list.parentNode); // parent node (can be Document or Fragment)
console.log(list.children); // HTMLCollection of element children
console.log([...list.children]); // convert to array
console.log(list.childNodes); // NodeList including text nodes

// First / last child
console.log(list.firstElementChild); // first <li>
console.log(list.lastElementChild); // last <li>

// Siblings
const item = list.firstElementChild;
console.log(item.nextElementSibling); // next <li>
console.log(item.previousElementSibling); // null (first child)

// Closest ancestor matching a selector
const btn = document.querySelector('.btn-delete');
const card = btn.closest('.card'); // walks up the DOM tree
```

## Creating and Modifying Elements

```js
// createElement
const article = document.createElement('article');
const h2 = document.createElement('h2');
const p = document.createElement('p');

// Set content
h2.textContent = 'New Article'; // safe — no HTML parsing
p.textContent = 'Article content here.';

// Set attributes
article.id = 'new-article';
article.className = 'article featured';
article.setAttribute('data-author', 'Alice');
article.setAttribute('aria-label', 'Featured article');

// Build tree before inserting (minimize reflows)
article.appendChild(h2);
article.appendChild(p);

// Inserting into the document
const main = document.querySelector('main');
main.appendChild(article); // appends as last child
main.prepend(article); // inserts as first child
main.insertBefore(article, main.firstChild); // before a reference node

// insertAdjacentElement — precise placement
const reference = document.querySelector('#hero');
reference.insertAdjacentElement('beforebegin', article); // before the element
reference.insertAdjacentElement('afterbegin', article); // first child
reference.insertAdjacentElement('beforeend', article); // last child
reference.insertAdjacentElement('afterend', article); // after the element

// insertAdjacentHTML — insert raw HTML string (faster for complex structures)
reference.insertAdjacentHTML('beforeend', '<span class="badge">New</span>');

// replaceWith — replace an element
const oldEl = document.querySelector('.old');
const newEl = document.createElement('div');
newEl.className = 'new';
oldEl.replaceWith(newEl);

// remove — remove from DOM
document.querySelector('.to-remove')?.remove();

// cloneNode
const original = document.querySelector('.card');
const clone = original.cloneNode(true); // true = deep clone (includes children)
clone.id = 'card-clone';
main.appendChild(clone);
```

## dataset and Attributes

```js
// HTML: <div id="user" data-user-id="42" data-role="admin" data-is-verified="true">
const el = document.querySelector('#user');

// dataset maps data-* attributes using camelCase
console.log(el.dataset.userId);      // output: '42' (always a string)
console.log(el.dataset.role);        // output: 'admin'
console.log(el.dataset.isVerified);  // output: 'true'

// Set data attributes
el.dataset.lastSeen = new Date().toISOString();
el.dataset.score = '95';

// Delete
delete el.dataset.role;

// Generic attribute methods
el.getAttribute('data-user-id');    // '42'
el.setAttribute('hidden', '');      // add boolean attribute
el.removeAttribute('hidden');
el.hasAttribute('data-role');       // false (we deleted it)
el.toggleAttribute('disabled');     // toggle presence
````

## innerHTML vs textContent

:::caution[XSS Risk with innerHTML]
Never insert user-provided content via `innerHTML` — it parses and executes HTML, enabling cross-site scripting (XSS) attacks. Use `textContent` for plain text, or sanitize with a library like DOMPurify before using `innerHTML`.
:::

```js
const el = document.querySelector('#output');
const userInput = '<img src="x" onerror="alert(\'XSS\')">';

// UNSAFE — executes the onerror handler
el.innerHTML = userInput;

// SAFE — treats content as plain text, no HTML parsing
el.textContent = userInput;
// Renders: <img src="x" onerror="alert('XSS')"> (as visible text)

// For trusted HTML content
el.innerHTML = '<strong>Safe author-controlled content</strong>';

// outerHTML — includes the element itself
console.log(el.outerHTML); // output: <div id="output"><strong>...</strong></div>

// Reading content
console.log(el.textContent); // text content of element and all descendants
console.log(el.innerText); // like textContent but respects CSS visibility
```

## Event Listeners

```js
const btn = document.querySelector('#submit');

// Add event listener
function handleClick(event) {
  console.log('Clicked!');
  console.log(event.type); // output: click
  console.log(event.target); // element that was clicked
  console.log(event.currentTarget); // element listener is attached to
  console.log(event.clientX, event.clientY); // mouse coordinates
  console.log(event.timeStamp); // when the event occurred
}

btn.addEventListener('click', handleClick);

// Remove — must pass the SAME function reference
btn.removeEventListener('click', handleClick);

// Anonymous function — cannot be removed! Prefer named functions.
btn.addEventListener('click', () => console.log('cannot remove'));

// Options
btn.addEventListener('click', handleClick, { once: true }); // fires only once
btn.addEventListener('click', handleClick, { passive: true }); // won't call preventDefault
btn.addEventListener('scroll', handleScroll, { passive: true }); // improves scroll performance
btn.addEventListener('click', handleClick, { capture: true }); // fire in capture phase
```

### Preventing Default Behavior

```js
// Prevent form submission
const form = document.querySelector('form');
form.addEventListener('submit', (event) => {
  event.preventDefault(); // stop the browser from navigating
  const data = new FormData(form);
  console.log(Object.fromEntries(data));
});

// Prevent link navigation
const link = document.querySelector('a.modal-trigger');
link.addEventListener('click', (event) => {
  event.preventDefault();
  openModal(link.dataset.target);
});

// Prevent context menu
document.addEventListener('contextmenu', (event) => {
  event.preventDefault();
  showCustomMenu(event.clientX, event.clientY);
});
```

## Event Bubbling, Capturing, and Delegation

Events bubble up from the target element through its ancestors to the document root. Capturing is the reverse — top-down phase.

```js
// Bubbling example
document.querySelector('.child').addEventListener('click', (e) => {
  console.log('child clicked'); // output: 1st
});
document.querySelector('.parent').addEventListener('click', (e) => {
  console.log('parent clicked'); // output: 2nd (bubble)
});
document.addEventListener('click', (e) => {
  console.log('document clicked'); // output: 3rd
});

// Stop propagation — prevents further bubbling
document.querySelector('.child').addEventListener('click', (e) => {
  e.stopPropagation(); // parent and document listeners won't fire
});

// stopImmediatePropagation — also prevents other listeners on the SAME element
btn.addEventListener('click', (e) => {
  e.stopImmediatePropagation();
  console.log('This fires');
});
btn.addEventListener('click', () => {
  console.log('This does NOT fire');
});

// Capture phase — fires before bubbling
document.addEventListener(
  'click',
  (e) => {
    console.log('capturing!');
  },
  { capture: true },
); // fires FIRST, before child listeners
```

### Event Delegation

Attach one listener to a parent element and handle events from all descendants. Efficient for dynamic lists.

```js
// Instead of adding a listener to every button:
const list = document.querySelector('#todo-list');

list.addEventListener('click', (event) => {
  const btn = event.target.closest('[data-action]');
  if (!btn) return; // click was not on an action button

  const action = btn.dataset.action;
  const item = btn.closest('.todo-item');
  const id = item?.dataset.id;

  if (action === 'delete') deleteTodo(id);
  if (action === 'complete') completeTodo(id);
  if (action === 'edit') editTodo(id);
});

// Works for dynamically added items too — no need to re-attach listeners
function addTodo(text) {
  list.insertAdjacentHTML(
    'beforeend',
    `
    <li class="todo-item" data-id="${crypto.randomUUID()}">
      <span>${text}</span>
      <button data-action="complete">Done</button>
      <button data-action="edit">Edit</button>
      <button data-action="delete">Delete</button>
    </li>
  `,
  );
}
```

## MutationObserver

Watch for changes to the DOM — added/removed nodes, attribute changes, text changes.

```js
const observer = new MutationObserver((mutations) => {
  for (const mutation of mutations) {
    if (mutation.type === 'childList') {
      mutation.addedNodes.forEach((node) => {
        if (node.nodeType === Node.ELEMENT_NODE) {
          console.log('Added:', node.tagName); // output: Added: LI
        }
      });
      mutation.removedNodes.forEach((node) => {
        console.log('Removed:', node.textContent);
      });
    }

    if (mutation.type === 'attributes') {
      console.log(
        `Attribute "${mutation.attributeName}" changed to:`,
        mutation.target.getAttribute(mutation.attributeName),
      );
    }
  }
});

observer.observe(document.querySelector('#container'), {
  childList: true, // watch for added/removed children
  subtree: true, // watch all descendants, not just direct children
  attributes: true, // watch attribute changes
  attributeFilter: ['class', 'data-state'], // only watch specific attributes
  characterData: true, // watch text content changes
});

// Stop observing
observer.disconnect();
```

## IntersectionObserver

Efficiently detect when elements enter or leave the viewport — perfect for lazy loading and infinite scroll.

```js
const observer = new IntersectionObserver(
  (entries) => {
    entries.forEach((entry) => {
      if (entry.isIntersecting) {
        const img = entry.target;
        img.src = img.dataset.src; // lazy load the image
        observer.unobserve(img); // stop watching after loading
      }
    });
  },
  {
    root: null, // viewport
    rootMargin: '100px', // trigger 100px before entering viewport
    threshold: 0, // 0% visibility triggers callback
  },
);

// Observe all lazy images
document
  .querySelectorAll('img[data-src]')
  .forEach((img) => observer.observe(img));

// Infinite scroll
const sentinel = document.querySelector('#load-more-sentinel');
const scrollObserver = new IntersectionObserver(
  (entries) => {
    if (entries[0].isIntersecting) {
      loadNextPage();
    }
  },
  { threshold: 1.0 },
); // 100% visible

scrollObserver.observe(sentinel);
```

## CustomEvent

Dispatch and listen for custom application events.

```js
// Dispatch a custom event
const productCard = document.querySelector('.product-card');
productCard.addEventListener('click', () => {
  const event = new CustomEvent('product:added-to-cart', {
    bubbles: true, // bubbles up through the DOM
    cancelable: true,
    detail: {
      productId: productCard.dataset.productId,
      name: productCard.querySelector('h2').textContent,
      price: parseFloat(productCard.dataset.price),
    },
  });

  const cancelled = !productCard.dispatchEvent(event);
  if (!cancelled) {
    showAddedAnimation(productCard);
  }
});

// Listen anywhere in the tree (uses bubbling)
document.addEventListener('product:added-to-cart', (event) => {
  console.log('Product added:', event.detail.name);
  updateCartCount(event.detail);
});

// Allow listeners to cancel the action
document.addEventListener('product:added-to-cart', (event) => {
  if (isCartFull()) {
    event.preventDefault(); // cancels if cancelable: true
    showCartFullMessage();
  }
});
```

## Form Events

```js
const form = document.querySelector('#registration-form');

// submit — fires when form is submitted
form.addEventListener('submit', async (event) => {
  event.preventDefault();

  const data = Object.fromEntries(new FormData(form));
  console.log(data);
  // output: { username: 'alice', email: 'alice@example.com', ... }

  try {
    await submitRegistration(data);
    form.reset(); // clear all form fields
  } catch (err) {
    showError(err.message);
  }
});

// input — fires on every keystroke (real-time)
const emailInput = document.querySelector('#email');
emailInput.addEventListener('input', (event) => {
  const value = event.target.value;
  const isValid = /^[^@]+@[^@]+\.[^@]+$/.test(value);
  emailInput.classList.toggle('invalid', !isValid && value.length > 0);
});

// change — fires when value changes AND element loses focus
const select = document.querySelector('#country');
select.addEventListener('change', (event) => {
  updateRegionOptions(event.target.value);
});

// Checkbox and radio
const checkbox = document.querySelector('#agree');
checkbox.addEventListener('change', (event) => {
  console.log('Checked:', event.target.checked); // output: true or false
});

// focus / blur
emailInput.addEventListener('focus', () => emailInput.classList.add('focused'));
emailInput.addEventListener('blur', () => {
  emailInput.classList.remove('focused');
  validateEmail(emailInput);
});

// Keyboard events on inputs
emailInput.addEventListener('keydown', (event) => {
  if (event.key === 'Enter') form.requestSubmit(); // programmatically submit
  if (event.key === 'Escape') emailInput.blur();
});
```

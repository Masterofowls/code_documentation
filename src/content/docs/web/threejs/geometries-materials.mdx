In Three.js, a **geometry** defines shape (vertex positions, normals, UVs) and a **material** defines appearance (colour, roughness, reflectivity). Together they form a `Mesh`.

---

## Built-in Geometries

### Common Geometries

```js
import * as THREE from 'three';

// Box — (width, height, depth, widthSegs, heightSegs, depthSegs)
const box = new THREE.BoxGeometry(1, 1, 1);
const boxDetailed = new THREE.BoxGeometry(2, 2, 2, 4, 4, 4);

// Sphere — (radius, widthSegs, heightSegs, phiStart, phiLength, thetaStart, thetaLength)
const sphere    = new THREE.SphereGeometry(1, 32, 32);
const halfSphere = new THREE.SphereGeometry(1, 32, 16, 0, Math.PI * 2, 0, Math.PI / 2);

// Cylinder — (radiusTop, radiusBottom, height, radialSegs, heightSegs, openEnded)
const cylinder = new THREE.CylinderGeometry(0.5, 0.5, 2, 32);
const cone     = new THREE.CylinderGeometry(0, 1, 2, 32);   // cone
const tube     = new THREE.CylinderGeometry(1, 1, 3, 32, 1, true); // open tube

// Plane — (width, height, widthSegs, heightSegs)
const plane = new THREE.PlaneGeometry(10, 10);
const subdivPlane = new THREE.PlaneGeometry(10, 10, 50, 50); // for displacement

// Torus — (radius, tube, radialSegs, tubularSegs, arc)
const torus     = new THREE.TorusGeometry(1, 0.3, 16, 100);
const semiTorus = new THREE.TorusGeometry(1, 0.3, 16, 100, Math.PI);

// Torus Knot — (radius, tube, tubularSegs, radialSegs, p, q)
const torusKnot = new THREE.TorusKnotGeometry(1, 0.3, 200, 20, 2, 3);

// Cone — (radius, height, radialSegs, heightSegs, openEnded)
const coneGeo = new THREE.ConeGeometry(1, 2, 32);

// Circle — (radius, segments, thetaStart, thetaLength)
const circle = new THREE.CircleGeometry(1, 64);
const arc    = new THREE.CircleGeometry(1, 64, 0, Math.PI);

// Ring — (innerRadius, outerRadius, thetaSegs, phiSegs)
const ring = new THREE.RingGeometry(0.5, 1, 64);

// Dodecahedron, Icosahedron, Octahedron, Tetrahedron
const dodec  = new THREE.DodecahedronGeometry(1);
const icosa  = new THREE.IcosahedronGeometry(1, 2);   // detail level
const octa   = new THREE.OctahedronGeometry(1);
const tetra  = new THREE.TetrahedronGeometry(1);

// Capsule — (radius, length, capSegs, radialSegs)
const capsule = new THREE.CapsuleGeometry(0.5, 2, 8, 16);
```

### Lathe, Tube, and Extrude

```js
// LatheGeometry — revolve a 2D profile around the Y axis
const points = [];
for (let i = 0; i < 10; i++) {
  points.push(new THREE.Vector2(
    Math.sin(i * 0.2) * 0.5 + 0.5,
    (i - 5) * 0.2
  ));
}
const lathe = new THREE.LatheGeometry(points, 32);

// TubeGeometry — extrude geometry along a 3D curve
class HelixCurve extends THREE.Curve {
  getPoint(t) {
    const angle = t * Math.PI * 4;
    return new THREE.Vector3(
      Math.cos(angle),
      t * 3 - 1.5,
      Math.sin(angle),
    );
  }
}
const tubeGeo = new THREE.TubeGeometry(new HelixCurve(), 200, 0.1, 12, false);

// ShapeGeometry — fill a 2D shape
const shape = new THREE.Shape();
shape.moveTo(0, 0);
shape.lineTo(1, 0);
shape.lineTo(1, 1);
shape.lineTo(0, 1);
shape.closePath();
const shapeGeo = new THREE.ShapeGeometry(shape);

// ExtrudeGeometry — extrude a 2D shape into 3D
const extrudeSettings = {
  steps: 2,
  depth: 0.5,
  bevelEnabled: true,
  bevelThickness: 0.1,
  bevelSize: 0.1,
  bevelOffset: 0,
  bevelSegments: 5,
};
const extruded = new THREE.ExtrudeGeometry(shape, extrudeSettings);
```

### Custom BufferGeometry

```js
const geo = new THREE.BufferGeometry();

// 3 vertices of a triangle
const vertices = new Float32Array([
  -1, -1, 0,   // vertex 0
   1, -1, 0,   // vertex 1
   0,  1, 0,   // vertex 2
]);

const normals = new Float32Array([
  0, 0, 1,
  0, 0, 1,
  0, 0, 1,
]);

const uvs = new Float32Array([
  0, 0,
  1, 0,
  0.5, 1,
]);

// Indexed geometry — reuse vertices
const indices = new Uint16Array([0, 1, 2]);

geo.setAttribute('position', new THREE.BufferAttribute(vertices, 3));
geo.setAttribute('normal',   new THREE.BufferAttribute(normals,  3));
geo.setAttribute('uv',       new THREE.BufferAttribute(uvs,      2));
geo.setIndex(new THREE.BufferAttribute(indices, 1));

// Compute normals automatically if not set
geo.computeVertexNormals();

// Vertex colours
const colors = new Float32Array([
  1, 0, 0,  // red
  0, 1, 0,  // green
  0, 0, 1,  // blue
]);
geo.setAttribute('color', new THREE.BufferAttribute(colors, 3));
// Then use vertexColors: true on the material
```

## Materials

### MeshBasicMaterial

Unlit — ignores all lights. Fast but flat-looking. Use for debug, wireframes, or UI elements.

```js
const mat = new THREE.MeshBasicMaterial({
  color: '#4f46e5',
  wireframe: false,
  transparent: false,
  opacity: 1.0,
  side: THREE.FrontSide,        // THREE.BackSide | THREE.DoubleSide
  depthTest: true,
  depthWrite: true,
  vertexColors: false,
  fog: true,
});
```

### MeshStandardMaterial (PBR)

The standard PBR (Physically Based Rendering) material. Responds to all lights.

```js
const mat = new THREE.MeshStandardMaterial({
  // Base colour
  color: '#ffffff',
  map: colorTexture,           // albedo texture

  // Surface properties
  roughness: 0.5,              // 0 = mirror, 1 = fully diffuse
  roughnessMap: roughnessTexture,
  metalness: 0.0,              // 0 = dielectric, 1 = full metal
  metalnessMap: metalnessTexture,

  // Normal mapping
  normalMap: normalTexture,
  normalScale: new THREE.Vector2(1, 1),

  // Ambient occlusion
  aoMap: aoTexture,
  aoMapIntensity: 1.0,         // requires uv2 attribute on geometry

  // Emission (self-illuminated glow — not a light source)
  emissive: '#000000',
  emissiveMap: emissiveTexture,
  emissiveIntensity: 1.0,

  // Displacement (actually moves vertices)
  displacementMap: dispTexture,
  displacementScale: 0.1,
  displacementBias: 0,

  // Alpha / transparency
  transparent: true,
  opacity: 0.8,
  alphaMap: alphaTexture,
  alphaToCoverage: false,

  // Environment map (auto-set if scene.environment is set)
  envMap: envTexture,
  envMapIntensity: 1.0,

  side: THREE.FrontSide,
  wireframe: false,
  flatShading: false,
});
```

### MeshPhysicalMaterial

Extends `MeshStandardMaterial` with additional optical effects for high-fidelity rendering:

```js
const mat = new THREE.MeshPhysicalMaterial({
  // All MeshStandardMaterial properties, plus:

  // Transmission (glass / water)
  transmission: 1.0,           // 0 = opaque, 1 = fully transparent
  transmissionMap: null,
  ior: 1.5,                    // index of refraction (glass ≈ 1.5)
  thickness: 0.5,              // physical thickness for refraction
  thicknessMap: null,
  attenuationDistance: Infinity,
  attenuationColor: '#ffffff',

  // Clearcoat (car paint, varnished wood)
  clearcoat: 1.0,              // 0–1 intensity
  clearcoatMap: null,
  clearcoatRoughness: 0.1,
  clearcoatRoughnessMap: null,
  clearcoatNormalMap: null,

  // Iridescence (soap bubbles, beetle shells, oil slicks)
  iridescence: 1.0,
  iridescenceMap: null,
  iridescenceIOR: 1.3,
  iridescenceThicknessRange: [100, 400],

  // Sheen (velvet, fabric)
  sheen: 1.0,
  sheenColor: '#ffffff',
  sheenColorMap: null,
  sheenRoughness: 0.5,

  // Specular (for non-metallic specular intensity)
  specularIntensity: 1.0,
  specularColor: '#ffffff',
});
```

:::tip[MeshStandardMaterial vs MeshPhysicalMaterial]
`MeshStandardMaterial` covers 95% of use cases and is cheaper to render. Use `MeshPhysicalMaterial` only when you need glass/transmission, clearcoat, iridescence, or sheen effects.
:::

### MeshToonMaterial

Cartoon / cel-shading material with discrete shading steps:

```js
// Gradient texture controls shading steps
const gradientMap = new THREE.TextureLoader().load('/textures/gradients/5.jpg');
gradientMap.minFilter = THREE.NearestFilter;
gradientMap.magFilter = THREE.NearestFilter;

const mat = new THREE.MeshToonMaterial({
  color: '#4f46e5',
  gradientMap,
});
```

### ShaderMaterial

Full control with custom GLSL vertex and fragment shaders:

```js
const mat = new THREE.ShaderMaterial({
  uniforms: {
    uTime:  { value: 0.0 },
    uColor: { value: new THREE.Color('#4f46e5') },
  },
  vertexShader: /* glsl */`
    uniform float uTime;
    varying vec2 vUv;
    varying float vElevation;

    void main() {
      vUv = uv;
      vec3 pos = position;
      pos.z += sin(pos.x * 4.0 + uTime) * 0.2;
      pos.z += sin(pos.y * 4.0 + uTime) * 0.2;
      vElevation = pos.z;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
    }
  `,
  fragmentShader: /* glsl */`
    uniform vec3 uColor;
    varying vec2 vUv;
    varying float vElevation;

    void main() {
      float brightness = vElevation * 2.0 + 0.5;
      gl_FragColor = vec4(uColor * brightness, 1.0);
    }
  `,
  wireframe: false,
  transparent: false,
});

// Update uniform each frame
function animate() {
  requestAnimationFrame(animate);
  mat.uniforms.uTime.value = clock.getElapsedTime();
  renderer.render(scene, camera);
}
```

### Other Materials

```js
// Points (particle systems)
const pointsMat = new THREE.PointsMaterial({
  size: 0.05,
  color: '#ffffff',
  sizeAttenuation: true,     // smaller when farther away
  map: particleTexture,
  alphaMap: particleTexture,
  transparent: true,
  depthWrite: false,         // fixes sorting artefacts
  blending: THREE.AdditiveBlending,
});
const points = new THREE.Points(geometry, pointsMat);

// Lines
const lineMat = new THREE.LineBasicMaterial({ color: '#ff0000', linewidth: 1 });
const lineSeg = new THREE.LineSegments(geometry, lineMat);
const line    = new THREE.Line(geometry, lineMat);
const loop    = new THREE.LineLoop(geometry, lineMat);

// Wireframe helper (visual only, not a renderable mesh)
const helper = new THREE.WireframeGeometry(geometry);
const wireframe = new THREE.LineSegments(helper, new THREE.LineBasicMaterial({ color: '#00ff00' }));
scene.add(wireframe);
```

### Vertex Colors

```js
const geo = new THREE.BoxGeometry(1, 1, 1);
const count = geo.attributes.position.count;
const colors = new Float32Array(count * 3);

for (let i = 0; i < count; i++) {
  colors[i * 3]     = Math.random(); // R
  colors[i * 3 + 1] = Math.random(); // G
  colors[i * 3 + 2] = Math.random(); // B
}

geo.setAttribute('color', new THREE.BufferAttribute(colors, 3));

const mat = new THREE.MeshBasicMaterial({ vertexColors: true });
```

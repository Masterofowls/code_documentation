## Full `tsconfig.json` Reference

A production-ready `tsconfig.json` with every significant option documented:

```ts
// tsconfig.json
{
  "compilerOptions": {
    // ─── Language and Environment ────────────────────────────────
    "target": "ES2022",
    // JavaScript version for emitted output.
    // ES2022 is safe for Node.js 18+ and modern browsers.
    // Options: ES5 | ES2015 | ES2016 | ... | ES2022 | ESNext

    "lib": ["ES2022", "DOM", "DOM.Iterable"],
    // Built-in type declarations to include.
    // Defaults to target-appropriate set if omitted.

    "useDefineForClassFields": true,
    // Use Object.defineProperty for class fields (TC39 semantics).
    // Required for native private fields (#) to work correctly.

    // ─── Modules ─────────────────────────────────────────────────
    "module": "NodeNext",
    // Output module format.
    // Options: CommonJS | ESNext | NodeNext | Bundler | AMD | UMD

    "moduleResolution": "NodeNext",
    // Algorithm for resolving module specifiers.
    // Options: Node | Node16 | NodeNext | Bundler | Classic

    "baseUrl": ".",
    // Root directory for non-relative module name resolution.
    // Required if using "paths".

    "paths": {
      "@/*": ["src/*"]
    },
    // Import path aliases. Also configure in your bundler/runtime.

    "rootDir": "./src",
    // Constrains the input root. Prevents files outside src/ from
    // being compiled into dist/.

    "outDir": "./dist",
    // Destination directory for all emitted files.

    "rootDirs": ["src", "generated"],
    // Multiple source roots treated as a single virtual root.
    // Useful when code generation puts files outside src/.

    "resolveJsonModule": true,
    // Allow importing .json files:
    // import config from './config.json';

    "allowImportingTsExtensions": false,
    // Allow `import foo from './foo.ts'` (requires noEmit or bundler).

    // ─── JavaScript Support ──────────────────────────────────────
    "allowJs": false,
    // Allow .js and .jsx files to be compiled.
    // Useful for gradual migration from JS to TS.

    "checkJs": false,
    // Report errors in .js files (requires allowJs).

    // ─── Declaration Files ───────────────────────────────────────
    "declaration": true,
    // Emit .d.ts declaration files alongside .js output.
    // Required for publishing libraries.

    "declarationMap": true,
    // Emit .d.ts.map files enabling "Go to Definition" to jump
    // to .ts source instead of .d.ts declarations.

    "declarationDir": "./types",
    // Separate output directory for .d.ts files (optional).

    // ─── Source Maps ─────────────────────────────────────────────
    "sourceMap": true,
    // Emit .js.map files. Enables debugger breakpoints in .ts files.

    "inlineSourceMap": false,
    // Embed source maps in .js files instead of separate .map files.

    "inlineSources": false,
    // Include original .ts source inside source maps.

    // ─── Strict Type Checking ────────────────────────────────────
    "strict": true,
    // Master switch enabling ALL strict flags listed below.

    "noImplicitAny": true,
    // Error when TypeScript infers `any` type.
    // Example: function f(x) {} — x implicitly has type `any`.

    "strictNullChecks": true,
    // null and undefined are NOT assignable to other types.
    // Example: const s: string = null; // Error

    "strictFunctionTypes": true,
    // Enable strict checking of function type compatibility.
    // Makes function parameters contravariant (safer callbacks).

    "strictBindCallApply": true,
    // .bind(), .call(), .apply() are type-checked against the
    // target function's signature.

    "strictPropertyInitialization": true,
    // Class properties must be initialized in the constructor.
    // Example: class C { name: string; } // Error

    "noImplicitThis": true,
    // Error when `this` has type `any` in function bodies.

    "useUnknownInCatchVariables": true,
    // Catch clause variables are typed `unknown` instead of `any`.
    // Forces you to narrow before accessing properties.

    // ─── Additional Checks ───────────────────────────────────────
    "noUnusedLocals": true,
    // Error on unused local variables.

    "noUnusedParameters": true,
    // Error on unused function parameters.
    // Prefix with _ to suppress: function f(_unused: string) {}

    "noImplicitReturns": true,
    // Error when not all code paths return a value.
    // function f(x: number): string {
    //   if (x > 0) return 'positive'; // Error: missing return for else
    // }

    "noFallthroughCasesInSwitch": true,
    // Error on switch case fallthrough (missing break/return).

    "noUncheckedIndexedAccess": true,
    // Array and index signature access returns T | undefined.
    // const arr: number[] = [1, 2, 3];
    // const x = arr[0]; // type: number | undefined

    "exactOptionalPropertyTypes": true,
    // Optional properties may NOT be explicitly set to undefined.
    // interface I { name?: string }
    // const i: I = { name: undefined }; // Error (with exactOptional)
    // Without: allows setting optional props to undefined explicitly.

    "noPropertyAccessFromIndexSignature": false,
    // Force bracket notation for index-signature properties.
    // When true: obj.dynamic is Error; obj['dynamic'] is OK.

    // ─── Interop ─────────────────────────────────────────────────
    "esModuleInterop": true,
    // Enables default imports from CommonJS modules.
    // import React from 'react'; // OK (instead of * as React)

    "allowSyntheticDefaultImports": true,
    // Allow default imports from modules with no default export.
    // Implied by esModuleInterop.

    "forceConsistentCasingInFileNames": true,
    // Error when file imports differ in casing from disk.
    // Prevents bugs on case-insensitive filesystems (macOS, Windows).

    "isolatedModules": true,
    // Error when TypeScript features require type information from
    // other files (required for transpile-only tools: esbuild, swc).
    // Each file must be independently transpilable.

    // ─── Output Control ──────────────────────────────────────────
    "noEmit": false,
    // Do not emit output files. Useful for type-checking only.
    // npx tsc --noEmit

    "removeComments": false,
    // Strip comments from emitted JavaScript.

    "importHelpers": true,
    // Import tslib helpers instead of inlining them.
    // Reduces output size. Requires: npm install tslib

    "downlevelIteration": true,
    // Emit correct (but verbose) code for iterating ES5 targets.

    // ─── Type Checking Performance ───────────────────────────────
    "skipLibCheck": true,
    // Skip type-checking of all .d.ts files.
    // Significantly speeds up compilation. Safe for most projects.

    "incremental": true,
    // Enable incremental compilation. Saves state to tsbuildinfo.
    // Only recompiles changed files.

    "tsBuildInfoFile": "./.tsbuildinfo",
    // Where to store incremental compilation state.

    // ─── Experimental ────────────────────────────────────────────
    "experimentalDecorators": false,
    // Enable legacy (pre-TC39) decorators. Required for NestJS/Angular.

    "emitDecoratorMetadata": false
    // Emit design-type metadata for decorators. Requires reflect-metadata.
  },

  "include": ["src/**/*", "tests/**/*"],
  "exclude": ["node_modules", "dist", "coverage"]
}
```

## Strict Mode Options in Detail

### `noImplicitAny`

```ts
// ERROR — implicit any
function processItems(items) {
  // Parameter 'items' implicitly has an 'any' type.
  return items.map((item) => item.id); // Parameter 'item' implicitly has an 'any' type.
}

// CORRECT — explicit types
function processItems(items: Array<{ id: number }>): number[] {
  return items.map((item) => item.id);
}
```

### `strictNullChecks`

```ts
// Without strictNullChecks — null/undefined are assignable to everything
let name: string = null; // OK (dangerous!)
let count: number = undefined; // OK (dangerous!)

// With strictNullChecks — must be explicit
let name: string | null = null; // OK
let count: number | undefined; // OK
let required: string = null; // Error: Type 'null' not assignable to 'string'.

// Practical effect — forces proper null handling:
function greet(name: string | null): string {
  // name.toUpperCase(); // Error: Object is possibly 'null'.
  return name?.toUpperCase() ?? 'Anonymous'; // correct
}
```

### `strictFunctionTypes`

```ts
type Handler = (event: MouseEvent) => void;

function addHandler(h: Handler): void {
  /* ... */
}

// MouseEvent extends Event
const h1: (e: MouseEvent) => void = (e) => e.clientX; // OK
const h2: (e: Event) => void = (e) => {}; // OK

// Function parameters are contravariant with strictFunctionTypes:
addHandler(h1); // OK — MouseEvent is the expected type
addHandler(h2); // Error — (Event) => void is NOT assignable to (MouseEvent) => void
// h2 might not handle MouseEvent-specific properties
```

### `noUncheckedIndexedAccess`

```ts
// Without noUncheckedIndexedAccess:
const nums: number[] = [1, 2, 3];
const first: number = nums[0]; // type: number (may be undefined in practice)
const oob: number = nums[999]; // type: number — but undefined at runtime!

// With noUncheckedIndexedAccess:
const nums2: number[] = [1, 2, 3];
const first2 = nums2[0]; // type: number | undefined
if (first2 !== undefined) {
  console.log(first2.toFixed(2)); // OK — narrowed to number
}

// Record index signatures also affected:
const map: Record<string, number> = { a: 1 };
const val = map['a']; // type: number | undefined (not just number)
const val2 = map['b']; // type: number | undefined — correct!
```

### `exactOptionalPropertyTypes`

```ts
interface UserPrefs {
  theme?: 'light' | 'dark';
  language?: string;
}

// Without exactOptionalPropertyTypes:
const prefs: UserPrefs = {
  theme: undefined, // OK — undefined is treated as "absent"
  language: 'en',
};

// With exactOptionalPropertyTypes:
const prefs2: UserPrefs = {
  theme: undefined, // Error: Type 'undefined' is not assignable to type 'light' | 'dark'.
  // To explicitly omit, just don't include the key:
  language: 'en',
};
```

### `useUnknownInCatchVariables`

```ts
// Without useUnknownInCatchVariables:
try {
  JSON.parse('{bad json');
} catch (error) {
  console.log(error.message); // error is 'any' — no type safety
}

// With useUnknownInCatchVariables:
try {
  JSON.parse('{bad json');
} catch (error) {
  // error is 'unknown' — must narrow before use
  // console.log(error.message); // Error: Object is of type 'unknown'.

  if (error instanceof Error) {
    console.log(error.message); // OK — narrowed to Error
  } else {
    console.log(String(error)); // fallback for non-Error throws
  }
}
```

## Project References and Composite Builds

Project references enable **multi-package** monorepos where each package is compiled independently with proper incremental caching:

```ts
// packages/core/tsconfig.json
{
  "compilerOptions": {
    "composite": true,      // Required for project references
    "declaration": true,    // Required for project references
    "declarationMap": true,
    "outDir": "./dist",
    "rootDir": "./src"
  },
  "include": ["src"]
}
```

```ts
// packages/api/tsconfig.json
{
  "compilerOptions": {
    "composite": true,
    "outDir": "./dist",
    "rootDir": "./src"
  },
  "references": [
    { "path": "../core" }  // depends on core
  ],
  "include": ["src"]
}
```

```ts
// tsconfig.json (root — solution file)
{
  "files": [],  // no direct sources
  "references": [
    { "path": "packages/core" },
    { "path": "packages/api" },
    { "path": "packages/frontend" }
  ]
}
```

```bash
# Build all packages respecting dependency order
npx tsc --build

# Build with verbose output
npx tsc --build --verbose

# Clean all build artifacts and tsbuildinfo files
npx tsc --build --clean

# Type-check without emitting
npx tsc --build --noEmit
```

## Incremental Compilation

```ts
// tsconfig.json
{
  "compilerOptions": {
    "incremental": true,
    "tsBuildInfoFile": "./.tsbuildinfo"
    // .tsbuildinfo stores a fingerprint of all input/output files
    // Add .tsbuildinfo to .gitignore (it's a build artifact)
  }
}
```

```bash
# First build: full compilation
npx tsc   # ~3s for a large project

# Subsequent builds: only recompile changed files
npx tsc   # ~300ms — only changed files are reprocessed
```

## ESLint with `@typescript-eslint`

<Steps>
1. **Install packages**
   ```bash
   npm install --save-dev \
     eslint \
     @eslint/js \
     typescript-eslint \
     @typescript-eslint/eslint-plugin \
     @typescript-eslint/parser
   ```

2. **Create `eslint.config.mjs`**

   ```ts
   // eslint.config.mjs
   import eslint from '@eslint/js';
   import tseslint from 'typescript-eslint';

   export default tseslint.config(
     eslint.configs.recommended,
     ...tseslint.configs.strictTypeChecked,
     ...tseslint.configs.stylisticTypeChecked,
     {
       languageOptions: {
         parserOptions: {
           project: true, // use tsconfig.json for type info
           tsconfigRootDir: import.meta.dirname,
         },
       },
       rules: {
         // ── TypeScript-specific rules ──────────────────────────
         '@typescript-eslint/no-explicit-any': 'error',
         '@typescript-eslint/no-unsafe-assignment': 'error',
         '@typescript-eslint/no-unsafe-call': 'error',
         '@typescript-eslint/no-unsafe-member-access': 'error',
         '@typescript-eslint/no-unsafe-return': 'error',
         '@typescript-eslint/no-unused-vars': [
           'error',
           {
             argsIgnorePattern: '^_',
             varsIgnorePattern: '^_',
           },
         ],
         '@typescript-eslint/prefer-nullish-coalescing': 'error',
         '@typescript-eslint/prefer-optional-chain': 'error',
         '@typescript-eslint/consistent-type-imports': [
           'error',
           {
             prefer: 'type-imports',
             fixStyle: 'inline-type-imports',
           },
         ],
         '@typescript-eslint/consistent-type-exports': 'error',
         '@typescript-eslint/await-thenable': 'error',
         '@typescript-eslint/no-floating-promises': 'error',
         '@typescript-eslint/no-misused-promises': 'error',
         '@typescript-eslint/require-await': 'error',

         // ── Base rules (override TS-aware versions) ────────────
         'no-unused-vars': 'off', // use @typescript-eslint version
         'require-await': 'off', // use @typescript-eslint version
       },
     },
   );
   ```

3. **Add scripts to `package.json`**

   ```ts
   // package.json
   {
     "scripts": {
       "lint": "eslint .",
       "lint:fix": "eslint . --fix"
     }
   }
   ```

4. **Run ESLint**
   ```bash
   npm run lint
   npm run lint:fix
   ```
   </Steps>

npm install --save-dev prettier eslint-config-prettier
```

// prettier.config.mjs
/** @type {import('prettier').Config} */
const config = {
  printWidth: 80,
  tabWidth: 2,
  useTabs: false,
  semi: true,
  singleQuote: true,
  quoteProps: 'as-needed',

  arrowParens: 'always',
  endOfLine: 'lf',
};

```

```ts
// eslint.config.mjs — add prettier last to disable conflicting rules
import prettierConfig from 'eslint-config-prettier';

export default tseslint.config(
  eslint.configs.recommended,
  ...tseslint.configs.strictTypeChecked,
  prettierConfig, // must be last
  {
    rules: {
      /* your custom rules */
    },
  },
);
```

```bash
# Check formatting
npx prettier --check .

# Fix formatting
npx prettier --write .

# Format + lint in one pass
npm run lint:fix && npx prettier --write .
```

## Testing with TypeScript

### ts-jest

```bash
npm install --save-dev jest ts-jest @types/jest
```

```ts
// jest.config.ts
import type { Config } from 'jest';

const config: Config = {
  preset: 'ts-jest',
  testEnvironment: 'node',
  roots: ['<rootDir>/src', '<rootDir>/tests'],
  testMatch: ['**/*.test.ts', '**/*.spec.ts'],
  transform: {
    '^.+\\.tsx?$': [
      'ts-jest',
      {
        tsconfig: {
          strict: true,
          noUnusedLocals: false, // relax for tests
        },
      },
    ],
  },
  moduleNameMapper: {
    '^@/(.*)$': '<rootDir>/src/$1',
  },
  coverageThreshold: {
    global: {
      branches: 80,
      functions: 80,
      lines: 80,
      statements: 80,
    },
  },
  collectCoverageFrom: ['src/**/*.ts', '!src/**/*.d.ts', '!src/**/index.ts'],
};

export default config;
```

```ts
// src/math.test.ts
import { add, multiply } from './math';

describe('math utilities', () => {
  describe('add', () => {
    it('should add two positive numbers', () => {
      expect(add(2, 3)).toBe(5);
    });

    it('should handle negative numbers', () => {
      expect(add(-1, 1)).toBe(0);
    });
  });

  describe('multiply', () => {
    it.each([
      [2, 3, 6],
      [0, 5, 0],
      [-2, 3, -6],
    ])('multiply(%i, %i) === %i', (a, b, expected) => {
      expect(multiply(a, b)).toBe(expected);
    });
  });
});
```

## Bundler Comparison

| Feature            | **Vite**    | **esbuild**      | **SWC**           | **tsc**     |
| ------------------ | ----------- | ---------------- | ----------------- | ----------- |
| **Type-checks?**   | ❌ No       | ❌ No            | ❌ No             | ✅ Yes      |
| **Speed**          | Very fast   | Fastest          | Very fast         | Moderate    |
| **Language**       | JS/Rust     | Go               | Rust              | TypeScript  |
| **Dev server**     | ✅ HMR      | ❌ Separate tool | ❌ Separate tool  | ❌ No       |
| **Tree-shaking**   | ✅ Yes      | ✅ Yes           | Partial           | ❌ No       |
| **Code splitting** | ✅ Yes      | ✅ Yes           | ❌ No             | ❌ No       |
| **CSS handling**   | ✅ Built-in | ❌ Plugin        | ❌ Plugin         | ❌ No       |
| **Decorators**     | ✅ Both     | Experimental     | Experimental      | ✅ Both     |
| **Primary use**    | Frontend    | Node bundles     | Babel replacement | Type safety |

### Vite Configuration

```ts
// vite.config.ts
import { defineConfig } from 'vite';
import path from 'path';

export default defineConfig({
  build: {
    target: 'ES2022',
    outDir: 'dist',
    lib: {
      entry: path.resolve(__dirname, 'src/index.ts'),
      name: 'MyLib',
      formats: ['es', 'cjs'],
      fileName: (format) => `index.${format}.js`,
    },
    rollupOptions: {
      external: ['react', 'react-dom'],
    },
  },
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
  },
});
```

### esbuild Configuration

```ts
// build.mjs
import * as esbuild from 'esbuild';

await esbuild.build({
  entryPoints: ['src/index.ts'],
  bundle: true,
  outdir: 'dist',
  format: 'esm',
  target: 'node18',
  platform: 'node',
  sourcemap: true,
  external: ['express', 'pg'], // don't bundle these
  define: {
    'process.env.NODE_ENV': '"production"',
  },
});
```

:::tip[Recommended Stack]
For a **Node.js library** or API: `tsc --noEmit` for type-checking in CI, `esbuild` or `tsup` for bundling.

For a **React frontend**: Vite for development and production bundling, `tsc --noEmit` in CI.

For a **full-stack monorepo**: TypeScript project references for the TS layer, Vite for frontend, esbuild for backend services.
:::

## CI Type-Checking Pipeline

```ts
// .github/workflows/typecheck.yml
// Note: This is YAML — shown as ts for syntax highlighting
{
  "name": "CI",
  "on": { "push": { "branches": ["main"] }, "pull_request": {} },
  "jobs": {
    "typecheck": {
      "runs-on": "ubuntu-latest",
      "steps": [
        { "uses": "actions/checkout@v4" },
        { "uses": "actions/setup-node@v4", "with": { "node-version": "20", "cache": "npm" } },
        { "run": "npm ci" },
        { "run": "npx tsc --noEmit", "name": "Type-check" },
        { "run": "npm run lint", "name": "Lint" },
        { "run": "npm test -- --coverage", "name": "Test" },
        { "run": "npm run build", "name": "Build" }
      ]
    }
  }
}
```

```bash
# Local pre-commit checks (add to .husky/pre-commit)
npx tsc --noEmit && npm run lint && npm test
```
